153 // AdaptiveTimeScaleProbe::calcValues calls targetProbe->getValues() and passes the result to
154 // mAdaptiveTimeScaleController->calcTimesteps() to use as timeScaleTrue.
155 // mAdaptiveTimeScaleController->calcTimesteps() returns timeScale and copies the result into
156 // probeValues. AdaptiveTimeScaleProbe also processes the triggering and only reads the
157 // mAdaptiveTimeScaleController when triggering doesn't happen.

159 int AdaptiveTimeScaleProbe::calcValues(double timeValue) {
160  std::vector<double> rawProbeValues;
161  if (triggerLayer != nullptr
162  && triggerLayer->needUpdate(timeValue + triggerOffset, parent->getDeltaTime())) {
163  rawProbeValues.assign(getNumValues(), -1.0);
164  }
165  else {
166  mTargetProbe->getValues(timeValue, &rawProbeValues);
167  }
168  pvAssert(rawProbeValues.size() == getNumValues()); // In allocateDataStructures, we checked that
169  // mTargetProbe has a compatible size.
170  std::vector<double> timeSteps =
171  mAdaptiveTimeScaleController->calcTimesteps(timeValue, rawProbeValues);
172  memcpy(getValuesBuffer(), timeSteps.data(), sizeof(double) * getNumValues());
173  return PV_SUCCESS;
174 }
