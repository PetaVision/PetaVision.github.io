108 int KernelProbe::outputState(double timed) {
109  InterColComm * icComm = parent->icCommunicator();
110  const int rank = icComm->commRank();
111  if( rank != 0 ) return PV_SUCCESS;
112  assert(getTargetConn()!=NULL);
113  int nxp = getTargetHyPerConn()->xPatchSize();
114  int nyp = getTargetHyPerConn()->yPatchSize();
115  int nfp = getTargetHyPerConn()->fPatchSize();
116  int patchSize = nxp*nyp*nfp;
117 
118  const pvwdata_t * wdata = getTargetHyPerConn()->get_wDataStart(arborID)+patchSize*kernelIndex;
119  const pvwdata_t * dwdata = outputPlasticIncr ?
120  getTargetHyPerConn()->get_dwDataStart(arborID)+patchSize*kernelIndex : NULL;
121  fprintf(outputstream->fp, "Time %f, Conn \"%s\", nxp=%d, nyp=%d, nfp=%d\n",
122  timed, getTargetConn()->getName(),nxp, nyp, nfp);
123  for(int f=0; f<nfp; f++) {
124  for(int y=0; y<nyp; y++) {
125  for(int x=0; x<nxp; x++) {
126  int k = kIndex(x,y,f,nxp,nyp,nfp);
127  fprintf(outputstream->fp, " x=%d, y=%d, f=%d (index %d):", x, y, f, k);
128  if(getOutputWeights()) {
129  fprintf(outputstream->fp, " weight=%f", (float)wdata[k]);
130  }
131  if(getOutputPlasticIncr()) {
132  fprintf(outputstream->fp, " dw=%f", (float)dwdata[k]);
133  }
134  fprintf(outputstream->fp,"\n");
135  }
136  }
137  }
138 
139  return PV_SUCCESS;
140 }

142 int KernelProbe::patchIndices(HyPerConn * conn) {
143  int nxp = conn->xPatchSize();
144  int nyp = conn->yPatchSize();
145  int nfp = conn->fPatchSize();
146  int nPreExt = conn->getNumWeightPatches();
147  assert(nPreExt == conn->preSynapticLayer()->getNumExtended());
148  const PVLayerLoc * loc = conn->preSynapticLayer()->getLayerLoc();
149  const PVHalo * halo = &loc->halo;
150  int nxPre = loc->nx;
151  int nyPre = loc->ny;
152  int nfPre = loc->nf;
153  int nxPreExt = nxPre+loc->halo.lt+loc->halo.rt;
154  int nyPreExt = nyPre+loc->halo.dn+loc->halo.up;
155  for( int kPre = 0; kPre < nPreExt; kPre++ ) {
156  PVPatch * w = conn->getWeights(kPre,arborID);
157  int xOffset = kxPos(w->offset, nxp, nyp, nfp);
158  int yOffset = kyPos(w->offset, nxp, nyp, nfp);
159  int kxPre = kxPos(kPre,nxPreExt,nyPreExt,nfPre)-loc->halo.lt;
160  int kyPre = kyPos(kPre,nxPreExt,nyPreExt,nfPre)-loc->halo.up;
161  int kfPre = featureIndex(kPre,nxPreExt,nyPreExt,nfPre);
162  fprintf(outputstream->fp," presynaptic neuron %d (x=%d, y=%d, f=%d) uses kernel index %d, starting at x=%d, y=%d\n",
163  kPre, kxPre, kyPre, kfPre, conn->patchIndexToDataIndex(kPre), xOffset, yOffset);
164  }
165  return PV_SUCCESS;
166 }
