15 Publisher::Publisher(MPIBlock const &mpiBlock, PVLayerCube *cube, int numLevels, bool isSparse) {
16  this->mLayerCube = cube;

18  int const numBuffers = cube->loc.nbatch;
19  int const numItems = cube->numItems / numBuffers; // number of items in one batch element.
20 
21  store = new DataStore(numBuffers, numItems, numLevels, isSparse);
22 
23  mBorderExchanger = new BorderExchange(mpiBlock, cube->loc);

25  mpiRequestsBuffer = new RingBuffer<std::vector<MPI_Request>>(numLevels, 1);

79 int Publisher::publish(double lastUpdateTime) {
80  //
81  // Everyone publishes border region to neighbors even if no subscribers.
82  // This means that everyone should wait as well.
83  //
84 
85  size_t dataSize = mLayerCube->numItems * sizeof(float);
86 
87  float const *sendBuf = mLayerCube->data;
88  float *recvBuf = recvBuffer(0); // Grab all of the buffer, allocated continuously
89 
90  memcpy(recvBuf, sendBuf, dataSize);
91  exchangeBorders(&mLayerCube->loc, 0);
92  store->setLastUpdateTime(0 /*bufferId*/, lastUpdateTime);
93 
94  for (int b = 0; b < store->getNumBuffers(); b++) {
95  store->markActiveIndicesOutOfSync(b, 0);
96  }
97  // Updating active indices is done after MPI wait in HyPerCol
98  // to avoid race condition because exchangeBorders mpi is async
99 
100  return PV_SUCCESS;
101 }

113 int Publisher::exchangeBorders(const PVLayerLoc *loc, int delay /*default 0*/) {
114  PVHalo const *halo = &loc->halo;
115  if (halo->lt == 0 && halo->rt == 0 && halo->dn == 0 && halo->up == 0) {
116  return PV_SUCCESS;
117  }
118  int status = PV_SUCCESS;
119 
120 #ifdef PV_USE_MPI
121  auto *requestsVector = mpiRequestsBuffer->getBuffer(delay, 0);
122  pvAssert(requestsVector->empty());
123 
124  // Loop through batch.
125  // The loop over batch elements probably belongs inside
126  // BorderExchange::exchange(), but for this to happen, exchange() would need
127  // to know how its data argument is organized with respect to batching.
128  int exchangeVectorSize = 2 * (mBorderExchanger->getNumNeighbors() - 1);
129  for (int b = 0; b < loc->nbatch; b++) {
130  // don't send interior
131  pvAssert(requestsVector->size() == b * exchangeVectorSize);
132 
133  float *data = recvBuffer(b, delay);
134  std::vector<MPI_Request> batchElementMPIRequest{};
135  mBorderExchanger->exchange(data, batchElementMPIRequest);
136  pvAssert(batchElementMPIRequest.size() == exchangeVectorSize);
137  requestsVector->insert(
138  requestsVector->end(), batchElementMPIRequest.begin(), batchElementMPIRequest.end());
139  pvAssert(requestsVector->size() == (b + 1) * exchangeVectorSize);
140  }
141 
142 #endif // PV_USE_MPI
143 
144  return status;
145 }

147 int Publisher::isExchangeFinished(int delay /* default 0*/) {
148  bool isReady;
149  auto *requestsVector = mpiRequestsBuffer->getBuffer(delay, 0);
150  if (requestsVector->empty()) {
151  isReady = true;
152  }
153  else {
154  int test;
155  MPI_Testall((int)requestsVector->size(), requestsVector->data(), &test, MPI_STATUSES_IGNORE);
156  if (test) {
157  requestsVector->clear();
158  updateActiveIndices(delay);
159  }
160  isReady = (bool)test;
161  }
162  return isReady;
163 }

168 int Publisher::wait(int delay /*default 0*/) {
169 #ifdef DEBUG_OUTPUT
170  InfoLog().printf(
171  "[%2d]: waiting for data, num_requests==%d\n", mBorderExchanger->getRank(), numRemote);
172  InfoLog().flush();
173 #endif // DEBUG_OUTPUT
174 
175  auto *requestsVector = mpiRequestsBuffer->getBuffer(delay, 0);
176  if (!requestsVector->empty()) {
177  mBorderExchanger->wait(*requestsVector);
178  pvAssert(requestsVector->empty());
179  }
180  updateActiveIndices(delay);
181 
182  return 0;
183 }
