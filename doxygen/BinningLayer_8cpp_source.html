105  if (originalLayer->getInitInfoCommunicatedFlag()==false) {

117  getKeyword(), name, originalLayerName);
118  errorMessage.printf(" original (nx=%d, ny=%d) versus (nx=%d, ny=%d)\n",
119  srcLoc->nxGlobal, srcLoc->nyGlobal, loc->nxGlobal, loc->nyGlobal);
120  }
121  MPI_Barrier(parent->icCommunicator()->communicator());
122  exit(EXIT_FAILURE);

170 int BinningLayer::doUpdateState(double timed, double dt, const PVLayerLoc * origLoc, const PVLayerLoc * currLoc, const pvdata_t * origData, pvdata_t * currA, float binMax, float binMin) {
171  int status = PV_SUCCESS;
172  //update_timer->start();
173  int numBins = currLoc->nf;
174 
175  int nx = currLoc->nx;
176  int ny = currLoc->ny;
177  //Check that both nb are the same
178  assert(origLoc->halo.lt == currLoc->halo.lt &&
179  origLoc->halo.rt == currLoc->halo.rt &&
180  origLoc->halo.dn == currLoc->halo.dn &&
181  origLoc->halo.up == currLoc->halo.up);
182  assert(origLoc->nf == 1);
183  PVHalo const * halo = &origLoc->halo;
184  float binRange = binMax - binMin;
185  float stepSize = float(binRange)/numBins;
186  int nbatch = currLoc->nbatch;
187 
188 
189 
190  for(int b = 0; b < nbatch; b++){
191  const pvdata_t * origDataBatch = origData + b * (origLoc->nx + origLoc->halo.lt + origLoc->halo.rt) * (origLoc->ny + origLoc->halo.up + origLoc->halo.dn) * origLoc->nf;
192  pvdata_t * currABatch = currA + b * (currLoc->nx + currLoc->halo.lt + currLoc->halo.rt) * (currLoc->ny + currLoc->halo.up + currLoc->halo.dn) * currLoc->nf;
193 
194  // each y value specifies a different target so ok to thread here (sum, sumsq are defined inside loop)
195 #ifdef PV_USE_OPENMP_THREADS
196 #pragma omp parallel for
197 #endif
198  for (int iY = 0; iY < (ny+halo->dn+halo->up); iY++){
199  for (int iX = 0; iX < (nx+halo->lt+halo->rt); iX++){

201  float inVal = origDataBatch[origIdx];
202  //If inVal is out of bounds in either binMax or binMin, set the value to be the maximum or minimum val
203  if(inVal < binMin){
204  inVal = binMin;
205  }
206  if(inVal > binMax){
207  inVal = binMax;
208  }
209 
210  if(zeroDCR && inVal == 0){
211  for(int iF = 0; iF < numBins; iF++){
212  int currIdx = kIndex(iX, iY, iF, nx+halo->lt+halo->rt, ny+halo->dn+halo->up, numBins);
213  currABatch[currIdx] = 0;
214  }
215  }
216  else{
217  //A sigma of zero means only the centered bin value should get input
218  int featureIdx = round((inVal-binMin)/stepSize);
219 
220  for(int iF = 0; iF < numBins; iF++){
221  if(binSigma == 0){
222  int currIdx = kIndex(iX, iY, iF, nx+halo->lt+halo->rt, ny+halo->dn+halo->up, numBins);
223  if(iF == featureIdx){
224  currABatch[currIdx] = 1;
225  }
226  //Resetting value
227  else{
228  if(zeroNeg){
229  currABatch[currIdx] = 0;
230  }
231  else{
232  currABatch[currIdx] = -1;
233  }
234  }
235  }
236  else{
237  //Calculate center value for featureIdx (the bin that the value belongs to without a sigma) is binning
238  float mean;
239  if(normalDist){
240  mean = featureIdx * stepSize + (stepSize/2);
241  }
242  else{
243  mean = featureIdx;
244  }
245  //Possible bins
246  int intSigma = ceil(binSigma);
247  int currIdx = kIndex(iX, iY, iF, nx+halo->lt+halo->rt, ny+halo->dn+halo->up, numBins);
248  if(iF >= featureIdx-intSigma && iF <= featureIdx+intSigma){
249  //Get center of that aBin for the x pos of the normal dist
250  float xVal;
251  if(normalDist){
252  xVal = iF * stepSize + (stepSize/2);
253  }
254  else{
255  xVal = iF;
256  }
257  //Calculate normal dist
258  float outVal = calcNormDist(xVal, mean, binSigma);
259  //Put into activity buffer

261  }
262  //Resetting value
263  else{
264  if(zeroNeg){
265  currABatch[currIdx] = 0;
266  }
267  else{
268  currABatch[currIdx] = -1;
269  }
270  }
271  }
272  }

274  }
