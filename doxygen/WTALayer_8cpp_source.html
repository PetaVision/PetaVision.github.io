31 int WTALayer::communicateInitInfo() {
32  int status = HyPerLayer::communicateInitInfo();
33  originalLayer = parent->getLayerFromName(originalLayerName);
34  if (originalLayer==NULL) {
35  if (parent->columnId()==0) {
36  fprintf(stderr, "%s \"%s\" error: originalLayerName \"%s\" is not a layer in the HyPerCol.\n",
37  getKeyword(), name, originalLayerName);
38  }
39  MPI_Barrier(parent->icCommunicator()->communicator());
40  exit(EXIT_FAILURE);
41  }
42  if (originalLayer->getInitInfoCommunicatedFlag()==false) {
43  return PV_POSTPONE;
44  }
45  originalLayer->synchronizeMarginWidth(this);
46  this->synchronizeMarginWidth(originalLayer);
47  const PVLayerLoc * srcLoc = originalLayer->getLayerLoc();
48  const PVLayerLoc * loc = getLayerLoc();
49  assert(srcLoc != NULL && loc != NULL);
50  if (srcLoc->nxGlobal != loc->nxGlobal || srcLoc->nyGlobal != loc->nyGlobal) {
51  if (parent->columnId()==0) {
52  fprintf(stderr, "%s \"%s\" error: originalLayerName \"%s\" does not have the same dimensions.\n",
53  getKeyword(), name, originalLayerName);
54  fprintf(stderr, " original (nx=%d, ny=%d) versus (nx=%d, ny=%d)\n",
55  srcLoc->nxGlobal, srcLoc->nyGlobal, loc->nxGlobal, loc->nyGlobal);
56  }
57  MPI_Barrier(parent->icCommunicator()->communicator());
58  exit(EXIT_FAILURE);
59  }
60  if(getLayerLoc()->nf != 1){
61  fprintf(stderr, "%s \"%s\" error: WTALayer can only have 1 feature.\n",
62  getKeyword(), name);
63  }
64  assert(srcLoc->nx==loc->nx && srcLoc->ny==loc->ny);
65  return status;

115 int WTALayer::updateState(double timef, double dt) {
116  pvdata_t * currA = getCLayer()->activity->data;
117  pvdata_t * srcA = originalLayer->getCLayer()->activity->data;
118 
119  const PVLayerLoc * loc = getLayerLoc();
120  const PVLayerLoc * srcLoc = originalLayer->getLayerLoc();
121 
122  //NF must be one
123  assert(loc->nf == 1);
124  float stepSize = (float)(binMax - binMin)/(float)srcLoc->nf;
125 
126  for(int b = 0; b < srcLoc->nbatch; b++){
127  pvdata_t * currABatch = currA + b * getNumExtended();
128  pvdata_t * srcABatch = srcA + b * originalLayer->getNumExtended();
129  //Loop over x and y of the src layer
130  for(int yi = 0; yi < srcLoc->ny; yi++){
131  for(int xi = 0; xi < srcLoc->nx; xi++){
132  //Find maximum output value in nf direction
133  float maxInput = -99999999;
134  float maxIndex = -99999999;
135  for(int fi = 0; fi < srcLoc->nf; fi++){
136  int kExt = kIndex(xi, yi, fi,
137  srcLoc->nx+srcLoc->halo.lt+srcLoc->halo.rt,
138  srcLoc->ny+srcLoc->halo.up+srcLoc->halo.dn,
139  srcLoc->nf);
140  if(srcABatch[kExt] > maxInput || fi == 0){
141  maxInput = srcABatch[kExt];
142  maxIndex = fi;
143  }
144  }
145  //Found max index, set output to value
146  float outVal = (maxIndex * stepSize) + binMin;
147  int kExtOut = kIndex(xi, yi, 0,
148  loc->nx+loc->halo.lt+loc->halo.rt,
149  loc->ny+loc->halo.up+loc->halo.dn,
150  loc->nf);
151  currABatch[kExtOut] = outVal;
152  }
153  }
154  }
155  return PV_SUCCESS;
156 }
