187 int TransposeConn::communicateInitInfo() {
188  int status = PV_SUCCESS;
189  BaseConnection * originalConnBase = parent->getConnFromName(this->originalConnName);
190  if (originalConnBase==NULL) {
191  if (parent->columnId()==0) {
192  pvErrorNoExit().printf("%s \"%s\": originalConnName \"%s\" does not refer to any connection in the column.\n", this->getKeyword(), name, this->originalConnName);
193  }
194  MPI_Barrier(parent->icCommunicator()->communicator());
195  exit(EXIT_FAILURE);
196  }
197  this->originalConn = dynamic_cast<HyPerConn *>(originalConnBase);
198  if (originalConn == NULL) {
199  if (parent->columnId()==0) {
200  pvErrorNoExit().printf("TransposeConn \"%s\": originalConnName \"%s\" is not an existing connection.\n", name, originalConnName);
201  status = PV_FAILURE;
202  }
203  }
204  if (status != PV_SUCCESS) return status;
205 
206  if (!originalConn->getInitInfoCommunicatedFlag()) {
207  if (parent->columnId()==0) {
208  const char * connectiontype = this->getKeyword();
209  pvInfo().printf("%s \"%s\" must wait until original connection \"%s\" has finished its communicateInitInfo stage.\n", connectiontype, name, originalConn->getName());
210  }
211  return PV_POSTPONE;
212  }
213 
214  sharedWeights = originalConn->usingSharedWeights();
215  parent->parameters()->handleUnnecessaryParameter(name, "sharedWeights", sharedWeights);
216 
217  numAxonalArborLists = originalConn->numberOfAxonalArborLists();
218  parent->parameters()->handleUnnecessaryParameter(name, "numAxonalArbors", numAxonalArborLists);
219 
220  //plasticityFlag = originalConn->getPlasticityFlag();
221  //parent->parameters()->handleUnnecessaryParameter(name, "plasticityFlag", plasticityFlag);
222 
223  if(originalConn->getShrinkPatches_flag()) {
224  if (parent->columnId()==0) {
225  pvErrorNoExit().printf("TransposeConn \"%s\": original conn \"%s\" has shrinkPatches set to true. TransposeConn has not been implemented for that case.\n", name, originalConn->getName());
226  }
227  MPI_Barrier(parent->icCommunicator()->communicator());
228  exit(EXIT_FAILURE);
229  }
230 
231  status = HyPerConn::communicateInitInfo(); // calls setPatchSize()
232  if (status != PV_SUCCESS) return status;
233 
234  const PVLayerLoc * preLoc = pre->getLayerLoc();
235  const PVLayerLoc * origPostLoc = originalConn->postSynapticLayer()->getLayerLoc();
236  if (preLoc->nx != origPostLoc->nx || preLoc->ny != origPostLoc->ny || preLoc->nf != origPostLoc->nf) {
237  if (parent->columnId()==0) {
238  pvErrorNoExit(errorMessage);
239  errorMessage.printf("%s \"%s\": transpose's pre layer and original connection's post layer must have the same dimensions.\n", this->getKeyword(), name);
240  errorMessage.printf(" (x=%d, y=%d, f=%d) versus (x=%d, y=%d, f=%d).\n", preLoc->nx, preLoc->ny, preLoc->nf, origPostLoc->nx, origPostLoc->ny, origPostLoc->nf);
241  }
242  MPI_Barrier(parent->icCommunicator()->communicator());
243  exit(EXIT_FAILURE);
244  }
245  const PVLayerLoc * postLoc = pre->getLayerLoc();
246  const PVLayerLoc * origPreLoc = originalConn->postSynapticLayer()->getLayerLoc();
247  if (postLoc->nx != origPreLoc->nx || postLoc->ny != origPreLoc->ny || postLoc->nf != origPreLoc->nf) {
248  if (parent->columnId()==0) {
249  pvErrorNoExit(errorMessage);
250  errorMessage.printf("%s \"%s\": transpose's post layer and original connection's pre layer must have the same dimensions.\n", this->getKeyword(), name);
251  errorMessage.printf(" (x=%d, y=%d, f=%d) versus (x=%d, y=%d, f=%d).\n", postLoc->nx, postLoc->ny, postLoc->nf, origPreLoc->nx, origPreLoc->ny, origPreLoc->nf);
252  }
253  MPI_Barrier(parent->icCommunicator()->communicator());
254  exit(EXIT_FAILURE);
255  }
256 
257  originalConn->setNeedPost(true);
258 
259 #if defined(PV_USE_OPENCL) || defined(PV_USE_CUDA)
260  if((updateGSynFromPostPerspective && receiveGpu) || allocPostDeviceWeights){
261  originalConn->setAllocDeviceWeights();
262  }
263  if((!updateGSynFromPostPerspective && receiveGpu) || allocDeviceWeights){
264  originalConn->setAllocPostDeviceWeights();
265  }
266 #endif
267 
268 
269  //Synchronize margines of this post and orig pre, and vice versa
270  originalConn->preSynapticLayer()->synchronizeMarginWidth(post);
271  post->synchronizeMarginWidth(originalConn->preSynapticLayer());
272 
273  originalConn->postSynapticLayer()->synchronizeMarginWidth(pre);
274  pre->synchronizeMarginWidth(originalConn->postSynapticLayer());
275 
276 
277  return status;
278 }

280 int TransposeConn::setPatchSize() {
281  // If originalConn is many-to-one, the transpose connection is one-to-many; then xscaleDiff > 0.
282  // Similarly, if originalConn is one-to-many, xscaleDiff < 0.
283 
284  // Some of the code duplication might be eliminated by adding some functions to convert.h
285 
286  assert(pre && post);
287  assert(originalConn);
288 
289  int xscaleDiff = pre->getXScale() - post->getXScale();
290  int nxp_orig = originalConn->xPatchSize();
291  int nyp_orig = originalConn->yPatchSize();
292  nxp = nxp_orig;
293  if(xscaleDiff > 0 ) {
294  nxp *= (int) pow( 2, xscaleDiff );
295  }
296  else if(xscaleDiff < 0) {
297  nxp /= (int) pow(2,-xscaleDiff);
298  assert(nxp_orig==nxp*pow( 2, (float) (-xscaleDiff) ));
299  }
300 
301  int yscaleDiff = pre->getYScale() - post->getYScale();
302  nyp = nyp_orig;
303  if(yscaleDiff > 0 ) {
304  nyp *= (int) pow( 2, yscaleDiff );
305  }
306  else if(yscaleDiff < 0) {
307  nyp /= (int) pow(2,-yscaleDiff);
308  assert(nyp_orig==nyp*pow( 2, (float) (-yscaleDiff) ));
309  }
310 
311  nfp = post->getLayerLoc()->nf;
312  // post->getLayerLoc()->nf must be the same as originalConn->preSynapticLayer()->getLayerLoc()->nf.
313  // This requirement is checked in communicateInitInfo
314 
315  parent->parameters()->handleUnnecessaryParameter(name, "nxp", nxp);
316  parent->parameters()->handleUnnecessaryParameter(name, "nyp", nyp);
317  parent->parameters()->handleUnnecessaryParameter(name, "nfp", nfp);
318  return PV_SUCCESS;
319 }
