60 int InitSpreadOverArborsWeights::spreadOverArborsWeights(/* PVPatch * patch */ pvdata_t * dataStart, int arborId,
61  InitSpreadOverArborsWeightsParams * weightParamPtr) {
62 
63 
64 
65  //load necessary params:
66  int nfPatch_tmp = weightParamPtr->getnfPatch();
67  int nyPatch_tmp = weightParamPtr->getnyPatch();
68  int nxPatch_tmp = weightParamPtr->getnxPatch();
69  int sx_tmp=weightParamPtr->getsx();
70  int sy_tmp=weightParamPtr->getsy();
71  int sf_tmp=weightParamPtr->getsf();
72 
73  const float iWeight = weightParamPtr->getInitWeight();
74  const int nArbors = callingConn->numberOfAxonalArborLists();
75 
76 
77  // pvdata_t * w_tmp = patch->data;
78 
79  // loop over all post-synaptic cells in temporary patch
80  for (int fPost = 0; fPost < nfPatch_tmp; fPost++) {
81  float thPost = weightParamPtr->calcThPost(fPost);
82  if(weightParamPtr->checkThetaDiff(thPost)) continue;
83  for (int jPost = 0; jPost < nyPatch_tmp; jPost++) {
84  float yDelta = weightParamPtr->calcYDelta(jPost);
85  for (int iPost = 0; iPost < nxPatch_tmp; iPost++) {
86  float xDelta = weightParamPtr->calcXDelta(iPost);
87 
88 
89  // rotate the reference frame by th (change sign of thPost?)
90  float xp = +xDelta * cosf(thPost) + yDelta * sinf(thPost);
91  float yp = -xDelta * sinf(thPost) + yDelta * cosf(thPost);
92 
93 
94  float weight = 0;
95  if (xp*xp+yp*yp<1e-4) {
96  weight = iWeight/nArbors;
97  }
98  else {
99  float theta2pi = atan2f(yp, xp)/(2*PI);
100  unsigned int xpraw, ypraw, atanraw;
101  union u { float f; unsigned int i;};
102  union u f2u;
103  f2u.f = xp; xpraw = f2u.i;
104  f2u.f = yp; ypraw = f2u.i;
105  f2u.f = theta2pi; atanraw = f2u.i;
106  if(theta2pi<0) theta2pi+=1;
107  if(theta2pi>=1) theta2pi-=1; // theta2pi should be in the range [0,1) but roundoff could make it exactly 1
108  float zone = theta2pi*nArbors;
109 
110  float intpart;
111  float fracpart = modff(zone, &intpart);
112  assert(intpart>=0 && intpart<nArbors && fracpart>=0 && fracpart<1);
113 
114  if (intpart==arborId) {
115  weight = iWeight*(1-fracpart);
116  }
117  else if ( (int) (intpart-arborId+1) % nArbors == 0) {
118  weight = iWeight*fracpart;
119  }
120  }
121  // if (arborId == zone)
122  // weight = iWeight;
123 
124  int index = iPost * sx_tmp + jPost * sy_tmp + fPost * sf_tmp;
125  dataStart[index] = weight;
126  }
127  }
128  }
129 
130  return PV_SUCCESS;
131 }
