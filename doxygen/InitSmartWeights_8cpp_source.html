49 int InitSmartWeights::smartWeights(/* PVPatch * wp */ pvdata_t * dataStart, int k, InitWeightsParams *weightParams) {
50  // pvdata_t * w = wp->data;
51 
52  const int nxp = weightParams->getnxPatch(); // wp->nx;
53  const int nyp = weightParams->getnyPatch(); // wp->ny;
54  const int nfp = weightParams->getnfPatch(); //wp->nf;
55 
56  const int sxp = weightParams->getsx(); //wp->sx;
57  const int syp = weightParams->getsy(); //wp->sy;
58  const int sfp = weightParams->getsf(); //wp->sf;
59 
60  // loop over all post-synaptic cells in patch
61  for (int y = 0; y < nyp; y++) {
62  for (int x = 0; x < nxp; x++) {
63  for (int f = 0; f < nfp; f++) {
64  dataStart[x * sxp + y * syp + f * sfp] = weightParams->getParentConn()->dataIndexToUnitCellIndex(k);
65  }
66  }
67  }
68 
69  return 0;
70 }
