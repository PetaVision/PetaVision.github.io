143 // triggerFlag was deprecated Oct 7, 2015, and marked obsolete Jun 9, 2017.
144 // Setting triggerLayerName to a nonempty string has the effect of triggerFlag=true, and
145 // setting triggerLayerName to NULL or "" has the effect of triggerFlag=false.
146 // For a reasonable fade-out time, it is an error for triggerFlag to be defined in params.
147 void BaseProbe::ioParam_triggerFlag(enum ParamsIOFlag ioFlag) {
148  assert(!parent->parameters()->presentAndNotBeenRead(name, "triggerLayerName"));
149  if (ioFlag == PARAMS_IO_READ && parent->parameters()->present(name, "triggerFlag")) {
150  bool flagFromParams = false;
151  parent->parameters()->ioParamValue(
152  ioFlag, name, "triggerFlag", &flagFromParams, flagFromParams);
153  if (parent->columnId() == 0) {
154  Fatal(triggerFlagDeprecated);
155  triggerFlagDeprecated.printf(
156  "%s: triggerFlag is obsolete for probes.\n", getDescription_c());
157  triggerFlagDeprecated.printf(
158  " If triggerLayerName is a nonempty string, triggering will be on;\n");
159  triggerFlagDeprecated.printf(
160  " if triggerLayerName is empty or null, triggering will be off.\n");
161  }
162  }
163 }

182 void BaseProbe::initOutputStreams(const char *filename, Checkpointer *checkpointer) {
183  MPIBlock const *mpiBlock = checkpointer->getMPIBlock();
184  int blockColumnIndex = mpiBlock->getColumnIndex();
185  int blockRowIndex = mpiBlock->getRowIndex();
186  if (blockColumnIndex == 0 and blockRowIndex == 0) {
187  int localBatchWidth = parent->getNBatch();
188  int mpiBatchIndex = mpiBlock->getStartBatch() + mpiBlock->getBatchIndex();
189  int localBatchOffset = localBatchWidth * mpiBatchIndex;
190  mOutputStreams.resize(localBatchWidth);
191  char const *probeOutputFilename = getProbeOutputFilename();
192  if (probeOutputFilename) {
193  std::string path(probeOutputFilename);
194  auto extensionStart = path.rfind('.');
195  std::string extension;
196  if (extensionStart != std::string::npos) {
197  extension = path.substr(extensionStart);
198  path = path.substr(0, extensionStart);
199  }
200  std::ios_base::openmode mode = std::ios_base::out;
201  if (!checkpointer->getCheckpointReadDirectory().empty()) {
202  mode |= std::ios_base::app;
203  }
204  for (int b = 0; b < localBatchWidth; b++) {
205  int globalBatchIndex = b + localBatchOffset;
206  std::string batchPath = path;
207  std::string batchIndexString = std::to_string(globalBatchIndex);
208  batchPath.append("_batchElement_").append(batchIndexString).append(extension);
209  if (batchPath[0] != '/') {
210  batchPath = checkpointer->makeOutputPathFilename(batchPath);
211  }
212  auto fs = new FileStream(batchPath.c_str(), mode, checkpointer->doesVerifyWrites());
213  mOutputStreams[b] = fs;
214  }
215  }
216  else {
217  for (int b = 0; b < localBatchWidth; b++) {
218  mOutputStreams[b] = new PrintStream(PV::getOutputStream());
219  }
220  }
221  }
222  else {
223  mOutputStreams.clear();
224  }
225 }

250 int BaseProbe::communicateInitInfo(CommunicateInitInfoMessage const *message) {
251  int status = PV_SUCCESS;
252 
253  // Set up triggering.
254  if (triggerFlag) {
255  triggerLayer = message->lookup<HyPerLayer>(std::string(triggerLayerName));
256  if (triggerLayer == NULL) {
257  if (parent->columnId() == 0) {
258  ErrorLog().printf(
259  "%s \"%s\": triggerLayer \"%s\" is not a layer in the HyPerCol.\n",
260  parent->parameters()->groupKeywordFromName(name),
261  name,
262  triggerLayerName);
263  }
264  MPI_Barrier(parent->getCommunicator()->communicator());
265  exit(EXIT_FAILURE);
266  }
267  }
268 
269  // Add the probe to the ColumnEnergyProbe, if there is one.
270  if (energyProbe && energyProbe[0]) {
271  ColumnEnergyProbe *probe = message->lookup<ColumnEnergyProbe>(std::string(energyProbe));
272  if (probe == NULL) {
273  if (parent->columnId() == 0) {
274  ErrorLog().printf(
275  "%s \"%s\": energyProbe \"%s\" is not a ColumnEnergyProbe in the "
276  "column.\n",
277  parent->parameters()->groupKeywordFromName(getName()),
278  getName(),
279  energyProbe);
280  }
281  MPI_Barrier(parent->getCommunicator()->communicator());
282  exit(EXIT_FAILURE);
283  }
284  status = probe->addTerm(this);
285  }
286  return status;
287 }

289 int BaseProbe::initMessage(const char *msg) {
290  assert(msgstring == NULL);
291  int status = PV_SUCCESS;
292  if (msg != NULL && msg[0] != '\0') {
293  size_t msglen = strlen(msg);
294  this->msgstring = (char *)calloc(
295  msglen + 2,
296  sizeof(char)); // Allocate room for colon plus null terminator
297  if (this->msgstring) {
298  memcpy(this->msgstring, msg, msglen);
299  this->msgstring[msglen] = ':';
300  this->msgstring[msglen + 1] = '\0';
301  }
302  }
303  else {
304  this->msgstring = (char *)calloc(1, sizeof(char));
305  if (this->msgstring) {
306  this->msgstring[0] = '\0';
307  }
308  }
309  if (!this->msgstring) {
310  ErrorLog().printf(
311  "%s \"%s\": Unable to allocate memory for probe's message.\n",
312  parent->parameters()->groupKeywordFromName(name),
313  name);
314  status = PV_FAILURE;
315  }
316  assert(status == PV_SUCCESS);
317  return status;
318 }

359 double BaseProbe::getValue(double timevalue, int index) {
360  if (index < 0 || index >= getNumValues()) {
361  return std::numeric_limits<double>::signaling_NaN();
362  }
363  else {
364  int status = PV_SUCCESS;
365  if (needRecalc(timevalue)) {
366  status = getValues(timevalue);
367  }
368  if (status != PV_SUCCESS) {
369  return std::numeric_limits<double>::signaling_NaN();
370  }
371  }
372  return probeValues[index];
373 }
