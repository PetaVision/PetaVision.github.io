156 int BaseProbe::initOutputStream(const char * filename) {
157  if( parent->columnId()==0 ) {
158  if( filename != NULL ) {
159  char * outputdir = parent->getOutputPath();
160  char * path = (char *) malloc(strlen(outputdir)+1+strlen(filename)+1);
161  sprintf(path, "%s/%s", outputdir, filename);
162  bool append = parent->getCheckpointReadFlag();
163  const char * fopenstring = append ? "a" : "w";
164  outputstream = PV_fopen(path, fopenstring, parent->getVerifyWrites());
165  if( !outputstream ) {
166  fprintf(stderr, "BaseProbe error opening \"%s\" for writing: %s\n", path, strerror(errno));
167  exit(EXIT_FAILURE);
168  }
169  free(path);
170  }
171  else {
172  outputstream = PV_stdout();
173  }
174  }
175  else {
176  outputstream = NULL; // Only root process writes; if other processes need something written it should be sent to root.
177  // Derived classes for which it makes sense for a different process to do the file i/o should override initOutputStream
178  }
179  return PV_SUCCESS;
180 }

182 int BaseProbe::communicateInitInfo() {
183  //Set up output stream
184  int status = initOutputStream(probeOutputFilename);
185  //communicate here only sets up triggering
186  if(triggerFlag){
187  triggerLayer = parent->getLayerFromName(triggerLayerName);
188  if (triggerLayer==NULL) {
189  if (parent->columnId()==0) {
190  fprintf(stderr, "%s \"%s\" error: triggerLayer \"%s\" is not a layer in the HyPerCol.\n",
191  parent->parameters()->groupKeywordFromName(name), name, triggerLayerName);
192  }
193 #ifdef PV_USE_MPI
194  MPI_Barrier(parent->icCommunicator()->communicator());
195 #endif
196  exit(EXIT_FAILURE);
197  }
198  }
199  if (energyProbe && energyProbe[0]) {
200  ColProbe * colprobe = getParent()->getColProbeFromName(energyProbe);
201  ColumnEnergyProbe * probe = dynamic_cast<ColumnEnergyProbe *>(colprobe);
202  if (probe==NULL) {
203  if (getParent()->columnId()==0) {
204  fprintf(stderr, "%s \"%s\" error: energyProbe \"%s\" is not a ColumnEnergyProbe in the column.\n",
205  getParent()->parameters()->groupKeywordFromName(getName()), getName(), energyProbe);
206  }
207 #ifdef PV_USE_MPI
208  MPI_Barrier(getParent()->icCommunicator()->communicator());
209 #endif
210  exit(EXIT_FAILURE);
211  }
212  status = probe->addTerm(this, coefficient, getParent()->getNBatch());
213  }
214  return status;
215 }

221 int BaseProbe::initMessage(const char * msg) {
222  assert(msgstring==NULL);
223  int status = PV_SUCCESS;
224  if( msg != NULL && msg[0] != '\0' ) {
225  size_t msglen = strlen(msg);
226  this->msgstring = (char *) calloc(msglen+2, sizeof(char)); // Allocate room for colon plus null terminator
227  if(this->msgstring) {
228  memcpy(this->msgstring, msg, msglen);
229  this->msgstring[msglen] = ':';
230  this->msgstring[msglen+1] = '\0';
231  }
232  }
233  else {
234  this->msgstring = (char *) calloc(1, sizeof(char));
235  if(this->msgstring) {
236  this->msgstring[0] = '\0';
237  }
238  }
239  if( !this->msgstring ) {
240  fprintf(stderr, "%s \"%s\": Unable to allocate memory for probe's message.\n",
241  parent->parameters()->groupKeywordFromName(name), name);
242  status = PV_FAILURE;
243  }
244  assert(status == PV_SUCCESS);
245  return status;
246 }

248 bool BaseProbe::needUpdate(double time, double dt){
249  if(triggerFlag){
250  assert(triggerLayer);
251  double updateTime;
252  //Update if trigger layer updated on this timestep
253  if(fabs(time - triggerLayer->getLastUpdateTime()) <= (dt/2)){
254  updateTime = triggerLayer->getLastUpdateTime();
255  }
256  else{
257  updateTime = triggerLayer->getNextUpdateTime();
258  }
259  //never update flag
260  if(updateTime == -1){
261  return false;
262  }
263  //Check for equality
264  if(fabs(time - (updateTime - triggerOffset)) < (dt/2)){
265  return true;
266  }
267  //If it gets to this point, don't update
268  return false;
269  }
270  //If no trigger, update every timestep
271  else{
272  return true;
273  }
274 }
