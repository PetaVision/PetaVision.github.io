363 void LIF::ioParam_method(enum ParamsIOFlag ioFlag) {
364  // Read the integration method: one of 'arma' (preferred), 'beginning' (deprecated), or 'original' (deprecated).
365  const char * default_method = "arma";
366  parent->ioParamString(ioFlag, name, "method", &methodString, default_method, true/*warnIfAbsent*/);
367  if (ioFlag != PARAMS_IO_READ) return;
368 
369  assert(methodString);
370  if (methodString[0] == '\0') {
371  free(methodString);
372  methodString = strdup(default_method);
373  if (methodString==NULL) {
374  fprintf(stderr, "%s \"%s\" error: unable to set method string: %s\n", getKeyword(), name, strerror(errno));
375  exit(EXIT_FAILURE);
376  }
377  }
378  method = methodString ? methodString[0] : 'a'; // Default is ARMA; 'beginning' and 'original' are deprecated.
379  if (method != 'o' && method != 'b' && method != 'a') {
380  if (getParent()->columnId()==0) {
381  fprintf(stderr, "LIF::setLIFParams error. Layer \"%s\" has method \"%s\". Allowable values are \"arma\", \"beginning\" and \"original\".", name, methodString);
382  }
383  MPI_Barrier(parent->icCommunicator()->communicator());
384  exit(EXIT_FAILURE);
385  }
386  if (method != 'a') {
387  if (getParent()->columnId()==0) {
388  fprintf(stderr, "Warning: LIF layer \"%s\" integration method \"%s\" is deprecated. Method \"arma\" is preferred.\n", name, methodString);
389  }
390  }

435 int LIF::allocateConductances(int num_channels) {
436  assert(num_channels>=3); // Need exc, inh, and inhb at a minimum.
437  const int numNeurons = getNumNeuronsAllBatches();
438  G_E = (pvdata_t *) calloc((size_t) (getNumNeuronsAllBatches()*numChannels), sizeof(pvdata_t));
439  if(G_E == NULL) {
440  fprintf(stderr, "LIF layer \"%s\" rank %d process unable to allocate memory for %d conductances: %s\n",
441  name, parent->columnId(), num_channels, strerror(errno));
442  exit(EXIT_FAILURE);
443  }
444 
445  G_I = G_E + 1*numNeurons;
446  G_IB = G_E + 2*numNeurons;
447  return PV_SUCCESS;
448 }
