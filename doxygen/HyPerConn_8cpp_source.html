290 {
291  int sx = nfp;
292  int sy = sx * nxp;
293  int sp = sy * nyp;
294  int nPatches = getNumDataPatches();
295  int status = PV_SUCCESS;
296 
297  //pvAssert(!parent->parameters()->presentAndNotBeenRead(name, "shrinkPatches"));
298 
299  // createArbors() uses the value of shrinkPatches. It should have already been read in ioParamsFillGroup.
300  //allocate the arbor arrays:
301  createArbors();
302 
303  setPatchStrides();
304 
307  wDataStart[0] = allocWeights(nPatches, nxp, nyp, nfp);
308  pvAssert(get_wDataStart(0) != NULL);
309 
310  for (int arborId=0;arborId<numAxonalArborLists;arborId++) {
311  status = createWeights(wPatches, arborId);
312  pvAssert(wPatches[arborId] != NULL);
313

315  wDataStart[arborId] = (get_wDataStart(0) + sp * nPatches * arborId);

344 int HyPerConn::shrinkPatch(int kExt, int arborId) {
345 
346  int kIndex = patchToDataLUT(kExt);
347 
348  PVPatch *weights = getWeights(kExt,arborId);
349 
350  pvwdata_t * w = &get_wDataStart(arborId)[patchStartIndex(kIndex)+weights->offset];
351 
352  int nx = weights->nx;
353  int ny = weights->ny;
354 
355  int maxnx = INT_MIN;
356  int minnx = INT_MAX;
357  int maxny = INT_MIN;
358  int minny = INT_MAX;
359 
360  bool nonZeroWeightFound = false;
361  // loop over all post-synaptic cells in patch
362  for (int y = 0; y < ny; y++) {
363  for (int x = 0; x < nx; x++) {
364  for (int f = 0; f < nfp; f++) {
365  if(fabs(w[x * sxp + y * syp + f * sfp]) <= shrinkPatchesThresh) {
366  nonZeroWeightFound=true;
367  maxnx = maxnx < x ? x : maxnx;
368  minnx = minnx > x ? x : minnx;
369  maxny = maxny < y ? y : maxny;
370  minny = minny > y ? y : minny;
371  }
372  }
373  }
374  }
375 
376  if(nonZeroWeightFound) {
377  //Plus one to capture all of the patch
378  int nxNew = maxnx+1 - minnx;
379  int nyNew = maxny+1 - minny;
380  int dxNew = minnx;
381  int dyNew = minny;
382 
383  // adjust patch size (shrink) to fit within interior of post-synaptic layer
384  //
385  pvpatch_adjust(weights, sxp, syp, nxNew, nyNew, dxNew, dyNew);
386 
387  gSynPatchStart[arborId][kExt] += dxNew*getPostNonextStrides()->sx + dyNew*getPostNonextStrides()->sy;
388  aPostOffset[arborId][kExt] += dxNew*getPostExtStrides()->sx + dyNew*getPostExtStrides()->sy; // Someone who uses these routines, please check that this is correct.
389  }
390  return 0;
391 }

395  // It is okay for either of weightInitializer or weightNormalizer to be null at this point, either because we're in a subclass that doesn't need it, or because we are allowing for
396  // backward compatibility.
397  // The two lines needs to be before the call to BaseConnection::initialize, because that function calls ioParamsFillGroup,
398  // which will call ioParam_weightInitType and ioParam_normalizeMethod, which for reasons of backward compatibility
399  // will create the initializer and normalizer if those member variables are null.
400  this->weightInitializer = weightInitializer;
401  normalizer = weightNormalizer;
402 
403  int status = BaseConnection::initialize(name, hc); // BaseConnection should *NOT* take weightInitializer or weightNormalizer as arguments, as it does not know about InitWeights or NormalizeBase
404 
405  pvAssert(parent);
406  PVParams * inputParams = parent->parameters();
407 
408  //set accumulateFunctionPointer
409  pvAssert(!inputParams->presentAndNotBeenRead(name, "pvpatchAccumulateType"));
410  switch (pvpatchAccumulateType) {
411  case CONVOLVE:
412  accumulateFunctionPointer = &pvpatch_accumulate;
413  accumulateFunctionFromPostPointer = &pvpatch_accumulate_from_post;
414  break;
415  case STOCHASTIC:
416  accumulateFunctionPointer = &pvpatch_accumulate_stochastic;
417  accumulateFunctionFromPostPointer = &pvpatch_accumulate_stochastic_from_post;
418  break;
419 #ifdef OBSOLETE // Marked obsolete May 3, 2016. HyPerConn defines AccumulateType and PoolingConn defines PoolingType
420  case ACCUMULATE_MAXPOOLING:
421  pvError().printf("ACCUMULATE_MAXPOOLING not allowed in HyPerConn, use PoolingConn instead");
422  //accumulateFunctionPointer = &pvpatch_max_pooling;
423  //accumulateFunctionFromPostPointer = &pvpatch_max_pooling_from_post;
424  break;
425  case ACCUMULATE_SUMPOOLING:
426  pvError().printf("ACCUMULATE_SUMPOOLING not allowed in HyPerConn, use PoolingConn instead");
427  //accumulateFunctionPointer = &pvpatch_sum_pooling;
428  //accumulateFunctionFromPostPointer = &pvpatch_accumulate_from_post;
429  break;
430  case ACCUMULATE_AVGPOOLING:
431  pvError().printf("ACCUMULATE_AVGPOOLING not allowed in HyPerConn, use PoolingConn instead");
432  //accumulateFunctionPointer = &pvpatch_sum_pooling;
433  //accumulateFunctionFromPostPointer = &pvpatch_accumulate_from_post;
434  break;
435 #endif // OBSOLETE // Marked obsolete May 3, 2016. HyPerConn defines AccumulateType and PoolingConn defines PoolingType

438  break;
439  }

444

515 int HyPerConn::initPlasticityPatches()
516 {
517  if (!plasticityFlag) return PV_SUCCESS;
518  int sx = nfp;
519  int sy = sx * nxp;
520  int sp = sy * nyp;
521  int nPatches = getNumDataPatches();
522 
523  const int numAxons = numberOfAxonalArborLists();
524 
525  if (combine_dW_with_W_flag){
526  dwDataStart = wDataStart;
527  return PV_SUCCESS;
528  }
529  dwDataStart[0] = allocWeights(nPatches, nxp, nyp, nfp);
530  pvAssert(get_dwDataStart(0) != NULL);
531  for (int arborId = 0; arborId < numAxons; arborId++) {
532  dwDataStart[arborId] = (dwDataStart[0] + sp * nPatches * arborId);
533  pvAssert(get_dwDataStart(arborId) != NULL);
534  } // loop over arbors
535 
536  if(sharedWeights && normalizeDwFlag){
537  numKernelActivations[0] = (long*) pvCalloc(nxp*nyp*nfp*nPatches, sizeof(long));
538  for (int arborId = 0; arborId < numAxons; arborId++) {
539  numKernelActivations[arborId] = (numKernelActivations[0] + sp * nPatches * arborId);
540  pvAssert(get_dwDataStart(arborId) != NULL);
541  } // loop over arbors
542  }
543 
544  return PV_SUCCESS;

1114 int HyPerConn::setPostPatchSize() {
1115  // If postConn is many-to-one, the transpose connection is one-to-many; then xscaleDiff > 0.
1116  // Similarly, if postConn is one-to-many, xscaleDiff < 0.
1117 
1118  // Some of the code duplication might be eliminated by adding some functions to convert.h
1119 
1120  pvAssert(pre && post);
1121 
1122  if (pre != NULL && post != NULL) {
1123  int xscaleDiff = post->getXScale() - pre->getXScale();
1124  int nxp_orig = xPatchSize();
1125  int nyp_orig = yPatchSize();
1126  nxpPost = nxp_orig;
1127  if(xscaleDiff > 0 ) {
1128  nxpPost *= (int) pow( 2, xscaleDiff );
1129  }
1130  else if(xscaleDiff < 0) {
1131  nxpPost /= (int) pow(2,-xscaleDiff);
1132  pvAssert(nxp_orig==nxpPost*pow( 2, (float) (-xscaleDiff) ));
1133  }
1134 
1135  int yscaleDiff = post->getYScale() - pre->getYScale();
1136  nypPost = nyp_orig;
1137  if(yscaleDiff > 0 ) {
1138  nypPost *= (int) pow( 2, yscaleDiff );
1139  }
1140  else if(yscaleDiff < 0) {
1141  nypPost /= (int) pow(2,-yscaleDiff);
1142  pvAssert(nyp_orig==nypPost*pow( 2, (float) (-yscaleDiff) ));

1151 int HyPerConn::communicateInitInfo() {

1153  // seeds they need. At the start of HyPerCol::run, the parent HyPerCol
1154  // calls each layer's and each connection's communicateInitInfo() sequentially in
1155  // a repeatable order (probably the order they appear in the params
1156  // file) to make sure that the same runs use the same RNG seeds in the
1157  // same way.
1158  //
1159  // HyPerConns need RNGs if they are using stochastic release flag, or if
1160  // their InitWeights method is random (e.g. UniformRandomWeights or
1161  // GaussianRandomWeights).

1164  // - its pre-synaptic layer how big a margin is needed

1166  // - its post-synaptic layer which channel it will deliver GSyn to.

1169  // the relative densities of the pre and post layers, and that nfp is
1170  // consistent with the number of features of post.
