96 int MoviePvp::initialize(const char * name, HyPerCol * hc) {
97  int status = ImagePvp::initialize(name, hc);
98  if (status != PV_SUCCESS) {
99  pvError().printf("Image::initialize failed on Movie layer \"%s\". Exiting.\n", name);
100  }
101 
102  //Update on first timestep
103  setNextUpdateTime(parent->simulationTime() + hc->getDeltaTime());
104 
105  // set output path for movie frames
106  if(writeImages){
107  status = parent->ensureDirExists(movieOutputPath);
108  }
109 
110  if(writeFrameToTimestamp){
111  std::string timestampFilename = std::string(parent->getOutputPath());
112  timestampFilename += "/timestamps/";
113  parent->ensureDirExists(timestampFilename.c_str());
114  timestampFilename += name;
115  timestampFilename += ".txt";
116  if(getParent()->icCommunicator()->commRank()==0){
117  //If checkpoint read is set, append, otherwise, clobber
118  if(getParent()->getCheckpointReadFlag()){
119  struct stat statbuf;
120  if (PV_stat(timestampFilename.c_str(), &statbuf) != 0) {
121  pvWarn().printf("%s \"%s\": timestamp file \"%s\" unable to be found. Creating new file.\n",
122  getKeyword(), name, timestampFilename.c_str());
123  timestampFile = PV::PV_fopen(timestampFilename.c_str(), "w", parent->getVerifyWrites());
124  }
125  else {
126  timestampFile = PV::PV_fopen(timestampFilename.c_str(), "r+", false/*verifyWrites*/);
127  }
128  }
129  else{
130  timestampFile = PV::PV_fopen(timestampFilename.c_str(), "w", parent->getVerifyWrites());
131  }
132  assert(timestampFile);
133  }
134  }
135  return PV_SUCCESS;
136 }

214 int MoviePvp::allocateDataStructures() {
215 
216  //Get file information
217  Communicator* comm = parent->icCommunicator();
218 
219  startFrameIndex = (int*)calloc(parent->getNBatch(), sizeof(int));
220  assert(startFrameIndex);
221  skipFrameIndex = (int*)calloc(parent->getNBatch(), sizeof(int));
222  assert(skipFrameIndex);
223 
224  int nbatch = parent->getNBatch();
225  assert(batchMethod);
226 
227  if(strcmp(batchMethod, "byImage") == 0){
228  //No skip here allowed
229  if(numSkipFrame != 0){
230  pvError() << "Movie layer " << name << " batchMethod of \"" << batchMethod << "\" sets skip_frame_index, do not specify.\n";
231  }
232 
233  int offset = 0;
234  //Default value
235  if(numStartFrame == 0){
236  }
237  //Uniform start array
238  else if(numStartFrame == 1){
239  offset = *paramsStartFrameIndex;
240  }
241  else{
242  pvError() << "Movie layer " << name << " batchMethod of \"" << batchMethod << "\" requires 0 or 1 start_frame_index values\n";
243  }
244  //Allocate and default
245  //Not done in allocate, as Image Allocate needs this parameter to be set
246  int kb0 = getLayerLoc()->kb0;
247  int nbatchGlobal = getLayerLoc()->nbatchGlobal;
248  for(int b = 0; b < nbatch; b++){
249  startFrameIndex[b] = offset + kb0 + b;
250  skipFrameIndex[b] = nbatchGlobal;
251  }
252  }
253  else if (strcmp(batchMethod, "byMovie") == 0){
254  //No skip here allowed
255  if(numSkipFrame != 0){
256  pvError() << "Movie layer " << name << " batchMethod of \"byImage\" sets skip_frame_index, do not specify.\n";
257  }
258 
259  int offset = 0;
260  //Default value
261  if(numStartFrame== 0){
262  }
263  //Uniform start array
264  else if(numStartFrame== 1){
265  offset = *paramsStartFrameIndex;
266  }
267  else{
268  pvError() << "Movie layer " << name << " batchMethod of \"byMovie\" requires 0 or 1 start_frame_index values\n";
269  }
270 
271  int nbatchGlobal = getLayerLoc()->nbatchGlobal;
272  int kb0 = getLayerLoc()->kb0;
273 
274  int framesPerBatch = floor(fileNumFrames/nbatchGlobal);
275  if(framesPerBatch < 1){
276  framesPerBatch = 1;
277  }
278  for(int b = 0; b < nbatch; b++){
279  //+1 for 1 indexed
280  startFrameIndex[b] = offset + ((b+kb0)*framesPerBatch);
281  skipFrameIndex[b] = 1;
282  }
283  }
284  else if(strcmp(batchMethod, "bySpecified") == 0){
285  if(numStartFrame != nbatch && numStartFrame != 0){
286  pvError() << "Movie layer " << name << " batchMethod of \"bySpecified\" requires 0 or " << nbatch << " start_frame_index values\n";
287  }
288  if(numSkipFrame != nbatch && numSkipFrame != 0){
289  pvError() << "Movie layer " << name << " batchMethod of \"bySpecified\" requires 0 or " << nbatch << " skip_frame_index values\n";
290  }
291  for(int b = 0; b < nbatch; b++){
292  if(numStartFrame == 0){
293  //+1 for 1 indexed
294  startFrameIndex[b] = 0;
295  }
296  else{
297  startFrameIndex[b] = paramsStartFrameIndex[b];
298  }
299  if(numSkipFrame == 0){
300  skipFrameIndex[b] = 1;
301  }
302  else{
303  skipFrameIndex[b] = paramsSkipFrameIndex[b];
304  }
305  }
306  }
307  else{
308  //This should never excute, as this check was done in the reading of this parameter
309  assert(0);
310  }
311 
312  //Allocate and set frameNumbers
313  frameNumbers = (int*) malloc(parent->getNBatch() * sizeof(int));
314  for(int b = 0; b < nbatch; b++){
315  frameNumbers[b] = startFrameIndex[b];
316  }
317 
318 
319 
320  int status = ImagePvp::allocateDataStructures();
321  assert(status == PV_SUCCESS);
322 
323  return status;
324 }

391 bool MoviePvp::updateImage(double time, double dt)
392 {
393  InterColComm * icComm = getParent()->icCommunicator();
394 
395  if(fabs(time - (parent->getStartTime() + parent->getDeltaTime())) > (parent->getDeltaTime()/2)){
396  int status = getFrame(time, dt);
397  if( status != PV_SUCCESS ) {
398  pvError().printf("Movie %s: Error reading file \"%s\"\n", name, inputPath);
399  }
400  }
401  if(writePosition && icComm->commRank()==0){
402  fprintf(fp_pos->fp,"%f %s: \n",time,inputPath);
403  }
404  //Set frame number (member variable in Image)
405  //Write to timestamp file here when updated
406  if( icComm->commRank()==0 ) {
407  //Only write if the parameter is set
408  if(timestampFile){
409  std::ostringstream outStrStream;
410  outStrStream.precision(15);
411  int kb0 = getLayerLoc()->kb0;
412  for(int b = 0; b < parent->getNBatch(); b++){
413  outStrStream << time << "," << b+kb0 << "," << frameNumbers[b] << "," << inputPath << "\n";
414  }
415  size_t len = outStrStream.str().length();
416  int status = PV_fwrite(outStrStream.str().c_str(), sizeof(char), len, timestampFile)==len ? PV_SUCCESS : PV_FAILURE;
417  if (status != PV_SUCCESS) {
418  pvError().printf("%s \"%s\" error: Movie::updateState failed to write to timestamp file.\n", getKeyword(), name);
419  }
420  //Flush buffer
421  fflush(timestampFile->fp);
422  }
423  }
424 
425  return true;
426 }
