87 int ImprintConn::imprintFeature(int arbor_ID, int batch_ID, int kExt){
88  const pvdata_t * postactbufAll = postSynapticLayer()->getLayerData();
89  const pvdata_t * postactbuf = postactbufAll + batch_ID * postSynapticLayer()->getNumExtended();
90 
91  const PVLayerLoc * postLoc = post->getLayerLoc();
92 
93  PVPatch * weights = getWeights(kExt,arbor_ID);
94 
95  int sya = (post->getLayerLoc()->nf * (post->getLayerLoc()->nx + post->getLayerLoc()->halo.lt + post->getLayerLoc()->halo.rt));
96 
97  size_t offset = getAPostOffset(kExt, arbor_ID);
98  const pvdata_t * postactRef = &postactbuf[offset];
99 
100  int sym = 0;
101  const pvdata_t * maskactRef = NULL;
102  if(useMask){
103  const pvdata_t * maskactbufAll = mask->getLayerData();
104  const pvdata_t * maskactbuf = maskactbufAll + batch_ID * mask->getNumExtended();
105  maskactRef = &maskactbuf[offset];
106  sym = (mask->getLayerLoc()->nf * (mask->getLayerLoc()->nx + mask->getLayerLoc()->halo.lt + mask->getLayerLoc()->halo.rt));
107  }
108 
109  int ny = weights->ny;
110  int nk = weights->nx * nfp;
111  int kernelIndex = patchIndexToDataIndex(kExt);
112 
113 
114  pvwdata_t * dwdata = get_dwData(arbor_ID, kExt);
115  long * activations = get_activations(arbor_ID, kExt);
116  int lineoffsetw = 0;
117  int lineoffseta = 0;
118  int lineoffsetm = 0;
119 
120  for( int y=0; y<ny; y++ ) {
121  for( int k=0; k<nk; k++ ) {
122  pvdata_t aPost = postactRef[lineoffseta+k];
123  //calculate contribution to dw unless masked out
124  assert(!useMask || maskactRef!=NULL); // if useMask is true, maskactRef must not be null
125  float maskVal = 1;
126  if(useMask){
127  if(mask->getLayerLoc()->nf == 1){
128  maskVal = maskactRef[lineoffsetm+((int)k/postLoc->nf)];
129  }
130  else{
131  maskVal = maskactRef[lineoffsetm+k];
132  }
133  }
134  if (maskVal != 0){
135  assert(sharedWeights);
136  //Offset in the case of a shrunken patch, where dwdata is applying when calling get_dwData
137  activations[lineoffsetw + k]++;
138  //Set actual values to dwData. The imprinted buffer will tell updateWeights to update this kernel by setting to dwWeight
139  dwdata[lineoffsetw + k] += aPost;
140  }
141  }
142  lineoffsetw += syp;
143  lineoffseta += sya;
144  lineoffsetm += sym;
145  }
146  return PV_SUCCESS;
147 }

156 int ImprintConn::update_dW(int arbor_ID){
157  // That takes place in reduceKernels, so that the output is
158  // independent of the number of processors.
159  int nExt = preSynapticLayer()->getNumExtended();
160  int numKernelIndices = getNumDataPatches();
161  const PVLayerLoc * loc = pre->getLayerLoc();
162  const pvdata_t * preactbuf = preSynapticLayer()->getLayerData(getDelay(arbor_ID));
163  int arborStart = arbor_ID * numKernelIndices;
164  int patchSize = nxp * nyp * nfp;
165  int nbatch = loc->nbatch;
166 
167  //Calculate x and y cell size
168  int xCellSize = zUnitCellSize(pre->getXScale(), post->getXScale());
169  int yCellSize = zUnitCellSize(pre->getYScale(), post->getYScale());
170  int nxExt = loc->nx + loc->halo.lt + loc->halo.rt;
171  int nyExt = loc->ny + loc->halo.up + loc->halo.dn;
172  int nf = loc->nf;
173  int numKernels = getNumDataPatches();
174 
175  for(int b = 0; b < nbatch; b++){
176  //Shared weights done in parallel, parallel in numkernels
177 #ifdef PV_USE_OPENMP_THREADS
178 #pragma omp parallel for
179 #endif
180  for(int kernelIdx = 0; kernelIdx < numKernels; kernelIdx++){
181  //Calculate xCellIdx, yCellIdx, and fCellIdx from kernelIndex
182  int kxCellIdx = kxPos(kernelIdx, xCellSize, yCellSize, nf);
183  int kyCellIdx = kyPos(kernelIdx, xCellSize, yCellSize, nf);
184  int kfIdx = featureIndex(kernelIdx, xCellSize, yCellSize, nf);
185 
186  if (parent->simulationTime() - lastActiveTime[arborStart + kernelIdx] > imprintTimeThresh){
187  imprinted[arborStart + kernelIdx] = true;
188  lastActiveTime[arborStart + kernelIdx] = parent->simulationTime();
189  std::cout << "Imprinted feature: Arbor " << arbor_ID << " kernel " << kernelIdx << "\n";
190  }
191 
192  //Loop over all cells in pre ext
193  int kyIdx = kyCellIdx;
194  int yCellIdx = 0;
195  while(kyIdx < nyExt){
196  int kxIdx = kxCellIdx;
197  int xCellIdx = 0;
198  while(kxIdx < nxExt){
199  //Calculate kExt from ky, kx, and kf
200  int kExt = kIndex(kxIdx, kyIdx, kfIdx, nxExt, nyExt, nf);
201  if(imprinted[arborStart + kernelIdx]){
202  imprintFeature(arbor_ID, b, kExt);
203  }
204  else{
205  int status = updateInd_dW(arbor_ID, b, kExt);
206  //Status will be PV_CONTINUE if preact is 0 (not active)
207  if(status == PV_SUCCESS){
208  lastActiveTime[arborStart + kernelIdx] = parent->simulationTime();
209  }
210  }
211  xCellIdx++;
212  kxIdx = kxCellIdx + xCellIdx * xCellSize;
213  }
214  yCellIdx++;
215  kyIdx = kyCellIdx + yCellIdx * yCellSize;
216  }
217  }
218 
219  //If update from clones, update dw here as well
220  //Updates on all PlasticClones
221  for(int clonei = 0; clonei < clones.size(); clonei++){
222  assert(clones[clonei]->preSynapticLayer()->getNumExtended() == nExt);
223  for(int kExt=0; kExt<nExt;kExt++) {
224  int kernelIndex = clones[clonei]->patchIndexToDataIndex(kExt);
225  if(!imprinted[arborStart + kernelIndex]){
226  clones[clonei]->updateInd_dW(arbor_ID, b, kExt);
227  }
228  }
229  }
230 
231  }
232  //Do mpi to update lastActiveTime
233  int ierr = MPI_Allreduce(MPI_IN_PLACE, lastActiveTime, numKernelIndices * numberOfAxonalArborLists(), MPI_DOUBLE, MPI_MAX, parent->icCommunicator()->communicator());
234 
235  return PV_SUCCESS;
236 }

238 int ImprintConn::updateWeights(int arbor_ID){
239  int numKernelIndices = getNumDataPatches();
240  for(int kArbor = 0; kArbor < this->numberOfAxonalArborLists(); kArbor++){
241  int arborStart = kArbor * numKernelIndices;
242  pvwdata_t * w_data_start = get_wDataStart(kArbor);
243  for(int kKernel = 0; kKernel < numKernelIndices; kKernel++){
244  //Regular weight update
245  for(int kPatch = 0; kPatch < nxp*nyp*nfp; kPatch++){
246  int k = kKernel * nxp*nyp*nfp + kPatch;
247  //If imprinte, dw buffer is set to post
248  if(imprinted[arborStart + kKernel]){
249  w_data_start[k] = get_dwDataStart(kArbor)[k];
250  }
251  //Otherwise, accumulate
252  else{
253  w_data_start[k] += get_dwDataStart(kArbor)[k];
254  }
255  }
256  }
257  }
258  return PV_BREAK;
259 }

261 int ImprintConn::checkpointRead(const char * cpDir, double * timeptr) {
262  int status = HyPerConn::checkpointRead(cpDir, timeptr);
263  long numBuf = getNumDataPatches() * numberOfAxonalArborLists();
264  if( parent->columnId() == 0 ) {
265  char * filename = parent->pathInCheckpoint(cpDir, getName(), "_ImprintState.bin");
266  PV_Stream * pvstream = PV_fopen(filename, "r", false/*verifyWrites*/);
267  if( pvstream != NULL ) {
268  status = PV_fread(lastActiveTime, sizeof(double), numBuf, pvstream) == 1 ? status : PV_FAILURE;
269  PV_fclose(pvstream);
270  }
271  else {
272  fprintf(stderr, "Unable to read from \"%s\"\n", filename);
273  exit(-1);
274  }
275  free(filename);
276  }
277 
278  if (parent->icCommunicator()->commSize()>1) {
279  //Communicate buffer size to rest of processes
280  MPI_Bcast(lastActiveTime, numBuf, MPI_DOUBLE, 0, parent->icCommunicator()->communicator());
281  }
282  return status;
283 }

296  if( pvstream != NULL ) {
297  status = PV_fwrite(lastActiveTime, sizeof(double), numBuf, pvstream) == numBuf ? status : PV_FAILURE;
298  PV_fclose(pvstream);
299  }
300  else {
301  fprintf(stderr, "Unable to write to \"%s\"\n", filename);
302  exit(-1);
303  }

306  exit(EXIT_FAILURE);

309  }
