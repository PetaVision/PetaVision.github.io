31 int FeedbackConn::setPreAndPostLayerNames() {
32  int status = PV_SUCCESS;
33  PVParams *params = parent->parameters();
34  if (params->stringPresent(name, "preLayerName")
35  || params->stringPresent(name, "postLayerName")) {
36  if (parent->columnId() == 0) {
37  ErrorLog().printf(
38  "%s: FeedbackConn does not use preLayerName or postLayerName.\n",
39  getDescription_c());
40  }
41  status = PV_FAILURE;
42  }
43  MPI_Barrier(parent->getCommunicator()->communicator());
44  if (status != PV_SUCCESS)
45  exit(EXIT_FAILURE);
46  return status;
47 }
