35 int FeedbackConn::setPreAndPostLayerNames() {
36  int status = PV_SUCCESS;
37  PVParams * params = parent->parameters();
38  if (params->stringPresent(name, "preLayerName") || params->stringPresent(name, "postLayerName")) {
39  if (parent->columnId()==0) {
40  fprintf(stderr, "%s \"%s\": FeedbackConn does not use preLayerName or postLayerName.\n", this->getKeyword(), name);
41  }
42  status = PV_FAILURE;
43  }
44  MPI_Barrier(parent->icCommunicator()->communicator());
45  if (status != PV_SUCCESS) exit(EXIT_FAILURE);
46  return status;
47 }
