91 int MaskLayer::communicateInitInfo() {
92  int status = ANNLayer::communicateInitInfo();
93  if(strcmp(maskMethod, "layer") == 0){
94  maskLayer = parent->getLayerFromName(maskLayerName);
95  if (maskLayer==NULL) {
96  if (parent->columnId()==0) {
97  fprintf(stderr, "%s \"%s\" error: maskLayerName \"%s\" is not a layer in the HyPerCol.\n",
98  getKeyword(), name, maskLayerName);
99  }
100 #ifdef PV_USE_MPI
101  MPI_Barrier(parent->icCommunicator()->communicator());
102 #endif
103  exit(EXIT_FAILURE);
104  }
105 
106  const PVLayerLoc * maskLoc = maskLayer->getLayerLoc();
107  const PVLayerLoc * loc = getLayerLoc();
108  assert(maskLoc != NULL && loc != NULL);
109  if (maskLoc->nxGlobal != loc->nxGlobal || maskLoc->nyGlobal != loc->nyGlobal) {
110  if (parent->columnId()==0) {
111  fprintf(stderr, "%s \"%s\" error: maskLayerName \"%s\" does not have the same x and y dimensions.\n",
112  getKeyword(), name, maskLayerName);
113  fprintf(stderr, " original (nx=%d, ny=%d, nf=%d) versus (nx=%d, ny=%d, nf=%d)\n",
114  maskLoc->nxGlobal, maskLoc->nyGlobal, maskLoc->nf, loc->nxGlobal, loc->nyGlobal, loc->nf);
115  }
116 #ifdef PV_USE_MPI
117  MPI_Barrier(parent->icCommunicator()->communicator());
118 #endif
119  exit(EXIT_FAILURE);
120  }
121 
122  if(maskLoc->nf != 1 && maskLoc->nf != loc->nf){
123  if (parent->columnId()==0) {
124  fprintf(stderr, "%s \"%s\" error: maskLayerName \"%s\" must either have the same number of features as this layer, or one feature.\n",
125  getKeyword(), name, maskLayerName);
126  fprintf(stderr, " original (nx=%d, ny=%d, nf=%d) versus (nx=%d, ny=%d, nf=%d)\n",
127  maskLoc->nxGlobal, maskLoc->nyGlobal, maskLoc->nf, loc->nxGlobal, loc->nyGlobal, loc->nf);
128  }
129 #ifdef PV_USE_MPI
130  MPI_Barrier(parent->icCommunicator()->communicator());
131 #endif
132  exit(EXIT_FAILURE);
133  }
134 
135  assert(maskLoc->nx==loc->nx && maskLoc->ny==loc->ny);
136  }
137  else{
138  //Check for in bounds featureIdxs
139  assert(features);
140  const PVLayerLoc * loc = getLayerLoc();
141  for(int f = 0; f < numSpecifiedFeatures; f++){
142  if(features[f] < 0 || features[f] >= loc->nf){
143  std::cout << "Specified feature " << features[f] << "out of bounds\n";
144  exit(-1);
145  }
146 
147  }
148  }
149 
150  return status;
151 }

153 int MaskLayer::updateState(double time, double dt)
154 {
155  ANNLayer::updateState(time, dt);
156  const PVLayerLoc * loc = getLayerLoc();
157  const PVLayerLoc * maskLoc = maskLayer->getLayerLoc();
158  pvdata_t * maskActivity = maskLayer->getActivity();
159  pvdata_t * A = getActivity();
160 
161  int nx = loc->nx;
162  int ny = loc->ny;
163  int nf = loc->nf;
164  int num_neurons = nx*ny*nf;
165  int nbatch = loc->nbatch;
166 
167  for(int b = 0; b < nbatch; b++){
168  pvdata_t * maskActivityBatch = maskActivity + b * getNumExtended();
169  pvdata_t * ABatch = A + b * getNumExtended();
170 #ifdef PV_USE_OPENMP_THREADS
171 #pragma omp parallel for
172 #endif
173  for(int ni = 0; ni < num_neurons; ni++){
174  int kThisRes = ni;
175  int kThisExt = kIndexExtended(ni, nx, ny, nf, loc->halo.lt, loc->halo.rt, loc->halo.dn, loc->halo.up);
176  int maskVal;
177  if(strcmp(maskMethod, "layer") == 0){
178  int kMaskRes;
179  if(maskLoc->nf == 1){
180  kMaskRes = ni/nf;
181  }
182  else{
183  kMaskRes = ni;
184  }
185  int kMaskExt = kIndexExtended(ni, nx, ny, maskLoc->nf, maskLoc->halo.lt, maskLoc->halo.rt, maskLoc->halo.dn, maskLoc->halo.up);
186  maskVal = maskActivityBatch[kMaskExt];
187  }
188  else if(strcmp(maskMethod, "maskFeatures") == 0){
189  //Calculate feature index of ni
190  int featureNum = featureIndex(ni, nx, ny, nf);
191  maskVal = 1; //If nothing specified, copy everything
192  for(int specF = 0; specF < numSpecifiedFeatures; specF++){
193  if(featureNum == features[specF]){
194  maskVal = 0;
195  break;
196  }
197  }
198  }
199  else if(strcmp(maskMethod, "noMaskFeatures") == 0){
200  //Calculate feature index of ni
201  int featureNum = featureIndex(ni, nx, ny, nf);
202  maskVal = 0; //If nothing specified, copy nothing
203  for(int specF = 0; specF < numSpecifiedFeatures; specF++){
204  if(featureNum == features[specF]){
205  maskVal = 1;
206  break;
207  }
208  }
209  }
210 
211  //Set value to 0, otherwise, updateState from ANNLayer should have taken care of it
212  if(maskVal == 0){
213  ABatch[kThisExt] = 0;
214  }
215  }
216  }
217  return PV_SUCCESS;
218 }
