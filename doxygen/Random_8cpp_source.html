59 //Initialize with repsect to nbatch, nx, ny, nf in extended space
60 int Random::initializeFromLoc(HyPerCol* hc, const PVLayerLoc* locptr, bool isExtended) {
61  int status = PV_SUCCESS;
62  parentHyPerCol = hc;
63 
64  PVHalo halo;
65  if (isExtended) {
66  memcpy(&halo, &(locptr->halo), sizeof(halo));
67  }
68  else {
69  halo.lt = 0;
70  halo.rt = 0;
71  halo.dn = 0;
72  halo.up = 0;
73  }
74  int nxExt = locptr->nx + halo.lt + halo.rt;
75  int nyExt = locptr->ny + halo.up + halo.dn;
76  int nf = locptr->nf;
77  int nbatch = locptr->nbatch;
78  rngArraySize = nxExt * nyExt * nf * nbatch;
79  //Calculate global size
80  int nxGlobalExt = locptr->nxGlobal + halo.lt + halo.rt;
81  int nyGlobalExt = locptr->nyGlobal + halo.up + halo.dn;
82  int nbatchGlobal = locptr->nbatchGlobal;
83  int numTotalSeeds = nxGlobalExt * nyGlobalExt * nf * nbatchGlobal;
84  //Allocate buffer to store rngArraySize
85  rngArray = (uint4 *) malloc(rngArraySize*sizeof(uint4));
86  if (rngArray==NULL) {
87  fprintf(stderr, "Random::initialize error: rank %d process unable to allocate memory for %lu RNGs.\n", hc->columnId(), rngArraySize);
88  status = PV_FAILURE;
89  }
90  if (status == PV_SUCCESS) {
91  unsigned int seedBase = hc->getObjectSeed(numTotalSeeds);
92  int sb = nxExt * nyExt * nf;
93  int sy = nxExt * nf;
94  int sbGlobal = nxGlobalExt * nyGlobalExt * nf;
95  int syGlobal = nxGlobalExt * nf;
96 
97  //Only thing that is continuous in memory is nx and ny, so loop over batch and y
98  for(int kb = 0; kb < nbatch; kb++){
99  for(int ky = 0; ky < nyExt; ky++){
100  //Calculate start index into local rngArray
101  int localExtStart = kb * sb + ky * sy;
102  //Calculate offset of the seedBase
103  int globalExtStart = (kb + locptr->kb0) * sbGlobal + (ky + locptr->ky0) * syGlobal + locptr->kx0;
104  //Testing: this line should break
105  //int globalExtStart = (kb ) * sbGlobal + (ky ) * syGlobal;
106  size_t count = nxExt * nf;
107  cl_random_init(&(rngArray[localExtStart]), count, seedBase + globalExtStart);
108  }
109  }
110  }
111  return status;
112 }
