187 int NormalizeBase::addConnToList(HyPerConn * newConn) {
188  HyPerConn ** newList = NULL;
189  if (connectionList) {
190  newList = (HyPerConn **) realloc(connectionList, sizeof(*connectionList)*(numConnections+1));
191  }
192  else {
193  newList = (HyPerConn **) malloc(sizeof(*connectionList)*(numConnections+1));
194  }
195  if (newList==NULL) {
196  pvError().printf("Normalizer \"%s\" unable to add connection \"%s\" as connection number %d : %s\n", name, newConn->getName(), numConnections+1, strerror(errno));
197  }
198  connectionList = newList;
199  connectionList[numConnections] = newConn;
200  numConnections++;
201  if (parent->columnId()==0) {
202  pvInfo().printf("Adding connection \"%s\" to normalizer group \"%s\".\n", newConn->getName(), this->getName());
203  }
204  return PV_SUCCESS;
205 }
