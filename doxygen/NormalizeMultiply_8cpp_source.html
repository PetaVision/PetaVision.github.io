72 int NormalizeMultiply::normalizeWeights() {
73  int status = PV_SUCCESS;
74 
75  // All connections in the group must have the same values of sharedWeights, numArbors, and numDataPatches
76  HyPerConn * conn0 = connectionList[0];
77  for (int c=1; c<numConnections; c++) {
78  HyPerConn * conn = connectionList[c];
79  // Do we need to require sharedWeights be the same for all connections in the group?
80  if (conn->usingSharedWeights()!=conn0->usingSharedWeights()) {
81  if (parent()->columnId() == 0) {
82  fprintf(stderr, "Normalizer %s: All connections in the normalization group must have the same sharedWeights (Connection \"%s\" has %d; connection \"%s\" has %d).\n",
83  this->getName(), conn0->getName(), conn0->usingSharedWeights(), conn->getName(), conn->usingSharedWeights());
84  }
85  status = PV_FAILURE;
86  }
87  if (conn->numberOfAxonalArborLists() != conn0->numberOfAxonalArborLists()) {
88  if (parent()->columnId() == 0) {
89  fprintf(stderr, "Normalizer %s: All connections in the normalization group must have the same number of arbors (Connection \"%s\" has %d; connection \"%s\" has %d).\n",
90  this->getName(), conn0->getName(), conn0->numberOfAxonalArborLists(), conn->getName(), conn->numberOfAxonalArborLists());
91  }
92  status = PV_FAILURE;
93  }
94  if (conn->getNumDataPatches() != conn0->getNumDataPatches()) {
95  if (parent()->columnId() == 0) {
96  fprintf(stderr, "Normalizer %s: All connections in the normalization group must have the same number of data patches (Connection \"%s\" has %d; connection \"%s\" has %d).\n",
97  this->getName(), conn0->getName(), conn0->getNumDataPatches(), conn->getName(), conn->getNumDataPatches());
98  }
99  status = PV_FAILURE;
100  }
101  if (status==PV_FAILURE) {
102  MPI_Barrier(parent()->icCommunicator()->communicator());
103  exit(EXIT_FAILURE);
104  }
105  }
106 
107  // Apply rMinX and rMinY
108  if (rMinX > 0.5f && rMinY > 0.5f) {
109  for (int c=0; c<numConnections; c++) {
110  HyPerConn * conn = connectionList[c];
111  int num_arbors = conn->numberOfAxonalArborLists();
112  int num_patches = conn->getNumDataPatches();
113  int num_weights_in_patch = conn->xPatchSize()*conn->yPatchSize()*conn->fPatchSize();
114  for (int arbor=0; arbor<num_arbors; arbor++) {
115  pvwdata_t * dataPatchStart = conn->get_wDataStart(arbor);
116  for (int patchindex=0; patchindex<num_patches; patchindex++) {
117  applyRMin(dataPatchStart+patchindex*num_weights_in_patch, rMinX, rMinY,
118  conn->xPatchSize(), conn->yPatchSize(), conn->xPatchStride(), conn->yPatchStride());
119  }
120  }
121  }
122  }
123 
124  // Apply nonnegativeConstraintFlag
125  if (nonnegativeConstraintFlag) {
126  for (int c=0; c<numConnections; c++) {
127  HyPerConn * conn = connectionList[c];
128  int num_arbors = conn->numberOfAxonalArborLists();
129  int num_patches = conn->getNumDataPatches();
130  int num_weights_in_patch = conn->xPatchSize()*conn->yPatchSize()*conn->fPatchSize();
131  int num_weights_in_arbor = num_patches * num_weights_in_patch;
132  for (int arbor=0; arbor<num_arbors; arbor++) {
133  pvwdata_t * dataPatchStart = conn->get_wDataStart(arbor);
134  for (int weightindex=0; weightindex<num_weights_in_arbor; weightindex++) {
135  pvwdata_t * w = &dataPatchStart[weightindex];
136  if (*w<0) { *w = 0; }
137  }
138  }
139  }
140  }
141 
142  // Apply normalize_cutoff
143  if (normalize_cutoff>0) {
144  float max = 0.0f;
145  for (int c=0; c<numConnections; c++) {
146  HyPerConn * conn = connectionList[c];
147  int num_arbors = conn->numberOfAxonalArborLists();
148  int num_patches = conn->getNumDataPatches();
149  int num_weights_in_patch = conn->xPatchSize()*conn->yPatchSize()*conn->fPatchSize();
150  for (int arbor=0; arbor<num_arbors; arbor++) {
151  pvwdata_t * dataStart = conn->get_wDataStart(arbor);
152  for (int patchindex=0; patchindex<num_patches; patchindex++) {
153  accumulateMaxAbs(dataStart+patchindex*num_weights_in_patch, num_weights_in_patch, &max);
154  }
155  }
156  }
157  for (int c=0; c<numConnections; c++) {
158  HyPerConn * conn = connectionList[c];
159  int num_arbors = conn->numberOfAxonalArborLists();
160  int num_patches = conn->getNumDataPatches();
161  int num_weights_in_patch = conn->xPatchSize()*conn->yPatchSize()*conn->fPatchSize();
162  for (int arbor=0; arbor<num_arbors; arbor++) {
163  pvwdata_t * dataStart = conn->get_wDataStart(arbor);
164  for (int patchindex=0; patchindex<num_patches; patchindex++) {
165  applyThreshold(dataStart+patchindex*num_weights_in_patch, num_weights_in_patch, max);
166  }
167  }
168  }
169  }
170 
171  return PV_SUCCESS;
172 }

180 int NormalizeMultiply::applyThreshold(pvwdata_t * dataPatchStart, int weights_in_patch, float wMax) {
181  assert(normalize_cutoff>0); // Don't call this routine unless normalize_cutoff was set
182  float threshold = wMax * normalize_cutoff;
183  for (int k=0; k<weights_in_patch; k++) {
184  if (fabsf(dataPatchStart[k])<threshold) dataPatchStart[k] = 0;
185  }
186  return PV_SUCCESS;
187 }
188 
189 // dataPatchStart points to head of full-sized patch
190 // rMinX, rMinY are the minimum radii from the center of the patch,
191 // all weights inside (non-inclusive) of this radius are set to zero
192 // the diameter of the central exclusion region is truncated to the nearest integer value, which may be zero
193 int NormalizeMultiply::applyRMin(pvwdata_t * dataPatchStart, float rMinX, float rMinY,
194  int nxp, int nyp, int xPatchStride, int yPatchStride) {
195  if(rMinX == 0 && rMinY == 0) return PV_SUCCESS;
196  int fullWidthX = floor(2 * rMinX);
197  int fullWidthY = floor(2 * rMinY);
198  int offsetX = ceil((nxp - fullWidthX) / 2.0);
199  int offsetY = ceil((nyp - fullWidthY) / 2.0);
200  int widthX = nxp - 2 * offsetX;
201  int widthY = nyp - 2 * offsetY;
202  pvwdata_t * rMinPatchStart = dataPatchStart + offsetY * yPatchStride + offsetX * xPatchStride;
203  int weights_in_row = xPatchStride * widthX;
204  for (int ky = 0; ky<widthY; ky++){
205  for (int k=0; k<weights_in_row; k++) {
206  rMinPatchStart[k] = 0;
207  }
208  rMinPatchStart += yPatchStride;
209  }
210  return PV_SUCCESS;
211 }
