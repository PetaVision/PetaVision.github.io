39 int RescaleConn::deliverPresynapticPerspective(PVLayerCube const *activity, int arborID) {
40  // Largely a duplicate of IdentConn::deliverPresynapticPerspective, except
41  // for two lines inside for-loops with large numbers of iterations.
42  // We're discussing ways to eliminate code duplication like this without
43  // incurring added computational costs. For now, leaving the duplicate
44  // code as is. --peteschultz, April 15, 2016
45 
46  // Check if we need to update based on connection's channel
47  if (getChannel() == CHANNEL_NOUPDATE) {
48  return PV_SUCCESS;
49  }
50  assert(post->getChannel(getChannel()));
51  assert(getConvertToRateDeltaTimeFactor() == 1.0);
52 
53  const PVLayerLoc *preLoc = preSynapticLayer()->getLayerLoc();
54  const PVLayerLoc *postLoc = postSynapticLayer()->getLayerLoc();
55 
56  assert(arborID == 0); // IdentConn can only have one arbor
57  const int numExtended = activity->numItems;
58 
59 #ifdef DEBUG_OUTPUT
60  int rank;
61  MPI_Comm_rank(parent->icCommunicator()->communicator(), &rank);
62  DebugLog(debugMessage);
63  debugMessage.printf(
64  "[%d]: HyPerLayr::recvSyn: neighbor=%d num=%d actv=%p this=%p conn=%p\n",
65  rank,
66  0,
67  numExtended,
68  activity,
69  this,
70  conn);
71  debugMessage.flush();
72 #endif // DEBUG_OUTPUT
73 
74  for (int b = 0; b < parent->getNBatch(); b++) {
75  float *activityBatch = activity->data
76  + b * (preLoc->nx + preLoc->halo.rt + preLoc->halo.lt)
77  * (preLoc->ny + preLoc->halo.up + preLoc->halo.dn)
78  * preLoc->nf;
79  float *gSynPatchHeadBatch =
80  post->getChannel(getChannel()) + b * postLoc->nx * postLoc->ny * postLoc->nf;
81 
82  if (activity->isSparse) {
83  SparseList<float>::Entry const *activeIndicesBatch =
84  (SparseList<float>::Entry *)activity->activeIndices
85  + b * (preLoc->nx + preLoc->halo.rt + preLoc->halo.lt)
86  * (preLoc->ny + preLoc->halo.up + preLoc->halo.dn) * preLoc->nf;
87  int numLoop = activity->numActive[b];
88 #ifdef PV_USE_OPENMP_THREADS
89 #pragma omp parallel for
90 #endif
91  for (int loopIndex = 0; loopIndex < numLoop; loopIndex++) {
92  int kPre = activeIndicesBatch[loopIndex].index;
93  float a = scale * activeIndicesBatch[loopIndex].value;
94  PVPatch *weights = getWeights(kPre, arborID);
95  if (weights->nx > 0 && weights->ny > 0) {
96  int f = featureIndex(kPre, preLoc->nx, preLoc->ny, preLoc->nf); // Not taking halo
97  // into account, but
98  // for feature index,
99  // shouldn't matter.
100  float *postPatchStart = gSynPatchHeadBatch + getGSynPatchStart(kPre, arborID) + f;
101  *postPatchStart += a;
102  }
103  }
104  }
105  else {
106  PVLayerLoc const *loc = &activity->loc;
107  PVHalo const *halo = &loc->halo;
108  int lineSizeExt = (loc->nx + halo->lt + halo->rt) * loc->nf;
109 #ifdef PV_USE_OPENMP_THREADS
110 #pragma omp parallel for
111 #endif
112  for (int y = 0; y < loc->ny; y++) {
113  float *lineStartPreActivity =
114  &activityBatch[(y + halo->up) * lineSizeExt + halo->lt * loc->nf];
115  int nk = loc->nx * loc->nf;
116  float *lineStartPostGSyn = &gSynPatchHeadBatch[y * nk];
117  for (int k = 0; k < nk; k++) {
118  lineStartPostGSyn[k] += scale * lineStartPreActivity[k];
119  }
120  }
121  }
122  }
123  return PV_SUCCESS;
124 }
