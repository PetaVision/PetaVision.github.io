40 int RescaleConn::deliverPresynapticPerspective(PVLayerCube const * activity, int arborID) {
41  //Largely a duplicate of IdentConn::deliverPresynapticPerspective, except
42  //for two lines inside for-loops with large numbers of iterations.
43  //We're discussing ways to eliminate code duplication like this without
44  //incurring added computational costs. For now, leaving the duplicate
45  //code as is. --peteschultz, April 15, 2016
46 
47  //Check if we need to update based on connection's channel
48  if(getChannel() == CHANNEL_NOUPDATE){
49  return PV_SUCCESS;
50  }
51  assert(post->getChannel(getChannel()));
52 
53  float dt_factor = getConvertToRateDeltaTimeFactor();
54  assert(dt_factor==1.0);
55 
56  const PVLayerLoc * preLoc = preSynapticLayer()->getLayerLoc();
57  const PVLayerLoc * postLoc = postSynapticLayer()->getLayerLoc();
58 
59  assert(arborID==0); // IdentConn can only have one arbor
60  const int numExtended = activity->numItems;
61 
62 #ifdef DEBUG_OUTPUT
63  int rank;
64  MPI_Comm_rank(parent->icCommunicator()->communicator(), &rank);
65  pvDebug(debugMessage);
66  debugMessage.printf("[%d]: HyPerLayr::recvSyn: neighbor=%d num=%d actv=%p this=%p conn=%p\n", rank, 0, numExtended, activity, this, conn);
67  debugMessage.flush();
68 #endif // DEBUG_OUTPUT
69 
70  for(int b = 0; b < parent->getNBatch(); b++){
71  pvdata_t * activityBatch = activity->data + b * (preLoc->nx + preLoc->halo.rt + preLoc->halo.lt) * (preLoc->ny + preLoc->halo.up + preLoc->halo.dn) * preLoc->nf;
72  pvdata_t * gSynPatchHeadBatch = post->getChannel(getChannel()) + b * postLoc->nx * postLoc->ny * postLoc->nf;
73 
74  if (activity->isSparse) {
75  unsigned int * activeIndicesBatch = activity->activeIndices + b * (preLoc->nx + preLoc->halo.rt + preLoc->halo.lt) * (preLoc->ny + preLoc->halo.up + preLoc->halo.dn) * preLoc->nf;
76  int numLoop = activity->numActive[b];
77 #ifdef PV_USE_OPENMP_THREADS
78 #pragma omp parallel for
79 #endif
80  for (int loopIndex = 0; loopIndex < numLoop; loopIndex++) {
81  int kPre = activeIndicesBatch[loopIndex];
82 
83  float a = scale * activityBatch[kPre];
84  // if (a == 0.0f) continue;
85  PVPatch * weights = getWeights(kPre, arborID);
86  if (weights->nx>0 && weights->ny>0) {
87  int f = featureIndex(kPre, preLoc->nx, preLoc->ny, preLoc->nf); // Not taking halo into account, but for feature index, shouldn't matter.
88  pvgsyndata_t * postPatchStart = gSynPatchHeadBatch + getGSynPatchStart(kPre, arborID) + f;
89  *postPatchStart += a;
90  }
91  }
92  }
93  else {
94  PVLayerLoc const * loc = &activity->loc;
95  PVHalo const * halo = &loc->halo;
96  int lineSizeExt = (loc->nx+halo->lt+halo->rt)*loc->nf;
97 #ifdef PV_USE_OPENMP_THREADS
98 #pragma omp parallel for
99 #endif
100  for (int y=0; y<loc->ny; y++) {
101  pvdata_t * lineStartPreActivity = &activityBatch[(y+halo->up)*lineSizeExt+halo->lt*loc->nf];
102  int nk = loc->nx*loc->nf;
103  pvdata_t * lineStartPostGSyn = &gSynPatchHeadBatch[y*nk];
104  for (int k=0; k<nk; k++) {
105  lineStartPostGSyn[k] += scale * lineStartPreActivity[k];
106  }
107  }
108  }
109  }
110  return PV_SUCCESS;
111 }
