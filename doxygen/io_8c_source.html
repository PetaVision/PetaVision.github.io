188 int pv_getopt_str(int argc, char * argv[], const char * opt, char ** sVal, bool * paramusage)
189 {
190  // sVal can be NULL. If sVal is not null and the option is found,
191  // the value of the option is put into sVal and the calling routine is
192  // responsible for freeing it.
193  // Example: if argv[1] is "-p" and argv[2] is "params.pv", and opt is "-p",
194  // *sVal will be "params.pv".
195  int i;
196  for (i = 1; i < argc; i += 1) {
197  if (i+1 < argc && strcmp(argv[i], opt) == 0) {
198  if( sVal != NULL ) *sVal = strdup(argv[i+1]);
199  if (paramusage) { paramusage[i] = true; paramusage[i+1] = true; }
200  return 0;
201  }
202  }
203  if (sVal != NULL) *sVal = NULL;
204  return -1; // not found
205 }

260 char * expandLeadingTilde(char const * path) {
261  char * newpath = NULL;
262  if (path != NULL) {
263  int len = strlen(path);
264  if (len==1 && path[0]=='~') {
265  newpath = strdup(getenv("HOME"));
266  if (newpath==NULL) {
267  fprintf(stderr, "Error expanding \"%s\": home directory not defined\n", path);
268  exit(EXIT_FAILURE);
269  }
270  }
271  else if (len>1 && path[0]=='~' && path[1]=='/') {
272  char * homedir = getenv("HOME");
273  if (homedir==NULL) {
274  fprintf(stderr, "Error expanding \"%s\": home directory not defined\n", path);
275  }
276  char dummy;
277  int chars_needed = snprintf(&dummy, 0, "%s/%s", homedir, &path[2]);
278  newpath = (char *) malloc(chars_needed+1);
279  if (newpath==NULL) {
280  fprintf(stderr, "Unable to allocate memory for path \"%s/%s\"\n", homedir, &path[2]);
281  exit(EXIT_FAILURE);
282  }
283  int chars_used = snprintf(newpath, chars_needed+1, "%s/%s", homedir, &path[2]);
284  assert(chars_used == chars_needed);
285  }
286  else {
287  newpath = strdup(path);
288  }
289  }
290  return newpath;
291 }
