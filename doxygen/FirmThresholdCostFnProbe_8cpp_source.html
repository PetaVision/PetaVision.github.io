77 double FirmThresholdCostFnProbe::getValueInternal(double timevalue, int index) {
78  if (index < 0 || index >= parent->getNBatch()) {
79  return PV_FAILURE;
80  }
81  PVLayerLoc const *loc = getTargetLayer()->getLayerLoc();
82  int const nx = loc->nx;
83  int const ny = loc->ny;
84  int const nf = loc->nf;
85  PVHalo const *halo = &loc->halo;
86  int const lt = halo->lt;
87  int const rt = halo->rt;
88  int const dn = halo->dn;
89  int const up = halo->up;
90  double sum = 0.0;
91  double VThreshPlusVWidth = VThresh + VWidth;
92  double amax = 0.5 * VThreshPlusVWidth;
93  double a2 = 0.5 / VThreshPlusVWidth;
94  float const *aBuffer =
95  getTargetLayer()->getLayerData() + index * getTargetLayer()->getNumExtended();
96 
97  if (getMaskLayer()) {
98  PVLayerLoc const *maskLoc = getMaskLayer()->getLayerLoc();
99  PVHalo const *maskHalo = &maskLoc->halo;
100  float const *maskLayerData =
101  getMaskLayer()->getLayerData()
102  + index * getMaskLayer()->getNumExtended(); // Is there a DataStore method to return the
103  // part of the layer data for a given batch
104  // index?
105  int const maskLt = maskHalo->lt;
106  int const maskRt = maskHalo->rt;
107  int const maskDn = maskHalo->dn;
108  int const maskUp = maskHalo->up;
109  if (maskHasSingleFeature()) {
110  assert(getTargetLayer()->getNumNeurons() == nx * ny * nf);
111  int nxy = nx * ny;
112 #ifdef PV_USE_OPENMP_THREADS
113 #pragma omp parallel for reduction(+ : sum)
114 #endif // PV_USE_OPENMP_THREADS
115  for (int kxy = 0; kxy < nxy; kxy++) {
116  int kexMask = kIndexExtended(kxy, nx, ny, 1, maskLt, maskRt, maskDn, maskUp);
117  if (maskLayerData[kexMask] == 0) {
118  continue;
119  }
120  int featureBase = kxy * nf;
121  for (int f = 0; f < nf; f++) {
122  int kex = kIndexExtended(featureBase++, nx, ny, nf, lt, rt, dn, up);
123  double a = (double)fabs(aBuffer[kex]);
124  if (a >= VThreshPlusVWidth) {
125  sum += amax;
126  }
127  else {
128  sum += a * (1.0 - a2 * a);
129  }
130  }
131  }
132  }
133  else {
134 #ifdef PV_USE_OPENMP_THREADS
135 #pragma omp parallel for reduction(+ : sum)
136 #endif // PV_USE_OPENMP_THREADS
137  for (int k = 0; k < getTargetLayer()->getNumNeurons(); k++) {
138  int kex = kIndexExtended(k, nx, ny, nf, lt, rt, dn, up);
139  double a = (double)fabs(aBuffer[kex]);
140  if (a == 0) {
141  continue;
142  }
143  int kexMask = kIndexExtended(k, nx, ny, nf, maskLt, maskRt, maskDn, maskUp);
144  if (maskLayerData[kexMask] == 0) {
145  continue;
146  }
147  if (a >= VThreshPlusVWidth) {
148  sum += amax;
149  }
150  else {
151  sum += a * (1.0 - a2 * a);
152  }
153  }
154  }
155  }
156  else {
157  if (getTargetLayer()->getSparseFlag()) {
158  PVLayerCube cube = getTargetLayer()->getPublisher()->createCube();
159  long int numActive = cube.numActive[index];
160  int numItems = cube.numItems / cube.loc.nbatch;
161  SparseList<float>::Entry const *activeList =
162  &((SparseList<float>::Entry *)cube.activeIndices)[index * numItems];
163 #ifdef PV_USE_OPENMP_THREADS
164 #pragma omp parallel for reduction(+ : sum)
165 #endif // PV_USE_OPENMP_THREADS
166  for (int k = 0; k < numActive; k++) {
167  int extIndex = activeList[k].index;
168  int inRestricted = !extendedIndexInBorderRegion(
169  extIndex, nx, ny, nf, halo->lt, halo->rt, halo->dn, halo->up);
170  double a = inRestricted * (double)fabs(aBuffer[extIndex]);
171  if (a >= VThreshPlusVWidth) {
172  sum += amax;
173  }
174  else {
175  sum += a * (1.0 - a2 * a);
176  }
177  }
178  }
179  else {
180 #ifdef PV_USE_OPENMP_THREADS
181 #pragma omp parallel for reduction(+ : sum)
182 #endif // PV_USE_OPENMP_THREADS
183  for (int k = 0; k < getTargetLayer()->getNumNeurons(); k++) {
184  int kex = kIndexExtended(k, nx, ny, nf, lt, rt, dn, up);
185  double a = (double)fabs(aBuffer[kex]);
186  if (a == 0) {
187  continue;
188  }
189  if (a >= VThreshPlusVWidth) {
190  sum += amax;
191  }
192  else {
193  sum += a * (1.0 - a2 * a);
194  }
195  }
196  }
197  }
198 
199  return sum;
200 }
