310 static inline int nearby_neighbor(int kzPre, int zScaleLog2Pre, int zScaleLog2Post)
311 {
312  float a = powf(2.0f, (float) (zScaleLog2Pre - zScaleLog2Post));
313  int ia = (int) a;
314 
315  int k0 = (ia < 2) ? 0 : ia/2 - 1;
316 
317  // negative kzPre is different if density of post-synaptic layer decreases
318  int k = (a < 1.0f && kzPre < 0) ? kzPre - (int) (1.0f/a) + 1 : kzPre;
319 
320  return k0 + (int) (a * k);
321 }

509 static inline float deltaWithPBC(float x1, float x2, float max)
510 {
511  float dx = x2 - x1;
512  float abs_dx = fabsf(dx);
513 
514  // Apply periodic boundary conditions
515  dx = abs_dx > max ? sign(dx) * (abs_dx - 2.0f*max) : dx;
516 
517  return dx;
518 }

542 //Converts an index from one layer to the other in the extended space
543 //Warning: function will return center point in a one to many conversion
544 //Conversion in feature space does not exist, output will be first feature
545 //If outside the area of out layer, will move to the clostest avaliable position in out layer
546 static inline int layerIndexExt(int kPreExt, const PVLayerLoc * inLoc, const PVLayerLoc * outLoc){
547  //Calculate scale factor based on restricted
548  float scaleFactorX = (float)outLoc->nxGlobal / inLoc->nxGlobal;
549  float scaleFactorY = (float)outLoc->nyGlobal / inLoc->nyGlobal;
550  //Calculate x and y in extended space
551  int kPreX = kxPos(kPreExt, inLoc->nx + inLoc->halo.lt + inLoc->halo.rt, inLoc->ny+ inLoc->halo.dn + inLoc->halo.up, inLoc->nf);
552  int kPreY = kyPos(kPreExt, inLoc->nx + inLoc->halo.lt + inLoc->halo.rt, inLoc->ny+ inLoc->halo.dn + inLoc->halo.up, inLoc->nf);
553  //Subtract margin to set 0 to the beginning of the restricted space
554  kPreX -= inLoc->halo.lt;
555  kPreY -= inLoc->halo.up;
556  int kPostX, kPostY, half;
557  //If one to many, scale factor is greater than 1
558  if (scaleFactorX > 1){
559  half = floor(scaleFactorX / 2);
560  kPostX = kPreX * scaleFactorX + half;
561  }
562  else{
563  kPostX = floor(kPreX * scaleFactorX);
564  }
565  if (scaleFactorY > 1){
566  half = floor(scaleFactorY / 2);
567  kPostY = kPreY * scaleFactorY + half;
568  }
569  else{
570  kPostY = floor(kPreY * scaleFactorY);
571  }
572 
573  //Change back to ext points
574  kPostX += outLoc->halo.lt;
575  kPostY += outLoc->halo.up;
576 
577  //If outside of out layer margins, shrink
578  //Left margin
579  if (kPostX < 0){
580  kPostX = 0;
581  }
582  //Right Margin
583  else if(kPostX >= outLoc->nx + outLoc->halo.lt + outLoc->halo.rt){
584  kPostX = outLoc->nx + outLoc->halo.dn + outLoc->halo.up - 1;
585  }
586  //Top margin
587  if (kPostY < 0){
588  kPostY = 0;
589  }
590  //Bottom Margin
591  else if(kPostY >= outLoc->ny + outLoc->halo.lt + outLoc->halo.rt){
592  kPostY = outLoc->ny + outLoc->halo.dn + outLoc->halo.up - 1;
593  }
594  //Change back to index
595  //Using feature of 0
596  return kIndex(kPostX, kPostY, 0, outLoc->nx + outLoc->halo.lt + outLoc->halo.rt, outLoc->ny + outLoc->halo.dn + outLoc->halo.up, outLoc->nf);
597 }

610 //Converts a local ext index into a global res index
611 //Returns -1 if localExtK is in extended space
612 static inline int localExtToGlobalRes(int localExtK, const PVLayerLoc * loc){
613  //Change local ext indicies to global res index
614  int localExtX = kxPos(localExtK, loc->nx+loc->halo.lt+loc->halo.rt, loc->ny+loc->halo.up+loc->halo.dn, loc->nf);
615  int localExtY = kyPos(localExtK, loc->nx+loc->halo.lt+loc->halo.rt, loc->ny+loc->halo.up+loc->halo.dn, loc->nf);
616  int localF = featureIndex(localExtK, loc->nx+loc->halo.lt+loc->halo.rt, loc->ny+loc->halo.up+loc->halo.dn, loc->nf);
617 
618  //Check if restricted
619  if(localExtX < loc->halo.lt || localExtX >= loc->nx+loc->halo.lt ||
620  localExtY < loc->halo.up || localExtY >= loc->ny+loc->halo.up){
621  return -1;
622  }
623 
624  //Change ext to res
625  int globalResX = localExtX - loc->halo.lt + loc->kx0;
626  int globalResY = localExtY - loc->halo.up + loc->ky0;
627 
628  //Get final globalResK
629  int globalResK = kIndex(globalResX, globalResY, localF, loc->nxGlobal, loc->nyGlobal, loc->nf);
630  return globalResK;
631 }
