264 static inline int nearby_neighbor(int kzPre, int zScaleLog2Pre, int zScaleLog2Post) {
265  float a = powf(2.0f, (float)(zScaleLog2Pre - zScaleLog2Post));
266  int ia = (int)a;
267 
268  int k0 = (ia < 2) ? 0 : ia / 2 - 1;
269 
270  // negative kzPre is different if density of post-synaptic layer decreases
271  int k = (a < 1.0f && kzPre < 0) ? kzPre - (int)(1.0f / a) + 1 : kzPre;
272 
273  return k0 + (int)(a * k);
274 }

461 static inline float deltaWithPBC(float x1, float x2, float max) {
462  float dx = x2 - x1;
463  float abs_dx = fabsf(dx);
464 
465  // Apply periodic boundary conditions
466  dx = abs_dx > max ? sign(dx) * (abs_dx - 2.0f * max) : dx;
467 
468  return dx;

501 // Converts an index from one layer to the other in the extended space
502 // Warning: function will return center point in a one to many conversion
503 // Conversion in feature space does not exist, output will be first feature
504 // If outside the area of out layer, will move to the clostest avaliable position in out layer
505 static inline int layerIndexExt(int kPreExt, const PVLayerLoc *inLoc, const PVLayerLoc *outLoc) {
506  // Calculate scale factor based on restricted
507  float scaleFactorX = (float)outLoc->nxGlobal / inLoc->nxGlobal;
508  float scaleFactorY = (float)outLoc->nyGlobal / inLoc->nyGlobal;
509  // Calculate x and y in extended space
510  int kPreX =
511  kxPos(kPreExt,
512  inLoc->nx + inLoc->halo.lt + inLoc->halo.rt,
513  inLoc->ny + inLoc->halo.dn + inLoc->halo.up,
514  inLoc->nf);
515  int kPreY =
516  kyPos(kPreExt,
517  inLoc->nx + inLoc->halo.lt + inLoc->halo.rt,
518  inLoc->ny + inLoc->halo.dn + inLoc->halo.up,
519  inLoc->nf);
520  // Subtract margin to set 0 to the beginning of the restricted space
521  kPreX -= inLoc->halo.lt;
522  kPreY -= inLoc->halo.up;
523  int kPostX, kPostY, half;
524  // If one to many, scale factor is greater than 1
525  if (scaleFactorX > 1) {
526  half = floor(scaleFactorX / 2);
527  kPostX = kPreX * scaleFactorX + half;
528  }
529  else {
530  kPostX = floor(kPreX * scaleFactorX);
531  }
532  if (scaleFactorY > 1) {
533  half = floor(scaleFactorY / 2);
534  kPostY = kPreY * scaleFactorY + half;
535  }
536  else {
537  kPostY = floor(kPreY * scaleFactorY);
538  }
539 
540  // Change back to ext points
541  kPostX += outLoc->halo.lt;
542  kPostY += outLoc->halo.up;
543 
544  // If outside of out layer margins, shrink
545  // Left margin
546  if (kPostX < 0) {
547  kPostX = 0;
548  }
549  // Right Margin
550  else if (kPostX >= outLoc->nx + outLoc->halo.lt + outLoc->halo.rt) {
551  kPostX = outLoc->nx + outLoc->halo.dn + outLoc->halo.up - 1;
552  }
553  // Top margin
554  if (kPostY < 0) {
555  kPostY = 0;
556  }
557  // Bottom Margin
558  else if (kPostY >= outLoc->ny + outLoc->halo.lt + outLoc->halo.rt) {
559  kPostY = outLoc->ny + outLoc->halo.dn + outLoc->halo.up - 1;
560  }
561  // Change back to index
562  // Using feature of 0
563  return kIndex(
564  kPostX,
565  kPostY,
566  0,
567  outLoc->nx + outLoc->halo.lt + outLoc->halo.rt,
568  outLoc->ny + outLoc->halo.dn + outLoc->halo.up,
569  outLoc->nf);
570 }

601 // Converts a local ext index into a global res index
602 // Returns -1 if localExtK is in extended space
603 static inline int localExtToGlobalRes(int localExtK, const PVLayerLoc *loc) {
604  // Change local ext indicies to global res index
605  int localExtX =
606  kxPos(localExtK,
607  loc->nx + loc->halo.lt + loc->halo.rt,
608  loc->ny + loc->halo.up + loc->halo.dn,
609  loc->nf);
610  int localExtY =
611  kyPos(localExtK,
612  loc->nx + loc->halo.lt + loc->halo.rt,
613  loc->ny + loc->halo.up + loc->halo.dn,
614  loc->nf);
615  int localF = featureIndex(
616  localExtK,
617  loc->nx + loc->halo.lt + loc->halo.rt,
618  loc->ny + loc->halo.up + loc->halo.dn,
619  loc->nf);
620 
621  // Check if restricted
622  if (localExtX < loc->halo.lt || localExtX >= loc->nx + loc->halo.lt || localExtY < loc->halo.up
623  || localExtY >= loc->ny + loc->halo.up) {
624  return -1;
625  }
626 
627  // Change ext to res
628  int globalResX = localExtX - loc->halo.lt + loc->kx0;
629  int globalResY = localExtY - loc->halo.up + loc->ky0;
630 
631  // Get final globalResK
632  int globalResK = kIndex(globalResX, globalResY, localF, loc->nxGlobal, loc->nyGlobal, loc->nf);
633  return globalResK;
634 }

696 static inline int requiredConvolveMargin(int nPre, int nPost, int patchSize) {
697  int margin = 0;
698  if (nPre == nPost) {
699  assert(patchSize % 2 == 1);
700  margin = (patchSize - 1) / 2;
701  }
702  else if (nPre > nPost) { // many-to-one
703  assert(nPre % nPost == 0);
704  int densityRatio = nPre / nPost;
705  assert(densityRatio % 2 == 0);
706  assert(pow(2.0, nearbyint(log2(densityRatio))) == (double)densityRatio);
707  margin = (patchSize - 1) * densityRatio / 2;
708  }
709  else {
710  assert(nPre < nPost); // one-to-many
711  assert(nPost % nPre == 0);
712  int densityRatio = nPost / nPre;
713  assert(densityRatio % 2 == 0);
714  assert(pow(2.0, nearbyint(log2(densityRatio))) == (double)densityRatio);
715  assert(patchSize % densityRatio == 0);
716  int numCells = patchSize / densityRatio;
717  margin = numCells / 2;
718  // integer division is correct, no matter whether numCells is even or odd
719  }
720  return margin;
721 }
