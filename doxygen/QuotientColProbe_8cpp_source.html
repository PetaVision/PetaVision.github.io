72 int QuotientColProbe::communicateInitInfo() {
73  int status = ColProbe::communicateInitInfo();
74  numerProbe = findProbe(numerator);
75  denomProbe = findProbe(denominator);
76  if (numerProbe==NULL || denomProbe==NULL) {
77  status = PV_FAILURE;
78  if (parent->columnId()==0) {
79  if (numerProbe==NULL) {
80  pvErrorNoExit().printf("%s \"%s\": numerator probe \"%s\" could not be found.\n", getKeyword(), getName(), numerator);
81  }
82  if (denomProbe==NULL) {
83  pvErrorNoExit().printf("%s \"%s\": denominator probe \"%s\" could not be found.\n", getKeyword(), getName(), denominator);
84  }
85  }
86  }
87  if (status == PV_SUCCESS) {
88  int nNumValues = numerProbe->getNumValues();
89  int dNumValues = denomProbe->getNumValues();
90  if (nNumValues != dNumValues) {
91  if (parent->columnId()==0) {
92  pvErrorNoExit().printf("%s \"%s\": numerator probe \"%s\" and denominator probe \"%s\" have differing numbers of values (%d vs. %d)\n",
93  getKeyword(), getName(), numerator, denominator, nNumValues, dNumValues);
94  }
95  MPI_Barrier(this->getParent()->icCommunicator()->communicator());
96  exit(EXIT_FAILURE);
97  }
98  status = setNumValues(nNumValues);
99  if (status != PV_SUCCESS) {
100  pvErrorNoExit().printf("%s \"%s\": unable to allocate memory for %d values: %s\n",
101  this->getKeyword(), this->getName(), nNumValues, strerror(errno));
102  exit(EXIT_FAILURE);
103  }
104  }
105  if (status != PV_SUCCESS) {
106  MPI_Barrier(parent->icCommunicator()->communicator());
107  exit(EXIT_FAILURE);
108  }
109  return status;
110 }

121 int QuotientColProbe::calcValues(double timeValue) {
122  int numValues = this->getNumValues();
123  double * valuesBuffer = getValuesBuffer();
124  if (parent->simulationTime()==parent->getStartTime()) {
125  for (int b=0; b<numValues; b++) {
126  valuesBuffer[b] = parent->getTimeScaleMin();
127  }
128  return PV_SUCCESS;
129  }
130  double n[numValues];
131  numerProbe->getValues(timeValue, n);
132  double d[numValues];
133  denomProbe->getValues(timeValue, d);
134  for (int b=0; b<numValues; b++) {
135  valuesBuffer[b] = n[b]/d[b];
136  }
137  return PV_SUCCESS;
138 }
