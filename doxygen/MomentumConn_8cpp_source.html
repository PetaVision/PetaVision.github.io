217 int MomentumConn::calc_dW() {
218  assert(plasticityFlag);
219  int status;
220  timeBatchIdx = (timeBatchIdx + 1) % timeBatchPeriod;
221 
222  //Clear at time 0, update at time timeBatchPeriod - 1
223  bool need_update_w = false;
224  bool need_clear_dw = false;
225  if(timeBatchIdx == 0){
226  need_clear_dw = true;
227  }
228 
229  //If updating next timestep, update weights here
230  if((timeBatchIdx + 1) % timeBatchPeriod == 0){
231  need_update_w = true;
232  }
233 
234  for(int arborId=0;arborId<numberOfAxonalArborLists();arborId++) {
235  //Clear every batch period
236  if(need_clear_dw){
237  status = initialize_dW(arborId);
238  if (status==PV_BREAK) { break; }
239  assert(status == PV_SUCCESS);
240  }
241  }
242 
243  for(int arborId=0;arborId<numberOfAxonalArborLists();arborId++) {
244  //Sum up parts every timestep
245  status = update_dW(arborId);
246  if (status==PV_BREAK) { break; }
247  assert(status == PV_SUCCESS);
248  }
249 
250  for(int arborId=0;arborId<numberOfAxonalArborLists();arborId++) {
251  //Reduce only when we need to update
252  if(need_update_w){
253  status = reduce_dW(arborId);
254  if (status==PV_BREAK) { break; }
255  assert(status == PV_SUCCESS);
256  }
257  }
258 
259  for(int arborId=0;arborId<numberOfAxonalArborLists();arborId++) {
260  //Normalize only when reduced
261  if(need_update_w){
262  status = normalize_dW(arborId);
263  if (status==PV_BREAK) { break; }
264  assert(status == PV_SUCCESS);
265  }
266  }
267  return PV_SUCCESS;
268 }

270 int MomentumConn::updateWeights(int arborId){
271  if(timeBatchIdx != timeBatchPeriod - 1){
272  return PV_SUCCESS;
273  }
274  //Add momentum right before updateWeights
275  for(int kArbor = 0; kArbor < this->numberOfAxonalArborLists(); kArbor++){
276  applyMomentum(arborId);
277  }
278 
279  //Saved to prevweights
280  assert(prev_dwDataStart);
281  std::memcpy(*prev_dwDataStart, *get_dwDataStart(),
282  sizeof(pvwdata_t) *
283  numberOfAxonalArborLists() *
284  nxp * nyp * nfp *
285  getNumDataPatches());
286 
287 
288  // add dw to w
289  for(int kArbor = 0; kArbor < this->numberOfAxonalArborLists(); kArbor++){
290  pvwdata_t * w_data_start = get_wDataStart(kArbor);
291  for( long int k=0; k<patchStartIndex(getNumDataPatches()); k++ ) {
292  w_data_start[k] += get_dwDataStart(kArbor)[k];
293  }
294  }
295  return PV_BREAK;
296 }

373 int MomentumConn::applyMomentum(int arbor_ID){
374  int nExt = preSynapticLayer()->getNumExtended();
375  const PVLayerLoc * loc = preSynapticLayer()->getLayerLoc();
376  if(sharedWeights){
377  int numKernels = getNumDataPatches();
378  //Shared weights done in parallel, parallel in numkernels
379 #ifdef PV_USE_OPENMP_THREADS
380 #pragma omp parallel for
381 #endif
382  for(int kernelIdx = 0; kernelIdx < numKernels; kernelIdx++){
383  pvwdata_t * dwdata_start = get_dwDataHead(arbor_ID, kernelIdx);
384  pvwdata_t * prev_dw_start = get_prev_dwDataHead(arbor_ID, kernelIdx);
385  pvwdata_t * wdata_start = get_wDataHead(arbor_ID, kernelIdx);
386  if(!strcmp(momentumMethod, "simple")){
387  for(int k = 0; k < nxp*nyp*nfp; k++){
388  dwdata_start[k] += momentumTau * prev_dw_start[k] - momentumDecay*wdata_start[k];
389  }
390  }
391  else if(!strcmp(momentumMethod, "viscosity")){
392  for(int k = 0; k < nxp*nyp*nfp; k++){
393  //dwdata_start[k] = momentumTau * (prev_dw_start[k] + dwdata_start[k]) * (1 - exp(-1.0/ momentumTau)) - momentumDecay*wdata_start[k];
394  dwdata_start[k] = (prev_dw_start[k] * exp(-1.0/ momentumTau)) + dwdata_start[k] - momentumDecay*wdata_start[k];
395  }
396  }
397  else if(!strcmp(momentumMethod, "alex")){
398  for(int k = 0; k < nxp*nyp*nfp; k++){
399  //weight_inc[i] := momW * weight_inc[i-1] - wc * epsW * weights[i-1] + epsW * weight_grads[i]
400  // weights[i] := weights[i-1] + weight_inc[i]
401  dwdata_start[k] = momentumTau * prev_dw_start[k] - momentumDecay * getDWMax()* wdata_start[k] + dwdata_start[k];
402  }
403  }
404  }
405  }
406  else{
407  std::cout << "Warning: Momentum not implemented for non-shared weights, not implementing momentum\n";
408  }
409  return PV_SUCCESS;
410 }
