38 int MomentumConn::allocateDataStructures() {
39  int status = HyPerConn::allocateDataStructures();
40  if (status == PV_POSTPONE) {
41  return status;
42  }
43  if (!plasticityFlag)
44  return status;
45  int sx = nfp;
46  int sy = sx * nxp;
47  int sp = sy * nyp;
48  int nPatches = getNumDataPatches();
49 
50  const int numAxons = numberOfAxonalArborLists();
51 
52  // Allocate dw buffer for previous dw
53  prev_dwDataStart = (float **)pvCalloc(numAxons, sizeof(float *));
54  std::size_t numWeights = (std::size_t)(numAxons * nxp * nyp * nfp) * (std::size_t)nPatches;
55  prev_dwDataStart[0] = (float *)pvCalloc(numWeights, sizeof(float));
56  for (int arborId = 0; arborId < numAxons; arborId++) {
57  prev_dwDataStart[arborId] = (prev_dwDataStart[0] + sp * nPatches * arborId);
58  pvAssert(prev_dwDataStart[arborId] != NULL);
59  } // loop over arbors
60 
61  return PV_SUCCESS;
62 }

154 void MomentumConn::applyMomentum(int arbor_ID) {
155  // Shared weights done in parallel, parallel in numkernels
156  switch (method) {
157  case SIMPLE: applyMomentum(arbor_ID, momentumTau, momentumDecay); break;
158  case VISCOSITY: applyMomentum(arbor_ID, std::exp(-1.0f / momentumTau), momentumDecay); break;
159  case ALEX: applyMomentum(arbor_ID, momentumTau, momentumDecay * getDWMax()); break;
160  default: pvAssertMessage(0, "Unrecognized momentumMethod\n"); break;
161  }
162 }

164 void MomentumConn::applyMomentum(int arbor_ID, float dwFactor, float wFactor) {
165  int const numKernels = getNumDataPatches();
166 #ifdef PV_USE_OPENMP_THREADS
167 #pragma omp parallel for
168 #endif
169  for (int kernelIdx = 0; kernelIdx < numKernels; kernelIdx++) {
170  float *dwdata_start = get_dwDataHead(arbor_ID, kernelIdx);
171  float const *prev_dw_start = get_prev_dwDataHead(arbor_ID, kernelIdx);
172  float const *wdata_start = get_wDataHead(arbor_ID, kernelIdx);
173  for (int k = 0; k < nxp * nyp * nfp; k++) {
174  dwdata_start[k] += dwFactor * prev_dw_start[k] - wFactor * wdata_start[k];
175  }
176  }
177 }
