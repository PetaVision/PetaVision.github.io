130 int CopyConn::communicateInitInfo(CommunicateInitInfoMessage const *message) {
131  int status = PV_SUCCESS;
132  this->originalConn = message->lookup<HyPerConn>(std::string(originalConnName));
133  if (originalConn == nullptr) {
134  if (parent->columnId() == 0) {
135  ErrorLog().printf(
136  "%s: originalConnName \"%s\" is not an connection in the column.\n",
137  getDescription_c(),
138  originalConnName);
139  status = PV_FAILURE;
140  }
141  }
142  if (status != PV_SUCCESS)
143  return status;
144 
145  if (!originalConn->getInitInfoCommunicatedFlag()) {
146  if (parent->columnId() == 0) {
147  InfoLog().printf(
148  "%s must wait until original connection \"%s\" has "
149  "finished its communicateInitInfo "
150  "stage.\n",
151  getDescription_c(),
152  originalConn->getName());
153  }
154  return PV_POSTPONE;
155  }
156 
157  sharedWeights = originalConn->usingSharedWeights();
158  parent->parameters()->handleUnnecessaryParameter(name, "sharedWeights", sharedWeights);
159 
160  numAxonalArborLists = originalConn->numberOfAxonalArborLists();
161  parent->parameters()->handleUnnecessaryParameter(name, "numAxonalArbors", numAxonalArborLists);
162 
163  plasticityFlag = originalConn->getPlasticityFlag();
164  parent->parameters()->handleUnnecessaryParameter(name, "plasticityFlag", plasticityFlag);
165 
166  status = HyPerConn::communicateInitInfo(message);
167 
168  return status;
169 }

225 int CopyConn::copy(int arborId) {
226  size_t arborsize =
227  (size_t)(xPatchSize() * yPatchSize() * fPatchSize() * getNumDataPatches()) * sizeof(float);
228  memcpy(get_wDataStart(arborId), originalConn->get_wDataStart(arborId), arborsize);
229  return PV_SUCCESS;
