136 int CopyConn::communicateInitInfo() {
137  int status = PV_SUCCESS;
138  BaseConnection * originalConnBase = parent->getConnFromName(this->originalConnName);
139  if (originalConnBase==NULL) {
140  if (parent->columnId()==0) {
141  pvErrorNoExit().printf("%s \"%s\": originalConnName \"%s\" does not refer to any connection in the column.\n", this->getKeyword(), name, this->originalConnName);
142  }
143  MPI_Barrier(parent->icCommunicator()->communicator());
144  exit(EXIT_FAILURE);
145  }
146  this->originalConn = dynamic_cast<HyPerConn *>(originalConnBase);
147  if (originalConn == NULL) {
148  if (parent->columnId()==0) {
149  pvErrorNoExit().printf("CopyConn \"%s\": originalConnName \"%s\" is not an existing connection.\n", name, originalConnName);
150  status = PV_FAILURE;
151  }
152  }
153  if (status != PV_SUCCESS) return status;
154 
155  if (!originalConn->getInitInfoCommunicatedFlag()) {
156  if (parent->columnId()==0) {
157  const char * connectiontype = this->getKeyword();
158  pvInfo().printf("%s \"%s\" must wait until original connection \"%s\" has finished its communicateInitInfo stage.\n", connectiontype, name, originalConn->getName());
159  }
160  return PV_POSTPONE;
161  }
162 
163  sharedWeights = originalConn->usingSharedWeights();
164  parent->parameters()->handleUnnecessaryParameter(name, "sharedWeights", sharedWeights);
165 
166  numAxonalArborLists = originalConn->numberOfAxonalArborLists();
167  parent->parameters()->handleUnnecessaryParameter(name, "numAxonalArbors", numAxonalArborLists);
168 
169  plasticityFlag = originalConn->getPlasticityFlag();
170  parent->parameters()->handleUnnecessaryParameter(name, "plasticityFlag", plasticityFlag);
171 
172  status = HyPerConn::communicateInitInfo();
173 
174  return status;
175 }

198 PVPatch*** CopyConn::initializeWeights(PVPatch*** patches, pvwdata_t** dataStart) {
199  assert(originalConn->getInitialValuesSetFlag()); // setInitialValues shouldn't call this function unless original conn has set its own initial values
200  assert(dataStart == get_wDataStart());
201  assert(patches==NULL || patches==get_wPatches());

203  copy(arbor);
204  }
205  return patches;
