53 int LayerProbe::communicateInitInfo(CommunicateInitInfoMessage const *message) {
54  BaseProbe::communicateInitInfo(message);
55  // Set target layer
56  targetLayer = message->lookup<HyPerLayer>(std::string(targetName));
57  if (targetLayer == NULL) {
58  if (parent->columnId() == 0) {
59  ErrorLog().printf(
60  "%s: targetLayer \"%s\" is not a layer in the column.\n",
61  getDescription_c(),
62  targetName);
63  }
64  MPI_Barrier(parent->getCommunicator()->communicator());
65  exit(EXIT_FAILURE);
66  }
