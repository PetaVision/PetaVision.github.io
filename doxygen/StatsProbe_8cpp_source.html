149 void StatsProbe::ioParam_buffer(enum ParamsIOFlag ioFlag) {
150  char * buffer = NULL;
151  if (ioFlag == PARAMS_IO_WRITE) {
152  switch(type) {
153  case BufV:
154  buffer = strdup("MembranePotential");
155  break;
156  case BufActivity:
157  buffer = strdup("Activity");
158  }
159  }
160  getParent()->ioParamString(ioFlag, getName(), "buffer", &buffer, "Activity", true/*warnIfAbsent*/);
161  if (ioFlag == PARAMS_IO_READ) {
162  assert(buffer);
163  size_t len = strlen(buffer);
164  for (size_t c=0; c<len; c++) {
165  buffer[c] = (char) tolower((int) buffer[c]);
166  }
167  if (!strcmp(buffer, "v") || !strcmp(buffer, "membranepotential")) {
168  type = BufV;
169  }
170  else if (!strcmp(buffer, "a") || !strcmp(buffer, "activity")) {
171  type = BufActivity;
172  }
173  else {
174  if (getParent()->columnId()==0) {
175  const char * bufnameinparams = getParent()->parameters()->stringValue(getName(), "buffer");
176  assert(bufnameinparams);
177  fprintf(stderr, "%s \"%s\" error: buffer \"%s\" is not recognized.\n",
178  this->getKeyword(), getName(), bufnameinparams);
179  }
180  MPI_Barrier(getParent()->icCommunicator()->communicator());
181  exit(EXIT_FAILURE);
182  }
183  }
184  free(buffer); buffer = NULL;
185 }

199 int StatsProbe::outputState(double timed)
200 {
201 #ifdef PV_USE_MPI
202  InterColComm * icComm = getTargetLayer()->getParent()->icCommunicator();
203  MPI_Comm comm = icComm->communicator();
204  int rank = icComm->commRank();
205  const int rcvProc = 0;
206 #endif // PV_USE_MPI
207 
208  int nk;
209  const pvdata_t * buf;
210  resetStats();
211 
212  nk = getTargetLayer()->getNumNeurons();
213 
214  int nbatch = getTargetLayer()->getParent()->getNBatch();
215 
216  switch (type) {
217  case BufV:
218  comptimer->start();
219  for(int b = 0; b < nbatch; b++){
220  buf = getTargetLayer()->getV() + b * getTargetLayer()->getNumNeurons();
221  if( buf == NULL ) {
222 #ifdef PV_USE_MPI
223  if( rank != rcvProc ) return 0;
224 #endif // PV_USE_MPI
225  fprintf(outputstream->fp, "%sV buffer is NULL\n", getMessage());
226  return 0;
227  }
228  for( int k=0; k<nk; k++ ) {
229  pvdata_t a = buf[k];
230  sum[b] += a;
231  sum2[b] += a*a;
232  if (fabs((double) a)>(double) nnzThreshold){
233  nnz[b]++;
234  } // Optimize for different datatypes of a?
235  if (a < fMin[b]) fMin[b] = a;
236  if (a > fMax[b]) fMax[b] = a;
237  }
238  }
239  comptimer->stop();
240  break;
241  case BufActivity:
242  comptimer->start();
243  for(int b = 0; b < nbatch; b++){
244  buf = getTargetLayer()->getLayerData() + b * getTargetLayer()->getNumExtended();
245  assert(buf != NULL);
246  for( int k=0; k<nk; k++ ) {
247  const PVLayerLoc * loc = getTargetLayer()->getLayerLoc();
248  int kex = kIndexExtended(k, loc->nx, loc->ny, loc->nf, loc->halo.lt, loc->halo.rt, loc->halo.dn, loc->halo.up); // TODO: faster to use strides and a double-loop than compute kIndexExtended for every neuron?
249  pvdata_t a = buf[kex];
250  sum[b] += a;
251  sum2[b] += a*a;
252  if (fabs((double) a)>(double) nnzThreshold) {
253  nnz[b]++; // Optimize for different datatypes of a?
254  }
255  if( a < fMin[b] ) fMin[b] = a;
256  if( a > fMax[b] ) fMax[b] = a;
257  }
258  }
259  comptimer->stop();
260  break;
261  default:
262  assert(0);
263  break;
264  }
265 
266 #ifdef PV_USE_MPI
267  mpitimer->start();
268  int ierr;
269 
270  //In place reduction to prevent allocating a temp recv buffer
271  if(rank == rcvProc){
272  ierr = MPI_Reduce(MPI_IN_PLACE, sum, nbatch, MPI_DOUBLE, MPI_SUM, rcvProc, comm);
273  ierr = MPI_Reduce(MPI_IN_PLACE, sum2, nbatch, MPI_DOUBLE, MPI_SUM, rcvProc, comm);
274  ierr = MPI_Reduce(MPI_IN_PLACE, nnz, nbatch, MPI_INT, MPI_SUM, rcvProc, comm);
275  ierr = MPI_Reduce(MPI_IN_PLACE, fMin, nbatch, MPI_FLOAT, MPI_MIN, rcvProc, comm);
276  ierr = MPI_Reduce(MPI_IN_PLACE, fMax, nbatch, MPI_FLOAT, MPI_MAX, rcvProc, comm);
277  ierr = MPI_Reduce(MPI_IN_PLACE, &nk, 1, MPI_INT, MPI_SUM, rcvProc, comm);
278  }
279  else{
280  ierr = MPI_Reduce(sum, sum, nbatch, MPI_DOUBLE, MPI_SUM, rcvProc, comm);
281  ierr = MPI_Reduce(sum2, sum2, nbatch, MPI_DOUBLE, MPI_SUM, rcvProc, comm);
282  ierr = MPI_Reduce(nnz, nnz, nbatch, MPI_INT, MPI_SUM, rcvProc, comm);
283  ierr = MPI_Reduce(fMin, fMin, nbatch, MPI_FLOAT, MPI_MIN, rcvProc, comm);
284  ierr = MPI_Reduce(fMax, fMax, nbatch, MPI_FLOAT, MPI_MAX, rcvProc, comm);
285  ierr = MPI_Reduce(&nk, &nk, 1, MPI_INT, MPI_SUM, rcvProc, comm);
286  return 0;
287  }
288 
289  mpitimer->stop();
290 #endif // PV_USE_MPI
291  double divisor = nk;
292 
293  iotimer->start();
294  for(int b = 0; b < nbatch; b++){
295  avg[b] = sum[b]/divisor;
296  sigma[b] = sqrt(sum2[b]/divisor - avg[b]*avg[b]);
297  if ( type == BufActivity && getTargetLayer()->getSparseFlag() ) {
298  float freq = 1000.0 * avg[b];
299  fprintf(outputstream->fp, "%st==%6.1f b==%d N==%d Total==%f Min==%f Avg==%f Hz (/dt ms) Max==%f sigma==%f nnz==%d\n", getMessage(), (double)timed, (int)b, (int)divisor, (double)sum[b], (double)fMin[b], (double)freq, (double)fMax[b], (double)sigma[b], (int)nnz[b]);
300  }
301  else {
302  fprintf(outputstream->fp, "%st==%6.1f b==%d N==%d Total==%f Min==%f Avg==%f Max==%f sigma==%f nnz==%d\n", getMessage(), (double)timed, (int)b, (int)divisor, (double)sum[b], (double)fMin[b], (double) avg[b], (double)fMax[b], (double) sigma[b], (int)nnz[b]);
303  }
304  fflush(outputstream->fp);
305  }
306 
307  iotimer->stop();
308 
309  return PV_SUCCESS;
310 }
