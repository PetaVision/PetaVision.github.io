117  }
118  MPI_Barrier(this->getParent()->icCommunicator()->communicator());
119  exit(EXIT_FAILURE);
120  }
121  assert(numVertices==0 || numVertices==numVerticesV);
122  numVertices = numVerticesV;

250 int PtwiseLinearTransferLayer::checkVertices() {
251  assert(numVertices>0);
252  int status = PV_SUCCESS;
253  for (int v=1; v<numVertices; v++) {
254  if (verticesV[v] < verticesV[v-1]) {
255  status = PV_FAILURE;
256  if (this->getParent()->columnId()==0) {
257  fprintf(stderr, "%s \"%s\" error: vertices %d and %d: V-coordinates decrease from %f to %f.\n",
258  this->getKeyword(), this->getName(), v, v+1, verticesV[v-1], verticesV[v]);
259  }
260  }
261  }
262  return status;
263 }

265 int PtwiseLinearTransferLayer::setSlopes() {
266  assert(numVertices>0);
267  assert(verticesA!=NULL);
268  assert(verticesV!=NULL);
269  slopes = (float *) malloc((size_t)(numVertices+1)*sizeof(*slopes));
270  if (slopes == NULL) {
271  fprintf(stderr, "%s \"%s\" error: unable to allocate memory for transfer function slopes: %s\n",
272  this->getKeyword(), name, strerror(errno));
273  exit(EXIT_FAILURE);
274 
275  }
276  slopes[0] = slopeNegInf;
277  slopes[numVertices] = slopePosInf;
278  for (int k=1; k<numVertices; k++) {
279  float V1 = verticesV[k-1];
280  float V2 = verticesV[k];
281  if (V1!=V2) {
282  slopes[k] = (verticesA[k]-verticesA[k-1])/(V2-V1);
283  }
284  else {
285  slopes[k] = verticesA[k]>verticesA[k-1] ? std::numeric_limits<float>::infinity() :
286  verticesA[k]<verticesA[k-1] ? -std::numeric_limits<float>::infinity() :
287  std::numeric_limits<float>::quiet_NaN();
288  }
289  }
290  return PV_SUCCESS;
291 }
