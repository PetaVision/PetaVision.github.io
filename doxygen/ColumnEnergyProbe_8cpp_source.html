67 int ColumnEnergyProbe::addTerm(BaseProbe *probe) {
68  if (probe == NULL) {
69  return PV_FAILURE;
70  }
71  int status = PV_SUCCESS;
72  if (numTerms == 0) {
73  int newNumValues = probe->getNumValues();
74  if (newNumValues < 0) {
75  status = PV_FAILURE;
76  if (parent->columnId() == 0) {
77  ErrorLog().printf(
78  "%s: %s cannot be used as a term of the energy "
79  "probe (getNumValue() returned a "
80  "negative number).\n",
81  getDescription_c(),
82  probe->getDescription_c());
83  }
84  MPI_Barrier(parent->getCommunicator()->communicator());
85  exit(EXIT_FAILURE);
86  }
87  if (newNumValues != this->getNumValues()) {
88  status = setNumValues(newNumValues);
89  if (status != PV_SUCCESS) {
90  ErrorLog().printf(
91  "%s: unable to allocate memory for %d probe values: %s\n",
92  getDescription_c(),
93  newNumValues,
94  strerror(errno));
95  exit(EXIT_FAILURE);
96  }
97  }
98  }
99  else {
100  if (probe->getNumValues() != this->getNumValues()) {
101  if (this->parent->columnId() == 0) {
102  ErrorLog().printf(
103  "Failed to add terms to %s: new probe \"%s\" "
104  "returns %d values, but previous "
105  "probes return %d values\n",
106  getDescription_c(),
107  probe->getName(),
108  probe->getNumValues(),
109  this->getNumValues());
110  }
111  MPI_Barrier(this->parent->getCommunicator()->communicator());
112  exit(EXIT_FAILURE);
113  }
114  }
115  assert(probe->getNumValues() == getNumValues());
116  int newNumTerms = numTerms + (size_t)1;
117  if (newNumTerms <= numTerms) {
118  if (this->parent->columnId() == 0) {
119  ErrorLog().printf(
120  "How did you manage to add %zu terms to %s? "
121  "Unable to add any more!\n",
122  numTerms,
123  getDescription_c());
124  }
125  MPI_Barrier(this->parent->getCommunicator()->communicator());
126  exit(EXIT_FAILURE);
127  }
128  BaseProbe **newTermsArray =
129  (BaseProbe **)realloc(terms, (numTerms + (size_t)1) * sizeof(BaseProbe *));
130  if (newTermsArray == NULL) {
131  ErrorLog().printf(
132  "%s: unable to add term %zu (\"%s\"): %s\n",
133  getDescription_c(),
134  numTerms + (size_t)1,
135  probe->getName(),
136  strerror(errno));
137  exit(EXIT_FAILURE);
138  }
139  terms = newTermsArray;
140  terms[numTerms] = probe;
141  numTerms = newNumTerms;
142  return PV_SUCCESS;
143 } // end ColumnEnergyProbe::addTerm(BaseProbe *, double)

149 int ColumnEnergyProbe::calcValues(double timevalue) {
150  if (mLastTimeValue == timevalue || --mSkipTimer > 0) {
151  mLastTimeValue = timevalue;
152  return PV_SUCCESS;
153  }
154  mSkipTimer = mSkipInterval + 1;
155  double *valuesBuffer = getValuesBuffer();
156  int numValues = this->getNumValues();
157  memset(valuesBuffer, 0, numValues * sizeof(*valuesBuffer));
158  double energy1[numValues];
159  for (int n = 0; n < numTerms; n++) {
160  BaseProbe *p = terms[n];
161  p->getValues(timevalue, energy1);
162  double coeff = p->getCoefficient();
163  for (int b = 0; b < numValues; b++) {
164  valuesBuffer[b] += coeff * energy1[b];
165  }
166  }
167  return PV_SUCCESS;
168 }

184  }
185  stream.printf("%10f, %d, %10.9f\n", timevalue, b, valuesBuffer[b]);
186  stream.flush();
187  }
