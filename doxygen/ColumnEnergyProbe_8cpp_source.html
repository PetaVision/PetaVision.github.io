48 int ColumnEnergyProbe::addTerm(BaseProbe * probe, double coefficient, size_t vectorSize) {
49  if (probe==NULL) { return PV_FAILURE; }
50  if (numTerms>0 && vectorSize != this->vectorSize) {
51  if (this->getParent()->columnId()==0) {
52  fprintf(stderr, "Error adding terms to %s \%s\": vector size %zu of new probe \"%s\" does not agree with existing vector size %zu\n",
53  getKeyword(), getName(), vectorSize, probe->getName(), this->vectorSize);
54  }
55  exit(EXIT_FAILURE);
56  }
57  this->vectorSize = vectorSize;
58  size_t newNumTerms = numTerms+(size_t) 1;
59  if (newNumTerms<=numTerms) {
60  if (this->getParent()->columnId()==0) {
61  fprintf(stderr, "How did you manage to add %zu terms to %s \"%s\"? Unable to add any more!\n",
62  numTerms, getKeyword(), getName());
63  }
64  MPI_Barrier(this->getParent()->icCommunicator()->communicator());
65  exit(EXIT_FAILURE);
66  }
67  energyTerm * newTermsArray = (energyTerm *) realloc(terms, (numTerms+(size_t) 1)*sizeof(energyTerm));
68  if (newTermsArray==NULL) {
69  fprintf(stderr, "%s \"%s\" error: unable to add term %zu (\"%s\"): %s\n",
70  getKeyword(), getName(), numTerms+(size_t) 1, probe->getName(),
71  strerror(errno));
72  exit(EXIT_FAILURE);
73  }
74  terms = newTermsArray;
75  terms[numTerms].probe = probe;
76  terms[numTerms].coeff = coefficient;
77  numTerms = newNumTerms;
78  return PV_SUCCESS;
79 } // end ColumnEnergyProbe::addTerm(BaseProbe *, double)

81 int ColumnEnergyProbe::getValues(double timevalue, std::vector<double> * values) {
82  values->assign(vectorSize, 0.0);
83  for (int n=0; n<numTerms; n++) {
84  std::vector<double> energy1;
85  energyTerm * p = &terms[n];
86  p->probe->getValues(timevalue, &energy1);
87  for (int b=0; b<vectorSize; b++) {
88  values->at(b) += p->coeff * energy1.at(b);
89  }
90  }
91  return PV_SUCCESS;
92 }

94 double ColumnEnergyProbe::getValue(double timevalue, int index) {
95  if (index<0 || index>=vectorSize) {
96  return std::numeric_limits<double>::signaling_NaN();
97  }
98  double sum = 0;
99  for (int n=0; n<numTerms; n++) {
100  energyTerm * p = &terms[n];
101  sum += p->coeff * p->probe->getValue(timevalue, index);
102  }
103  return sum;
104 } // end ColumnEnergyProbe::evaluate(float)
