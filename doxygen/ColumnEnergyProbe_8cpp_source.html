48 int ColumnEnergyProbe::addTerm(BaseProbe * probe) {
49  if (probe==NULL) { return PV_FAILURE; }
50  int status = PV_SUCCESS;
51  if (numTerms==0) {
52  int newNumValues = probe->getNumValues();
53  if (newNumValues < 0) {
54  status = PV_FAILURE;
55  if (parent->columnId()==0) {
56  fprintf(stderr, "%s \"%s\" error: probe \"%s\" cannot be used as a term of the energy probe (getNumValue() returned a negative number).\n",
57  getKeyword(), getName(), probe->getName());
58  }
59  MPI_Barrier(parent->icCommunicator()->communicator());
60  exit(EXIT_FAILURE);
61  }
62  if (newNumValues != this->getNumValues()) {
63  status = setNumValues(newNumValues);
64  if (status != PV_SUCCESS) {
65  fprintf(stderr, "%s \"%s\" error: unable to allocate memory for %d probe values: %s\n",
66  this->getKeyword(), this->getName(), newNumValues, strerror(errno));
67  exit(EXIT_FAILURE);
68  }
69  }
70  }
71  else {
72  if (probe->getNumValues() != this->getNumValues()) {
73  if (this->getParent()->columnId()==0) {
74  fprintf(stderr, "Error adding terms to %s \%s\": new probe \"%s\" returns %d values, but previous probes return %d values\n",
75  getKeyword(), getName(), probe->getName(), probe->getNumValues(), this->getNumValues());
76  }
77  MPI_Barrier(this->getParent()->icCommunicator()->communicator());
78  exit(EXIT_FAILURE);
79  }
80  }
81  assert(probe->getNumValues()==getNumValues());
82  int newNumTerms = numTerms+(size_t) 1;
83  if (newNumTerms<=numTerms) {
84  if (this->getParent()->columnId()==0) {
85  fprintf(stderr, "How did you manage to add %zu terms to %s \"%s\"? Unable to add any more!\n",
86  numTerms, getKeyword(), getName());
87  }
88  MPI_Barrier(this->getParent()->icCommunicator()->communicator());
89  exit(EXIT_FAILURE);
90  }
91  BaseProbe ** newTermsArray = (BaseProbe **) realloc(terms, (numTerms+(size_t) 1)*sizeof(BaseProbe *));
92  if (newTermsArray==NULL) {
93  fprintf(stderr, "%s \"%s\" error: unable to add term %zu (\"%s\"): %s\n",
94  getKeyword(), getName(), numTerms+(size_t) 1, probe->getName(),
95  strerror(errno));
96  exit(EXIT_FAILURE);
97  }
98  terms = newTermsArray;
99  terms[numTerms] = probe;
100  numTerms = newNumTerms;
101  return PV_SUCCESS;
102 } // end ColumnEnergyProbe::addTerm(BaseProbe *, double)

108 int ColumnEnergyProbe::calcValues(double timevalue) {
109  double * valuesBuffer = getValuesBuffer();
110  int numValues = this->getNumValues();
111  memset(valuesBuffer, 0, numValues*sizeof(*valuesBuffer));
112  double energy1[numValues];
113  for (int n=0; n<numTerms; n++) {
114  BaseProbe * p = terms[n];
115  p->getValues(timevalue, energy1);
116  double coeff = p->getCoefficient();
117  for (int b=0; b<numValues; b++) {
118  valuesBuffer[b] += coeff * energy1[b];
119  }
120  }
121  return PV_SUCCESS;
122 }
