146 HyPerCol * build(PV_Init* initObj,
147  void * (*customgroups)(const char *, const char *, HyPerCol *)) {
148  HyPerCol * hc = new HyPerCol("column", initObj);
149  if( hc == NULL ) {
150  fprintf(stderr, "Unable to create HyPerCol\n");
151  return NULL;
152  }
153  PVParams * hcparams = hc->parameters();
154  int numGroups = hcparams->numberOfGroups();
155 
156  // Make sure first group defines a column
157  if( strcmp(hcparams->groupKeywordFromIndex(0), "HyPerCol") ) {
158  fprintf(stderr, "First group of params file did not define a HyPerCol.\n");
159  delete hc;
160  return NULL;
161  }
162 
163  CoreParamGroupHandler * handler = new CoreParamGroupHandler();
164  for( int k=0; k<numGroups; k++ ) {
165  const char * kw = hcparams->groupKeywordFromIndex(k);
166  const char * name = hcparams->groupNameFromIndex(k);
167  ParamGroupType groupType = handler->getGroupType(kw);
168  void * addedObject = NULL;
169  InitWeights * weightInitializer = NULL;
170  NormalizeBase * weightNormalizer = NULL;
171  switch(groupType) {
172  case UnrecognizedGroupType:
173  if (customgroups != NULL) {
174  addedObject = customgroups(kw, name, hc); // Handler for custom groups
175  }
176  break;
177  case HyPerColGroupType:
178  addedObject = handler->createHyPerCol(kw, name, hc);
179  break;
180  case LayerGroupType:
181  addedObject = handler->createLayer(kw, name, hc);
182  break;
183  case ConnectionGroupType:
184  addedObject = createConnection(handler, NULL, 0, kw, name, hc);
185  break;
186  case ProbeGroupType:
187  addedObject = handler->createProbe(kw, name, hc);
188  break;
189  case ColProbeGroupType: // TODO: make ColProbe a subclass of BaseProbe and eliminate this group type.
190  addedObject = handler->createColProbe(kw, name, hc);
191  break;
192  default:
193  assert(0); // All possibilities for groupType are handled above
194  }
195  if(addedObject == NULL) {
196  if (hc->globalRank()==0) {
197  fprintf(stderr, "Parameter group \"%s\": %s could not be created.\n", name, kw);
198  }
199  MPI_Barrier(hc->icCommunicator()->globalCommunicator());
200  exit(EXIT_FAILURE);
201  }
202  }
203  delete handler;
204 
205  if( hc->numberOfLayers() == 0 ) {
206  fprintf(stderr, "HyPerCol \"%s\" does not have any layers.\n", hc->getName());
207  delete hc;
208  return NULL;
209  }
210  return hc;
211 }

254 HyPerCol * build(PV_Init* initObj, ParamGroupHandler ** groupHandlerList, int numGroupHandlers) {
255  HyPerCol * hc = new HyPerCol("column", initObj);
256  if( hc == NULL ) {
257  fprintf(stderr, "Unable to create HyPerCol\n");
258  return NULL;
259  }
260  PVParams * hcparams = hc->parameters();
261  int numGroups = hcparams->numberOfGroups();
262 
263  // Make sure first group defines a column
264  if( strcmp(hcparams->groupKeywordFromIndex(0), "HyPerCol") ) {
265  fprintf(stderr, "First group of params file did not define a HyPerCol.\n");
266  delete hc;
267  return NULL;
268  }
269 
270  CoreParamGroupHandler * coreHandler = new CoreParamGroupHandler(); // The ParamGroupHandler for keywords known by trunk
271  for( int k=0; k<numGroups; k++ ) {
272  const char * kw = hcparams->groupKeywordFromIndex(k);
273  const char * name = hcparams->groupNameFromIndex(k);
274  void * addedObject = NULL;
275  ParamGroupType groupType = UnrecognizedGroupType;
276  ParamGroupHandler * handler = getGroupHandlerFromList(kw, coreHandler, groupHandlerList, numGroupHandlers, &groupType);
277  if (handler == NULL) {
278  if (hc->globalRank()==0) {
279  fprintf(stderr, "Error: parameter group \"%s\": %s is not recognized by any known ParamGroupHandler.\n", name, kw);
280  }
281  MPI_Barrier(hc->icCommunicator()->globalCommunicator());
282  exit(EXIT_FAILURE);
283  }
284  switch (groupType) {
285  case UnrecognizedGroupType:
286  assert(0); // UnrecognizedGroupType should only happen if handler==NULL, and vice versa.
287  break;
288  case HyPerColGroupType:
289  addedObject = handler->createHyPerCol(kw, name, hc);
290  break;
291  case LayerGroupType:
292  addedObject = handler->createLayer(kw, name, hc);
293  break;
294  case ConnectionGroupType:
295  addedObject = createConnection(coreHandler, groupHandlerList, numGroupHandlers, kw, name, hc);
296  break;
297  case ProbeGroupType:
298  addedObject = handler->createProbe(kw, name, hc);
299  break;
300  case ColProbeGroupType: // TODO: make ColProbe a subclass of BaseProbe and eliminate this group type.
301  addedObject = handler->createColProbe(kw, name, hc);
302  break;
303  case WeightInitializerGroupType:
304  addedObject = handler->createWeightInitializer(kw, name, hc);
305  break;
306  case WeightNormalizerGroupType:
307  addedObject = handler->createWeightNormalizer(kw, name, hc);
308  break;
309  default:
310  assert(0); // All possibilities for groupType are handled above
311  }
312  if (addedObject==NULL) {
313  if (hc->globalRank()==0) {
314  fprintf(stderr, "Error creating %s \"%s\".\n", kw, name);
315  }
316  MPI_Barrier(hc->icCommunicator()->globalCommunicator());
317  exit(EXIT_FAILURE);
318  }
319  }
320  delete coreHandler;
321 
322  if( hc->numberOfLayers() == 0 ) {
323  fprintf(stderr, "HyPerCol \"%s\" does not have any layers.\n", hc->getName());
324  delete hc;
325  return NULL;
326  }
327  return hc;
328 }

330 ParamGroupHandler * getGroupHandlerFromList(char const * keyword, CoreParamGroupHandler * coreHandler, ParamGroupHandler ** groupHandlerList, int numGroupHandlers, ParamGroupType * foundGroupType) {
331  ParamGroupType groupType = coreHandler->getGroupType(keyword);
332  if (groupType != UnrecognizedGroupType) {
333  if (foundGroupType) { *foundGroupType = groupType; }
334  return coreHandler;
335  }
336  else {
337  for (int h=0; h<numGroupHandlers; h++) {
338  groupType = groupHandlerList[h]->getGroupType(keyword);
339  if (groupType != UnrecognizedGroupType) {
340  if (foundGroupType) { *foundGroupType = groupType; }
341  return groupHandlerList[h];
342  }
343  }
344  }
345  if (foundGroupType) { *foundGroupType = UnrecognizedGroupType; }
346  return NULL;
347 }

349 BaseConnection * createConnection(CoreParamGroupHandler * coreGroupHandler, ParamGroupHandler ** customHandlerList, int numGroupHandlers, char const * keyword, char const * groupname, HyPerCol * hc) {
350  // The basic logic is:
351  // get the weight initializer by reading the weightInitType parameter, and create an InitWeights object
352  // get the weight normalizer by reading the normalizeMethod parameter, and create a NormalizeBase object
353  // get the connection type from the keyword and create a BaseConnection object, passing the InitWeights and NormalizeBase objects to the connection's constructor.
354  //
355  // The complications are: the weightInitType, normalizeMethod, and connection type could in principle be handled by three different ParamGroupHandlers, so we have to call getGroupHandlerFromList for each one;
356  // and it is allowed for some subclasses not to have a weightInitType and/or a normalizeMethod parameter, so we have to allow for them to be null.
357  ParamGroupType groupType;
358  InitWeights * weightInitializer = NULL;
359  char const * weightInitStr = hc->parameters()->stringValue(groupname, "weightInitType", false/*warnIfAbsent*/);
360  if (weightInitStr!=NULL) {
361  ParamGroupHandler * weightInitHandler = getGroupHandlerFromList(weightInitStr, coreGroupHandler, customHandlerList, numGroupHandlers, &groupType);
362  if (weightInitHandler==NULL || groupType != WeightInitializerGroupType) {
363  fprintf(stderr, "Connection %s error: weightInitType \"%s\" is not recognized.\n", keyword, weightInitStr);
364  exit(EXIT_FAILURE);
365  }
366  weightInitializer = weightInitHandler->createWeightInitializer(weightInitStr, groupname, hc);
367  }
368  NormalizeBase * weightNormalizer = NULL;
369  char const * weightNormalizeStr = hc->parameters()->stringValue(groupname, "normalizeMethod", false/*warnIfAbsent*/);
370  if (weightNormalizeStr!=NULL) {
371  ParamGroupHandler * normalizeHandler = getGroupHandlerFromList(weightNormalizeStr, coreGroupHandler, customHandlerList, numGroupHandlers, &groupType);
372  if (normalizeHandler==NULL || groupType != WeightNormalizerGroupType) {
373  fprintf(stderr, "Connection %s error: normalizeMethod \"%s\" is not recognized.\n", keyword, weightNormalizeStr);
374  exit(EXIT_FAILURE);
375  }
376  weightNormalizer = normalizeHandler->createWeightNormalizer(weightNormalizeStr, groupname, hc);
377  }
378  ParamGroupHandler * connectionHandler = getGroupHandlerFromList(keyword, coreGroupHandler, customHandlerList, numGroupHandlers, &groupType);
379  if (connectionHandler==NULL || groupType != ConnectionGroupType) {
380  fprintf(stderr, "Connection %s error: connection type \"%s\" is not recognized.\n", keyword, weightNormalizeStr);
381  exit(EXIT_FAILURE);
382  }
383  BaseConnection * baseConn = connectionHandler->createConnection(keyword, groupname, hc, weightInitializer, weightNormalizer);
384  return baseConn;
385 }
