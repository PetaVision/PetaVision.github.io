1 /*
2  * buildandrun.cpp
3  *
4  * buildandrun() builds the layers, connections, and
5  * (to a limited extent) probes from the params file and then calls the hypercol's run method.
6  * It deletes the PV_Init and HyPerCol objects that it creates.
7  * Often, the main() function consists only of a call to buildandrun.
8  *
9  * outputParams(argc, argv, path, factory) builds the
10  * layers, connections, etc. and then calls the hypercol's processParams
11  * method, which fills in default parameters, ignores unnecessary parameters
12  * and sends the parameters to the file specified in the path argument.
13  * Relative paths are relative to the params file outputParams deletes the
14  * PV_Init and HyPerCol objects that it creates; it is written to be a
15  * stand-alone function to create a cleaned-up params file.
16  *
17  * build() builds the hypercol but does not run it. That way additional objects
18  * can be created and added by hand if they are not yet supported by build().
19  *
20  * Created on: May 27, 2011
21  * Author: peteschultz
22  */

41 int buildandrun(PV_Init * initObj,
42  int (*custominit)(HyPerCol *, int, char **),
43  int (*customexit)(HyPerCol *, int, char **)) {
44  if(initObj->isExtraProc()){
45  return 0;
46  }
47  PVParams * params = initObj->getParams();
48  if (params==NULL) {
49  if (initObj->getWorldRank()==0) {
50  char const * progName = initObj->getProgramName();
51  if (progName==NULL) { progName = "PetaVision"; }
52  pvErrorNoExit().printf("%s was called without having set a params file\n", progName);
53  }
54  MPI_Barrier(initObj->getComm()->communicator());
55  exit(EXIT_FAILURE);
56  }
57 
58  int numParamSweepValues = initObj->getParams()->getParameterSweepSize();
59 
60  int status = PV_SUCCESS;
61  if (numParamSweepValues) {
62  for (int k=0; k<numParamSweepValues; k++) {
63  if (initObj->getWorldRank()==0) {
64  pvInfo().printf("Parameter sweep: starting run %d of %d\n", k+1, numParamSweepValues);
65  }
66  status = buildandrun1paramset(initObj, custominit, customexit, k) == PV_SUCCESS ? status : PV_FAILURE;
67  }
68  }
69  else{
70  if(initObj->getComm()->numCommBatches() > 1){
71  initObj->getParams()->setBatchSweepValues();
72  }
73  status = buildandrun1paramset(initObj, custominit, customexit) == PV_SUCCESS ? status : PV_FAILURE;
74  }
75 
76  return status;
77 }

288 HyPerCol * build(PV_Init* initObj,
289  void * (*customgroups)(const char *, const char *, HyPerCol *)) {
290  initObj->printBuildAndRunDeprecationWarning("build", "initObj, customgroups");
291  HyPerCol * hc = new HyPerCol("column", initObj);
292  if( hc == NULL ) {
293  pvErrorNoExit().printf("Unable to create HyPerCol\n");
294  return NULL;
295  }
296  PVParams * hcparams = hc->parameters();
297  int numGroups = hcparams->numberOfGroups();
298 
299  // Make sure first group defines a column
300  if( strcmp(hcparams->groupKeywordFromIndex(0), "HyPerCol") ) {
301  pvErrorNoExit().printf("First group of params file did not define a HyPerCol.\n");
302  delete hc;
303  return NULL;
304  }
305 
306  CoreParamGroupHandler * handler = new CoreParamGroupHandler();
307  for( int k=0; k<numGroups; k++ ) {
308  const char * kw = hcparams->groupKeywordFromIndex(k);
309  const char * name = hcparams->groupNameFromIndex(k);
310  ParamGroupType groupType = handler->getGroupType(kw);
311  void * addedObject = NULL;
312  InitWeights * weightInitializer = NULL;
313  NormalizeBase * weightNormalizer = NULL;
314  switch(groupType) {
315  case UnrecognizedGroupType:
316  if (customgroups != NULL) {
317  addedObject = customgroups(kw, name, hc); // Handler for custom groups
318  }
319  break;
320  case HyPerColGroupType:
321  addedObject = handler->createHyPerCol(kw, name, hc);
322  break;
323  case LayerGroupType:
324  addedObject = handler->createLayer(kw, name, hc);
325  break;
326  case ConnectionGroupType:
327  addedObject = createConnection(handler, NULL, 0, kw, name, hc);
328  break;
329  case ProbeGroupType:
330  addedObject = handler->createProbe(kw, name, hc);
331  break;
332  case ColProbeGroupType: // TODO: make ColProbe a subclass of BaseProbe and eliminate this group type.
333  addedObject = handler->createColProbe(kw, name, hc);
334  break;
335  default:
336  assert(0); // All possibilities for groupType are handled above
337  }
338  if(addedObject == NULL) {
339  if (hc->globalRank()==0) {
340  pvErrorNoExit().printf("Parameter group \"%s\": %s could not be created.\n", name, kw);
341  }
342  MPI_Barrier(hc->icCommunicator()->globalCommunicator());
343  exit(EXIT_FAILURE);
344  }
345  }
346  delete handler;
347 
348  if( hc->numberOfLayers() == 0 ) {
349  pvErrorNoExit().printf("HyPerCol \"%s\" does not have any layers.\n", hc->getName());
350  delete hc;
351  return NULL;
352  }
353  return hc;
354 }

411 HyPerCol * build(PV_Init* initObj, ParamGroupHandler ** groupHandlerList, int numGroupHandlers) {
412  HyPerCol * hc = new HyPerCol("column", initObj);
413  if( hc == NULL ) {
414  pvErrorNoExit().printf("Unable to create HyPerCol\n");
415  return NULL;
416  }
417  PVParams * hcparams = hc->parameters();
418  int numGroups = hcparams->numberOfGroups();
419 
420  // Make sure first group defines a column
421  if( strcmp(hcparams->groupKeywordFromIndex(0), "HyPerCol") ) {
422  pvErrorNoExit().printf("First group of params file did not define a HyPerCol.\n");
423  delete hc;
424  return NULL;
425  }
426 
427  CoreParamGroupHandler * coreHandler = new CoreParamGroupHandler(); // The ParamGroupHandler for keywords known by trunk
428  for( int k=0; k<numGroups; k++ ) {
429  const char * kw = hcparams->groupKeywordFromIndex(k);
430  const char * name = hcparams->groupNameFromIndex(k);
431  void * addedObject = NULL;
432  ParamGroupType groupType = UnrecognizedGroupType;
433  ParamGroupHandler * handler = getGroupHandlerFromList(kw, coreHandler, groupHandlerList, numGroupHandlers, &groupType);
434  if (handler == NULL) {
435  if (hc->globalRank()==0) {
436  pvErrorNoExit().printf("Error: parameter group \"%s\": %s is not recognized by any known ParamGroupHandler.\n", name, kw);
437  }
438  MPI_Barrier(hc->icCommunicator()->globalCommunicator());
439  exit(EXIT_FAILURE);
440  }
441  switch (groupType) {
442  case UnrecognizedGroupType:
443  assert(0); // UnrecognizedGroupType should only happen if handler==NULL, and vice versa.
444  break;
445  case HyPerColGroupType:
446  addedObject = handler->createHyPerCol(kw, name, hc);
447  break;
448  case LayerGroupType:
449  addedObject = handler->createLayer(kw, name, hc);
450  break;
451  case ConnectionGroupType:
452  addedObject = createConnection(coreHandler, groupHandlerList, numGroupHandlers, kw, name, hc);
453  break;
454  case ProbeGroupType:
455  addedObject = handler->createProbe(kw, name, hc);
456  break;
457  case ColProbeGroupType: // TODO: make ColProbe a subclass of BaseProbe and eliminate this group type.
458  addedObject = handler->createColProbe(kw, name, hc);
459  break;
460  case WeightInitializerGroupType:
461  addedObject = handler->createWeightInitializer(kw, name, hc);
462  break;
463  case WeightNormalizerGroupType:
464  addedObject = handler->createWeightNormalizer(kw, name, hc);
465  break;
466  default:
467  assert(0); // All possibilities for groupType are handled above
468  }
469  if (addedObject==NULL) {
470  if (hc->globalRank()==0) {
471  pvErrorNoExit().printf("Error creating %s \"%s\".\n", kw, name);
472  }
473  MPI_Barrier(hc->icCommunicator()->globalCommunicator());
474  exit(EXIT_FAILURE);
475  }
476  }
477  delete coreHandler;
478 
479  if( hc->numberOfLayers() == 0 ) {
480  pvErrorNoExit().printf("HyPerCol \"%s\" does not have any layers.\n", hc->getName());
481  delete hc;
482  return NULL;
483  }
484  return hc;
485 }

487 ParamGroupHandler * getGroupHandlerFromList(char const * keyword, CoreParamGroupHandler * coreHandler, ParamGroupHandler ** groupHandlerList, int numGroupHandlers, ParamGroupType * foundGroupType) {
488  ParamGroupType groupType = coreHandler->getGroupType(keyword);
489  if (groupType != UnrecognizedGroupType) {
490  if (foundGroupType) { *foundGroupType = groupType; }
491  return coreHandler;
492  }
493  else {
494  for (int h=0; h<numGroupHandlers; h++) {
495  groupType = groupHandlerList[h]->getGroupType(keyword);
496  if (groupType != UnrecognizedGroupType) {
497  if (foundGroupType) { *foundGroupType = groupType; }
498  return groupHandlerList[h];
499  }
500  }
501  }
502  if (foundGroupType) { *foundGroupType = UnrecognizedGroupType; }
503  return NULL;

506 BaseConnection * createConnection(CoreParamGroupHandler * coreGroupHandler, ParamGroupHandler ** customHandlerList, int numGroupHandlers, char const * keyword, char const * groupname, HyPerCol * hc) {
507  // The basic logic is:
508  // get the weight initializer by reading the weightInitType parameter, and create an InitWeights object
509  // get the weight normalizer by reading the normalizeMethod parameter, and create a NormalizeBase object
510  // get the connection type from the keyword and create a BaseConnection object, passing the InitWeights and NormalizeBase objects to the connection's constructor.
511  //
512  // The complications are: the weightInitType, normalizeMethod, and connection type could in principle be handled by three different ParamGroupHandlers, so we have to call getGroupHandlerFromList for each one;
513  // and it is allowed for some subclasses not to have a weightInitType and/or a normalizeMethod parameter, so we have to allow for them to be null.
514  ParamGroupType groupType;
515  InitWeights * weightInitializer = NULL;
516  char const * weightInitStr = hc->parameters()->stringValue(groupname, "weightInitType", false/*warnIfAbsent*/);
517  if (weightInitStr!=NULL) {
518  ParamGroupHandler * weightInitHandler = getGroupHandlerFromList(weightInitStr, coreGroupHandler, customHandlerList, numGroupHandlers, &groupType);
519  if (weightInitHandler==NULL || groupType != WeightInitializerGroupType) {
520  pvError().printf("Connection %s error: weightInitType \"%s\" is not recognized.\n", keyword, weightInitStr);
521  }
522  weightInitializer = weightInitHandler->createWeightInitializer(weightInitStr, groupname, hc);
523  }
524  NormalizeBase * weightNormalizer = NULL;
525  char const * weightNormalizeStr = hc->parameters()->stringValue(groupname, "normalizeMethod", false/*warnIfAbsent*/);
526  if (weightNormalizeStr!=NULL) {
527  ParamGroupHandler * normalizeHandler = getGroupHandlerFromList(weightNormalizeStr, coreGroupHandler, customHandlerList, numGroupHandlers, &groupType);
528  if (normalizeHandler==NULL || groupType != WeightNormalizerGroupType) {
529  pvError().printf("Connection %s error: normalizeMethod \"%s\" is not recognized.\n", keyword, weightNormalizeStr);
530  }
531  weightNormalizer = normalizeHandler->createWeightNormalizer(weightNormalizeStr, groupname, hc);
532  }
533  ParamGroupHandler * connectionHandler = getGroupHandlerFromList(keyword, coreGroupHandler, customHandlerList, numGroupHandlers, &groupType);
534  if (connectionHandler==NULL || groupType != ConnectionGroupType) {
535  pvError().printf("Connection %s error: connection type \"%s\" is not recognized.\n", keyword, weightNormalizeStr);
536  }
537  BaseConnection * baseConn = connectionHandler->createConnection(keyword, groupname, hc, weightInitializer, weightNormalizer);
538  return baseConn;
539 }
