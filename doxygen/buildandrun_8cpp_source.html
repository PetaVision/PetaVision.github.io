2  * buildandrun.cpp
3  *
4  * buildandrun() builds the layers, connections, and
5  * (to a limited extent) probes from the params file and then calls the
6  * hypercol's run method.
7  * It deletes the PV_Init and HyPerCol objects that it creates.
8  * Often, the main() function consists only of a call to buildandrun.
9  *
10  * outputParams(argc, argv, path, factory) builds the
11  * layers, connections, etc. and then calls the hypercol's processParams
12  * method, which fills in default parameters, ignores unnecessary parameters
13  * and sends the parameters to the file specified in the path argument.
14  * Relative paths are relative to the params file outputParams deletes the
15  * PV_Init and HyPerCol objects that it creates; it is written to be a
16  * stand-alone function to create a cleaned-up params file.
17  *
18  * build() builds the hypercol but does not run it. That way additional objects
19  * can be created and added by hand if they are not yet supported by build().
20  *
21  * Created on: May 27, 2011
22  * Author: peteschultz
23  */

29 // The buildandrun, rebuildandrun, and buildandrun1paramset functions below
30 // automate creating the HyPerCol, filling it with layers, connections, etc.,
31 // and running it. To add custom groups, instantiate a PV_Init object
32 // and call PV_Init::registerKeyword with the create function (in most cases,
33 // the static function template PV::Factory::create<CustomClass>.
34 //
35 // Older versions of buildandrun, that used the now-obsolete customgroups
36 // function
37 // pointer system or the now-obsolete ParamGroupHandler class, were removed
38 // July 27, 2016.
39 int buildandrun(
40  int argc,
41  char *argv[],
42  int (*custominit)(HyPerCol *, int, char **),
43  int (*customexit)(HyPerCol *, int, char **)) {
44  PV_Init initObj(&argc, &argv, false /*value of allowUnrecognizedArguments*/);
45  int status = buildandrun(&initObj, custominit, customexit);
46  return status;
47 }

49 int buildandrun(
50  PV_Init *initObj,
51  int (*custominit)(HyPerCol *, int, char **),
52  int (*customexit)(HyPerCol *, int, char **)) {
53  if (initObj->isExtraProc()) {
54  return 0;
55  }
56  PVParams *params = initObj->getParams();
57  if (params == NULL) {
58  if (initObj->getWorldRank() == 0) {
59  char const *progName = initObj->getProgramName();
60  if (progName == NULL) {
61  progName = "PetaVision";
62  }
63  ErrorLog().printf("%s was called without having set a params file\n", progName);
64  }
65  MPI_Barrier(initObj->getCommunicator()->communicator());
66  exit(EXIT_FAILURE);
67  }
68 
69  int numParamSweepValues = initObj->getParams()->getParameterSweepSize();
70 
71  int status = PV_SUCCESS;
72  if (numParamSweepValues) {
73  for (int k = 0; k < numParamSweepValues; k++) {
74  if (initObj->getWorldRank() == 0) {
75  InfoLog().printf(
76  "Parameter sweep: starting run %d of %d\n", k + 1, numParamSweepValues);
77  }
78  status = buildandrun1paramset(initObj, custominit, customexit, k) == PV_SUCCESS
79  ? status
80  : PV_FAILURE;
81  }
82  }
83  else {
84  status = buildandrun1paramset(initObj, custominit, customexit) == PV_SUCCESS ? status
85  : PV_FAILURE;
86  }
87 
88  return status;
89 }

100 int buildandrun1paramset(
101  PV_Init *initObj,
102  int (*custominit)(HyPerCol *, int, char **),
103  int (*customexit)(HyPerCol *, int, char **),
104  int sweepindex) {
105  if (sweepindex >= 0) {
106  initObj->getParams()->setParameterSweepValues(sweepindex);
107  }
108  HyPerCol *hc = createHyPerCol(initObj);
109  if (hc == nullptr)
110  return PV_FAILURE; // createHyPerCol() prints any error message
111 
112  int status = PV_SUCCESS;
113  int argc = 0;
114  char **argv = NULL;
115  if (custominit || customexit) {
116  argc = initObj->getNumArgs();
117  argv = initObj->getArgsCopy();
118  }
119  if (custominit != NULL) {
120  status = (*custominit)(hc, argc, argv);
121  if (status != PV_SUCCESS) {
122  ErrorLog().printf("custominit function failed with return value %d\n", status);
123  }
124  }
125 
126  if (status == PV_SUCCESS && hc->getInitialStep() < hc->getFinalStep()) {
127  status = hc->run();
128  if (status != PV_SUCCESS) {
129  ErrorLog().printf("HyPerCol::run() returned with error code %d\n", status);
130  }
131  }
132  if (status == PV_SUCCESS && customexit != NULL) {
133  status = (*customexit)(hc, argc, argv);
134  if (status != PV_SUCCESS) {
135  ErrorLog().printf("customexit function failed with return value %d\n", status);
136  }
137  }
138  if (custominit || customexit) {
139  initObj->freeArgs(argc, argv);
140  }
141  delete hc; /* HyPerCol's destructor takes care of deleting layers and
142  connections */
143  return status;
144 }
