149 int Patterns::stringMatch(const char ** allowed_values, const char * stopstring, const char * string_to_match) {
150  int match = -1;
151  for( int i=0; strcmp(allowed_values[i], stopstring); i++ ) {
152  const char * current_string = allowed_values[i];
153  if( !strcmp(string_to_match, current_string) ) {
154  match = i;
155  break;
156  }
157  }
158  return match;
159 }

161 void Patterns::ioParam_patternType(enum ParamsIOFlag ioFlag) {

164  const char * allowed_pattern_types[] = { // these strings should correspond to the types in enum PatternType in Patterns.hpp

484 int Patterns::communicateInitInfo() {
485  int status = BaseInput::communicateInitInfo();
486 
487  patternRandState = new Random(parent, 1);
488 #ifndef NDEBUG
489  // This should put the RNG into the same state across MPI, but let's check.
490  uint4 * state = patternRandState->getRNG(0);
491  uint4 checkState;
492  memcpy(&checkState, state, sizeof(uint4));
493  MPI_Bcast(&checkState, sizeof(uint4), MPI_CHAR, 0, parent->icCommunicator()->communicator());
494  assert(!memcmp(state, &checkState, sizeof(uint4)));
495 #endif // NDEBUG
496 
497  const PVLayerLoc * loc = getLayerLoc();
498  if(dropPosition == -1){
499  nextPosChangeFrame = nextDropFrame + dropPositionRandomMin + floor((dropPositionRandomMax - dropPositionRandomMin) * patternRandState->uniformRandom());
500  xPos = (int)floor(loc->nxGlobal * patternRandState->uniformRandom());
501  yPos = (int)floor(loc->nyGlobal * patternRandState->uniformRandom());
502  }
503  else if(dropPosition == 0){
504  xPos = (int)floor((loc->nxGlobal - 1) / 2);
505  yPos = (int)floor((loc->nyGlobal - 1) / 2);
506  }
507  else{
508  nextPosChangeFrame = nextDropFrame + dropPosition;
509  xPos = (int)floor(loc->nxGlobal * patternRandState->uniformRandom());
510  yPos = (int)floor(loc->nyGlobal * patternRandState->uniformRandom());
511  }
512  MPI_Bcast(&nextDropFrame, 1, MPI_DOUBLE, 0, parent->icCommunicator()->communicator());
513 
514  return status;
515 }

531 int Patterns::drawPattern(float val)
532 {
533  // extended frame
534  const PVLayerLoc * loc = getLayerLoc();
535 
536  const int nx = loc->nx + loc->halo.lt + loc->halo.rt;
537  const int ny = loc->ny + loc->halo.dn + loc->halo.up;
538 
539  // reset data buffer
540  const int nk = nx * ny;
541 
542  float neutralval;
543  if (onOffFlag){
544  neutralval = .5;
545  }
546  else{
547  neutralval = 0.0;
548  }
549 
550  for (int k = 0; k < nk; k++) {
551  data[k] = neutralval;
552  }
553 
554  int status = PV_SUCCESS;
555  if (type == RECTANGLES) {
556  status = drawRectangles(val);
557  }
558  else if (type == BARS) { // type is bars
559  status = drawBars(orientation, data, nx, ny, val);
560  }
561  else if((type == COSWAVE)||(type == SINEWAVE)||
562  (type == COSV)||(type == SINEV)) {
563  status = drawWaves(val);
564  }
565  else if (type == IMPULSE) {
566  status = drawImpulse();
567  }
568  else if (type == DROP){
569  status = drawDrops();
570  }
571 
572  if (normalizeLuminanceFlag) { // Copied from Image::readImage except for names of a couple variables. Make a normalizeLuminance() function?
573  int n=getNumNeurons();
574  double image_sum = 0.0f;
575  for (int k=0; k<n; k++) {
576  image_sum += data[k];
577  }
578  double image_ave = image_sum / n;
579 #ifdef PV_USE_MPI
580  MPI_Allreduce(MPI_IN_PLACE, &image_ave, 1, MPI_DOUBLE, MPI_SUM, parent->icCommunicator()->communicator());
581  image_ave /= parent->icCommunicator()->commSize();
582 #endif // PV_USE_MPI
583  float image_shift = 0.5f - image_ave;
584  for (int k=0; k<n; k++) {
585  data[k] += image_shift;
586  }
587  }
588 
589  return 0;
590 }

592 int Patterns::drawBars(OrientationMode ormode, pvdata_t * buf, int nx, int ny, float val) {
593  int crossstride, alongstride; // strides in the direction across the bar and along the bar, respectively
594  int crosssize, alongsize; // Size of buffer in the direction across the bar and along the bar, respectively
595  int wavelength;
596  int width;
597  switch(ormode) {
598  case vertical:
599  crossstride = 1; // assumes number of features = 1;
600  alongstride = nx;
601  crosssize = nx;
602  alongsize = ny;
603  wavelength = wavelengthVert;
604  width = maxWidth;
605  break;
606  case horizontal:
607  crossstride = nx;
608  alongstride = 1; // assumes number of features = 1;
609  crosssize = ny;
610  alongsize = nx;
611  wavelength = wavelengthHoriz;
612  width = maxHeight;
613  break;
614  case mixed:
615  default:
616  assert(0);
617  break;
618  }
619 
620  // data is val on [position,position+width), discretized by pixels
621  // Set the pixel at floor(position) to maxVal*( 1-(mod(position,1)) ).
622  // Subsequent pixels are set to val (in the while loop), until the
623  // remaining mass is less than maxVal;
624  // The next pixel after that is set to the remainder.
625  // The pattern is then repeated with period wavelength in the direction
626  // across the bar; and repeated with period one in the direction along
627  // the bar.
628  float mass = val*width;
629  float point = position;
630  int k = ((int) floor(point)) % wavelength;
631  float dm = val*(1-(point-k));
632  buf[k*crossstride] = dm;
633  mass -= dm;
634  dm = val;
635  while( mass > dm ) {
636  k++; k %= wavelength;
637  data[k*crossstride] = dm;
638  mass -= dm;
639  }
640  k++; k %= wavelength;
641  data[k*crossstride] = mass;
642 
643  // Repeat with period wavelength
644  for( k = wavelength; k < crosssize; k++ ) {
645  data[k*crossstride] = data[(k-wavelength)*crossstride];
646  }
647 
648  // Repeat in the direction along the bar
649  for( int m = 0; m < crosssize; m++ ) {
650  int idxsrc = m*crossstride;
651  for( k = 1; k < alongsize; k++ ) {
652  int idxdest = idxsrc + k*alongstride;
653  data[idxdest] = data[idxsrc];
654  }
655  }
656 
657  return PV_SUCCESS;
658 }

771 int Patterns::drawDrops() {
772  int status = PV_SUCCESS;
773  const PVLayerLoc * loc = getLayerLoc();
774  const int nx = loc->nx + loc->halo.lt + loc->halo.rt;
775  const int ny = loc->ny + loc->halo.dn + loc->halo.up;
776  const int sx = 1;
777  const int sy = sx * nx;
778  const int kx0 = loc->kx0;
779  const int ky0 = loc->ky0;
780  const int nxgl = loc->nxGlobal;
781  const int nygl = loc->nyGlobal;
782 
783  // For Drops, jittering moves the centers of the drops by offsetX,offsetY.
784  // Since we always redraw the pattern, we don't need jitter's return value.
785  // jitter should be added to other pattern types, in which case the call
786  // to jitter() should be moved to updatePattern.
787  if (jitterFlag) {
788  jitter();
789  // Because the Patterns layer has its own uint4 random state, and all MPI processes seeded it the same way,
790  // all MPI processes have the same offset and bias, without needing to do an MPI call.
791  }
792 
793  //Max radius at corner of screen
794  float max_radius = sqrt(nxgl * nxgl + nygl * nygl);
795 
796  //Using iterators to iterate while removing from loop
797  for(std::vector<Drop>::iterator dropIt = vDrops.begin(); dropIt < vDrops.end(); dropIt++){
798  //Update radius
799  dropIt->radius += dropIt->speed;
800  //If no longer in the frame either in or out
801  if(dropIt->radius >= max_radius || dropIt->radius < 0){
802 
803  //Erase from vector, erase returns next iterator object
804  dropIt = vDrops.erase(dropIt);
805  }
806  }
807 
808  //Change x and y position if needed
809  if(framenumber >= nextPosChangeFrame && dropPosition != 0){
810  if(dropPosition == -1){
811  nextPosChangeFrame += dropPositionRandomMin + floor((dropPositionRandomMax - dropPositionRandomMin) * patternRandState->uniformRandom());
812  xPos = (int)(floor(loc->nxGlobal * patternRandState->uniformRandom()));
813  yPos = (int)(floor(loc->nyGlobal * patternRandState->uniformRandom()));
814  }
815  else{
816  nextPosChangeFrame += dropPosition;
817  xPos = (int)(floor(loc->nxGlobal * patternRandState->uniformRandom()));
818  yPos = (int)(floor(loc->nyGlobal * patternRandState->uniformRandom()));
819  }
820  //No need to communicate it since drop creator will decide where to drop
821  }
822 
823  //Add new circles
824  if(framenumber >= nextDropFrame && framenumber <= endFrame){
825  if(dropPeriod == -1){
826  nextDropFrame = framenumber + dropPeriodRandomMin + floor((dropPeriodRandomMax - dropPeriodRandomMin) * patternRandState->uniformRandom());
827  }
828  else{
829  nextDropFrame = framenumber + dropPeriod;
830  }
831  //Create new structure
832  Drop newDrop;
833  //Random drop speed
834  if(dropSpeed == -1){
835  newDrop.speed = dropSpeedRandomMin + (dropSpeedRandomMax - dropSpeedRandomMin) * patternRandState->uniformRandom();
836  }
837  else{
838  newDrop.speed = dropSpeed;
839  }
840  newDrop.centerX = xPos;
841  newDrop.centerY = yPos;
842  //Random on/off input
843  if(patternRandState->uniformRandom() < .5){
844  newDrop.on = true;
845  }
846  else{
847  newDrop.on = false;
848  }
849  memset(&newDrop.padding, 0, 3);
850  //If out is true, out drop
851  bool out;
852  //Convert inOut into a scale between 0 and 1, where 0 maps to in and 1 maps to out
853  float inOutChance = (inOut + 1)/2;
854  if(patternRandState->uniformRandom() < inOutChance){
855  out = true;
856  }
857  else{
858  out = false;
859  }
860 
861  if (out){
862  newDrop.radius = 0;
863  }
864  else{
865  //Start at max radius
866  newDrop.radius = max_radius;
867  //Reverse sign of speed
868  newDrop.speed = -newDrop.speed;
869  }
870 
871  //Communicate to rest of processors
872  MPI_Bcast(&nextDropFrame, 1, MPI_DOUBLE, 0, parent->icCommunicator()->communicator());
873  MPI_Bcast(&newDrop, sizeof(Drop), MPI_BYTE, 0, parent->icCommunicator()->communicator());
874  vDrops.push_back(newDrop);
875  }
876 
877  //Draw circle
878  for(int i = 0; i < (int)vDrops.size(); i++){
879  double delta_theta = fabs(atan(0.1/(double) vDrops[i].radius));
880  for (double theta = 0; theta < 2*PI; theta += delta_theta){
881  int ix = (int)(round(getOffsetX(this->offsetAnchor, this->offsets[0]) + vDrops[i].centerX + vDrops[i].radius * cos(theta)));
882  int iy = (int)(round(getOffsetY(this->offsetAnchor, this->offsets[1]) + vDrops[i].centerY + vDrops[i].radius * sin(theta)));
883  //Check edge bounds based on nx/ny size
884  if(ix < nx + kx0 && iy < ny + ky0 && ix >= kx0 && iy >= ky0){
885  //Random either on circle or off circle
886  if(vDrops[i].on){
887  data[(ix - kx0) * sx + (iy - ky0) * sy] = maxVal;
888  }
889  else{
890  data[(ix - kx0) * sx + (iy - ky0) * sy] = minVal;
891  }
892  }
893  }
894  }//End radius for loop
895  return status;
896 }

1011 float Patterns::calcPosition(float pos, int step)
1012 {
1013  double p = patternRandState->uniformRandom();
1014 
1015  switch (movementType) {
1016  case MOVEFORWARD:
1017  pos = (pos+movementSpeed) ;
1018  if(pos>step) {pos -= step;}
1019  if(pos<0) {pos += step;}
1020  if(pos>step) {pos -= step;}
1021  break;
1022  case MOVEBACKWARD:
1023  pos = (pos-movementSpeed) ;
1024  if(pos<0) {pos += step;}
1025  if(pos>step) {pos -= step;}
1026  break;
1027  case RANDOMJUMP:
1028  pos = floor(p * step);
1029  break;
1030  case RANDOMWALK:
1031  default: //in case of any problems with setting the movementType var, just use the
1032  //random walk as default
1033  if (p < 0.5){
1034  pos = ((int)(pos+movementSpeed)) % step;
1035  } else {
1036  pos = ((int)(pos-movementSpeed)) % step;
1037  }
1038  break;
1039  }
1040  if (patternsFile != NULL) {
1041  assert(parent->columnId()==0);
1042  fprintf(patternsFile->fp, "Time %f, position %f\n", parent->simulationTime(), pos);
1043  }
1044 
1045  return pos;
