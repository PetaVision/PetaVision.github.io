282  // Bicubic convolution kernel with a=-1
283  inline static pvadata_t bicubic(pvadata_t x) {
284  pvadata_t const absx = fabsf(x); // assumes pvadata_t is float ; ideally should generalize
285  return absx < 1 ? 1 + absx*absx*(-2 + absx) : absx < 2 ? 4 + absx*(-8 + absx*(5-absx)) : 0;
286 
287  }
