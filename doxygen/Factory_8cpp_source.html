222 int Factory::registerKeyword(char const * keyword, ObjectCreateFn creator) {
223  KeywordHandler const * keywordHandler = getKeywordHandler(keyword);
224  if (keywordHandler != NULL) {
225  return PV_FAILURE;
226  }
227  KeywordHandler * newKeyword = new KeywordHandler(keyword, creator);
228  keywordHandlerList.push_back(newKeyword);
229  return PV_SUCCESS;
230 }

237 KeywordHandler const * Factory::getKeywordHandler(char const * keyword) const {
238  for (std::vector<KeywordHandler*>::const_iterator iter = keywordHandlerList.begin(); iter < keywordHandlerList.end(); iter++) {
239  KeywordHandler * typeCreator = *iter;
240  if (!strcmp(typeCreator->getKeyword(), keyword)) {

243  }
244  return NULL;
