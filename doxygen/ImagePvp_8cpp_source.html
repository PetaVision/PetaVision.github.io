111 int ImagePvp::readPvp(const char * filename, int frameNumber) {
112 
113  //Root process reads a PVP file into imageData.
114  //All processes must be called because pvp_read_header requires that,
115  //but only root process does actual I/O.
116  //Scattering takes place in BaseInput::scatterInput.
117 
118  int status = PV_SUCCESS;
119  int rootproc = 0;
120  int rank = parent->columnId();
121  Communicator * comm = parent->icCommunicator();
122  PV_Stream * pvstream = PV::pvp_open_read_file(filename, comm);
123  int numParams = NUM_BIN_PARAMS;
124  int params[numParams];
125  PV::pvp_read_header(pvstream, comm, params, &numParams);
126  assert (pvstream==NULL ^ rank==rootproc); // root process needs non-null pvstream; all other processes should have null pvstream.
127  if (frameNumber < 0 || frameNumber >= fileNumFrames) {
128  if (rank==rootproc) {
129  pvErrorNoExit().printf("scatterImageFilePVP: requested frameNumber %d but file \"%s\" only has frames numbered 0 through %d.\n", frameNumber, filename, params[INDEX_NBANDS]-1);
130  }
131  return PV_FAILURE;
132  }
133 
134  if (rank!=rootproc) { return PV_SUCCESS; }
135 
136  PVLayerLoc fileloc;
137  fileloc.nx = params[INDEX_NX];
138  fileloc.ny = params[INDEX_NY];
139  fileloc.nf = params[INDEX_NF];
140  //fileloc.nbatch = params[INDEX_NBATCH];
141  fileloc.nxGlobal = params[INDEX_NX_GLOBAL];
142  fileloc.nyGlobal = params[INDEX_NY_GLOBAL];
143  fileloc.kx0 = params[INDEX_KX0];
144  fileloc.ky0 = params[INDEX_KY0];
145  int nxProcs = params[INDEX_NX_PROCS];
146  int nyProcs = params[INDEX_NY_PROCS];
147  if (fileloc.nx != fileloc.nxGlobal || fileloc.ny != fileloc.nyGlobal ||
148  nxProcs != 1 || nyProcs != 1 ||
149  fileloc.kx0 != 0 || fileloc.ky0 != 0) {
150  pvError().printf("File \"%s\" appears to be in an obsolete version of the .pvp format.\n", filename);
151  }
152 
153  int bufferSize = fileloc.nx*fileloc.ny*fileloc.nf;
154  if (bufferSize != imageLoc.nxGlobal*imageLoc.nyGlobal*imageLoc.nf) {
155  free(imageData);
156  imageData = (pvadata_t *) malloc(bufferSize*sizeof(pvadata_t));
157  }
158  imageLoc.nxGlobal = fileloc.nx;
159  imageLoc.nyGlobal = fileloc.ny;
160  imageLoc.nf = fileloc.nf;
161  this->imageColorType = COLORTYPE_UNRECOGNIZED; // TODO: recognize RGB, YUV, etc.
162 
163  bool spiking = false;
164  double timed = 0.0;
165  int filetype = params[INDEX_FILE_TYPE];
166 
167  switch (filetype) {
168  case PVP_FILE_TYPE:
169  assert(0); // Is PVP_FILE_TYPE ever used?
170  break;
171 
172  case PVP_ACT_FILE_TYPE:
173  status = readSparseBinaryActivityFrame(numParams, params, pvstream, frameNumber);
174  break;
175  case PVP_ACT_SPARSEVALUES_FILE_TYPE:
176  status = readSparseValuesActivityFrame(numParams, params, pvstream, frameNumber);
177  break;
178  case PVP_NONSPIKING_ACT_FILE_TYPE:
179  status = readNonspikingActivityFrame(numParams, params, pvstream, frameNumber);
180  break;
181  case PVP_WGT_FILE_TYPE:
182  pvErrorNoExit().printf("scatterImageFilePVP failure opening \"%s\": file is a weight file, not an image file.\n", filename);
183  break;
184  case PVP_KERNEL_FILE_TYPE:
185  pvErrorNoExit().printf("scatterImageFilePVP failure opening \"%s\": file is a weight file, not an image file.\n", filename);
186  break;
187 
188  default:
189  pvErrorNoExit().printf("scatterImageFilePVP failure opening \"%s\": filetype %d is unrecognized.\n", filename ,filetype);
190  status = PV_FAILURE;
191  break;
192  }
193  if (status != PV_SUCCESS) {
194  exit(EXIT_FAILURE);
195  }
196  pvp_close_file(pvstream, comm);
197 
198  pvpFileTime = timed;
199  //This is being printed twice: track down
200  //pvInfo() << "Rank " << rank << " Reading pvpFileTime " << pvpFileTime << " at timestep " << parent->simulationTime() << " with offset (" << xOffset << "," << yOffset << ")\n";
201  return status;
202 }

204 int ImagePvp::readSparseBinaryActivityFrame(int numParams, int * params, PV_Stream * pvstream, int frameNumber) {
205 
206  //Allocate the byte positions in file where each frame's data starts and the number of active neurons in each frame
207  //Only need to do this once
208  int status = PV_SUCCESS;
209  if (needFrameSizesForSpiking) {
210  pvInfo() << "Calculating file positions\n";
211  frameStartBuf = (long *) calloc(params[INDEX_NBANDS] ,sizeof(long));
212  if (frameStartBuf==NULL) {
213  status = PV_FAILURE;
214  pvError().printf("scatterImageFilePVP unable to allocate memory for frameStart.\n");
215  }
216 
217  countBuf = (int *) calloc(params[INDEX_NBANDS] ,sizeof(int));
218  if (countBuf==NULL) {
219  status = PV_FAILURE;
220  pvError().printf("scatterImageFilePVP unable to allocate memory for frameLength.\n");
221  }
222 
223  //Fseek past the header and first timestamp
224  PV::PV_fseek(pvstream, (long)8 + (long)params[INDEX_HEADER_SIZE], SEEK_SET);
225  int percent = 0;
226  for (int f = 0; f<params[INDEX_NBANDS]; f++) {
227  int newpercent = 100*(f/(params[INDEX_NBANDS]));
228  if(percent != newpercent){
229  percent = newpercent;
230  pvInfo() << "\r" << percent << "\% Done";
231  pvInfo().flush();
232  }
233  //First byte position should always be 92
234  if (f == 0) {
235  frameStartBuf[f] = (long)92;
236  }
237  //Read in the number of active neurons for that frame and calculate byte position
238  else {
239  PV::PV_fread(&countBuf[f-1], sizeof(int), 1, pvstream);
240  frameStartBuf[f] = frameStartBuf[f-1] + (long)countBuf[f-1]*(long)params[INDEX_DATA_SIZE] + (long)12;
241  PV::PV_fseek(pvstream, frameStartBuf[f] - (long)4, SEEK_SET);
242  }
243  }
244  pvInfo() << "\r" << percent << "% Done\n";
245  pvInfo().flush();
246  //We still need the last count
247  PV::PV_fread(&countBuf[(params[INDEX_NBANDS])-1], sizeof(int), 1, pvstream);
248 
249  //So we don't have to calculate frameStart and count again
250  needFrameSizesForSpiking = false;
251  }
252 
253  long framepos = (long)frameStartBuf[frameNumber];
254  unsigned int length = countBuf[frameNumber];
255  PV::PV_fseek(pvstream, framepos-sizeof(double)-sizeof(unsigned int), SEEK_SET);
256  PV::PV_fread(&pvpFileTime, sizeof(double), 1, pvstream);
257 
258  unsigned int dropLength;
259  PV::PV_fread(&dropLength, sizeof(unsigned int), 1, pvstream);
260  assert(dropLength == length);
261 
262  memset(imageData, 0, sizeof(*imageData)*(size_t) (imageLoc.nxGlobal*imageLoc.nyGlobal*imageLoc.nf));
263  int locations[length];
264  PV::PV_fread(locations, sizeof(int), length, pvstream);
265  for (unsigned int l=0; l<length; l++) {
266  imageData[locations[l]]=(pvadata_t) 1;
267  }
268 
269  return status;
270 }
