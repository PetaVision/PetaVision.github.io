107 int ANNLayer::setVertices() {
108  if (VWidth<0) {
109  VThresh += VWidth;
110  VWidth = -VWidth;
111  if (parent->columnId()==0) {
112  pvWarn().printf("%s \"%s\": interpreting negative VWidth as setting VThresh=%f and VWidth=%f\n",
113  getKeyword(), name, VThresh, VWidth);
114  }
115  }
116 
117  pvdata_t limfromright = VThresh+VWidth-AShift;
118  if (AMax < limfromright) limfromright = AMax;
119 
120  if (AMin > limfromright) {
121  if (parent->columnId()==0) {
122  if (VWidth==0) {
123  pvWarn().printf("%s \"%s\": nonmonotonic transfer function, jumping from %f to %f at Vthresh=%f\n",
124  getKeyword(), name, AMin, limfromright, VThresh);
125  }
126  else {
127  pvWarn().printf("%s \"%s\": nonmonotonic transfer function, changing from %f to %f as V goes from VThresh=%f to VThresh+VWidth=%f\n",
128  getKeyword(), name, AMin, limfromright, VThresh, VThresh+VWidth);
129  }
130  }
131  }
132 
133  // Initialize slopes to NaN so that we can tell whether they've been initialized.
134  slopeNegInf = std::numeric_limits<double>::quiet_NaN();
135  slopePosInf = std::numeric_limits<double>::quiet_NaN();
136  std::vector<pvpotentialdata_t> vectorV;
137  std::vector<pvadata_t> vectorA;
138 
139  slopePosInf = 1.0f;
140  if (VThresh <= -0.999*max_pvadata_t) {
141  numVertices = 1;
142  vectorV.push_back((pvpotentialdata_t) 0);
143  vectorA.push_back(-AShift);
144  slopeNegInf = 1.0f;
145  }
146  else {
147  assert(VWidth >= (pvpotentialdata_t) 0);
148  if (VWidth == (pvpotentialdata_t) 0 && (pvadata_t) VThresh - AShift == AMin) { // Should there be a tolerance instead of strict ==?
149  numVertices = 1;
150  vectorV.push_back(VThresh);
151  vectorA.push_back(AMin);
152  }
153  else {
154  numVertices = 2;
155  vectorV.push_back(VThresh);
156  vectorV.push_back(VThresh+VWidth);
157  vectorA.push_back(AMin);
158  vectorA.push_back(VThresh+VWidth-AShift);
159  }
160  slopeNegInf = 0.0f;
161  }
162  if (AMax < 0.999*max_pvadata_t) {
163  assert(slopePosInf == 1.0f);
164  if (vectorA[numVertices-1] < AMax) {
165  pvadata_t interval = AMax - vectorA[numVertices-1];
166  vectorV.push_back(vectorV[numVertices-1]+(pvpotentialdata_t) interval);
167  vectorA.push_back(AMax);
168  numVertices++;
169  }
170  else {
171  // find the last vertex where A < AMax.
172  bool found = false;
173  int v;
174  for (v=numVertices-1; v>=0; v--) {
175  if (vectorA[v] < AMax) { found = true; break; }
176  }
177  if (found) {
178  assert(v+1 < numVertices && vectorA[v] < AMax && vectorA[v+1] >= AMax);
179  pvadata_t interval = AMax - vectorA[v];
180  numVertices = v+1;
181  vectorA.resize(numVertices);
182  vectorV.resize(numVertices);
183  vectorV.push_back(vectorV[v]+(pvpotentialdata_t) interval);
184  vectorA.push_back(AMax);
185  // In principle, there could be a case where a vertex n has A[n]>AMax but A[n-1] and A[n+1] are both < AMax.
186  // But with the current ANNLayer parameters, that won't happen.
187  }
188  else {
189  // All vertices have A>=AMax.
190  // If slopeNegInf is positive, transfer function should increase from -infinity to AMax, and then stays constant.
191  // If slopeNegInf is negative or zero,
192  numVertices = 1;
193  vectorA.resize(numVertices);
194  vectorV.resize(numVertices);
195  if (slopeNegInf > 0) {
196  pvadata_t intervalA = vectorA[0]-AMax;
197  pvpotentialdata_t intervalV = (pvpotentialdata_t) (intervalA / slopeNegInf);
198  vectorV[0] = vectorV[0] - intervalV;
199  vectorA[0] = AMax;
200  }
201  else {
202  // Everything everywhere is above AMax, so make the transfer function a constant A=AMax.
203  vectorA.resize(1);
204  vectorV.resize(1);
205  vectorV[0] = (pvpotentialdata_t) 0;
206  vectorA[0] = AMax;
207  numVertices = 1;
208  slopeNegInf = 0;
209  }
210  }
211 
212  }
213  slopePosInf = 0.0f;
214  }
215  assert(!isnan(slopeNegInf) && !isnan(slopePosInf) && numVertices > 0);
216  assert(vectorA.size()==numVertices && vectorV.size()==numVertices);
217  verticesV = (pvpotentialdata_t *) malloc((size_t) numVertices * sizeof(*verticesV));
218  verticesA = (pvadata_t *) malloc((size_t) numVertices * sizeof(*verticesA));
219  if (verticesV==NULL || verticesA==NULL) {
220  pvErrorNoExit().printf("%s \"%s\": unable to allocate memory for vertices:%s\n",
221  getKeyword(), name, strerror(errno));
222  exit(EXIT_FAILURE);
223  }
224  memcpy(verticesV, &vectorV[0], numVertices * sizeof(*verticesV));
225  memcpy(verticesA, &vectorA[0], numVertices * sizeof(*verticesA));
226 
227  return PV_SUCCESS;
228 }
