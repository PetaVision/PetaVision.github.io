121 int pv_getoptionalopt_int(int argc, char * argv[], const char * opt, int * iVal, bool * useDefaultVal, bool * paramusage)
122 {
123  int i;
124  for (i = 1; i < argc; i += 1) {
125  if(strcmp(argv[i], opt) == 0){
126  //Default parameter
127  if (i+1 >= argc || argv[i+1][0] == '-') {
128  if (iVal != NULL) { *iVal = -1; }
129  if (useDefaultVal != NULL) { *useDefaultVal = true;}
130  if (paramusage) { paramusage[i] = true; }
131  }

133  if (iVal != NULL) { *iVal = atoi(argv[i+1]); }
134  if (useDefaultVal != NULL) { *useDefaultVal = false; }

189 int pv_getopt_str(int argc, char * argv[], const char * opt, char ** sVal, bool * paramusage)
190 {
191  // sVal can be NULL. If sVal is not null and the option is found,
192  // the value of the option is put into sVal and the calling routine is
193  // responsible for freeing it.
194  // Example: if argv[1] is "-p" and argv[2] is "params.pv", and opt is "-p",
195  // *sVal will be "params.pv".
196  int i;
197  for (i = 1; i < argc; i += 1) {
198  if (i+1 < argc && strcmp(argv[i], opt) == 0) {
199  if( sVal != NULL ) *sVal = strdup(argv[i+1]);
200  if (paramusage) { paramusage[i] = true; paramusage[i+1] = true; }
201  return 0;
202  }
203  }
204  if (sVal != NULL) *sVal = NULL;
205  return -1; // not found
206 }

215 int pv_center_image(float * V, int nx0, int ny0, int nx, int ny)
216 {
217  int i0, j0, i, j, ii;
218 
219  float * buf = (float *) malloc(nx0 * ny0 * sizeof(float));
220  assert(buf != NULL);
221 
222  assert(nx0 <= nx);
223  assert(ny0 <= ny);
224 
225  memcpy(buf, V, nx0 * nx0 * sizeof(float));
226 
227  i0 = nx/2 - nx0/2;
228  j0 = ny/2 - ny0/2;
229 
230  for (i = 0; i < nx*ny; i++) {
231  V[i] = 0;
232  }
233 
234  ii = 0;
235  for (j = j0; j < j0+ny0; j++) {
236  for (i = i0; i < i0+nx0; i++) {
237  V[i+nx*j] = buf[ii++];
238  }
239  }
240  free(buf);
241 
242  return 0;
243 }

245 char * expandLeadingTilde(char const * path) {
246  char * newpath = NULL;
247  if (path != NULL) {
248  int len = strlen(path);
249  if (len==1 && path[0]=='~') {
250  newpath = strdup(getenv("HOME"));
251  if (newpath==NULL) {
252  pvError().printf("Unable to expand \"%s\": home directory not defined\n", path);
253  exit(EXIT_FAILURE);
254  }
255  }
256  else if (len>1 && path[0]=='~' && path[1]=='/') {
257  char * homedir = getenv("HOME");
258  if (homedir==NULL) {
259  pvError().printf("Unable to expand \"%s\": home directory not defined\n", path);
260  }
261  char dummy;
262  int chars_needed = snprintf(&dummy, 0, "%s/%s", homedir, &path[2]);
263  newpath = (char *) malloc(chars_needed+1);
264  if (newpath==NULL) {
265  pvError().printf("Unable to allocate memory for path \"%s/%s\"\n", homedir, &path[2]);
266  }
267  int chars_used = snprintf(newpath, chars_needed+1, "%s/%s", homedir, &path[2]);
268  assert(chars_used == chars_needed);
269  }
270  else {
271  newpath = strdup(path);
272  }
273  }
274  return newpath;
275 }
