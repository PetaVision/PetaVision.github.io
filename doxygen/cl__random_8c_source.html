42 int cl_random_init(taus_uint4 *state, size_t count, unsigned int seed) {
43  int i;
44 
45  // a zero seed can cause problems (see taus_set)
46  seed = (seed == 0) ? 1 : seed;
47 
48  // initialize state array using a separate seed for each element
49  //
50  for (i = 0; i < count; i++) {
51  taus_set(&state[i].state, i + seed);
52  state[i].s0 = (state[i].state.s1 ^ state[i].state.s2 ^ state[i].state.s3);
53  }
54 
55  return 0;
56 }

58 static void taus_set(taus_state_t *state, unsigned int s) {
59  if (s == 0) {
60  s = 1; /* default seed is 1 */
61  }
62 
63 // original for unsigned long int
64 //#define LCG(n) ((69069 * n) & 0xffffffffUL)
65 #define LCG(n) ((69069 * n))
66 
67  state->s1 = LCG(s);
68  state->s2 = LCG(state->s1);
69  state->s3 = LCG(state->s2);
70 
71  /* "warm it up" */
72  taus_get(state);
73  taus_get(state);
74  taus_get(state);
75  taus_get(state);
76  taus_get(state);
77  taus_get(state);
78  return;
79 }

81 static inline unsigned int taus_get(taus_state_t *state) {
82 
83 //#define MASK 0xffffffffUL
84 //#define TAUSWORTHE(s,a,b,c,d) (((s &c) <<d) &MASK) ^ ((((s <<a) &MASK)^s) >>b)
85 #define TAUSWORTHE(s, a, b, c, d) (((s & c) << d)) ^ ((((s << a)) ^ s) >> b)
86 
87  state->s1 = TAUSWORTHE(state->s1, 13, 19, 4294967294, 12);
88  state->s2 = TAUSWORTHE(state->s2, 2, 25, 4294967288, 4);
89  state->s3 = TAUSWORTHE(state->s3, 3, 11, 4294967280, 17);
90 
91  return (state->s1 ^ state->s2 ^ state->s3);
92 }

95  Transformation

103 // Argument box_muller_state * bm_state added so that cl_random_prob() could be used in place of
104 // Carter's ranf().
105 // use_last and y2 changed to elements of bm_state so that several independent RNGs can be used.
106 float cl_box_muller(
107  float m,
108  float s,
109  struct box_muller_state *bm_state) /* normal random variate generator */
110 { /* mean m, standard deviation s */
111  float x1, x2, w, y1;
112 
113  if (bm_state->use_last) /* use value from previous call */
114  {
115  y1 = bm_state->last_value;
116  bm_state->use_last = 0;
117  }
118  else {
119  do {
120  *(bm_state->state) = cl_random_get(*(bm_state->state));
121  x1 = 2.0 * bm_state->state->s0 / (double)CL_RANDOM_MAX - 1.0;
122  *(bm_state->state) = cl_random_get(*(bm_state->state));
123  x2 = 2.0 * bm_state->state->s0 / (double)CL_RANDOM_MAX - 1.0;
124  w = x1 * x1 + x2 * x2;
125  } while (w >= 1.0);
126 
127  w = sqrt((-2.0 * log(w)) / w);
128  y1 = x1 * w;
129  bm_state->last_value = x2 * w;
130  bm_state->use_last = 1;
131  }
132 
133  return (m + y1 * s);
134 }
