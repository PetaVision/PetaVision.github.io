222 int IdentConn::communicateInitInfo() {
223  int status = HyPerConn::communicateInitInfo();
224  assert(pre && post);
225  const PVLayerLoc * preLoc = pre->getLayerLoc();
226  const PVLayerLoc * postLoc = post->getLayerLoc();
227  if( preLoc->nx != postLoc->nx || preLoc->ny != postLoc->ny || preLoc->nf != postLoc->nf ) {
228  if (parent->columnId()==0) {
229  pvErrorNoExit().printf("IdentConn \"%s\" Error: %s and %s do not have the same dimensions.\n Dims: %dx%dx%d vs. %dx%dx%d\n",
230  name, preLayerName,postLayerName,preLoc->nx,preLoc->ny,preLoc->nf,postLoc->nx,postLoc->ny,postLoc->nf);
231  }
232  exit(EXIT_FAILURE);
233  }
234  parent->parameters()->handleUnnecessaryParameter(name, "nfp", nfp); // nfp is set during call to HyPerConn::communicateInitInfo, so don't check for unnecessary int parameter until after that.
235  return status;
236 }

278 int IdentConn::deliverPresynapticPerspective(PVLayerCube const * activity, int arborID) {
279 
280  //Check if we need to update based on connection's channel
281  if(getChannel() == CHANNEL_NOUPDATE){
282  return PV_SUCCESS;
283  }
284  assert(post->getChannel(getChannel()));
285 
286  float dt_factor = getConvertToRateDeltaTimeFactor();
287  assert(dt_factor==1.0);
288 
289  const PVLayerLoc * preLoc = preSynapticLayer()->getLayerLoc();
290  const PVLayerLoc * postLoc = postSynapticLayer()->getLayerLoc();
291 
292  assert(arborID==0); // IdentConn can only have one arbor
293  const int numExtended = activity->numItems;
294 
295 #ifdef DEBUG_OUTPUT
296  int rank;
297  MPI_Comm_rank(parent->icCommunicator()->communicator(), &rank);
298  pvDebug(debugMessage);
299  debugMessage.printf("[%d]: HyPerLayr::recvSyn: neighbor=%d num=%d actv=%p this=%p conn=%p\n", rank, 0, numExtended, activity, this, conn);
300  debugMessage.flush();
301 #endif // DEBUG_OUTPUT
302 
303  for(int b = 0; b < parent->getNBatch(); b++){
304  pvdata_t * activityBatch = activity->data + b * (preLoc->nx + preLoc->halo.rt + preLoc->halo.lt) * (preLoc->ny + preLoc->halo.up + preLoc->halo.dn) * preLoc->nf;
305  pvdata_t * gSynPatchHeadBatch = post->getChannel(getChannel()) + b * postLoc->nx * postLoc->ny * postLoc->nf;
306 
307  if (activity->isSparse) {
308  unsigned int * activeIndicesBatch = activity->activeIndices + b * (preLoc->nx + preLoc->halo.rt + preLoc->halo.lt) * (preLoc->ny + preLoc->halo.up + preLoc->halo.dn) * preLoc->nf;
309  int numLoop = activity->numActive[b];
310 #ifdef PV_USE_OPENMP_THREADS
311 #pragma omp parallel for
312 #endif
313  for (int loopIndex = 0; loopIndex < numLoop; loopIndex++) {
314  int kPre = activeIndicesBatch[loopIndex];
315 
316  float a = activityBatch[kPre];
317  // if (a == 0.0f) continue;
318  PVPatch * weights = getWeights(kPre, arborID);
319  if (weights->nx>0 && weights->ny>0) {
320  int f = featureIndex(kPre, preLoc->nx, preLoc->ny, preLoc->nf); // Not taking halo into account, but for feature index, shouldn't matter.
321  pvgsyndata_t * postPatchStart = gSynPatchHeadBatch + getGSynPatchStart(kPre, arborID) + f;
322  *postPatchStart += a;
323  }
324  }
325  }
326  else {
327  PVLayerLoc const * loc = &activity->loc;
328  PVHalo const * halo = &loc->halo;
329  // The code below is a replacement for the block marked obsolete below it. Jan 5, 2016
330  int lineSizeExt = (loc->nx+halo->lt+halo->rt)*loc->nf;
331 #ifdef PV_USE_OPENMP_THREADS
332 #pragma omp parallel for
333 #endif
334  for (int y=0; y<loc->ny; y++) {
335  pvdata_t * lineStartPreActivity = &activityBatch[(y+halo->up)*lineSizeExt+halo->lt*loc->nf];
336  int nk = loc->nx*loc->nf;
337  pvdata_t * lineStartPostGSyn = &gSynPatchHeadBatch[y*nk];
338  for (int k=0; k<nk; k++) {
339  lineStartPostGSyn[k] += lineStartPreActivity[k];
340  }
341  }
342 #ifdef OBSOLETE // Marked obsolete Jan 5, 2016. IdentConn is simple enough that we shouldn't need to call kIndexExtended inside the inner loop.
343  int numRestricted = loc->nx * loc->ny * loc->nf;
344 #ifdef PV_USE_OPENMP_THREADS
345 #pragma omp parallel for
346 #endif
347  for (int kRestricted = 0; kRestricted < numRestricted; kRestricted++) {
348  int kExtended = kIndexExtended(kRestricted, loc->nx, loc->ny, loc->nf, loc->halo.lt, loc->halo.rt, loc->halo.dn, loc->halo.up);
349  float a = activityBatch[kExtended];
350  // if (a == 0.0f) continue;
351  gSynPatchHeadBatch[kRestricted] += a;
352  }
353 #endif // OBSOLETE // Marked obsolete Jan 5, 2016
354  }
355  }
356  return PV_SUCCESS;
357 }
