224 int IdentConn::communicateInitInfo(CommunicateInitInfoMessage const *message) {
225  int status = HyPerConn::communicateInitInfo(message);
226  assert(pre && post);
227  const PVLayerLoc *preLoc = pre->getLayerLoc();
228  const PVLayerLoc *postLoc = post->getLayerLoc();
229  if (preLoc->nx != postLoc->nx || preLoc->ny != postLoc->ny || preLoc->nf != postLoc->nf) {
230  if (parent->columnId() == 0) {
231  ErrorLog().printf(
232  "IdentConn \"%s\" Error: %s and %s do not have the same dimensions.\n Dims: "
233  "%dx%dx%d vs. %dx%dx%d\n",
234  name,
235  preLayerName,
236  postLayerName,
237  preLoc->nx,
238  preLoc->ny,
239  preLoc->nf,
240  postLoc->nx,
241  postLoc->ny,
242  postLoc->nf);
243  }
244  exit(EXIT_FAILURE);
245  }
246  parent->parameters()->handleUnnecessaryParameter(
247  name, "nfp", nfp); // nfp is set during call to HyPerConn::communicateInitInfo, so don't
248  // check for unnecessary int parameter until after that.
249  return status;

259 int IdentConn::deliverPresynapticPerspective(PVLayerCube const *activity, int arborID) {
260 
261  // Check if we need to update based on connection's channel
262  if (getChannel() == CHANNEL_NOUPDATE) {
263  return PV_SUCCESS;
264  }
265  assert(post->getChannel(getChannel()));
266  assert(getConvertToRateDeltaTimeFactor() == 1.0);
267 
268  const PVLayerLoc *preLoc = preSynapticLayer()->getLayerLoc();
269  const PVLayerLoc *postLoc = postSynapticLayer()->getLayerLoc();
270 
271  assert(arborID == 0); // IdentConn can only have one arbor
272  const int numExtended = activity->numItems;
273 
274 #ifdef DEBUG_OUTPUT
275  int rank;
276  MPI_Comm_rank(parent->getCommunicator()->communicator(), &rank);
277  DebugLog(debugMessage);
278  debugMessage.printf(
279  "[%d]: HyPerLayr::recvSyn: neighbor=%d num=%d actv=%p this=%p conn=%p\n",
280  rank,
281  0,
282  numExtended,
283  activity,
284  this,
285  conn);
286  debugMessage.flush();
287 #endif // DEBUG_OUTPUT
288 
289  for (int b = 0; b < parent->getNBatch(); b++) {
290  float *activityBatch = activity->data
291  + b * (preLoc->nx + preLoc->halo.rt + preLoc->halo.lt)
292  * (preLoc->ny + preLoc->halo.up + preLoc->halo.dn)
293  * preLoc->nf;
294  float *gSynPatchHeadBatch =
295  post->getChannel(getChannel()) + b * postLoc->nx * postLoc->ny * postLoc->nf;
296 
297  if (activity->isSparse) {
298  SparseList<float>::Entry const *activeIndicesBatch =
299  (SparseList<float>::Entry *)activity->activeIndices
300  + b * (preLoc->nx + preLoc->halo.rt + preLoc->halo.lt)
301  * (preLoc->ny + preLoc->halo.up + preLoc->halo.dn) * preLoc->nf;
302  int numLoop = activity->numActive[b];
303 #ifdef PV_USE_OPENMP_THREADS
304 #pragma omp parallel for
305 #endif
306  for (int loopIndex = 0; loopIndex < numLoop; loopIndex++) {
307  int kPre = activeIndicesBatch[loopIndex].index;
308 
309  float a = activeIndicesBatch[loopIndex].value; // activityBatch[kPre];
310  PVPatch *weights = getWeights(kPre, arborID);
311  if (weights->nx > 0 && weights->ny > 0) {
312  int f = featureIndex(kPre, preLoc->nx, preLoc->ny, preLoc->nf); // Not taking halo
313  // into account, but
314  // for feature index,
315  // shouldn't matter.
316  float *postPatchStart = gSynPatchHeadBatch + getGSynPatchStart(kPre, arborID) + f;
317  *postPatchStart += a;
318  }
319  }
320  }
321  else {
322  PVLayerLoc const *loc = &activity->loc;
323  PVHalo const *halo = &loc->halo;
324  int lineSizeExt = (loc->nx + halo->lt + halo->rt) * loc->nf;
325 #ifdef PV_USE_OPENMP_THREADS
326 #pragma omp parallel for
327 #endif
328  for (int y = 0; y < loc->ny; y++) {
329  float *lineStartPreActivity =
330  &activityBatch[(y + halo->up) * lineSizeExt + halo->lt * loc->nf];
331  int nk = loc->nx * loc->nf;
332  float *lineStartPostGSyn = &gSynPatchHeadBatch[y * nk];
333  for (int k = 0; k < nk; k++) {
334  lineStartPostGSyn[k] += lineStartPreActivity[k];
335  }
336  }
337  }
338  }
339  return PV_SUCCESS;
340 }
