49 void GapConn::ioParam_sharedWeights(enum ParamsIOFlag ioFlag) {
50  // Default of true for sharedWeights for GapConns was deprecated Aug 11, 2014.
51  // This default was chosen for backwards compatibility because GapConn used to require sharedWeights be true.
52  // Now GapConn can be used with or without shared weights, so eventually the default will false as it is for other HyPerConns.
53  parent->ioParamValue(ioFlag, name, "sharedWeights", &sharedWeights, true/*default*/, true/*warn if absent*/);
54  if (ioFlag==PARAMS_IO_READ && !parent->parameters()->present(name, "sharedWeights")) {
55  sharedWeights = true;
56  if (parent->columnId()==0) {
57  fprintf(stderr, "%s \"%s\" warning: sharedWeights defaults to true for GapConns, but the default may be changed to false in a future release, to be consistent with other HyPerConns.\n", this->getKeyword(), name);
58  }
59  return;
60  }
61  HyPerConn::ioParam_sharedWeights(ioFlag);
62 }

64 void GapConn::ioParam_normalizeMethod(enum ParamsIOFlag ioFlag) {
65  // Default of normalizeSum for normalizeMethod for GapConns was deprecated Aug 11, 2014.
66  // This default was chosen for backwards compatibility because GapConn used to require normalizeMethod be normalizeSum.
67  // Now GapConn can be normalized using any method, so eventually the default will be removed and the parameter required as is for other HyPerConns.
68  if (ioFlag==PARAMS_IO_READ && !parent->parameters()->stringPresent(name, "normalizeMethod")) {
69  normalizeMethod = strdup("normalizeSum");
70  GapConn * conn = this;
71  normalizer = new NormalizeGap(name, parent);
72  if (parent->columnId()==0) {
73  fprintf(stderr, "%s \"%s\" warning: normalizeMethod defaults to normalizeSum for GapConns, but this parameter may be required in a future release, to be consistent with other HyPerConns.\n", this->getKeyword(), name);
74  }
75  return;
76  }
77  HyPerConn::ioParam_normalizeMethod(ioFlag);
