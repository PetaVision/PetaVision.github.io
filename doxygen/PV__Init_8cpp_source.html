82 int PV_Init::commInit(int* argc, char*** argv)
83 {
84 #ifdef PV_USE_MPI
85  int mpiInit;
86  // If MPI wasn't initialized, initialize it.
87  // Remember if it was initialized on entry; the destructor will only finalize if the constructor init'ed.
88  // This way, you can do several simulations sequentially by initializing MPI before creating
89  // the first HyPerCol; after running the first simulation the MPI environment will still exist and you
90  // can run the second simulation, etc.
91  MPI_Initialized(&mpiInit);
92  if( !mpiInit) {
93  assert((*argv)[*argc]==NULL); // Open MPI 1.7 assumes this.
94  MPI_Init(argc, argv);
95  }
96  else{
97  std::cout << "Error: PV_Init communicator already initialized\n";
98  exit(-1);
99  }
100 #endif
101 
102  return 0;
103 }
