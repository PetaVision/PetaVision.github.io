33 int PV_Init::initSignalHandler()
34 {
35  // Block SIGUSR1. root process checks for SIGUSR1 during advanceTime() and broadcasts sends to all processes,
36  // which saves the result in the checkpointSignal member variable.
37  // When run() checks whether to call checkpointWrite, it looks at checkpointSignal, and writes a
38  // checkpoint if checkpointWriteFlag is true, regardless of whether the next scheduled checkpoint time has arrived.
39  //
40  // This routine must be called before MPI_Initialize; otherwise a thread created by MPI will not get the signal handler
41  // but will get the signal and the job will terminate.
42  sigset_t blockusr1;
43  sigemptyset(&blockusr1);
44  sigaddset(&blockusr1, SIGUSR1);
45  sigprocmask(SIG_BLOCK, &blockusr1, NULL);
46  return 0;
47 }

49 int PV_Init::initialize() {
50  char const * params_file = arguments->getParamsFile();

53  exit(PV_FAILURE);
54  }
55  if (icComm) { delete icComm; }
56  icComm = new InterColComm(arguments);
57  if (params) { delete params; }
58  params = new PVParams(params_file, 2*(INITIAL_LAYER_ARRAY_SIZE+INITIAL_CONNECTION_ARRAY_SIZE), icComm);
59  initialized = true;
60  return PV_SUCCESS;

63 int PV_Init::commInit(int* argc, char*** argv)
64 {
65  int mpiInit;
66  // If MPI wasn't initialized, initialize it.
67  // Remember if it was initialized on entry; the destructor will only finalize if the constructor init'ed.
68  // This way, you can do several simulations sequentially by initializing MPI before creating
69  // the first HyPerCol; after running the first simulation the MPI environment will still exist and you
70  // can run the second simulation, etc.
71  MPI_Initialized(&mpiInit);
72  if( !mpiInit) {
73  assert((*argv)[*argc]==NULL); // Open MPI 1.7 assumes this.
74  MPI_Init(argc, argv);
75  }
76  else{
77  std::cout << "Error: PV_Init communicator already initialized\n";
78  exit(-1);
79  }
80 
81  return 0;
82 }
