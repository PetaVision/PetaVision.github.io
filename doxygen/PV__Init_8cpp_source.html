39 int PV_Init::initSignalHandler()
40 {
41  // Block SIGUSR1. root process checks for SIGUSR1 during advanceTime() and broadcasts sends to all processes,
42  // which saves the result in the checkpointSignal member variable.
43  // When run() checks whether to call checkpointWrite, it looks at checkpointSignal, and writes a
44  // checkpoint if checkpointWriteFlag is true, regardless of whether the next scheduled checkpoint time has arrived.
45  //
46  // This routine must be called before MPI_Initialize; otherwise a thread created by MPI will not get the signal handler
47  // but will get the signal and the job will terminate.
48  sigset_t blockusr1;
49  sigemptyset(&blockusr1);
50  sigaddset(&blockusr1, SIGUSR1);
51  sigprocmask(SIG_BLOCK, &blockusr1, NULL);
52  return 0;
53 }

82 int PV_Init::commInit(int* argc, char*** argv)
83 {
84  int mpiInit;
85  // If MPI wasn't initialized, initialize it.
86  // Remember if it was initialized on entry; the destructor will only finalize if the constructor init'ed.
87  // This way, you can do several simulations sequentially by initializing MPI before creating
88  // the first HyPerCol; after running the first simulation the MPI environment will still exist and you
89  // can run the second simulation, etc.
90  MPI_Initialized(&mpiInit);
91  if( !mpiInit) {
92  pvAssert((*argv)[*argc]==NULL); // Open MPI 1.7 assumes this.
93  MPI_Init(argc, argv);
94  }
95  else{
96  pvError() << "PV_Init communicator already initialized\n";
97  }
98 
99  return 0;
100 }

102 void PV_Init::initLogFile() {
103  // TODO: Under MPI, non-root processes should send messages to root process.
104  // Currently, if logFile is directory/filename.txt, the root process writes to that path,
105  // and nonroot processes write to directory/filename_<rank>.txt, where <rank> is replaced with the global rank.
106  // If filename does not have an extension, _<rank> is appended.
107  // Note that the global rank zero process does not insert _<rank>. This is deliberate, as the nonzero ranks
108  // should be MPI-ing the data to the zero rank.
109  char const * logFile = arguments->getLogFile();
110  std::ios_base::openmode const mode = std::ios_base::out; // TODO: Provide control over whether to truncate or append
111  int const globalRootProcess = 0;
112  int globalRank;
113  MPI_Comm_rank(MPI_COMM_WORLD, &globalRank);
114  if (logFile && globalRank != globalRootProcess) {
115  // To prevent collisions caused by multiple processes opening the same file for logging,
116  // processes with global rank other than zero append the rank to the log filename.
117  // If the logfile has an extension (e.g. ".log", ".txt"), the rank is appended before the
118  // period separating the extension.
119  std::string logFileString(logFile);
120  size_t finalSlash = logFileString.rfind('/');
121  size_t finalDot = logFileString.rfind('.');
122  size_t insertionPoint;
123  if (finalDot == std::string::npos || (finalSlash!=std::string::npos && finalDot < finalSlash) ) {
124  insertionPoint = logFileString.length();
125  }
126  else {
127  insertionPoint = finalDot;
128  }
129  logFileString.insert(finalDot, std::to_string(globalRank)).insert(finalDot, "_");
130  PV::setLogFile(logFileString.c_str(), mode);
131  }
132  else {
133  PV::setLogFile(logFile, mode);
134  }
135 }

188 HyPerCol * PV_Init::build() {
189  HyPerCol * hc = new HyPerCol("column", this);
190  if( hc == NULL ) {
191  pvErrorNoExit().printf("Unable to create HyPerCol\n");
192  return NULL;
193  }
194  PVParams * hcparams = hc->parameters();
195  int numGroups = hcparams->numberOfGroups();
196 
197  // Make sure first group defines a column
198  if( strcmp(hcparams->groupKeywordFromIndex(0), "HyPerCol") ) {
199  pvErrorNoExit().printf("First group in the params file did not define a HyPerCol.\n");
200  delete hc;
201  return NULL;
202  }
203 
204  for (int k=0; k<numGroups; k++) {
205  const char * kw = hcparams->groupKeywordFromIndex(k);
206  const char * name = hcparams->groupNameFromIndex(k);
207  if (!strcmp(kw, "HyPerCol")) {
208  if (k==0) { continue; }
209  else {
210  if (hc->columnId()==0) {
211  pvErrorNoExit().printf("Group %d in params file (\"%s\") is a HyPerCol; the HyPerCol must be the first group.\n",
212  k+1, name);
213  }
214  delete hc;
215  return NULL;
216  }
217  }
218  else {
219  BaseObject * addedObject = factory->create(kw, name, hc);
220  if (addedObject==NULL) {
221  if (hc->globalRank()==0) {
222  pvErrorNoExit().printf("Unable to create %s \"%s\".\n", kw, name);
223  }
224  delete hc;
225  return NULL;
226  }
227  }
228  }
229 
230  if( hc->numberOfLayers() == 0 ) {
231  pvErrorNoExit().printf("HyPerCol \"%s\" does not have any layers.\n", hc->getName());
232  delete hc;
233  return NULL;
234  }
235  return hc;
236 }
