50 char ** PV_Arguments::copyArgs(int argc, char const * const * argv) {
51  char ** argumentArray = (char **) malloc((size_t) (argc+1) * sizeof(char *));
52  if (argumentArray==NULL) {
53  fprintf(stderr, "PV_Arguments error: unable to allocate memory for %d arguments: %s\n",
54  argc, strerror(errno));
55  exit(EXIT_FAILURE);
56  }
57  for (int a=0; a<argc; a++) {
58  char const * arga = argv[a];
59  if (arga) {
60  char * copied = strdup(arga);
61  if (!copied) {
62  fprintf(stderr, "PV_Arguments unable to store argument %d: %s\n", a, strerror(errno));
63  fprintf(stderr, "Argument was \"%s\".\n", arga);
64  exit(EXIT_FAILURE);
65  }
66  argumentArray[a] = copied;
67  }
68  else {
69  argumentArray[a] = NULL;
70  }
71  }
72  argumentArray[argc] = NULL;
73  return argumentArray;
74 }

188 int PV_Arguments::setStateFromCmdLineArgs(bool allowUnrecognizedArguments) {
189  bool * usedArgArray = (bool *) calloc((size_t) numArgs, sizeof(bool)); // Because, C++ expert, I'm *trying* to cause you pain.
190  if (usedArgArray==NULL) {
191  fprintf(stderr, "PV_Arguments::setStateFromCmdLineArgs unable to allocate memory for usedArgArray: %s\n", strerror(errno));
192  exit(EXIT_FAILURE);
193  }
194 
195  int restart = (int) restartFlag;
196  int status = parse_options(numArgs, args,
197  usedArgArray, &requireReturnFlag, &outputPath, &paramsFile, &logFile,
198  &gpuDevices, &randomSeed, &workingDir,
199  &restart, &checkpointReadDir, &numThreads,
200  &numRows, &numColumns, &batchWidth);
201  restartFlag = restart!=0;
202 
203 #ifdef PV_USE_OPENMP_THREADS
204  // If "-t" appeared as the last argument, set threads to the max possible.
205  if (numThreads==0 && pv_getopt(numArgs, args, "-t", usedArgArray)==0) {
206  numThreads = omp_get_max_threads();
207  }
208 #else // PV_USE_OPENMP_THREADS
209  numThreads = 1;
210 #endif // PV_USE_OPENMP_THREADS
211 
212  // Error out if both -r and -c are used.
213  if (errorChecking()) {
214  exit(EXIT_FAILURE);
215  }
216 
217  if (!allowUnrecognizedArguments) {
218  bool anyUnusedArgs = false;
219  for (int a=0; a<numArgs; a++) {
220  if(!usedArgArray[a]) {
221  fprintf(stderr, "%s: argument %d, \"%s\", is not recognized.\n",
222  getProgramName(), a, args[a]);
223  anyUnusedArgs = true;
224  }
225  }
226  if (anyUnusedArgs) {
227  exit(EXIT_FAILURE);
228  }
229  }
230  free(usedArgArray);
231  return PV_SUCCESS;
232 }

234 int PV_Arguments::printState() {
235  printf("%s", getProgramName());
236  if (requireReturnFlag) { printf(" --require-return"); }
237  if (outputPath) { printf(" -o %s", outputPath); }
238  if (paramsFile) { printf(" -p %s", paramsFile); }
239  if (logFile) { printf(" -l %s", logFile); }
240  if (gpuDevices) { printf(" -d %s", gpuDevices); }
241  if (randomSeed) { printf(" -s %u", randomSeed); }
242  if (workingDir) { printf(" -w %s", workingDir); }
243  assert(!(restartFlag && checkpointReadDir));
244  if (restartFlag) { printf(" -r"); }
245  if (checkpointReadDir) { printf(" -c %s", checkpointReadDir); }
246  if (numThreads) { printf(" -t %d", numThreads); }
247  if (numRows) { printf(" -rows %d", numRows); }
248  if (numColumns) { printf(" -columns %d", numColumns); }
249  if (batchWidth) { printf(" -batchwidth %d", batchWidth); }
250  printf("\n");
251  return PV_SUCCESS;
252 }
