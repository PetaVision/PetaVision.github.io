38 int BaseConnectionProbe::communicateInitInfo(CommunicateInitInfoMessage const *message) {
39  int status = BaseProbe::communicateInitInfo(message);
40  if (status != PV_SUCCESS) {
41  return status;
42  }
43  targetConn = message->lookup<BaseConnection>(std::string(targetName));
44  if (targetConn == nullptr) {
45  ErrorLog().printf(
46  "%s, rank %d process: targetConnection \"%s\" is "
47  "not a connection in the column.\n",
48  getDescription_c(),
49  parent->columnId(),
50  targetName);
51  status = PV_FAILURE;
52  }
53  MPI_Barrier(parent->getCommunicator()->communicator());
54  if (status != PV_SUCCESS) {
55  exit(EXIT_FAILURE);
56  }
57  targetConn->insertProbe(this);
58  return status;
59 }
