154 void ShuffleLayer::readFreq(){ // TODO: Add MPI Bcast so that only root proc does this
155  int nf = getLayerLoc()->nf;
156  int nbatch = getLayerLoc()->nbatch;
157  string line;
158  ifstream freqFile(freqFilename);
159  if (freqFile.is_open()){
160  for (int b = 0; b < nbatch; b++){
161  for (int kf = 0; kf < nf; kf++){
162  getline (freqFile,line);
163  if (freqFile.fail()){
164  fprintf(stderr, "Shuffle Layer: Unable to read from frequency file %s\n",freqFilename);
165  exit(PV_FAILURE);
166  }
167  featureFreqCount[b][kf] = atol(line.c_str());
168  if(featureFreqCount[b][kf] > maxCount[b]){
169  maxCount[b] = featureFreqCount[b][kf];
170  }
171  if (freqFile.eof()){
172  fprintf(stderr, "Shuffle Layer: Invalid frequency file %s: EOF before %d nf, %d batches.\n ",freqFilename, nf, nbatch);
173  exit(PV_FAILURE);
174  }
175  }
176  if (getline(freqFile, line)){
177  fprintf(stderr, "Shuffle Layer: Invalid frequency file: %s contains > %d nf, %d batches.\n ",freqFilename, nf, nbatch);
178  exit(PV_FAILURE);
179  }
180  }
181  freqFile.close();
182  }
183  else{
184  fprintf(stderr, "Shuffle Layer: Unable to open frequency file %s\n",freqFilename);
185  exit(PV_FAILURE);
186  }

189 void ShuffleLayer::collectFreq(const pvdata_t * sourceData){
190  PVHalo const * haloOrig = &originalLayer->getLayerLoc()->halo;
191  int nx = getLayerLoc()->nx;
192  int ny = getLayerLoc()->ny;
193  int nxExt = nx + haloOrig->lt + haloOrig->rt;
194  int nyExt = ny + haloOrig->dn + haloOrig->up;
195  int nf = getLayerLoc()->nf;
196  int nbatch = getLayerLoc()->nbatch;
197  for(int b = 0; b < nbatch; b++){
198  const pvdata_t * sourceDataBatch = sourceData + b * nxExt * nyExt * nf;
199  //Reset currFeatureFreqCount
200  for(int kf = 0; kf < nf; kf++){
201  currFeatureFreqCount[b][kf] = 0;
202  }
203  for (int ky = 0; ky < ny; ky++){
204  for (int kx = 0; kx < nx; kx++){
205  for (int kf = 0; kf < nf; kf++){
206  int extIdx = kIndex(kx+haloOrig->lt, ky+haloOrig->up, kf, nxExt, nyExt, nf);
207  float inData = sourceDataBatch[extIdx];
208  if(inData > 0){ //Really use 0? Or should there be a threshold parameter
209  currFeatureFreqCount[b][kf]++;
210  }
211  }
212  }
213  }
214 
215  //Collect over mpi
216  MPI_Allreduce(MPI_IN_PLACE, currFeatureFreqCount[b], nf, MPI_LONG, MPI_SUM, parent->icCommunicator()->communicator());
217 
218  for (int kf = 0; kf < nf; kf++){
219  featureFreqCount[b][kf] += currFeatureFreqCount[b][kf];
220  if(featureFreqCount[b][kf] > maxCount[b]){
221  maxCount[b] = featureFreqCount[b][kf];
222  }
223  }
224  }

227 void ShuffleLayer::rejectionShuffle(const pvdata_t * sourceData, pvdata_t * activity){
228  const PVLayerLoc * loc = getLayerLoc();
229  PVHalo const * haloOrig = &originalLayer->getLayerLoc()->halo;
230  PVHalo const * halo = &loc->halo;
231  int nx = loc->nx;
232  int ny = loc->ny;
233  int nf = loc->nf;
234  int numextended = getNumExtended();
235  int nbatch = loc->nbatch;
236  int rndIdx, rndIdxOrig;
237  if(!readFreqFromFile && parent->simulationTime() <= freqCollectTime){
238  //Collect maxVActivity and featureFreq
239  collectFreq(sourceData);
240  }
241  else{
242  for (int i = 0; i < numextended*nbatch; i++) { //Zero activity array for shuffling activity
243  activity[i] = 0;
244  }
245  //NOTE: The following code assumes that the active features are sparse.
246  // If the number of active features in sourceData is greater than 1/2 of nf, while will loop infinitely
247  for(int b = 0; b < nbatch; b++){
248  pvdata_t * activityBatch = activity + b * numextended;
249  const pvdata_t * sourceDataBatch = sourceData + b * numextended;
250  for (int ky = 0; ky < ny; ky++){
251  for (int kx = 0; kx < nx; kx++){
252  int extIdx = kIndex(kx+halo->lt, ky+halo->up, 0, nx+halo->lt+halo->rt, ny+halo->dn+halo->up, nf);
253  int extIdxOrig = kIndex(kx+haloOrig->lt, ky+haloOrig->up, 0, nx+haloOrig->lt+haloOrig->rt, ny+haloOrig->dn+haloOrig->up, nf);
254  // Assumes stride in features is 1 when computing indices for features other than kf=0
255  for (int kf = 0; kf < nf; kf++){
256  float inData = sourceDataBatch[extIdxOrig+kf];
257  //If no activity, reject and continue
258  if(inData <= 0){
259  continue;
260  }
261  bool rejectFlag = true;
262  while(rejectFlag){
263  //Grab random feature index
264  int rdf = rand() % nf;
265  rndIdx = extIdx + rdf;
266  rndIdxOrig = extIdxOrig + rdf;
267  //Reject if random feature is active, or has been swapped
268  if(sourceDataBatch[rndIdxOrig] || activityBatch[rndIdx]){
269  continue;
270  }
271  //Grab random index from 0 to 1
272  float prd = (float)rand() / (float)RAND_MAX;
273  //Compare frequency
274  if(prd <= (float)featureFreqCount[b][rdf]/(float)maxCount[b]){
275  //accepted
276  rejectFlag = false;
277  }
278  }
279  //rdf is now the random index to shuffle with
280  activity[rndIdx] = sourceData[extIdxOrig+kf];
281  activity[extIdx+kf] = sourceData[rndIdxOrig]; // << This doesn't do anything
282  }
283  }
284  }
285  }
286  }

289 void ShuffleLayer::randomShuffle(const pvdata_t * sourceData, pvdata_t * activity){
290  const PVLayerLoc * loc = getLayerLoc();
291  PVHalo const * haloOrig = &originalLayer->getLayerLoc()->halo;
292  PVHalo const * halo = &loc->halo;
293  int nx = loc->nx;
294  int ny = loc->ny;
295  int nf = loc->nf;
296  int nbatch = loc->nbatch;
297  int numextended = getNumExtended();
298  int rndIdx, rndIdxOrig;
299  for (int i = 0; i < numextended * nbatch; i++) { //Zero activity array for shuffling activity
300  activity[i] = 0;
301  }
302  //NOTE: The following code assumes that the active features are sparse.
303  // If the number of active features in sourceData is greater than 1/2 of nf, do..while will loop infinitely
304  for(int b = 0; b < nbatch; b++){
305  const pvdata_t * sourceDataBatch = sourceData + b * numextended;
306  pvdata_t * activityBatch = activity + b * numextended;
307  for (int ky = 0; ky < ny; ky++){
308  for (int kx = 0; kx < nx; kx++){
309  int extIdx = kIndex(kx+halo->lt, ky+halo->rt, 0, nx+halo->lt+halo->rt, ny+halo->dn+halo->up, nf);
310  int extIdxOrig = kIndex(kx+haloOrig->lt, ky+haloOrig->up, 0, nx+haloOrig->lt+haloOrig->rt, ny+haloOrig->dn+haloOrig->up, nf);
311  // Assumes stride in features is 1 when computing indices for features other than kf=0
312  for (int kf = 0; kf < nf; kf++){
313  float inData = sourceDataBatch[extIdxOrig+kf];
314  if (inData != 0) { //Features with 0 activity are not changed
315  do {
316  int rd = rand() % nf; //TODO: Improve PRNG
317  rndIdx = extIdx + rd;
318  rndIdxOrig = extIdxOrig + rd;
319  } while(sourceDataBatch[rndIdxOrig] || activityBatch[rndIdx]);
320  activityBatch[rndIdx] = sourceDataBatch[extIdxOrig+kf];
321  activityBatch[extIdx+kf] = sourceDataBatch[rndIdxOrig];
322  }
323  }
324  }
325  }
326  }

329 int ShuffleLayer::updateState(double timef, double dt) {
330  int status = PV_SUCCESS;
331  //sourceData is extended
332  const pvdata_t * sourceData = originalLayer->getLayerData();
333  pvdata_t * A = getActivity();
334  const PVLayerLoc * loc = getLayerLoc();
335  const PVLayerLoc * sourceLoc = originalLayer->getLayerLoc();
336 
337  //Make sure layer loc and source layer loc is equivelent
338  assert(loc->nx == sourceLoc->nx);
339  assert(loc->ny == sourceLoc->ny);
340  assert(loc->nf == sourceLoc->nf);
341  //assert(loc->nb == sourceLoc->nb);
342 
343  //Create a one to one mapping of neuron to neuron
344  if (strcmp(shuffleMethod, "random") == 0){
345  randomShuffle(sourceData, A);
346  }
347  else if(strcmp(shuffleMethod, "rejection") == 0){
348  rejectionShuffle(sourceData, A);
349  }
350 
351  return status;
352 }
