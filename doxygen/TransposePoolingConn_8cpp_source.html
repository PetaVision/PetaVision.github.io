45 int TransposePoolingConn::initialize(const char *name, HyPerCol *hc) {
46  // It is okay for either of weightInitializer or weightNormalizer to be null at this point,
47  // either because we're in a subclass that doesn't need it, or because we are allowing for //
48  // backward compatibility.
49  // The two lines needs to be before the call to BaseConnection::initialize, because that function
50  // calls ioParamsFillGroup,
51  // which will call ioParam_weightInitType and ioParam_normalizeMethod, which for reasons of
52  // backward compatibility
53  // will create the initializer and normalizer if those member variables are null.
54  this->weightInitializer = NULL;
55  this->normalizer = NULL;
56 
57  int status = BaseConnection::initialize(name, hc); // BaseConnection should *NOT* take
58  // weightInitializer or weightNormalizer as
59  // arguments, as it does not know about
60  // InitWeights or NormalizeBase
61 
62  assert(parent);
63  PVParams *inputParams = parent->parameters();
64 
65  // set accumulateFunctionPointer
66  assert(!inputParams->presentAndNotBeenRead(name, "pvpatchAccumulateType"));
67  switch (mPoolingType) {
68  case PoolingConn::MAX:
69  accumulateFunctionPointer = &pvpatch_max_pooling;
70  accumulateFunctionFromPostPointer = &pvpatch_max_pooling_from_post;
71  break;
72  case PoolingConn::SUM:
73  accumulateFunctionPointer = &pvpatch_sum_pooling;
74  accumulateFunctionFromPostPointer = &pvpatch_sumpooling_from_post;
75  break;
76  case PoolingConn::AVG:
77  accumulateFunctionPointer = &pvpatch_sum_pooling;
78  accumulateFunctionFromPostPointer = &pvpatch_sumpooling_from_post;
79  break;
80  default: pvAssert(0); break;
81  }
82 
83  this->io_timer = new Timer(getName(), "conn", "io ");
84  this->update_timer = new Timer(getName(), "conn", "update ");
85 
86  return status;
87 }

239 int TransposePoolingConn::communicateInitInfo(CommunicateInitInfoMessage const *message) {
240  int status = PV_SUCCESS;
241  mOriginalConn = message->lookup<PoolingConn>(std::string(mOriginalConnName));
242  if (mOriginalConn == NULL) {
243  if (parent->columnId() == 0) {
244  ErrorLog().printf(
245  "%s: originalConnName \"%s\" is not a PoolingConn.\n",
246  getDescription_c(),
247  mOriginalConnName);
248  status = PV_FAILURE;
249  }
250  }
251  if (status != PV_SUCCESS)
252  return status;
253 
254  if (!mOriginalConn->getInitInfoCommunicatedFlag()) {
255  if (parent->columnId() == 0) {
256  InfoLog().printf(
257  "%s must wait until original connection \"%s\" has finished its communicateInitInfo "
258  "stage.\n",
259  getDescription_c(),
260  mOriginalConn->getName());
261  }
262  return PV_POSTPONE;
263  }
264 
265 #ifdef PV_USE_CUDA
266  receiveGpu = mOriginalConn->getReceiveGpu();
267  parent->parameters()->handleUnnecessaryParameter(name, "receiveGpu", receiveGpu);
268 #endif // PV_USE_CUDA
269 
270  sharedWeights = mOriginalConn->usingSharedWeights();
271  parent->parameters()->handleUnnecessaryParameter(name, "sharedWeights", sharedWeights);
272 
273  numAxonalArborLists = mOriginalConn->numberOfAxonalArborLists();
274  parent->parameters()->handleUnnecessaryParameter(name, "numAxonalArbors", numAxonalArborLists);
275 
276  plasticityFlag = mOriginalConn->getPlasticityFlag();
277  parent->parameters()->handleUnnecessaryParameter(name, "plasticityFlag", plasticityFlag);
278 
279  if (mOriginalConn->getShrinkPatches_flag()) {
280  if (parent->columnId() == 0) {
281  ErrorLog().printf(
282  "TransposePoolingConn \"%s\": original conn \"%s\" has shrinkPatches set to true. "
283  "TransposePoolingConn has not been implemented for that case.\n",
284  name,
285  mOriginalConn->getName());
286  }
287  MPI_Barrier(parent->getCommunicator()->communicator());
288  exit(EXIT_FAILURE);
289  }
290 
291  status = HyPerConn::communicateInitInfo(message); // calls setPatchSize()
292  if (status != PV_SUCCESS)
293  return status;
294 
295  // Check post layer phases to make sure it matches
296  if (mOriginalConn->postSynapticLayer()->getPhase() >= post->getPhase()) {
297  WarnLog().printf(
298  "TransposePoolingConn \"%s\": originalConn's post layer phase is greater or equal than "
299  "this layer's post. Behavior undefined.\n",
300  name);
301  }
302 
303  mPoolingType = mOriginalConn->getPoolingType();
304  if (parent->parameters()->stringPresent(name, "pvpatchAccumulateType")) {
305  char const *checkStringPresent =
306  parent->parameters()->stringValue(name, "pvpatchAccumulateType");
307  if (PoolingConn::parseAccumulateTypeString(checkStringPresent) != mPoolingType) {
308  Fatal() << getDescription()
309  << ": originalConn accumulateType does not match this layer's accumulate type.\n";
310  }
311  }
312  if (mPoolingType == PoolingConn::MAX && !mOriginalConn->needPostIndex()) {
313 #ifdef PV_USE_CUDA // Hopefully the awkwardness of this macro management will go away once I clean
314  // up this class.
315  if (!receiveGpu)
316 #endif // PV_USE_CUDA
317  {
318  if (parent->columnId() == 0) {
319  ErrorLog().printf(
320  "TransposePoolingConn \"%s\": original pooling conn \"%s\" needs to have a "
321  "postIndexLayer if unmax pooling.\n",
322  name,
323  mOriginalConnName);
324  status = PV_FAILURE;
325  }
326  }
327  }
328  if (status != PV_SUCCESS)
329  return status;
330 
331  const PVLayerLoc *preLoc = pre->getLayerLoc();
332  const PVLayerLoc *origPostLoc = mOriginalConn->postSynapticLayer()->getLayerLoc();
333  if (preLoc->nx != origPostLoc->nx || preLoc->ny != origPostLoc->ny
334  || preLoc->nf != origPostLoc->nf) {
335  if (parent->columnId() == 0) {
336  ErrorLog(errorMessage);
337  errorMessage.printf(
338  "%s: transpose's pre layer and original connection's post layer must have the same "
339  "dimensions.\n",
340  getDescription_c());
341  errorMessage.printf(
342  " (x=%d, y=%d, f=%d) versus (x=%d, y=%d, f=%d).\n",
343  preLoc->nx,
344  preLoc->ny,
345  preLoc->nf,
346  origPostLoc->nx,
347  origPostLoc->ny,
348  origPostLoc->nf);
349  }
350  MPI_Barrier(parent->getCommunicator()->communicator());
351  exit(EXIT_FAILURE);
352  }
353  const PVLayerLoc *postLoc = pre->getLayerLoc();
354  const PVLayerLoc *origPreLoc = mOriginalConn->postSynapticLayer()->getLayerLoc();
355  if (postLoc->nx != origPreLoc->nx || postLoc->ny != origPreLoc->ny
356  || postLoc->nf != origPreLoc->nf) {
357  if (parent->columnId() == 0) {
358  ErrorLog(errorMessage);
359  errorMessage.printf(
360  "%s: transpose's post layer and original connection's pre layer must have the same "
361  "dimensions.\n",
362  getDescription_c());
363  errorMessage.printf(
364  " (x=%d, y=%d, f=%d) versus (x=%d, y=%d, f=%d).\n",
365  postLoc->nx,
366  postLoc->ny,
367  postLoc->nf,
368  origPreLoc->nx,
369  origPreLoc->ny,
370  origPreLoc->nf);
371  }
372  MPI_Barrier(parent->getCommunicator()->communicator());
373  exit(EXIT_FAILURE);
374  }
375 
376  if (!updateGSynFromPostPerspective) {
377  mOriginalConn->setNeedPost();
378  }
379  mOriginalConn->setNeedAllocPostWeights(false);
380 
381  // Synchronize margines of this post and orig pre, and vice versa
382  mOriginalConn->preSynapticLayer()->synchronizeMarginWidth(post);
383  post->synchronizeMarginWidth(mOriginalConn->preSynapticLayer());
384 
385  mOriginalConn->postSynapticLayer()->synchronizeMarginWidth(pre);
386  pre->synchronizeMarginWidth(mOriginalConn->postSynapticLayer());
387 
388  if (mOriginalConn->getPostIndexLayer()) {
389  // Sync pre margins
390  // Synchronize margines of this post and the postIndexLayer, and vice versa
391  pre->synchronizeMarginWidth(mOriginalConn->getPostIndexLayer());
392  mOriginalConn->getPostIndexLayer()->synchronizeMarginWidth(pre);
393 
394  // Need to tell postIndexLayer the number of delays needed by this connection
395  int allowedDelay = mOriginalConn->getPostIndexLayer()->increaseDelayLevels(maxDelaySteps());
396  if (allowedDelay < getDelayArraySize()) {
397  if (this->parent->columnId() == 0) {
398  ErrorLog().printf(
399  "%s: attempt to set delay to %d, but the maximum allowed delay is %d. Exiting\n",
400  this->getDescription_c(),
401  getDelayArraySize(),
402  allowedDelay);
403  }
404  exit(EXIT_FAILURE);
405  }
406  }
407 
408  return status;
409 }

411 int TransposePoolingConn::setPatchSize() {
412  // If mOriginalConn is many-to-one, the transpose connection is one-to-many; then xscaleDiff > 0.
413  // Similarly, if mOriginalConn is one-to-many, xscaleDiff < 0.
414 
415  // Some of the code duplication might be eliminated by adding some functions to convert.h
416 
417  assert(pre && post);
418  assert(mOriginalConn);
419 
420  int xscaleDiff = pre->getXScale() - post->getXScale();
421  int nxp_orig = mOriginalConn->xPatchSize();
422  int nyp_orig = mOriginalConn->yPatchSize();
423  nxp = nxp_orig;
424  if (xscaleDiff > 0) {
425  nxp *= (int)pow(2, xscaleDiff);
426  }
427  else if (xscaleDiff < 0) {
428  nxp /= (int)pow(2, -xscaleDiff);
429  assert(nxp_orig == nxp * pow(2, (float)(-xscaleDiff)));
430  }
431 
432  int yscaleDiff = pre->getYScale() - post->getYScale();
433  nyp = nyp_orig;
434  if (yscaleDiff > 0) {
435  nyp *= (int)pow(2, yscaleDiff);
436  }
437  else if (yscaleDiff < 0) {
438  nyp /= (int)pow(2, -yscaleDiff);
439  assert(nyp_orig == nyp * pow(2, (float)(-yscaleDiff)));
440  }
441 
442  nfp = post->getLayerLoc()->nf;
443  // post->getLayerLoc()->nf must be the same as
444  // mOriginalConn->preSynapticLayer()->getLayerLoc()->nf.
445  // This requirement is checked in communicateInitInfo
446 
447  parent->parameters()->handleUnnecessaryParameter(name, "nxp", nxp);
448  parent->parameters()->handleUnnecessaryParameter(name, "nyp", nyp);
449  parent->parameters()->handleUnnecessaryParameter(name, "nfp", nfp);
450  return PV_SUCCESS;
451 }

584 int TransposePoolingConn::deliverPresynapticPerspective(PVLayerCube const *activity, int arborID) {
585  // Check if we need to update based on connection's channel
586  if (getChannel() == CHANNEL_NOUPDATE) {
587  return PV_SUCCESS;
588  }
589  assert(post->getChannel(getChannel()));
590 
591  const PVLayerLoc *preLoc = preSynapticLayer()->getLayerLoc();
592  const PVLayerLoc *postLoc = postSynapticLayer()->getLayerLoc();
593 
594  assert(arborID >= 0);
595  const int numExtended = activity->numItems;
596 
597  // Grab postIdxLayer's data
598  float *postIdxData = nullptr;
599  if (mPoolingType == PoolingConn::MAX) {
600  PoolingIndexLayer *postIndexLayer = mOriginalConn->getPostIndexLayer();
601  assert(postIndexLayer);
602  // Make sure this layer is an integer layer
603  assert(postIndexLayer->getDataType() == PV_INT);
604  int delay = getDelay(arborID);
605  PVLayerCube cube = postIndexLayer->getPublisher()->createCube(delay);
606  postIdxData = cube.data;
607  }
608 
609  for (int b = 0; b < parent->getNBatch(); b++) {
610  float *activityBatch = activity->data
611  + b * (preLoc->nx + preLoc->halo.rt + preLoc->halo.lt)
612  * (preLoc->ny + preLoc->halo.up + preLoc->halo.dn)
613  * preLoc->nf;
