49 int TransposePoolingConn::initialize(const char * name, HyPerCol * hc) {
50  // It is okay for either of weightInitializer or weightNormalizer to be null at this point, either because we're in a subclass that doesn't need it, or because we are allowing for // backward compatibility.
51  // The two lines needs to be before the call to BaseConnection::initialize, because that function calls ioParamsFillGroup,
52  // which will call ioParam_weightInitType and ioParam_normalizeMethod, which for reasons of backward compatibility
53  // will create the initializer and normalizer if those member variables are null.
54  this->weightInitializer = NULL;
55  this->normalizer = NULL;
56 
57  int status = BaseConnection::initialize(name, hc); // BaseConnection should *NOT* take weightInitializer or weightNormalizer as arguments, as it does not know about InitWeights or NormalizeBase
58 
59  assert(parent);
60  PVParams * inputParams = parent->parameters();
61 
62  //set accumulateFunctionPointer
63  assert(!inputParams->presentAndNotBeenRead(name, "pvpatchAccumulateType"));
64  switch (pvpatchAccumulateType) {
65  case ACCUMULATE_CONVOLVE:
66  std::cout << "ACCUMULATE_CONVOLVE not allowed in TransposePoolingConn\n";
67  exit(-1);
68  break;
69  case ACCUMULATE_STOCHASTIC:
70  std::cout << "ACCUMULATE_STOCASTIC not allowed in TransposePoolingConn\n";
71  exit(-1);
72  break;
73  case ACCUMULATE_MAXPOOLING:
74  accumulateFunctionPointer = &pvpatch_max_pooling;
75  accumulateFunctionFromPostPointer = &pvpatch_max_pooling_from_post;
76  break;
77  case ACCUMULATE_SUMPOOLING:
78  accumulateFunctionPointer = &pvpatch_sum_pooling;
79  accumulateFunctionFromPostPointer = &pvpatch_sumpooling_from_post;
80  break;
81  default:
82  assert(0);
83  break;
84  }
85 
86  //ioAppend = parent->getCheckpointReadFlag();
87 
88  this->io_timer = new Timer(getName(), "conn", "io ");
89  this->update_timer = new Timer(getName(), "conn", "update ");
90 
91  return status;
92 }

225 int TransposePoolingConn::communicateInitInfo() {
226  int status = PV_SUCCESS;
227  BaseConnection * originalConnBase = parent->getConnFromName(this->originalConnName);
228  if (originalConnBase==NULL) {
229  if (parent->columnId()==0) {
230  fprintf(stderr, "%s \"%s\" error: originalConnName \"%s\" does not refer to any connection in the column.\n", this->getKeyword(), name, this->originalConnName);
231  }
232  MPI_Barrier(parent->icCommunicator()->communicator());
233  exit(EXIT_FAILURE);
234  }
235  originalConn = dynamic_cast<PoolingConn *>(originalConnBase);
236  if (originalConn == NULL) {
237  if (parent->columnId()==0) {
238  fprintf(stderr, "TransposePoolingConn \"%s\" error: originalConnName \"%s\" is not a PoolingConn.\n", name, originalConnName);
239  status = PV_FAILURE;
240  }
241  }
242  if (status != PV_SUCCESS) return status;
243 
244  if (!originalConn->getInitInfoCommunicatedFlag()) {
245  if (parent->columnId()==0) {
246  const char * connectiontype = this->getKeyword();
247  printf("%s \"%s\" must wait until original connection \"%s\" has finished its communicateInitInfo stage.\n", connectiontype, name, originalConn->getName());
248  }
249  return PV_POSTPONE;
250  }
251 
252 
253  sharedWeights = originalConn->usingSharedWeights();
254  parent->parameters()->handleUnnecessaryParameter(name, "sharedWeights", sharedWeights);
255 
256  numAxonalArborLists = originalConn->numberOfAxonalArborLists();
257  parent->parameters()->handleUnnecessaryParameter(name, "numAxonalArbors", numAxonalArborLists);
258 
259  plasticityFlag = originalConn->getPlasticityFlag();
260  parent->parameters()->handleUnnecessaryParameter(name, "plasticityFlag", plasticityFlag);
261 
262  if(originalConn->getShrinkPatches_flag()) {
263  if (parent->columnId()==0) {
264  fprintf(stderr, "TransposePoolingConn \"%s\" error: original conn \"%s\" has shrinkPatches set to true. TransposePoolingConn has not been implemented for that case.\n", name, originalConn->getName());
265  }
266  MPI_Barrier(parent->icCommunicator()->communicator());
267  exit(EXIT_FAILURE);
268  }
269 
270  status = HyPerConn::communicateInitInfo(); // calls setPatchSize()
271  if (status != PV_SUCCESS) return status;
272 
273  if(!originalConn->needPostIndex() && pvpatchAccumulateType == ACCUMULATE_MAXPOOLING){
274  if (parent->columnId()==0) {
275  fprintf(stderr, "TransposePoolingConn \"%s\" error: original pooling conn \"%s\" needs to have a postIndexLayer if unmax pooling.\n", name, originalConnName);
276  status = PV_FAILURE;
277  }
278  }
279  if (status != PV_SUCCESS) return status;
280 
281  //Check post layer phases to make sure it matches
282  if(originalConn->postSynapticLayer()->getPhase() >= post->getPhase()){
283  fprintf(stderr, "TransposePoolingConn \"%s\" warning: originalConn's post layer phase is greater or equal than this layer's post. Behavior undefined.\n", name);
284  }
285 
286  if(originalConn->getPvpatchAccumulateType() != getPvpatchAccumulateType()){
287  fprintf(stderr, "TransposePoolingConn \"%s\" error: originalConn accumulateType does not match this layer's accumulate type.\n", name);
288  exit(-1);
289  }
290 
291  const PVLayerLoc * preLoc = pre->getLayerLoc();
292  const PVLayerLoc * origPostLoc = originalConn->postSynapticLayer()->getLayerLoc();
293  if (preLoc->nx != origPostLoc->nx || preLoc->ny != origPostLoc->ny || preLoc->nf != origPostLoc->nf) {
294  if (parent->columnId()==0) {
295  fprintf(stderr, "%s \"%s\" error: transpose's pre layer and original connection's post layer must have the same dimensions.\n", this->getKeyword(), name);
296  fprintf(stderr, " (x=%d, y=%d, f=%d) versus (x=%d, y=%d, f=%d).\n", preLoc->nx, preLoc->ny, preLoc->nf, origPostLoc->nx, origPostLoc->ny, origPostLoc->nf);
297  }
298  MPI_Barrier(parent->icCommunicator()->communicator());
299  exit(EXIT_FAILURE);
300  }
301  const PVLayerLoc * postLoc = pre->getLayerLoc();
302  const PVLayerLoc * origPreLoc = originalConn->postSynapticLayer()->getLayerLoc();
303  if (postLoc->nx != origPreLoc->nx || postLoc->ny != origPreLoc->ny || postLoc->nf != origPreLoc->nf) {
304  if (parent->columnId()==0) {
305  fprintf(stderr, "%s \"%s\" error: transpose's post layer and original connection's pre layer must have the same dimensions.\n", this->getKeyword(), name);
306  fprintf(stderr, " (x=%d, y=%d, f=%d) versus (x=%d, y=%d, f=%d).\n", postLoc->nx, postLoc->ny, postLoc->nf, origPreLoc->nx, origPreLoc->ny, origPreLoc->nf);
307  }
308  MPI_Barrier(parent->icCommunicator()->communicator());
309  exit(EXIT_FAILURE);
310  }
311 
312  originalConn->setNeedPost(true);
313  originalConn->setNeedAllocPostWeights(false);
314 
315  //Synchronize margines of this post and orig pre, and vice versa
316  originalConn->preSynapticLayer()->synchronizeMarginWidth(post);
317  post->synchronizeMarginWidth(originalConn->preSynapticLayer());
318 
319  originalConn->postSynapticLayer()->synchronizeMarginWidth(pre);
320  pre->synchronizeMarginWidth(originalConn->postSynapticLayer());
321 
322  if(pvpatchAccumulateType == ACCUMULATE_MAXPOOLING){
323  //Sync pre margins
324  //Synchronize margines of this post and the postIndexLayer, and vice versa
325  pre->synchronizeMarginWidth(originalConn->getPostIndexLayer());
326  originalConn->getPostIndexLayer()->synchronizeMarginWidth(pre);
327 
328  //Need to tell postIndexLayer the number of delays needed by this connection
329  int allowedDelay = originalConn->getPostIndexLayer()->increaseDelayLevels(getDelayArraySize());
330  if( allowedDelay < getDelayArraySize()) {
331  if( this->getParent()->columnId() == 0 ) {
332  fflush(stdout);
333  fprintf(stderr, "Connection \"%s\": attempt to set delay to %d, but the maximum allowed delay is %d. Exiting\n", this->getName(), getDelayArraySize(), allowedDelay);
334  }
335  exit(EXIT_FAILURE);
336  }
337  }
338 
339 
340  return status;
341 }

343 int TransposePoolingConn::setPatchSize() {
344  // If originalConn is many-to-one, the transpose connection is one-to-many; then xscaleDiff > 0.
345  // Similarly, if originalConn is one-to-many, xscaleDiff < 0.
346 
347  // Some of the code duplication might be eliminated by adding some functions to convert.h
348 
349  assert(pre && post);
350  assert(originalConn);
351 
352  int xscaleDiff = pre->getXScale() - post->getXScale();
353  int nxp_orig = originalConn->xPatchSize();
354  int nyp_orig = originalConn->yPatchSize();
355  nxp = nxp_orig;
356  if(xscaleDiff > 0 ) {
357  nxp *= (int) pow( 2, xscaleDiff );
358  }
359  else if(xscaleDiff < 0) {
360  nxp /= (int) pow(2,-xscaleDiff);
361  assert(nxp_orig==nxp*pow( 2, (float) (-xscaleDiff) ));
362  }
363 
364  int yscaleDiff = pre->getYScale() - post->getYScale();
365  nyp = nyp_orig;
366  if(yscaleDiff > 0 ) {
367  nyp *= (int) pow( 2, yscaleDiff );
368  }
369  else if(yscaleDiff < 0) {
370  nyp /= (int) pow(2,-yscaleDiff);
371  assert(nyp_orig==nyp*pow( 2, (float) (-yscaleDiff) ));
372  }
373 
374  nfp = post->getLayerLoc()->nf;
375  // post->getLayerLoc()->nf must be the same as originalConn->preSynapticLayer()->getLayerLoc()->nf.
376  // This requirement is checked in communicateInitInfo
377 
378  parent->parameters()->handleUnnecessaryParameter(name, "nxp", nxp);
379  parent->parameters()->handleUnnecessaryParameter(name, "nyp", nyp);
380  parent->parameters()->handleUnnecessaryParameter(name, "nfp", nfp);
381  return PV_SUCCESS;
382 
383 }

478 int TransposePoolingConn::deliverPresynapticPerspective(PVLayerCube const * activity, int arborID) {

480  if(getChannel() == CHANNEL_NOUPDATE){
481  return PV_SUCCESS;
482  }
483  assert(post->getChannel(getChannel()));
484 
485  const PVLayerLoc * preLoc = preSynapticLayer()->getLayerLoc();

487 
488  assert(arborID >= 0);
489  const int numExtended = activity->numItems;
490 
491  //Grab postIdxLayer's data
492  int* postIdxData = NULL;
493  if(pvpatchAccumulateType == ACCUMULATE_MAXPOOLING){
494  PoolingIndexLayer* postIndexLayer = originalConn->getPostIndexLayer();
495  assert(postIndexLayer);

503  }

505  for(int b = 0; b < parent->getNBatch(); b++){

507  pvdata_t * gSynPatchHeadBatch = post->getChannel(getChannel()) + b * postLoc->nx * postLoc->ny * postLoc->nf;
508  int * postIdxDataBatch = NULL;
509  if(pvpatchAccumulateType == ACCUMULATE_MAXPOOLING){
510  postIdxDataBatch = postIdxData + b * originalConn->getPostIndexLayer()->getNumExtended();
511  }
512

514  if(activity->isSparse){
515  activeIndicesBatch = activity->activeIndices + b * (preLoc->nx + preLoc->halo.rt + preLoc->halo.lt) * (preLoc->ny + preLoc->halo.up + preLoc->halo.dn) * preLoc->nf;
516  }

518  int numLoop;

521  }
522  else{
523  numLoop = numExtended;
524  }

537  }

539 #endif // PV_USE_OPENMP_THREADS

541

544 #endif
