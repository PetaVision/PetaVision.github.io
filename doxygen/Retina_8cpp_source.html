366 void Retina::ioParam_foregroundRate(enum ParamsIOFlag ioFlag) {
367  PVParams * params = parent->parameters();
368  if (ioFlag==PARAMS_IO_READ && !params->present(name, "foregroundRate")) {
369  if (params->present(name, "noiseOnFreq")) {
370  probStimParam = params->value(name, "noiseOnFreq");
371  if (parent->columnId()==0) {
372  fprintf(stderr, "Warning: noiseOnFreq is deprecated. Use foregroundRate instead.\n");
373  }
374  return;
375  }
376  if (params->present(name, "poissonEdgeProb")) {
377  probStimParam = params->value(name, "poissonEdgeProb");
378  if (parent->columnId()==0) {
379  fprintf(stderr, "Warning: poissonEdgeProb is deprecated. Use foregroundRate instead.\n");
380  }
381  return;
382  }
383  }
384  // noiseOnFreq and poissonEdgeProb were deprecated Jan 24, 2013
385  // When support for noiseOnFreq and poissonEdgeProb is removed entirely, remove the above if-statement and keep the ioParamValue call below.
386  parent->ioParamValue(ioFlag, name, "foregroundRate", &probStimParam, 1.0f);
387 }

389 void Retina::ioParam_backgroundRate(enum ParamsIOFlag ioFlag) {
390  PVParams * params = parent->parameters();
391  if (ioFlag==PARAMS_IO_READ && !params->present(name, "backgroundRate")) {
392  if (params->present(name, "noiseOffFreq")) {
393  probBaseParam = params->value(name, "noiseOffFreq");
394  if (parent->columnId()==0) {
395  fprintf(stderr, "Warning: noiseOffFreq is deprecated. Use backgroundRate instead.\n");
396  }
397  return;
398  }
399  if (params->present(name, "poissonBlankProb")) {
400  probBaseParam = params->value(name, "poissonBlankProb");
401  if (parent->columnId()==0) {
402  fprintf(stderr, "Warning: poissonEdgeProb is deprecated. Use backgroundRate instead.\n");
403  }
404  return;
405  }
406  }
407  // noiseOffFreq and poissonBlankProb was deprecated Jan 24, 2013
408  // When support for noiseOnFreq and poissonEdgeProb is removed entirely, remove the above if-statement and keep the ioParamValue call below.
409  parent->ioParamValue(ioFlag, name, "backgroundRate", &probBaseParam, 0.0f);
410  if (ioFlag==PARAMS_IO_READ) {
411  assert(!parent->parameters()->presentAndNotBeenRead(name, "foregroundRate"));
412  if (probBaseParam > probStimParam) {
413  fprintf(stderr, "Error in %s \"%s\": backgroundRate cannot be greater than foregroundRate.\n",
414  getKeyword(), name);
415  exit(EXIT_FAILURE);
416  }
417  }
418 }
