60 int FilenameParsingGroundTruthLayer::registerData(Checkpointer *checkpointer) {
61  // Belongs in registerData and not allocateDataStructures because the
62  // default path for the class list is the output path, which is available
63  // only in registerData through the checkpointer argument.
64  // But is this the best default?
65  int status = HyPerLayer::registerData(checkpointer);
66 
67  std::ifstream inputFile;
68  std::string outPath("");
69 
70  if (mClassListFileName != nullptr) {
71  outPath += std::string(mClassListFileName);
72  }
73  else {
74  outPath += checkpointer->getOutputPath();
75  outPath += "/classes.txt";
76  }
77 
78  inputFile.open(outPath.c_str(), std::ifstream::in);
79  FatalIf(!inputFile.is_open(), "%s: Unable to open file %s\n", getName(), outPath.c_str());
80 
81  mClasses.clear();
82  std::string line;
83  while (getline(inputFile, line)) {
84  mClasses.push_back(line);
85  }
86  inputFile.close();
87 
88  std::size_t numFeatures = (std::size_t)getLayerLoc()->nf;
89  FatalIf(
90  numFeatures != mClasses.size(),
91  "%s has %d features but classList \"%s\" has %zu categories.\n",
92  getDescription_c(),
93  getLayerLoc()->nf,
94  outPath.c_str(),
95  mClasses.size());
96  return status;
97 }

122 int FilenameParsingGroundTruthLayer::updateState(double time, double dt) {
123  update_timer->start();
124  float *A = getCLayer()->activity->data;
125  const PVLayerLoc *loc = getLayerLoc();
126  int numNeurons = getNumNeurons();
127  int const localBatchWidth = getLayerLoc()->nbatch;
128  int const blockBatchWidth = getMPIBlock()->getBatchDimension() * localBatchWidth;
129  for (int b = 0; b < blockBatchWidth; b++) {
130  int const mpiBlockBatchIndex = b / localBatchWidth; // integer division
131  int const localBatchIndex = b % localBatchWidth;
132 
133  std::vector<float> fileMatches(mClasses.size());
134  if (getMPIBlock()->getRank() == 0) {
135  std::string currentFilename =
136  mInputLayer->getCurrentFilename(localBatchIndex, mpiBlockBatchIndex).c_str();
137  for (auto ci = (std::size_t)0; ci < mClasses.size(); ci++) {
138  std::size_t match = currentFilename.find(mClasses.at(ci));
139  fileMatches[ci] = match != std::string::npos ? mGtClassTrueValue : mGtClassFalseValue;
140  }
141  }
142  // It seems clunky to send each process all the fileMatches, when
143  // they'll only use only the fileMatches for the correct MPIBlock
144  // batch index. Use MPI_Send/MPI_Recv? Create more MPI_Comm's?
145  MPI_Bcast(fileMatches.data(), fileMatches.size(), MPI_FLOAT, 0, getMPIBlock()->getComm());
146 
147  if (getMPIBlock()->getBatchIndex() != mpiBlockBatchIndex) {
148  continue;
149  }
150 
151  float *ABatch = A + localBatchIndex * getNumExtended();
152  for (int i = 0; i < numNeurons; i++) {
153  int nExt = kIndexExtended(
154  i,
155  loc->nx,
156  loc->ny,
157  loc->nf,
158  loc->halo.lt,
159  loc->halo.rt,
160  loc->halo.dn,
161  loc->halo.up);
162  int fi = featureIndex(
163  nExt,
164  loc->nx + loc->halo.rt + loc->halo.lt,
165  loc->ny + loc->halo.dn + loc->halo.up,
166  loc->nf);
167  ABatch[nExt] = fileMatches[fi];
168  }
169  }
170  update_timer->stop();
171  return PV_SUCCESS;
172 }
