97 int FilenameParsingGroundTruthLayer::updateState(double time, double dt)
98 {
99  update_timer->start();
100  pvdata_t * A = getCLayer()->activity->data;
101  const PVLayerLoc * loc = getLayerLoc();
102  int num_neurons = getNumNeurons();
103  if (num_neurons != numClasses)
104  {
105  std::cout << "The number of neurons in " << getName() << " is not equal to the number of classes specified in " << parent->getOutputPath() << "/classes.txt\n";
106  exit(EXIT_FAILURE);
107  }
108 
109  for(int b = 0; b < loc->nbatch; b++){
110  char * currentFilename = NULL;
111  int filenameLen = 0;
112  //TODO depending on speed of this layer, more efficient way would be to preallocate currentFilename buffer
113  if(parent->icCommunicator()->commRank()==0){
114  currentFilename = strdup(movieLayer->getFilename(b));
115  //Get length of currentFilename and broadcast
116  int filenameLen = (int) strlen(currentFilename) + 1; //+1 for the null terminator
117  //Using local communicator, as each batch MPI will handle it's own run
118  MPI_Bcast(&filenameLen, 1, MPI_INT, 0, parent->icCommunicator()->communicator());
119  //Braodcast filename to all other local processes
120  MPI_Bcast(currentFilename, filenameLen, MPI_CHAR, 0, parent->icCommunicator()->communicator());
121  }
122  else{
123  //Receive broadcast about length of filename
124  MPI_Bcast(&filenameLen, 1, MPI_INT, 0, parent->icCommunicator()->communicator());
125  currentFilename = (char*)calloc(sizeof(char), filenameLen);
126  //Receive filename
127  MPI_Bcast(currentFilename, filenameLen, MPI_CHAR, 0, parent->icCommunicator()->communicator());
128  }
129 
130  std::string fil = currentFilename;
131  pvdata_t * ABatch = A + b * getNumExtended();
132  for(int i = 0; i < num_neurons; i++){
133  int nExt = kIndexExtended(i, loc->nx, loc->ny, loc->nf, loc->halo.lt, loc->halo.rt, loc->halo.dn, loc->halo.up);
134  int fi = featureIndex(nExt, loc->nx+loc->halo.rt+loc->halo.lt, loc->ny+loc->halo.dn+loc->halo.up, loc->nf);
135  int match = fil.find(classes[i]);
136  if(0 <= match){
137  ABatch[nExt] = gtClassTrueValue;
138  }
139  else{
140  ABatch[nExt] = gtClassFalseValue;
141  }
142  }
143  //Free buffer, TODO, preallocate buffer to avoid this
144  free(currentFilename);
145  }
146  update_timer->stop();
147  return PV_SUCCESS;
148 }
