59 double L2NormProbe::getValueInternal(double timevalue, int index) {
60  if (index < 0 || index >= parent->getNBatch()) {
61  return PV_FAILURE;
62  }
63  PVLayerLoc const *loc = getTargetLayer()->getLayerLoc();
64  int const nx = loc->nx;
65  int const ny = loc->ny;
66  int const nf = loc->nf;
67  PVHalo const *halo = &loc->halo;
68  int const lt = halo->lt;
69  int const rt = halo->rt;
70  int const dn = halo->dn;
71  int const up = halo->up;
72  double l2normsq = 0.0;
73  float const *aBuffer =
74  getTargetLayer()->getLayerData() + index * getTargetLayer()->getNumExtended();
75 
76  if (getMaskLayer()) {
77  PVLayerLoc const *maskLoc = getMaskLayer()->getLayerLoc();
78  PVHalo const *maskHalo = &maskLoc->halo;
79  float const *maskLayerData =
80  getMaskLayer()->getLayerData()
81  + index * getMaskLayer()->getNumExtended(); // Is there a DataStore method to return the
82  // part of the layer data for a given batch
83  // index?
84  int const maskLt = maskHalo->lt;
85  int const maskRt = maskHalo->rt;
86  int const maskDn = maskHalo->dn;
87  int const maskUp = maskHalo->up;
88  if (maskHasSingleFeature()) {
89  assert(getTargetLayer()->getNumNeurons() == nx * ny * nf);
90  int nxy = nx * ny;
91 #ifdef PV_USE_OPENMP_THREADS
92 #pragma omp parallel for reduction(+ : l2normsq)
93 #endif // PV_USE_OPENMP_THREADS
94  for (int kxy = 0; kxy < nxy; kxy++) {
95  int kexMask = kIndexExtended(kxy, nx, ny, 1, maskLt, maskRt, maskDn, maskUp);
96  if (maskLayerData[kexMask]) {
97  int featureBase = kxy * nf;
98  for (int f = 0; f < nf; f++) {
99  int kex = kIndexExtended(featureBase++, nx, ny, nf, lt, rt, dn, up);
100  l2normsq += pow((double)aBuffer[kex], 2.0);
101  }
102  }
103  }
104  }
105  else {
106 #ifdef PV_USE_OPENMP_THREADS
107 #pragma omp parallel for reduction(+ : l2normsq)
108 #endif // PV_USE_OPENMP_THREADS
109  for (int k = 0; k < getTargetLayer()->getNumNeurons(); k++) {
110  int kex = kIndexExtended(k, nx, ny, nf, lt, rt, dn, up);
111  int kexMask = kIndexExtended(k, nx, ny, nf, maskLt, maskRt, maskDn, maskUp);
112  double val = aBuffer[kex];
113  l2normsq += (double)maskLayerData[kexMask] * val * val;
114  }
115  }
116  }
117  else {
118  if (getTargetLayer()->getSparseFlag()) {
119  PVLayerCube cube = getTargetLayer()->getPublisher()->createCube();
120  long int numActive = cube.numActive[index];
121  int numItems = cube.numItems / cube.loc.nbatch;
122  SparseList<float>::Entry const *activeList =
123  &((SparseList<float>::Entry *)cube.activeIndices)[index * numItems];
124 #ifdef PV_USE_OPENMP_THREADS
125 #pragma omp parallel for reduction(+ : l2normsq)
126 #endif // PV_USE_OPENMP_THREADS
127  for (int k = 0; k < numActive; k++) {
128  int extIndex = activeList[k].index;
129  int inRestricted = !extendedIndexInBorderRegion(
130  extIndex, nx, ny, nf, halo->lt, halo->rt, halo->dn, halo->up);
131  double val = inRestricted * fabs((double)aBuffer[extIndex]);
132  l2normsq += pow(val, 2.0);
133  }
134  }
135  else {
136 #ifdef PV_USE_OPENMP_THREADS
137 #pragma omp parallel for reduction(+ : l2normsq)
138 #endif // PV_USE_OPENMP_THREADS
139  for (int k = 0; k < getTargetLayer()->getNumNeurons(); k++) {
140  int kex = kIndexExtended(k, nx, ny, nf, lt, rt, dn, up);
141  l2normsq += pow((double)aBuffer[kex], 2.0);
142  }
143  }
144  }
145 
146  return l2normsq;
147 }
