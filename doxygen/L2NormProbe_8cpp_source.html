62 double L2NormProbe::getValueInternal(double timevalue, int index) {
63  if (index < 0 || index >= getParent()->getNBatch()) { return PV_FAILURE; }
64  PVLayerLoc const * loc = getTargetLayer()->getLayerLoc();
65  int const nx = loc->nx;
66  int const ny = loc->ny;
67  int const nf = loc->nf;
68  PVHalo const * halo = &loc->halo;
69  int const lt = halo->lt;
70  int const rt = halo->rt;
71  int const dn = halo->dn;
72  int const up = halo->up;
73  double l2normsq = 0.0;
74  pvadata_t const * aBuffer = getTargetLayer()->getLayerData() + index * getTargetLayer()->getNumExtended();
75 
76  if (getMaskLayer()) {
77  PVLayerLoc const * maskLoc = getMaskLayer()->getLayerLoc();
78  PVHalo const * maskHalo = &maskLoc->halo;
79  pvadata_t const * maskLayerData = getMaskLayer()->getLayerData() + index*getMaskLayer()->getNumExtended(); // Is there a DataStore method to return the part of the layer data for a given batch index?
80  int const maskLt = maskHalo->lt;
81  int const maskRt = maskHalo->rt;
82  int const maskDn = maskHalo->dn;
83  int const maskUp = maskHalo->up;
84  if (maskHasSingleFeature()) {
85  assert(getTargetLayer()->getNumNeurons()==nx*ny*nf);
86  int nxy = nx*ny;
87 #ifdef PV_USE_OPENMP_THREADS
88 #pragma omp parallel for reduction(+ : l2normsq)
89 #endif // PV_USE_OPENMP_THREADS
90  for (int kxy=0; kxy<nxy; kxy++) {
91  int kexMask = kIndexExtended(kxy, nx, ny, 1, maskLt, maskRt, maskDn, maskUp);
92  if (maskLayerData[kexMask]) {
93  int featureBase = kxy*nf;
94  for (int f=0; f<nf; f++) {
95  int kex = kIndexExtended(featureBase++, nx, ny, nf, lt, rt, dn, up);
96  pvadata_t val = aBuffer[kex];
97  l2normsq += fabs(val*val);
98  }
99  }
100  }
101  }
102  else {
103 #ifdef PV_USE_OPENMP_THREADS
104 #pragma omp parallel for reduction(+ : l2normsq)
105 #endif // PV_USE_OPENMP_THREADS
106  for (int k=0; k<getTargetLayer()->getNumNeurons(); k++) {
107  int kex = kIndexExtended(k, nx, ny, nf, lt, rt, dn, up);
108  int kexMask = kIndexExtended(k, nx, ny, nf, maskLt, maskRt, maskDn, maskUp);
109  if (maskLayerData[kexMask]) {
110  pvadata_t val = aBuffer[kex];
111  l2normsq += fabs(val*val);
112  }
113  }
114  }
115  }
116  else {
117 #ifdef PV_USE_OPENMP_THREADS
118 #pragma omp parallel for reduction(+ : l2normsq)
119 #endif // PV_USE_OPENMP_THREADS
120  for (int k=0; k<getTargetLayer()->getNumNeurons(); k++) {
121  int kex = kIndexExtended(k, nx, ny, nf, lt, rt, dn, up);
122  pvadata_t val = aBuffer[kex];
123  l2normsq += fabs(val*val);
124  }
125  }
126  return l2normsq;
127 }

129 int L2NormProbe::calcValues(double timevalue) {
