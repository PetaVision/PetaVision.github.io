35 int BackgroundLayer::communicateInitInfo(CommunicateInitInfoMessage const *message) {
36  int status = HyPerLayer::communicateInitInfo(message);
37  originalLayer = message->lookup<HyPerLayer>(std::string(originalLayerName));
38  if (originalLayer == NULL) {
39  if (parent->columnId() == 0) {
40  ErrorLog().printf(
41  "%s: originalLayerName \"%s\" is not a layer in the HyPerCol.\n",
42  getDescription_c(),
43  originalLayerName);
44  }
45  MPI_Barrier(parent->getCommunicator()->communicator());
46  exit(EXIT_FAILURE);
47  }
48  const PVLayerLoc *srcLoc = originalLayer->getLayerLoc();
49  const PVLayerLoc *loc = getLayerLoc();
50  assert(srcLoc != NULL && loc != NULL);
51  if (srcLoc->nxGlobal != loc->nxGlobal || srcLoc->nyGlobal != loc->nyGlobal) {
52  if (parent->columnId() == 0) {
53  ErrorLog(errorMessage);
54  errorMessage.printf(
55  "%s: originalLayerName \"%s\" does not have the same X/Y dimensions.\n",
56  getDescription_c(),
57  originalLayerName);
58  errorMessage.printf(
59  " original (nx=%d, ny=%d, nf=%d) versus (nx=%d, ny=%d, nf=%d)\n",
60  srcLoc->nxGlobal,
61  srcLoc->nyGlobal,
62  srcLoc->nf,
63  loc->nxGlobal,
64  loc->nyGlobal,
65  loc->nf);
66  }
67  MPI_Barrier(parent->getCommunicator()->communicator());
68  exit(EXIT_FAILURE);
69  }
70  if ((srcLoc->nf + 1) * repFeatureNum != loc->nf) {
71  if (parent->columnId() == 0) {
72  ErrorLog(errorMessage);
73  errorMessage.printf(
74  "%s: nf must have (n+1)*repFeatureNum (%d) features in BackgroundLayer \"%s\", "
75  "where n is the orig layer number of features.\n",
76  getDescription_c(),
77  (srcLoc->nf + 1) * repFeatureNum,
78  originalLayerName);
79  errorMessage.printf(
80  " original (nx=%d, ny=%d, nf=%d) versus (nx=%d, ny=%d, nf=%d)\n",
81  srcLoc->nxGlobal,
82  srcLoc->nyGlobal,
83  srcLoc->nf,
84  loc->nxGlobal,
85  loc->nyGlobal,
86  loc->nf);
87  }
88  MPI_Barrier(parent->getCommunicator()->communicator());
89  exit(EXIT_FAILURE);
90  }
91  assert(srcLoc->nx == loc->nx && srcLoc->ny == loc->ny);
92  return status;
93 }

121 int BackgroundLayer::updateState(double timef, double dt) {
122  int status = PV_SUCCESS;
123  float *A = clayer->activity->data;
124  const float *originalA = originalLayer->getCLayer()->activity->data;
125  const PVLayerLoc *loc = getLayerLoc();
126  const PVLayerLoc *locOriginal = originalLayer->getLayerLoc();
127 
128  // Make sure all sizes match
129  assert(locOriginal->nx == loc->nx);
130  assert(locOriginal->ny == loc->ny);
131  assert((locOriginal->nf + 1) * repFeatureNum == loc->nf);
132 
133  int nx = loc->nx;
134  int ny = loc->ny;
135  int origNf = locOriginal->nf;
136  int thisNf = loc->nf;
137  int nbatch = loc->nbatch;
138 
139  PVHalo const *halo = &loc->halo;
140  PVHalo const *haloOrig = &locOriginal->halo;
141 
142  for (int b = 0; b < nbatch; b++) {
143  float *ABatch = A + b * getNumExtended();
144  const float *originalABatch = originalA + b * originalLayer->getNumExtended();
145 
146 // Loop through all nx and ny
147 // each y value specifies a different target so ok to thread here (sum, sumsq are defined inside
148 // loop)
149 #ifdef PV_USE_OPENMP_THREADS
150 #pragma omp parallel for
151 #endif
152  for (int iY = 0; iY < ny; iY++) {
153  for (int iX = 0; iX < nx; iX++) {
154  // outVal stores the NOR of the other values
155  int outVal = 1;
156  // Shift all features down by one
157  for (int iF = 0; iF < origNf; iF++) {
158  int kextOrig = kIndex(
159  iX,
160  iY,
161  iF,
162  nx + haloOrig->lt + haloOrig->rt,
163  ny + haloOrig->dn + haloOrig->up,
164  origNf);
165  float origActivity = originalABatch[kextOrig];
166  // outVal is the final out value for the background
167  if (origActivity != 0) {
168  outVal = 0;
169  }
170  // Loop over replicated features
171  for (int repIdx = 0; repIdx < repFeatureNum; repIdx++) {
172  // Index iF one down, multiply by replicate feature number, add repIdx offset
173  int newFeatureIdx = ((iF + 1) * repFeatureNum) + repIdx;
174  assert(newFeatureIdx < thisNf);
175  int kext = kIndex(
176  iX,
177  iY,
178  newFeatureIdx,
179  nx + halo->lt + halo->rt,
180  ny + halo->dn + halo->up,
181  thisNf);
182  ABatch[kext] = origActivity;
183  }
184  }
185  // Set background indices to outVal
186  for (int repIdx = 0; repIdx < repFeatureNum; repIdx++) {
187  int kextBackground = kIndex(
188  iX, iY, repIdx, nx + halo->lt + halo->rt, ny + halo->dn + halo->up, thisNf);
189  ABatch[kextBackground] = outVal;
190  }
191  }
192  }
193  }
194  return status;
195 }
