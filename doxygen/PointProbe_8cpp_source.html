73 int PointProbe::initOutputStream(const char * filename) {

75  // Called by LayerProbe::initLayerProbe, which is called near the end of PointProbe::initPointProbe

99 int PointProbe::communicateInitInfo() {
100  int status = LayerProbe::communicateInitInfo();
101  assert(getTargetLayer());
102  const PVLayerLoc * loc = getTargetLayer()->getLayerLoc();
103  bool isRoot = getParent()->icCommunicator()->commRank()==0;
104  if( (xLoc < 0 || xLoc > loc->nxGlobal) && isRoot ) {
105  fprintf(stderr, "PointProbe on layer %s: xLoc coordinate %d is out of bounds (layer has %d neurons in the x-direction.\n", getTargetLayer()->getName(), xLoc, loc->nxGlobal);
106  status = PV_FAILURE;
107  }
108  if( (yLoc < 0 || yLoc > loc->nyGlobal) && isRoot ) {
109  fprintf(stderr, "PointProbe on layer %s: yLoc coordinate %d is out of bounds (layer has %d neurons in the y-direction.\n", getTargetLayer()->getName(), yLoc, loc->nyGlobal);
110  status = PV_FAILURE;
111  }
112  if( (fLoc < 0 || fLoc > loc->nf) && isRoot ) {
113  fprintf(stderr, "PointProbe on layer %s: fLoc coordinate %d is out of bounds (layer has %d features.\n", getTargetLayer()->getName(), fLoc, loc->nf);
114  status = PV_FAILURE;
115  }
116  if( (batchLoc < 0 || batchLoc > loc->nbatch) && isRoot ) {
117  fprintf(stderr, "PointProbe on layer %s: batchLoc coordinate %d is out of bounds (layer has %d batches.\n", getTargetLayer()->getName(), batchLoc, loc->nbatch);
118  status = PV_FAILURE;
119  }
120  if( status != PV_SUCCESS ) abort();
121  return status;
122 }

145 int PointProbe::calcValues(double timevalue) {
146  assert(this->getNumValues()==2);
147  double * valuesBuffer = this->getValuesBuffer();
148  //We need to calculate which mpi process contains the target point, and send that info to the root process
149  //Each process calculates local index
150  const PVLayerLoc * loc = getTargetLayer()->getLayerLoc();
151  //Calculate local cords from global
152  const int kx0 = loc->kx0;
153  const int ky0 = loc->ky0;
154  const int kb0 = loc->kb0;
155  const int nx = loc->nx;
156  const int ny = loc->ny;
157  const int nf = loc->nf;
158  const int nbatch = loc->nbatch;
159  const int xLocLocal = xLoc - kx0;
160  const int yLocLocal = yLoc - ky0;
161  const int nbatchLocal = batchLoc - kb0;
162 
163  //if in bounds
164  if( xLocLocal >= 0 && xLocLocal < nx &&
165  yLocLocal >= 0 && yLocLocal < ny &&
166  nbatchLocal >= 0 && nbatchLocal < nbatch){
167  const pvdata_t * V = getTargetLayer()->getV();
168  const pvdata_t * activity = getTargetLayer()->getLayerData();
169  //Send V and A to root
170  const int k = kIndex(xLocLocal, yLocLocal, fLoc, nx, ny, nf);
171  if(V){
172  valuesBuffer[0] = V[k + nbatchLocal*getTargetLayer()->getNumNeurons()];
173  }
174  else {
175  valuesBuffer[0] = 0.0;
176  }
177  if(activity){
178  const int kex = kIndexExtended(k, nx, ny, nf, loc->halo.lt, loc->halo.rt, loc->halo.dn, loc->halo.up);
179  valuesBuffer[1] = activity[kex + nbatchLocal * getTargetLayer()->getNumExtended()];
180  }
181  else {
182  valuesBuffer[1] = 0.0;
183  }
184  //If not in root process, send to root process
185  if(parent->columnId()!=0){
186  MPI_Send(&valuesBuffer, 2, MPI_DOUBLE, 0, 0, parent->icCommunicator()->communicator());
187  }
188  }
189 
190  //Root process
191  if(parent->columnId()==0){
192  //Calculate which rank target neuron is
193  //TODO we need to calculate rank from batch as well
194  int xRank = xLoc/nx;
195  int yRank = yLoc/ny;
196 
197  int srcRank = rankFromRowAndColumn(yRank, xRank, parent->icCommunicator()->numCommRows(), parent->icCommunicator()->numCommColumns());
198 
199  //If srcRank is not root process, MPI_Recv from that rank
200  if(srcRank != 0){
201  MPI_Recv(&valuesBuffer, 2, MPI_DOUBLE, srcRank, 0, parent->icCommunicator()->communicator(), MPI_STATUS_IGNORE);
202  }
203  }
204  return PV_SUCCESS;
205 }
