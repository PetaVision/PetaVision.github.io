67 int PointProbe::communicateInitInfo(CommunicateInitInfoMessage const *message) {
68  int status = LayerProbe::communicateInitInfo(message);
69  assert(getTargetLayer());
70  const PVLayerLoc *loc = getTargetLayer()->getLayerLoc();
71  bool isRoot = parent->getCommunicator()->commRank() == 0;
72  if ((xLoc < 0 || xLoc > loc->nxGlobal) && isRoot) {
73  ErrorLog().printf(
74  "PointProbe on layer %s: xLoc coordinate %d is out "
75  "of bounds (layer has %d neurons in "
76  "the x-direction.\n",
77  getTargetLayer()->getName(),
78  xLoc,
79  loc->nxGlobal);
80  status = PV_FAILURE;
81  }
82  if ((yLoc < 0 || yLoc > loc->nyGlobal) && isRoot) {
83  ErrorLog().printf(
84  "PointProbe on layer %s: yLoc coordinate %d is out "
85  "of bounds (layer has %d neurons in "
86  "the y-direction.\n",
87  getTargetLayer()->getName(),
88  yLoc,
89  loc->nyGlobal);
90  status = PV_FAILURE;
91  }
92  if ((fLoc < 0 || fLoc > loc->nf) && isRoot) {
93  ErrorLog().printf(
94  "PointProbe on layer %s: fLoc coordinate %d is out "
95  "of bounds (layer has %d features.\n",
96  getTargetLayer()->getName(),
97  fLoc,
98  loc->nf);
99  status = PV_FAILURE;
100  }
101  if ((batchLoc < 0 || batchLoc > loc->nbatch) && isRoot) {
102  ErrorLog().printf(
103  "PointProbe on layer %s: batchLoc coordinate %d is "
104  "out of bounds (layer has %d "
105  "batches.\n",
106  getTargetLayer()->getName(),
107  batchLoc,
108  loc->nbatch);
109  status = PV_FAILURE;
110  }
111  if (status != PV_SUCCESS)
112  abort();
113  return status;
114 }

116 void PointProbe::initOutputStreams(const char *filename, Checkpointer *checkpointer) {
117  PVLayerLoc const *loc = getTargetLayer()->getLayerLoc();
118  int xRank = xLoc / loc->nx; // integer division
119  int yRank = yLoc / loc->ny; // integer division
120  int batchRank = batchLoc / loc->nbatch; // integer division
121 
122  int blockColumnIndex = getMPIBlock()->getStartColumn() + getMPIBlock()->getColumnIndex();
123  int blockRowIndex = getMPIBlock()->getStartRow() + getMPIBlock()->getRowIndex();
124  int blockBatchIndex = getMPIBlock()->getStartBatch() + getMPIBlock()->getBatchIndex();
125 
126  if (xRank == blockColumnIndex and yRank == blockRowIndex and batchRank == blockBatchIndex) {
127  if (getProbeOutputFilename()) {
128  std::string path(getProbeOutputFilename());
129  std::ios_base::openmode mode = std::ios_base::out;
130  if (!checkpointer->getCheckpointReadDirectory().empty()) {
131  mode |= std::ios_base::app;
132  }
133  if (path[0] != '/') {
134  path = checkpointer->makeOutputPathFilename(path);
135  }
136  auto stream = new FileStream(path.c_str(), mode, checkpointer->doesVerifyWrites());
137  mOutputStreams.push_back(stream);
138  }
139  else {
140  auto stream = new PrintStream(PV::getOutputStream());
141  mOutputStreams.push_back(stream);
142  }
143  }
144  else {
145  mOutputStreams.clear();
146  }
147 }

164 int PointProbe::calcValues(double timevalue) {
165  assert(this->getNumValues() == 2);
166  double *valuesBuffer = this->getValuesBuffer();
167  // We need to calculate which mpi process contains the target point, and send
168  // that info to the root process Each process calculates local index
169  const PVLayerLoc *loc = getTargetLayer()->getLayerLoc();
170  // Calculate local coords from global
171  const int kx0 = loc->kx0;
172  const int ky0 = loc->ky0;
173  const int kb0 = loc->kb0;
174  const int nx = loc->nx;
175  const int ny = loc->ny;
176  const int nf = loc->nf;
177  const int nbatch = loc->nbatch;
178  const int xLocLocal = xLoc - kx0;
179  const int yLocLocal = yLoc - ky0;
180  const int nbatchLocal = batchLoc - kb0;
181 
182  // if in bounds
183  if (xLocLocal >= 0 && xLocLocal < nx && yLocLocal >= 0 && yLocLocal < ny && nbatchLocal >= 0
184  && nbatchLocal < nbatch) {
185  const float *V = getTargetLayer()->getV();
186  const float *activity = getTargetLayer()->getLayerData();
187  // Send V and A to root
188  const int k = kIndex(xLocLocal, yLocLocal, fLoc, nx, ny, nf);
189  if (V) {
190  valuesBuffer[0] = V[k + nbatchLocal * getTargetLayer()->getNumNeurons()];
191  }
192  else {
193  valuesBuffer[0] = 0.0;
194  }
195  if (activity) {
196  const int kex = kIndexExtended(
197  k, nx, ny, nf, loc->halo.lt, loc->halo.rt, loc->halo.dn, loc->halo.up);
198  valuesBuffer[1] = activity[kex + nbatchLocal * getTargetLayer()->getNumExtended()];
199  }
200  else {
201  valuesBuffer[1] = 0.0;
202  }
203  // If not in root process, send to root process
204  if (parent->columnId() != 0) {
205  MPI_Send(valuesBuffer, 2, MPI_DOUBLE, 0, 0, parent->getCommunicator()->communicator());
206  }
207  }
208 
209  // Root process
210  if (parent->columnId() == 0) {
211  // Calculate which rank target neuron is
212  // TODO we need to calculate rank from batch as well
213  int xRank = xLoc / nx;
214  int yRank = yLoc / ny;
215 
216  int srcRank = rankFromRowAndColumn(
217  yRank,
218  xRank,
219  parent->getCommunicator()->numCommRows(),
220  parent->getCommunicator()->numCommColumns());
221 
222  // If srcRank is not root process, MPI_Recv from that rank
223  if (srcRank != 0) {
224  MPI_Recv(
225  valuesBuffer,
226  2,
227  MPI_DOUBLE,
228  srcRank,
229  0,
230  parent->getCommunicator()->communicator(),
231  MPI_STATUS_IGNORE);
232  }
233  }
234  return PV_SUCCESS;
235 }
