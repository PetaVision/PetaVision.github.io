73 int PointProbe::initOutputStream(const char * filename) {
74  if(parent->columnId()==0){
75  // Called by LayerProbe::initLayerProbe, which is called near the end of PointProbe::initPointProbe
76  // So this->xLoc, yLoc, fLoc have been set.
77  if( filename != NULL ) {
78  char * outputdir = getParent()->getOutputPath();
79  char * path = (char *) malloc(strlen(outputdir)+1+strlen(filename)+1);
80  sprintf(path, "%s/%s", outputdir, filename);
81  outputstream = PV_fopen(path, "w", false/*verifyWrites*/);

83  fprintf(stderr, "LayerProbe error opening \"%s\" for writing: %s\n", path, strerror(errno));

87  }
88  else {
89  outputstream = PV_stdout();
90  }

95 int PointProbe::communicateInitInfo() {
96  int status = LayerProbe::communicateInitInfo();
97  assert(getTargetLayer());
98  const PVLayerLoc * loc = getTargetLayer()->getLayerLoc();
99  bool isRoot = getParent()->icCommunicator()->commRank()==0;
100  if( (xLoc < 0 || xLoc > loc->nxGlobal) && isRoot ) {
101  fprintf(stderr, "PointProbe on layer %s: xLoc coordinate %d is out of bounds (layer has %d neurons in the x-direction.\n", getTargetLayer()->getName(), xLoc, loc->nxGlobal);
102  status = PV_FAILURE;
103  }
104  if( (yLoc < 0 || yLoc > loc->nyGlobal) && isRoot ) {
105  fprintf(stderr, "PointProbe on layer %s: yLoc coordinate %d is out of bounds (layer has %d neurons in the y-direction.\n", getTargetLayer()->getName(), yLoc, loc->nyGlobal);
106  status = PV_FAILURE;
107  }
108  if( (fLoc < 0 || fLoc > loc->nf) && isRoot ) {
109  fprintf(stderr, "PointProbe on layer %s: fLoc coordinate %d is out of bounds (layer has %d features.\n", getTargetLayer()->getName(), fLoc, loc->nf);
110  status = PV_FAILURE;
111  }
112  if( (batchLoc < 0 || batchLoc > loc->nbatch) && isRoot ) {
113  fprintf(stderr, "PointProbe on layer %s: batchLoc coordinate %d is out of bounds (layer has %d batches.\n", getTargetLayer()->getName(), batchLoc, loc->nbatch);
114  status = PV_FAILURE;
115  }
116  if( status != PV_SUCCESS ) abort();
117  return status;
118 }

130 int PointProbe::outputState(double timef)
131 {
132  //We need to calculate which mpi process contains the target point, and send that info to the root process
133  //Each process calculates local index
134  const PVLayerLoc * loc = getTargetLayer()->getLayerLoc();
135  //Calculate local cords from global
136  const int kx0 = loc->kx0;
137  const int ky0 = loc->ky0;
138  const int kb0 = loc->kb0;
139  const int nx = loc->nx;
140  const int ny = loc->ny;
141  const int nf = loc->nf;
142  const int nbatch = loc->nbatch;
143  const int xLocLocal = xLoc - kx0;
144  const int yLocLocal = yLoc - ky0;
145  const int nbatchLocal = batchLoc - kb0;
146 
147  float vval = 0;
148  float aval = 0;
149  //if in bounds
150  if( xLocLocal >= 0 && xLocLocal < nx &&
151  yLocLocal >= 0 && yLocLocal < ny &&
152  nbatchLocal >= 0 && nbatchLocal < nbatch){
153  const pvdata_t * V = getTargetLayer()->getV();
154  const pvdata_t * activity = getTargetLayer()->getLayerData();
155  //Send V and A to root
156  const int k = kIndex(xLocLocal, yLocLocal, fLoc, nx, ny, nf);
157  const int kex = kIndexExtended(k, nx, ny, nf, loc->halo.lt, loc->halo.rt, loc->halo.dn, loc->halo.up);
158  if(V){
159  vval = V[k + nbatchLocal*getTargetLayer()->getNumNeurons()];
160  }
161  if(activity){
162  aval = activity[kex + nbatchLocal * getTargetLayer()->getNumExtended()];
163  }
164 #ifdef PV_USE_MPI
165  //If not in root process, send to root process
166  if(parent->columnId()!=0){
167  MPI_Send(&vval, 1, MPI_FLOAT, 0, 0, parent->icCommunicator()->communicator());
168  MPI_Send(&aval, 1, MPI_FLOAT, 0, 0, parent->icCommunicator()->communicator());
169  }
170 #endif
171  }
172 
173  //Root process
174  if(parent->columnId()==0){
175  //Calculate which rank target neuron is
176  //TODO we need to calculate rank from batch as well
177  int xRank = xLoc/nx;
178  int yRank = yLoc/ny;
179 
180  int srcRank = rankFromRowAndColumn(yRank, xRank, parent->icCommunicator()->numCommRows(), parent->icCommunicator()->numCommColumns());
181 
182 #ifdef PV_USE_MPI
183  //If srcRank is not root process, MPI_Recv from that rank
184  if(srcRank != 0){
185  MPI_Recv(&vval, 1, MPI_FLOAT, srcRank, 0, parent->icCommunicator()->communicator(), MPI_STATUS_IGNORE);
186  MPI_Recv(&aval, 1, MPI_FLOAT, srcRank, 0, parent->icCommunicator()->communicator(), MPI_STATUS_IGNORE);
187  }
188 #endif
189  return point_writeState(timef, vval, aval);
190  }
191  else{
192  return PV_SUCCESS;
193  }
194 }
