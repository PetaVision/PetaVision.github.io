10 inline static float bicubic(float x) {
11  float const absx = fabsf(x);
12  return absx < 1 ? 1 + absx * absx * (-2 + absx) : absx < 2 ? 4 + absx * (-8 + absx * (5 - absx))
13  : 0;
14 }

23  int bandStrideIn,

28  /* Interpolation using nearest neighbor interpolation */
29  int xinteger[widthOut];
30  float dx = (float)(widthIn - 1) / (float)(widthOut - 1);
31 
32  for (int kx = 0; kx < widthOut; kx++) {

43  }
44

58 void bicubicInterp(
59  float const *bufferIn,
60  int widthIn,
61  int heightIn,
62  int numBands,
63  int xStrideIn,
64  int yStrideIn,
65  int bandStrideIn,
66  float *bufferOut,
67  int widthOut,
68  int heightOut) {
69 
70  // Interpolation using bicubic convolution with a = -1
71  // (following Octave image toolbox's imremap function - change this?)
72  float xinterp[widthOut];
73  int xinteger[widthOut];
74  float xfrac[widthOut];
75  float dx = (float)(widthIn - 1) / (float)(widthOut - 1);
76 
77  for (int kx = 0; kx < widthOut; kx++) {
78  float x = dx * (float)kx;
79  xinterp[kx] = x;
80  float xfloor = floorf(x);
81  xinteger[kx] = (int)xfloor;
82  xfrac[kx] = x - xfloor;
83  }
84 
85  float yinterp[heightOut];
86  int yinteger[heightOut];
87  float yfrac[heightOut];
88  float dy = (float)(heightIn - 1) / (float)(heightOut - 1);
89 
90  for (int ky = 0; ky < heightOut; ky++) {
91  float y = dy * (float)ky;
92  yinterp[ky] = y;
93  float yfloor = floorf(y);
94  yinteger[ky] = (int)yfloor;
95  yfrac[ky] = y - yfloor;
96  }
97 
98  memset(bufferOut, 0, sizeof(*bufferOut) * size_t(widthOut * heightOut * numBands));
99 
100  for (int xOff = 2; xOff > -2; xOff--) {
101  for (int yOff = 2; yOff > -2; yOff--) {
102  for (int ky = 0; ky < heightOut; ky++) {
103  float ycoeff = bicubic(yfrac[ky] - (float)yOff);
104  int yfetch = yinteger[ky] + yOff;
105 
106  if (yfetch < 0)
107  yfetch = -yfetch;
108  if (yfetch >= heightIn)
109  yfetch = heightIn - (yfetch - heightIn) - 1;
110 
111  for (int kx = 0; kx < widthOut; kx++) {
112  float xcoeff = bicubic(xfrac[kx] - (float)xOff);
113  int xfetch = xinteger[kx] + xOff;
114 
115  if (xfetch < 0)
116  xfetch = -xfetch;
117  if (xfetch >= widthIn)
118  xfetch = widthIn - (xfetch - widthIn) - 1;
119 
120  assert(xfetch >= 0 && xfetch < widthIn && yfetch >= 0 && yfetch < heightIn);
121 
122  for (int f = 0; f < numBands; f++) {
123  int fetchIdx = yfetch * yStrideIn + xfetch * xStrideIn + f * bandStrideIn;
124  float p = bufferIn[fetchIdx];
125  int outputIdx = kIndex(kx, ky, f, widthOut, heightOut, numBands);
126  bufferOut[outputIdx] += xcoeff * ycoeff * p;
127  }
128  }
129  }
130  }
131  }

135 // Rescale a buffer, preserving aspect ratio
136 void rescale(
137  Buffer<float> &buffer,
138  int newWidth,
139  int newHeight,
140  enum RescaleMethod rescaleMethod,
141  enum InterpolationMethod interpMethod,
142  enum Buffer<float>::Anchor anchor) {
143  float xRatio = (float)newWidth / buffer.getWidth();
144  float yRatio = (float)newHeight / buffer.getHeight();
145  int resizedWidth = newWidth;
146  int resizedHeight = newHeight;
147  float resizeFactor = 1.0f;
148 
149  switch (rescaleMethod) {
150  case CROP: resizeFactor = xRatio < yRatio ? yRatio : xRatio; break;
151  case PAD: resizeFactor = xRatio < yRatio ? xRatio : yRatio; break;
152  }
153 
154  resizedWidth = (int)nearbyintf(resizeFactor * buffer.getWidth());
155  resizedHeight = (int)nearbyintf(resizeFactor * buffer.getHeight());
156 
157  std::vector<float> rawInput = buffer.asVector();
158  std::vector<float> scaledInput(resizedWidth * resizedHeight * buffer.getFeatures());
159  switch (interpMethod) {
160  case BICUBIC:
161  bicubicInterp(
162  rawInput.data(),
163  buffer.getWidth(),
164  buffer.getHeight(),
165  buffer.getFeatures(),
166  buffer.getFeatures(),
167  buffer.getFeatures() * buffer.getWidth(),
168  1,
169  scaledInput.data(),
170  resizedWidth,
171  resizedHeight);
172  break;
173  case NEAREST:
174  nearestNeighborInterp(
175  rawInput.data(),
176  buffer.getWidth(),
177  buffer.getHeight(),
178  buffer.getFeatures(),
179  buffer.getFeatures(),
180  buffer.getFeatures() * buffer.getWidth(),
181  1,
182  scaledInput.data(),
183  resizedWidth,
184  resizedHeight);
185  break;
186  }
187  buffer.set(scaledInput, resizedWidth, resizedHeight, buffer.getFeatures());
188 
189  // This final call resizes the buffer to our specified
190  // newWidth and newHeight. If our rescaleMethod was PAD,
191  // this actually grows the buffer to include the padded region.
192  switch (rescaleMethod) {
193  case CROP: buffer.crop(newWidth, newHeight, anchor); break;
194  case PAD: buffer.grow(newWidth, newHeight, anchor); break;
195  }
196 }
