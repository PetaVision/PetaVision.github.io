113 int Image::readImageFileGDAL(char const * filename, PVLayerLoc const * loc) {
114  assert(parent->columnId()==0);
115 
116  GDALAllRegister();
117 
118  GDALDataset * dataset = PV_GDALOpen(filename);
119  if (dataset==NULL) { return PV_FAILURE; } // PV_GDALOpen prints an error message.
120  GDALRasterBand * poBand = dataset->GetRasterBand(1);
121  GDALDataType dataType = poBand->GetRasterDataType();
122 
123  // GDAL defines whether a band is binary, not whether the image as a whole is binary.
124  // Set isBinary to false if any band is not binary (metadata doesn't have NBITS=1)
125  bool isBinary = true;
126  for(int iBand = 0; iBand < GDALGetRasterCount(dataset); iBand++){
127  GDALRasterBandH hBand = GDALGetRasterBand(dataset, iBand+1);
128  char ** metadata = GDALGetMetadata(hBand, "Image_Structure");
129  if(CSLCount(metadata) > 0){
130  bool found = false;
131  for(int i = 0; metadata[i] != NULL; i++){
132  if(strcmp(metadata[i], "NBITS=1") == 0){
133  found = true;
134  break;
135  }
136  }
137  if(!found){
138  isBinary = false;
139  }
140  }
141  else{
142  isBinary = false;
143  }
144  }
145 
146  int const xImageSize = imageLoc.nxGlobal;
147  int const yImageSize = imageLoc.nyGlobal;
148  int const bandsInFile = imageLoc.nf;
149 
150  int numTotal = xImageSize * yImageSize * bandsInFile;
151 
152  dataset->RasterIO(GF_Read, 0, 0, xImageSize, yImageSize, imageData,
153  xImageSize, yImageSize, GDT_Float32, bandsInFile, NULL,
154  bandsInFile*sizeof(float), bandsInFile*xImageSize*sizeof(float), sizeof(float));
155 
156 
157  GDALClose(dataset);
158  if(!isBinary){
159  pvadata_t fac;
160  if(dataType == GDT_Byte){
161  fac = 1.0f / 255.0f; // normalize to 1.0
162  }
163  else if(dataType == GDT_UInt16){
164  fac = 1.0f / 65535.0f; // normalize to 1.0
165  }
166  else{
167  pvError().printf("Image data type %s in file \"%s\" is not implemented.\n", GDALGetDataTypeName(dataType), filename);
168  }
169  for( int n=0; n<numTotal; n++ ) {
170  imageData[n] *= fac;
171  }
172  }
173  return EXIT_SUCCESS;
174 }

231  for(int iBand = 0; iBand < GDALGetRasterCount(dataset); iBand++){

256  int yImageSize = dataset->GetRasterYSize();

258

295 #endif

308  //? For the auto resize flag, PV checks which side (x or y) is the shortest, relative to the
309  //? hypercolumn size specified. Then it determines the largest chunk it can possibly take
310  //? from the image with the correct aspect ratio determined by hypercolumn. It then
311  //? determines the offset needed in the long dimension to center the cropped image,
312  //? and reads in that portion of the image. The offset can optionally be translated by
313  //? offset{X,Y} specified in the params file (values can be positive or negative).
314  //? If the specified offset takes the cropped image outside the image file, it uses the
315  //? largest-magnitude offset that stays within the image file's borders.
316

318

320  int new_y = int(round(ny*xImageSize/(double)xTotalSize));
321  int y_off = int(round((yImageSize - new_y*nyProcs)/2.0));
322 
323  int jitter_y = max(min(y_off,yOffset),-y_off);
324

329  GDT_Float32, bandsInFile, NULL, bandsInFile*sizeof(float),
330  bandsInFile*nx*sizeof(float), sizeof(float));
331  }
332  else{
333  int new_x = int(round(nx*yImageSize/(double)yTotalSize));
334  int x_off = int(round((xImageSize - new_x*nxProcs)/2.0));
335 
336  int jitter_x = max(min(x_off,xOffset),-x_off);
337 
338  kx = new_x * col;
339  ky = yImageSize/nyProcs * row;

342  GDT_Float32, bandsInFile, NULL, bandsInFile*sizeof(float),

344  }

348  //offset is in Image space

352  int img_bot = img_top + yImageSize;
353 
354  //Check if in bounds
355  if(img_left >= kx+nx || img_right <= kx || img_top >= ky+ny || img_bot <= ky){

357  if(dest > 0){

359  }

361  }
362 
363 
364  //start of the buffer on the left/top side
365  int buf_left = img_left > kx ? img_left-kx : 0;
366  int buf_top = img_top > ky ? img_top-ky : 0;
367  int buf_right = img_right < kx+nx ? img_right-kx : nx;
368  int buf_bot = img_bot < ky+ny ? img_bot-ky : ny;
369

371  int buf_ySize = buf_bot - buf_top;

459 int Image::readImage(const char * filename)
460 {
461  assert(parent->columnId()==0); // readImage is called by retrieveData, which only the root process calls. BaseInput::scatterInput does the scattering.
462  int status = 0;
463  PVLayerLoc * loc = & clayer->loc;
464 
465  char * path = NULL;
466  bool usingTempFile = false;
467 
468  if (strstr(filename, "://") != NULL) {
469  usingTempFile = true;
470  std::string pathstring = parent->getOutputPath();
471  pathstring += "/temp.XXXXXX";
472  const char * ext = strrchr(filename, '.');
473  if (ext) { pathstring += ext; }
474  path = strdup(pathstring.c_str());
475  int fid;
476  fid=mkstemps(path, strlen(ext));
477  if (fid<0) {
478  pvError().printf("Cannot create temp image file.\n");
479  }
480  close(fid);
481  std::string systemstring;
482  if (strstr(filename, "s3://") != NULL) {
483  systemstring = "aws s3 cp \'";
484  systemstring += filename;
485  systemstring += "\' ";
486  systemstring += path;
487  }
488  else { // URLs other than s3://
489  systemstring = "wget -O ";
490  systemstring += path;
491  systemstring += " \'";
492  systemstring += filename;
493  systemstring += "\'";
494  }
495 
496  int numAttempts = 5;
497  for(int attemptNum = 0; attemptNum < numAttempts; attemptNum++){
498  int status = system(systemstring.c_str());
499  if(status != 0){
500  if(attemptNum == numAttempts - 1){
501  pvError().printf("download command \"%s\" failed: %s. Exiting\n", systemstring.c_str(), strerror(errno));
502  }
503  else{
504  pvWarn().printf("download command \"%s\" failed: %s. Retrying %d out of %d.\n", systemstring.c_str(), strerror(errno), attemptNum+1, numAttempts);
505  sleep(1);
506  }
507  }
508  else{
509  break;
510  }
511  }
512  }
513  else {
514  path = strdup(filename);
515  }
516  GDALColorInterp * colorbandtypes = NULL;
517  status = getImageInfoGDAL(path, &imageLoc, &colorbandtypes);
518  calcColorType(imageLoc.nf, colorbandtypes);
519  free(colorbandtypes); colorbandtypes = NULL;
520 
521  if(status != 0) {
522  pvError().printf("Movie: Unable to get image info for \"%s\"\n", filename);
523  }
524 
525  delete[] imageData;
526  imageData = new pvadata_t[imageLoc.nxGlobal*imageLoc.nyGlobal*imageLoc.nf];
527 
528  status = readImageFileGDAL(path, &imageLoc);
529  if (status != PV_SUCCESS) {
530  pvError().printf("Image::readImage failed for layer \"%s\"\n", getName());
531  }
532 
533  if (usingTempFile) {
534  int rmstatus = remove(path);
535  if (rmstatus) {
536  pvError().printf("remove(\"%s\") failed. Exiting.\n", path);
537  }
538  }
539  free(path);
540 
541  return status;
542 }

544 int Image::calcColorType(int numBands, GDALColorInterp * colorbandtypes) {
545  int colortype = 0; // 1=grayscale(with or without alpha), return value 2=RGB(with or without alpha), 0=unrecognized
546  const GDALColorInterp grayalpha[2] = {GCI_GrayIndex, GCI_AlphaBand};
547  const GDALColorInterp rgba[4] = {GCI_RedBand, GCI_GreenBand, GCI_BlueBand, GCI_AlphaBand};
548  const float grayalphaweights[2] = {1.0, 0.0};
549  const float rgbaweights[4] = {0.30, 0.59, 0.11, 0.0}; // RGB weights from <https://en.wikipedia.org/wiki/Grayscale>, citing Pratt, Digital Image Processing
550  switch (numBands) {
551  case 1:
552  case 2:
553  imageColorType = memcmp(colorbandtypes, grayalpha, numBands*sizeof(GDALColorInterp)) ? COLORTYPE_UNRECOGNIZED : COLORTYPE_GRAYSCALE;
554  break;
555  case 3:
556  case 4:
557  imageColorType = memcmp(colorbandtypes, rgba, numBands*sizeof(GDALColorInterp)) ? COLORTYPE_UNRECOGNIZED : COLORTYPE_RGB;
558  break;
559  default:
560  imageColorType = COLORTYPE_UNRECOGNIZED;
561  break;
562  }
563  return PV_SUCCESS;
564 }
