98  // If WarmStart is set and CheckpointWrite is false, CheckpointReadDirectory
99  // is LastCheckpointDir. If WarmStart is set and CheckpointWrite is true, we
100  // CheckpointReadDirectory is the last checkpoint in CheckpointWriteDir.
101  // Hence we cannot set CheckpointReadDirectory until we've read the params.
102  if (mWarmStart and ioFlag == PARAMS_IO_READ) {
103  // Arguments class should prevent -r and -c from both being set.
104  pvAssert(mCheckpointReadDirectory.empty());
105  if (mCheckpointWriteFlag) {
106  // Set mCheckpointReadDirectory to the last checkpoint in the CheckpointWrite directory.
107  findWarmStartDirectory();
108  }
109  else {
110  mCheckpointReadDirectory = mLastCheckpointDir;
111  }
112  }

277 void Checkpointer::ioParam_checkpointWriteClockUnit(enum ParamsIOFlag ioFlag, PVParams *params) {
278  pvAssert(!params->presentAndNotBeenRead(mName.c_str(), "checkpointWrite"));
279  if (mCheckpointWriteFlag) {
280  pvAssert(!params->presentAndNotBeenRead(mName.c_str(), "checkpointWriteTriggerMode"));
281  if (mCheckpointWriteTriggerMode == WALLCLOCK) {
282  assert(
283  !params->presentAndNotBeenRead(
284  mName.c_str(), "checkpointWriteTriggerClockInterval"));
285  params->ioParamString(
286  ioFlag,
287  mName.c_str(),
288  "checkpointWriteClockUnit",
289  &mCheckpointWriteWallclockUnit,
290  "seconds");
291  if (ioFlag == PARAMS_IO_READ) {
292  pvAssert(mCheckpointWriteWallclockUnit);
293  for (size_t n = 0; n < strlen(mCheckpointWriteWallclockUnit); n++) {
294  mCheckpointWriteWallclockUnit[n] = tolower(mCheckpointWriteWallclockUnit[n]);
295  }
296  if (!strcmp(mCheckpointWriteWallclockUnit, "second")
297  || !strcmp(mCheckpointWriteWallclockUnit, "seconds")
298  || !strcmp(mCheckpointWriteWallclockUnit, "sec")
299  || !strcmp(mCheckpointWriteWallclockUnit, "s")) {
300  free(mCheckpointWriteWallclockUnit);
301  mCheckpointWriteWallclockUnit = strdup("seconds");
302  mCheckpointWriteWallclockIntervalSeconds = mCheckpointWriteWallclockInterval;
303  }
304  else if (
305  !strcmp(mCheckpointWriteWallclockUnit, "minute")
306  || !strcmp(mCheckpointWriteWallclockUnit, "minutes")
307  || !strcmp(mCheckpointWriteWallclockUnit, "min")
308  || !strcmp(mCheckpointWriteWallclockUnit, "m")) {
309  free(mCheckpointWriteWallclockUnit);
310  mCheckpointWriteWallclockUnit = strdup("minutes");
311  mCheckpointWriteWallclockIntervalSeconds =
312  mCheckpointWriteWallclockInterval * (time_t)60;
313  }
314  else if (
315  !strcmp(mCheckpointWriteWallclockUnit, "hour")
316  || !strcmp(mCheckpointWriteWallclockUnit, "hours")
317  || !strcmp(mCheckpointWriteWallclockUnit, "hr")
318  || !strcmp(mCheckpointWriteWallclockUnit, "h")) {
319  free(mCheckpointWriteWallclockUnit);
320  mCheckpointWriteWallclockUnit = strdup("hours");
321  mCheckpointWriteWallclockIntervalSeconds =
322  mCheckpointWriteWallclockInterval * (time_t)3600;
323  }
324  else if (
325  !strcmp(mCheckpointWriteWallclockUnit, "day")
326  || !strcmp(mCheckpointWriteWallclockUnit, "days")) {
327  free(mCheckpointWriteWallclockUnit);
328  mCheckpointWriteWallclockUnit = strdup("days");
329  mCheckpointWriteWallclockIntervalSeconds =
330  mCheckpointWriteWallclockInterval * (time_t)86400;
331  }
332  else {
333  if (mMPIBlock->getRank() == 0) {
334  ErrorLog().printf(
335  "checkpointWriteClockUnit \"%s\" is unrecognized. Use \"seconds\", "
336  "\"minutes\", \"hours\", or \"days\".\n",
337  mCheckpointWriteWallclockUnit);
338  }
339  MPI_Barrier(mMPIBlock->getComm());
340  exit(EXIT_FAILURE);
341  }
342  FatalIf(
343  mCheckpointWriteWallclockUnit == nullptr,
344  "Error in global rank %d process converting checkpointWriteClockUnit: %s\n",
345  mMPIBlock->getRank(),
346  strerror(errno));
347  }
348  }
349  }

525 void Checkpointer::findWarmStartDirectory() {
526  char warmStartDirectoryBuffer[PV_PATH_MAX];
527  if (mMPIBlock->getRank() == 0) {
528  if (mCheckpointWriteFlag) {
529  // Look for largest indexed Checkpointnnnnnn directory in checkpointWriteDir
530  pvAssert(mCheckpointWriteDir);
531  std::string cpDirString = mCheckpointWriteDir;
532  if (cpDirString.c_str()[cpDirString.length() - 1] != '/') {
533  cpDirString += "/";
534  }
535  struct stat statbuf;
536  int statstatus = PV_stat(cpDirString.c_str(), &statbuf);
537  if (statstatus == 0) {
538  if (statbuf.st_mode & S_IFDIR) {
539  char *dirs[] = {mCheckpointWriteDir, nullptr};
540  FTS *fts = fts_open(dirs, FTS_LOGICAL, nullptr);
541  FTSENT *ftsent = fts_read(fts);
542  bool found = false;
543  long int cpIndex = LONG_MIN;
544  std::string indexedDir;
545  for (ftsent = fts_children(fts, 0); ftsent != nullptr; ftsent = ftsent->fts_link) {
546  if (ftsent->fts_statp->st_mode & S_IFDIR) {
547  long int x;
548  int k = sscanf(ftsent->fts_name, "Checkpoint%ld", &x);
549  if (x > cpIndex) {
550  cpIndex = x;
551  indexedDir = ftsent->fts_name;
552  found = true;
553  }
554  }
555  }
556  FatalIf(
557  !found,
558  "restarting but checkpointWriteFlag is set and "
559  "checkpointWriteDir directory \"%s\" does not have any "
560  "checkpoints\n",
561  mCheckpointWriteDir);
562  mCheckpointReadDirectory = cpDirString;
563  mCheckpointReadDirectory.append(indexedDir);
564  }
565  else {
566  Fatal().printf(
567  "checkpoint read directory \"%s\" is "
568  "not a directory.\n",
569  mCheckpointWriteDir);
570  }
571  }
572  else if (errno == ENOENT) {
573  Fatal().printf(
574  "restarting but neither Last nor checkpointWriteDir "
575  "directory \"%s\" exists.\n",
576  mCheckpointWriteDir);
577  }
578  }
579  else {
580  pvAssert(mLastCheckpointDir);
581  FatalIf(
582  mLastCheckpointDir[0] == '\0',
583  "Restart flag set, but unable to determine restart directory.\n");
584  mCheckpointReadDirectory = strdup(mLastCheckpointDir);
585  }
586  FatalIf(
587  mCheckpointReadDirectory.size() >= PV_PATH_MAX,
588  "Restart flag set, but inferred checkpoint read directory is too long (%zu "
589  "bytes).\n",
590  mCheckpointReadDirectory.size());
591  memcpy(
592  warmStartDirectoryBuffer,
593  mCheckpointReadDirectory.c_str(),
594  mCheckpointReadDirectory.size());
595  warmStartDirectoryBuffer[mCheckpointReadDirectory.size()] = '\0';
596  }
597  MPI_Bcast(warmStartDirectoryBuffer, PV_PATH_MAX, MPI_CHAR, 0, mMPIBlock->getComm());
598  if (mMPIBlock->getRank() != 0) {
599  mCheckpointReadDirectory = warmStartDirectoryBuffer;
600  }
601 }

612 void Checkpointer::extractCheckpointReadDirectory() {
613  std::vector<std::string> checkpointReadDirs;
614  checkpointReadDirs.reserve(mMPIBlock->getBatchDimension());
615  std::size_t dirStart = (std::size_t)0;
616  while (dirStart < mCheckpointReadDirectory.size()) {
617  std::size_t dirStop = mCheckpointReadDirectory.find(':', dirStart);
618  if (dirStop == std::string::npos) {
619  dirStop = mCheckpointReadDirectory.size();
620  }
621  checkpointReadDirs.push_back(mCheckpointReadDirectory.substr(dirStart, dirStop - dirStart));
622  FatalIf(
623  checkpointReadDirs.size() > (std::size_t)mMPIBlock->getBatchDimension(),
624  "Checkpoint read parsing error: Too many colon separated "
625  "checkpoint read directories. "
626  "Only specify %d checkpoint directories.\n",
627  mMPIBlock->getBatchDimension());
628  dirStart = dirStop + 1;
629  }
630  // Make sure number matches up
631  int const count = (int)checkpointReadDirs.size();
632  FatalIf(
633  count != mMPIBlock->getBatchDimension() && count != 1,
634  "Checkpoint read parsing error: Not enough colon separated "
635  "checkpoint read directories. "
636  "Running with %d batch MPIs but only %d colon separated checkpoint "
637  "directories.\n",
638  mMPIBlock->getBatchDimension(),
639  count);
640  // Grab the directory for this rank and use as mCheckpointReadDir
641  int const checkpointIndex = count == 1 ? 0 : mMPIBlock->getBatchIndex();
642  std::string dirString = expandLeadingTilde(checkpointReadDirs[checkpointIndex].c_str());
643  mCheckpointReadDirectory = dirString.c_str();
644  pvAssert(!mCheckpointReadDirectory.empty());
645 
646  InfoLog().printf(
647  "Global Rank %d process setting CheckpointReadDirectory to %s.\n",
648  mMPIBlock->getGlobalRank(),
649  mCheckpointReadDirectory.c_str());

672 void Checkpointer::checkpointWrite(double simTime) {
673  mTimeInfo.mSimTime = simTime;
674  // set mSimTime here so that it is available in routines called by checkpointWrite.
675  if (!mCheckpointWriteFlag) {
676  return;
677  }
678  if (checkpointWriteSignal()) {
679  InfoLog().printf(
680  "Global rank %d: checkpointing in response to SIGUSR1 at time %f.\n",
681  mMPIBlock->getGlobalRank(),
682  simTime);
683  mCheckpointSignal = 0;
684  checkpointNow();
685  }
686  else {
687  switch (mCheckpointWriteTriggerMode) {
688  case NONE:
689  pvAssert(0);
690  break; // Only NONE if checkpointWrite is off, in which case this method should have
691  // returned above.
692  case STEP: checkpointWriteStep(); break;
693  case SIMTIME: checkpointWriteSimtime(); break;
694  case WALLCLOCK: checkpointWriteWallclock(); break;
695  default: pvAssert(0); break;
696  }
697  }
698  mTimeInfo.mCurrentCheckpointStep++;
699 }

701 bool Checkpointer::checkpointWriteSignal() {
702  if (mMPIBlock->getGlobalRank() == 0) {
703  int sigstatus = PV_SUCCESS;
704  sigset_t pollusr1;
705 
706  sigstatus = sigpending(&pollusr1);
707  assert(sigstatus == 0);
708  mCheckpointSignal = sigismember(&pollusr1, SIGUSR1);
709  assert(mCheckpointSignal == 0 || mCheckpointSignal == 1);
710  if (mCheckpointSignal) {
711  sigstatus = sigemptyset(&pollusr1);
712  assert(sigstatus == 0);
713  sigstatus = sigaddset(&pollusr1, SIGUSR1);
714  assert(sigstatus == 0);
715  int result = 0;
716  sigwait(&pollusr1, &result);
717  assert(result == SIGUSR1);
718  }
719  }
720  MPI_Bcast(&mCheckpointSignal, 1 /*count*/, MPI_INT, 0, mMPIBlock->getGlobalComm());
721  bool signaled = (mCheckpointSignal != 0);
722  if (signaled) {
723  InfoLog().printf(
724  "Global rank %d: checkpointing in response to SIGUSR1 at time %f.\n",
725  mMPIBlock->getGlobalRank(),
726  mTimeInfo.mSimTime);
727  mCheckpointSignal = 0;
728  checkpointNow();
729  }
730  return signaled;
731 }

762 void Checkpointer::checkpointNow() {
763  std::stringstream checkpointDirStream;
764  checkpointDirStream << mCheckpointWriteDir << "/Checkpoint";
765  int fieldWidth = mCheckpointIndexWidth < 0 ? mWidthOfFinalStepNumber : mCheckpointIndexWidth;
766  checkpointDirStream.fill('0');
767  checkpointDirStream.width(fieldWidth);
768  checkpointDirStream << mTimeInfo.mCurrentCheckpointStep;
769  std::string checkpointDirectory = checkpointDirStream.str();
770  if (checkpointDirectory != mCheckpointReadDirectory) {
771  /* Note: the strcmp isn't perfect, since there are multiple ways to specify a path that
772  * points to the same directory. Should use realpath, but that breaks under OS X. */
773  if (mMPIBlock->getGlobalRank() == 0) {
774  InfoLog() << "Checkpointing to \"" << checkpointDirectory
775  << "\", simTime = " << mTimeInfo.mSimTime << "\n";
776  }
777  }
778  else {
779  if (mMPIBlock->getGlobalRank() == 0) {
780  InfoLog().printf(
781  "Skipping checkpoint to \"%s\","
782  " which would clobber the checkpointRead checkpoint.\n",
783  checkpointDirectory.c_str());
784  }
785  return;
786  }
787  checkpointToDirectory(checkpointDirectory);
788  if (mMPIBlock->getRank() == 0) {
789  InfoLog().printf("checkpointWrite complete. simTime = %f\n", mTimeInfo.mSimTime);
790  }
791 
792  if (mDeleteOlderCheckpoints) {
793  rotateOldCheckpoints(checkpointDirectory);
794  }
795 }

797 void Checkpointer::checkpointToDirectory(std::string const &directory) {
798  std::string checkpointDirectory = generateBlockPath(directory);
799  mCheckpointTimer->start();
800  if (mMPIBlock->getRank() == 0) {
801  InfoLog() << "Checkpointing to directory \"" << checkpointDirectory
802  << "\" at simTime = " << mTimeInfo.mSimTime << "\n";
803  struct stat timeinfostat;
804  std::string timeinfoFilename(checkpointDirectory);
805  timeinfoFilename.append("/timeinfo.bin");
806  int statstatus = stat(timeinfoFilename.c_str(), &timeinfostat);
807  if (statstatus == 0) {
808  WarnLog() << "Checkpoint directory \"" << checkpointDirectory
809  << "\" has existing timeinfo.bin, which is now being deleted.\n";
810  mTimeInfoCheckpointEntry->remove(checkpointDirectory);
811  }
812  }
813  notify(
814  mObserverTable,
815  std::make_shared<PrepareCheckpointWriteMessage const>(checkpointDirectory),
816  mMPIBlock->getRank() == 0 /*printFlag*/);

838 void Checkpointer::rotateOldCheckpoints(std::string const &newCheckpointDirectory) {
839  std::string &oldestCheckpointDir = mOldCheckpointDirectories[mOldCheckpointDirectoriesIndex];
840  if (!oldestCheckpointDir.empty()) {
841  if (mMPIBlock->getRank() == 0) {
842  std::string targetDirectory = generateBlockPath(oldestCheckpointDir);
843  struct stat lcp_stat;
844  int statstatus = stat(targetDirectory.c_str(), &lcp_stat);
845  if (statstatus != 0 || !(lcp_stat.st_mode & S_IFDIR)) {
846  if (statstatus == 0) {
847  ErrorLog().printf(
848  "Failed to delete older checkpoint: failed to stat \"%s\": %s.\n",
849  targetDirectory.c_str(),
850  strerror(errno));
851  }
852  else {
853  ErrorLog().printf(
854  "Deleting older checkpoint: \"%s\" exists but is not a directory.\n",
855  targetDirectory.c_str());
856  }
857  }
858  sync();
859  std::string rmrf_string("");
860  rmrf_string = rmrf_string + "rm -r '" + targetDirectory + "'";
861  int rmrf_result = system(rmrf_string.c_str());
862  if (rmrf_result != 0) {
863  WarnLog().printf(
864  "unable to delete older checkpoint \"%s\": rm command returned %d\n",
865  targetDirectory.c_str(),
866  WEXITSTATUS(rmrf_result));
867  }
868  }
869  MPI_Barrier(mMPIBlock->getGlobalComm());
870  if (mMPIBlock->getGlobalRank() == 0) {
871  sync();
872  struct stat oldcp_stat;
873  int statstatus = stat(oldestCheckpointDir.c_str(), &oldcp_stat);
874  if (statstatus == 0 && (oldcp_stat.st_mode & S_IFDIR)) {
875  int rmdirstatus = rmdir(oldestCheckpointDir.c_str());
876  if (rmdirstatus) {
877  ErrorLog().printf(
878  "Unable to delete older checkpoint \"%s\": rmdir command returned %d "
879  "(%s)\n",
880  oldestCheckpointDir.c_str(),
881  errno,
882  std::strerror(errno));
883  }
884  }
885  }
886  }
887  mOldCheckpointDirectories[mOldCheckpointDirectoriesIndex] = newCheckpointDirectory;
888  mOldCheckpointDirectoriesIndex++;
889  if (mOldCheckpointDirectoriesIndex == mNumCheckpointsKept) {
890  mOldCheckpointDirectoriesIndex = 0;
891  }
892 }
