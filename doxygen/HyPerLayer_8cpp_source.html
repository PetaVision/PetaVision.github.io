299 
300  PVLayerLoc * loc = &clayer->loc;
301  setLayerLoc(loc, nxScale, nyScale, numFeatures, parent->getNBatch());

303

305 
306  clayer->numNeurons = loc->nx * loc->ny * loc->nf;
307  clayer->numExtended = clayer->numNeurons; // initially, margin is zero; it will be updated as needed during the communicateInitInfo stage.
308  clayer->numNeuronsAllBatches = nBatch * loc->nx * loc->ny * loc->nf;
309  clayer->numExtendedAllBatches = clayer->numNeuronsAllBatches;
310 
311  double xScaled = -log2( (double) nxScale);
312  double yScaled = -log2( (double) nyScale);
313 
314  int xScale = (int) nearbyint(xScaled);
315  int yScale = (int) nearbyint(yScaled);
316 
317  clayer->xScale = xScale;
318  clayer->yScale = yScale;
319 
320  // Other fields of clayer will be set in allocateClayerBuffers, or during updateState
321  return status;

546 int HyPerLayer::setLayerLoc(PVLayerLoc * layerLoc, float nxScale, float nyScale, int nf, int numBatches)
547 {
548  int status = PV_SUCCESS;
549 
550  InterColComm * icComm = parent->icCommunicator();
551 
552  float nxglobalfloat = nxScale * parent->getNxGlobal();
553  layerLoc->nxGlobal = (int) nearbyintf(nxglobalfloat);
554  if (fabs(nxglobalfloat-layerLoc->nxGlobal)>0.0001) {
555  if (parent->columnId()==0) {
556  pvErrorNoExit(errorMessage);
557  errorMessage.printf("nxScale of layer \"%s\" is incompatible with size of column.\n", getName());
558  errorMessage.printf("Column nx %d multiplied by nxScale %f must be an integer.\n", parent->getNxGlobal(), nxScale);
559  }
560  status = PV_FAILURE;
561  }
562 
563  float nyglobalfloat = nyScale * parent->getNyGlobal();
564  layerLoc->nyGlobal = (int) nearbyintf(nyglobalfloat);
565  if (fabs(nyglobalfloat-layerLoc->nyGlobal)>0.0001) {
566  if (parent->columnId()==0) {
567  pvErrorNoExit(errorMessage);
568  errorMessage.printf("nyScale of layer \"%s\" is incompatible with size of column.\n", getName());
569  errorMessage.printf("Column ny %d multiplied by nyScale %f must be an integer.\n", parent->getNyGlobal(), nyScale);
570  }
571  status = PV_FAILURE;
572  }
573 
574  // partition input space based on the number of processor
575  // columns and rows
576  //
577 
578  if (layerLoc->nxGlobal % icComm->numCommColumns() != 0) {
579  if (parent->columnId()==0) {
580  pvErrorNoExit(errorMessage);
581  errorMessage.printf("Size of HyPerLayer \"%s\" is not compatible with the mpi configuration.\n", name);
582  errorMessage.printf("The layer has %d pixels horizontally, and there are %d mpi processes in a row, but %d does not divide %d.\n",
583  layerLoc->nxGlobal, icComm->numCommColumns(), icComm->numCommColumns(), layerLoc->nxGlobal);
584  }
585  status = PV_FAILURE;
586  }
587  if (layerLoc->nyGlobal % icComm->numCommRows() != 0) {
588  if (parent->columnId()==0) {
589  pvErrorNoExit(errorMessage);
590  errorMessage.printf("Size of HyPerLayer \"%s\" is not compatible with the mpi configuration.\n", name);
591  errorMessage.printf("The layer has %d pixels vertically, and there are %d mpi processes in a column, but %d does not divide %d.\n",
592  layerLoc->nyGlobal, icComm->numCommRows(), icComm->numCommRows(), layerLoc->nyGlobal);
593  }
594  status = PV_FAILURE;
595  }
596  MPI_Barrier(icComm->communicator()); // If there is an error, make sure that MPI doesn't kill the run before process 0 reports the error.
597  if (status != PV_SUCCESS) {
598  if (parent->columnId()==0) {
599  pvErrorNoExit().printf("setLayerLoc failed for %s \"%s\".\n", getKeyword(), getName());
600  }
601  exit(EXIT_FAILURE);
602  }
603  layerLoc->nx = layerLoc->nxGlobal / icComm->numCommColumns();
604  layerLoc->ny = layerLoc->nyGlobal / icComm->numCommRows();
605  assert(layerLoc->nxGlobal == layerLoc->nx * icComm->numCommColumns());
606  assert(layerLoc->nyGlobal == layerLoc->ny * icComm->numCommRows());
607 
608  layerLoc->kx0 = layerLoc->nx * icComm->commColumn();
609  layerLoc->ky0 = layerLoc->ny * icComm->commRow();
610 
611  layerLoc->nf = nf;
612 
613  layerLoc->nbatch = numBatches;
614 
615  layerLoc->kb0 = parent->commBatch() * numBatches;
616  layerLoc->nbatchGlobal = parent->numCommBatches() * numBatches;
617 
618  // halo is set in calls to updateClayerMargin
619  layerLoc->halo.lt = 0; // margin;
620  layerLoc->halo.rt = 0; // margin;
621  layerLoc->halo.dn = 0; // margin;
622  layerLoc->halo.up = 0; // margin;
623 
624  return 0;
625 }

1091 int HyPerLayer::communicateInitInfo()
1092 {
1093  // HyPerLayers need to tell the parent HyPerCol how many random number
1094  // seeds they need. At the start of HyPerCol::run, the parent HyPerCol
1095  // calls each layer's communicateInitInfo() sequentially in a repeatable order
1096  // (probably the order the layers appear in the params file) to make sure
1097  // that the same runs use the same RNG seeds in the same way.
1098  //
1099  // If any other object in the column needs the layer to have a certain minimum
1100  // margin width (e.g. a HyPerConn with patch size bigger than one), it should
1101  // call the layer's requireMarginWidth() method during its communicateInitInfo
1102  // stage.
1103  //
1104  // Since all communicateInitInfo() methods are called before any allocateDataStructures()
1105  // methods, HyPerLayer knows its marginWidth before it has to allocate
1106  // anything. So the margin width does not have to be specified in params.
1107  if(triggerFlag){
1108  triggerLayer = parent->getLayerFromName(triggerLayerName);
1109  if (triggerLayer==NULL) {
1110  if (parent->columnId()==0) {
1111  pvErrorNoExit().printf("%s \"%s\": triggerLayerName \"%s\" is not a layer in the HyPerCol.\n",
1112  getKeyword(), name, triggerLayerName);
1113  }
1114  MPI_Barrier(parent->icCommunicator()->communicator());

1116  }
1117  nextTriggerTime = triggerLayer->getNextUpdateTime();
1118  if (triggerBehaviorType==RESETSTATE_TRIGGER) {
1119  char const * resetLayerName = NULL; // Will point to name of actual resetLayer, whether triggerResetLayerName is blank (in which case resetLayerName==triggerLayerName) or not
1120  if (triggerResetLayerName==NULL || triggerResetLayerName[0]=='\0') {
1121  resetLayerName = triggerLayerName;
1122  triggerResetLayer = triggerLayer;
1123  }
1124  else {
1125  resetLayerName = triggerResetLayerName;
1126  triggerResetLayer = parent->getLayerFromName(triggerResetLayerName);
1127  if (triggerResetLayer==NULL) {
1128  if (parent->columnId()==0) {
1129  pvErrorNoExit().printf("%s \"%s\": triggerResetLayerName \"%s\" is not a layer in the HyPerCol.\n",
1130  getKeyword(), name, triggerResetLayerName);
1131  }
1132  MPI_Barrier(parent->icCommunicator()->communicator());

1134  }
1135  }

1137  // Do we need to postpone until triggerResetLayer has finished its communicateInitInfo?
1138  PVLayerLoc const * triggerLoc = triggerResetLayer->getLayerLoc();

1142  pvError(errorMessage);
