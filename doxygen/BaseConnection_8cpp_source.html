111 int BaseConnection::inferPreAndPostFromConnName(
112  const char *name,
113  int rank,
114  char **preLayerNamePtr,
115  char **postLayerNamePtr) {
116  // If the connection name has the form "AbcToXyz", then *preLayerNamePtr will
117  // be Abc and
118  // *postLayerNamePtr will be Xyz.
119  // If either of the intended pre- or post-layer names contains the string
120  // "To", this method
121  // cannot be used to infer them:
122  // it returns PV_FAILURE if the string contains either more or less than one
123  // occurrence of the
124  // string "To", and does not change
125  // *preLayerNamePtr or *postLayerNamePtr
126  // This routine uses malloc to fill *{pre,post}LayerNamePtr, so the routine
127  // calling this one is
128  // responsible for freeing them.
129 
130  int status = PV_SUCCESS;
131  // Check to see if the string "To" appears exactly once in name
132  // If so, use part preceding "To" as pre-layer, and part after "To" as post.
133  const char *separator = "To";
134  const char *locto = strstr(name, separator);
135  if (locto != NULL) {
136  const char *nextto = strstr(locto + 1, separator); // Make sure "To" doesn't appear again.
137  if (nextto == NULL) {
138  int seplen = strlen(separator);
139 
140  int pre_len = locto - name;
141  *preLayerNamePtr = (char *)malloc((size_t)(pre_len + 1));
142  if (*preLayerNamePtr == NULL) {
143  Fatal().printf(
144  "Connection \"%s\": unable to allocate memory for "
145  "preLayerName: %s\n",
146  name,
147  strerror(errno));
148  }
149  const char *preInConnName = name;
150  memcpy(*preLayerNamePtr, preInConnName, pre_len);
151  (*preLayerNamePtr)[pre_len] = 0;
152 
153  int post_len = strlen(name) - pre_len - seplen;
154  *postLayerNamePtr = (char *)malloc((size_t)(post_len + 1));
155  if (*postLayerNamePtr == NULL) {
156  Fatal().printf(
157  "Connection \"%s\": unable to allocate memory for "
158  "postLayerName: %s\n",
159  name,
160  strerror(errno));
161  }
162  const char *postInConnName = &name[pre_len + seplen];
163  memcpy(*postLayerNamePtr, postInConnName, post_len);
164  (*postLayerNamePtr)[post_len] = 0;
165  }
166  else {
167  status = PV_FAILURE;
168  if (rank == 0) {
169  ErrorLog(errorMessage);
170  errorMessage.printf(
171  "Unable to infer pre and post from connection name \"%s\":\n", name);
172  errorMessage.printf("The string \"To\" cannot appear in the name more than once.\n");
173  }
174  }
175  }
176  else {
177  status = PV_FAILURE;
178  if (rank == 0) {
179  ErrorLog(errorMessage);
180  errorMessage.printf("Unable to infer pre and post from connection name \"%s\".\n", name);
181  errorMessage.printf(
182  "The connection name must have the form "
183  "\"AbcToXyz\", to infer the names,\n");
184  errorMessage.printf("but the string \"To\" does not appear.\n");
185  }
186  }
187  return status;
188 }

190 int BaseConnection::getPreAndPostLayerNames(
191  const char *name,
192  char **preLayerNamePtr,
193  char **postLayerNamePtr) {
194  // Retrieves preLayerName and postLayerName from parameter group whose name is
195  // given in the
196  // functions first argument.
197  // This routine uses strdup to fill *{pre,post}LayerNamePtr, so the routine
198  // calling this one is
199  // responsible for freeing them.
200  int status = PV_SUCCESS;
201  *preLayerNamePtr = NULL;
202  *postLayerNamePtr = NULL;
203  if (preLayerName == NULL && postLayerName == NULL) {
204  if (parent->getCommunicator()->commRank() == 0) {
205  InfoLog().printf(
206  "%s: preLayerName and postLayerName will be inferred in "
207  "the communicateInitInfo "
208  "stage.\n",
209  getDescription_c());
210  }
211  }
212  else if (preLayerName == NULL && postLayerName != NULL) {
213  status = PV_FAILURE;
214  if (parent->getCommunicator()->commRank() == 0) {
215  ErrorLog().printf(
216  "%s: if postLayerName is specified, preLayerName "
217  "must be specified as well.\n",
218  getDescription_c());
219  }
220  }
221  else if (preLayerName != NULL && postLayerName == NULL) {
222  status = PV_FAILURE;
223  if (parent->getCommunicator()->commRank() == 0) {
224  ErrorLog().printf(
225  "%s: if preLayerName is specified, postLayerName "
226  "must be specified as well.\n",
227  getDescription_c());
228  }
229  }
230  else {
231  assert(preLayerName != NULL && postLayerName != NULL);
232  }
233  MPI_Barrier(parent->getCommunicator()->communicator());
234  if (status != PV_SUCCESS) {
235  exit(EXIT_FAILURE);
236  }
237  return status;
238 }

387 int BaseConnection::insertProbe(BaseConnectionProbe *p) {
388  if (p->getTargetConn() != this) {
389  WarnLog().printf(
390  "%s: insertProbe called with probe %p, whose targetConn is "
391  "not this connection. Probe "
392  "was not inserted.\n",
393  getDescription_c(),
394  p);
395  return numProbes;
396  }
397  for (int i = 0; i < numProbes; i++) {
398  if (p == probes[i]) {
399  WarnLog().printf(
400  "%s: insertProbe called with probe %p, which has already "
401  "been inserted as probe "
402  "%d.\n",
403  getDescription_c(),
404  p,
405  i);
406  return numProbes;
407  }
408  }
409 
410  BaseConnectionProbe **tmp;
411  tmp = (BaseConnectionProbe **)malloc((numProbes + 1) * sizeof(BaseConnectionProbe *));
412  assert(tmp != NULL);
413 
414  for (int i = 0; i < numProbes; i++) {
415  tmp[i] = probes[i];
416  }
417  delete probes;
418 
419  probes = tmp;
420  probes[numProbes] = p;
421 
422  return ++numProbes;
423 }

460 int BaseConnection::communicateInitInfo(CommunicateInitInfoMessage const *message) {
461  int status = PV_SUCCESS;
462 
463  if (this->getPreLayerName() == NULL) {
464  assert(this->getPostLayerName() == NULL);
465  status = handleMissingPreAndPostLayerNames();
466  }
467  MPI_Barrier(this->parent->getCommunicator()->communicator());
468  if (status != PV_SUCCESS) {
469  assert(this->getPreLayerName() == NULL && this->getPostLayerName() == NULL);
470  if (this->parent->columnId() == 0) {
471  ErrorLog().printf(
472  "%s: Unable to determine pre- and post-layer names. Exiting.\n",
473  this->getDescription_c());
474  }
475  exit(EXIT_FAILURE);
476  }
477  this->pre = message->lookup<HyPerLayer>(std::string(this->getPreLayerName()));
478  this->post = message->lookup<HyPerLayer>(std::string(this->getPostLayerName()));
479  if (this->preSynapticLayer() == NULL) {
480  if (this->parent->columnId() == 0) {
481  ErrorLog().printf(
482  "%s: preLayerName \"%s\" does not correspond to a "
483  "layer in the column.\n",
484  getDescription_c(),
485  this->getPreLayerName());
486  }
487  status = PV_FAILURE;
488  }
489  if (this->postSynapticLayer() == NULL) {
490  if (this->parent->columnId() == 0) {
491  ErrorLog().printf(
492  "%s: postLayerName \"%s\" does not correspond to "
493  "a layer in the column.\n",
494  getDescription_c(),
495  this->getPostLayerName());
496  }
497  status = PV_FAILURE;
498  }
499  MPI_Barrier(this->parent->getCommunicator()->communicator());
500  if (status != PV_SUCCESS) {
501  exit(EXIT_FAILURE);
502  }
503 
504  initializeDelays(fDelayArray, delayArraySize);
505  int maxDelay = maxDelaySteps();
506  int allowedDelay = this->preSynapticLayer()->increaseDelayLevels(maxDelay);
507  if (allowedDelay < maxDelay) {
508  if (this->parent->columnId() == 0) {
509  ErrorLog().printf(
510  "%s: attempt to set delay to %d, but the maximum "
511  "allowed delay is %d. Exiting\n",
512  getDescription_c(),
513  maxDelay,
514  allowedDelay);
515  }
516  exit(EXIT_FAILURE);
517  }
518 
519  // Make sure post-synaptic layer has enough channels.
520  int num_channels_check = 0;
521  int ch = (int)this->getChannel();
522  if (ch >= 0) {
523  status = this->postSynapticLayer()->requireChannel(ch, &num_channels_check);
524  }
525  pvAssert(status != PV_SUCCESS || num_channels_check > (int)this->getChannel());
526  // if requireChannel passes, layer's numChannels should be large enough for the
527  // connection's channel
528  if (status != PV_SUCCESS) {
529  if (parent->columnId() == 0) {
530  ErrorLog().printf(
531  "%s: postsynaptic layer \"%s\" failed to add channel %d\n",
532  this->getDescription_c(),
533  this->postSynapticLayer()->getName(),
534  (int)this->getChannel());
535  }
536  }
537 
538  this->postSynapticLayer()->addRecvConn(this);
539  return status;
540 }

542 int BaseConnection::initializeDelays(const float *fDelayArray, int size) {
543  int status = PV_SUCCESS;
544  assert(!this->parent->parameters()->presentAndNotBeenRead(this->getName(), "numAxonalArbors"));
545  // Allocate delay data structure
546  delays = (int *)calloc(this->numberOfAxonalArborLists(), sizeof(int));
547  if (delays == NULL) {
548  Fatal().printf(
549  "%s: unable to allocate memory for %d delays: %s\n",
550  getDescription_c(),
551  size,
552  strerror(errno));
553  }
554 
555  // Initialize delays for each arbor
556  // Using setDelay to convert ms to timesteps
557  for (int arborId = 0; arborId < this->numberOfAxonalArborLists(); arborId++) {
558  if (size == 0) {
559  // No delay
560  setDelay(arborId, 0);
561  }
562  else if (size == 1) {
563  setDelay(arborId, fDelayArray[0]);
564  }
565  else if (size == this->numberOfAxonalArborLists()) {
566  setDelay(arborId, fDelayArray[arborId]);
567  }
568  else {
569  Fatal().printf(
570  "Delay must be either a single value or the same length "
571  "as the number of arbors\n");
572  }
573  }
574  return status;
575 }

577 int BaseConnection::maxDelaySteps() {
578  int maxDelay = 0;
579  for (int arborId = 0; arborId < numberOfAxonalArborLists(); arborId++) {
580  if (delays[arborId] > maxDelay) {
581  maxDelay = delays[arborId];
582  }
583  }
584  return maxDelay;

587 // Input delay is in ms
588 void BaseConnection::setDelay(int arborId, double delay) {
589  assert(arborId >= 0 && arborId < numberOfAxonalArborLists());
590  int intDelay = round(delay / parent->getDeltaTime());
591  if (fmod(delay, parent->getDeltaTime()) != 0) {
592  double actualDelay = intDelay * parent->getDeltaTime();
593  WarnLog() << getName() << ": A delay of " << delay << " will be rounded to " << actualDelay
594  << "\n";
595  }
596  delays[arborId] = (int)(intDelay);
597 }
