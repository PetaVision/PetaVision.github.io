170 int BaseConnection::inferPreAndPostFromConnName(const char * name, int rank, char ** preLayerNamePtr, char ** postLayerNamePtr) {
171  // If the connection name has the form "AbcToXyz", then *preLayerNamePtr will be Abc and *postLayerNamePtr will be Xyz.
172  // If either of the intended pre- or post-layer names contains the string "To", this method cannot be used to infer them:
173  // it returns PV_FAILURE if the string contains either more or less than one occurrence of the string "To", and does not change
174  // *preLayerNamePtr or *postLayerNamePtr
175  // This routine uses malloc to fill *{pre,post}LayerNamePtr, so the routine calling this one is responsible for freeing them.
176 
177  int status = PV_SUCCESS;
178  // Check to see if the string "To" appears exactly once in name
179  // If so, use part preceding "To" as pre-layer, and part after "To" as post.
180  const char * separator = "To";
181  const char * locto = strstr(name, separator);
182  if( locto != NULL ) {
183  const char * nextto = strstr(locto+1, separator); // Make sure "To" doesn't appear again.
184  if( nextto == NULL ) {
185  int seplen = strlen(separator);
186 
187  int pre_len = locto - name;
188  *preLayerNamePtr = (char *) malloc((size_t) (pre_len + 1));
189  if( *preLayerNamePtr==NULL) {
190  fprintf(stderr, "Error: unable to allocate memory for preLayerName in connection \"%s\": %s\n", name, strerror(errno));
191  exit(EXIT_FAILURE);
192  }
193  const char * preInConnName = name;
194  memcpy(*preLayerNamePtr, preInConnName, pre_len);
195  (*preLayerNamePtr)[pre_len] = 0;
196 
197  int post_len = strlen(name)-pre_len-seplen;
198  *postLayerNamePtr = (char *) malloc((size_t) (post_len + 1));
199  if( *postLayerNamePtr==NULL) {
200  fprintf(stderr, "Error: unable to allocate memory for postLayerName in connection \"%s\": %s\n", name, strerror(errno));
201  exit(EXIT_FAILURE);
202  }
203  const char * postInConnName = &name[pre_len+seplen];
204  memcpy(*postLayerNamePtr, postInConnName, post_len);
205  (*postLayerNamePtr)[post_len] = 0;
206  }
207  else {
208  status = PV_FAILURE;
209  if (rank==0) {
210  fprintf(stderr, "Unable to infer pre and post from connection name \"%s\":\n", name);
211  fprintf(stderr, "The string \"To\" cannot appear in the name more than once.\n");
212  }
213  }
214  }
215  else {
216  status = PV_FAILURE;
217  if (rank==0) {
218  fprintf(stderr, "Unable to infer pre and post from connection name \"%s\".\n", name);
219  fprintf(stderr, "The connection name must have the form \"AbcToXyz\", to infer the names,\n");
220  fprintf(stderr, "but the string \"To\" does not appear.\n");
221  }
222  }
223  return status;
224 }

226 int BaseConnection::getPreAndPostLayerNames(const char * name, char ** preLayerNamePtr, char ** postLayerNamePtr) {
227  // Retrieves preLayerName and postLayerName from parameter group whose name is given in the functions first argument.
228  // This routine uses strdup to fill *{pre,post}LayerNamePtr, so the routine calling this one is responsible for freeing them.
229  int status = PV_SUCCESS;
230  *preLayerNamePtr = NULL;
231  *postLayerNamePtr = NULL;
232  if (preLayerName == NULL && postLayerName == NULL) {
233  if (parent->icCommunicator()->commRank()==0) {
234  printf("Connection \"%s\": preLayerName and postLayerName will be inferred in the communicateInitInfo stage.\n", name);
235  }
236  }
237  else if (preLayerName==NULL && postLayerName!=NULL) {
238  status = PV_FAILURE;
239  if (parent->icCommunicator()->commRank()==0) {
240  fprintf(stderr, "Connection \"%s\" error: if postLayerName is specified, preLayerName must be specified as well.\n", name);
241  }
242  }
243  else if (preLayerName!=NULL && postLayerName==NULL) {
244  status = PV_FAILURE;
245  if (parent->icCommunicator()->commRank()==0) {
246  fprintf(stderr, "Connection \"%s\" error: if preLayerName is specified, postLayerName must be specified as well.\n", name);
247  }
248  }
249  else {
250  assert(preLayerName!=NULL && postLayerName!=NULL);
251  }
252  MPI_Barrier(parent->icCommunicator()->communicator());
253  if (status != PV_SUCCESS) {
254  exit(EXIT_FAILURE);
255  }
256  return status;
257 }

406 int BaseConnection::insertProbe(BaseConnectionProbe * p)
407 {
408  if(p->getTargetConn() != this) {
409  fprintf(stderr, "Connection \"%s\": insertProbe called with probe %p, whose targetConn is not this connection. Probe was not inserted.\n", this->getName(), p);
410  return numProbes;
411  }
412  for( int i=0; i<numProbes; i++ ) {
413  if( p == probes[i] ) {
414  fprintf(stderr, "Connection \"%s\": insertProbe called with probe %p, which has already been inserted as probe %d.\n", this->getName(), p, i);
415  return numProbes;
416  }
417  }
418 
419  BaseConnectionProbe ** tmp;
420  tmp = (BaseConnectionProbe **) malloc((numProbes + 1) * sizeof(BaseConnectionProbe *));
421  assert(tmp != NULL);
422 
423  for (int i = 0; i < numProbes; i++) {
424  tmp[i] = probes[i];
425  }
426  delete probes;
427 
428  probes = tmp;
429  probes[numProbes] = p;
430 
431  return ++numProbes;
432 }

443 int BaseConnection::communicateInitInfo() {
444  int status = PV_SUCCESS;
445 
446  if (this->getPreLayerName()==NULL) {
447  assert(this->getPostLayerName()==NULL);
448  status = handleMissingPreAndPostLayerNames();
449  }
450  MPI_Barrier(this->getParent()->icCommunicator()->communicator());
451  if (status != PV_SUCCESS) {
452  assert(this->getPreLayerName()==NULL && this->getPostLayerName()==NULL);
453  if (this->getParent()->columnId()==0) {
454  fprintf(stderr, "%s \"%s\" error: Unable to determine pre- and post-layer names. Exiting.\n", this->getKeyword(), this->getName());
455  }
456  exit(EXIT_FAILURE);
457  }
458  this->setPreSynapticLayer(this->getParent()->getLayerFromName(this->getPreLayerName()));
459  this->setPostSynapticLayer(this->getParent()->getLayerFromName(this->getPostLayerName()));
460  if (this->preSynapticLayer()==NULL) {
461  if (this->getParent()->columnId()==0) {
462  fprintf(stderr, "Connection \"%s\": preLayerName \"%s\" does not correspond to a layer in the column.\n", this->getName(), this->getPreLayerName());
463  }
464  status = PV_FAILURE;
465  }
466  if (this->postSynapticLayer()==NULL) {
467  if (this->getParent()->columnId()==0) {
468  fprintf(stderr, "Connection \"%s\": postLayerName \"%s\" does not correspond to a layer in the column.\n", this->getName(), this->getPostLayerName());
469  }
470  status = PV_FAILURE;
471  }
472  MPI_Barrier(this->getParent()->icCommunicator()->communicator());
473  if (status != PV_SUCCESS) {
474  exit(EXIT_FAILURE);
475  }
476 
477  // Find maximum delay over all the arbors and send it to the presynaptic layer
478  int maxdelay = 0;
479  for (int delayi = 0; delayi < delayArraySize; delayi++){
480  if (fDelayArray[delayi] > maxdelay){
481  maxdelay = fDelayArray[delayi];
482  }
483  }
484  int allowedDelay = this->preSynapticLayer()->increaseDelayLevels(maxdelay);
485  if( allowedDelay < maxdelay ) {
486  if( this->getParent()->columnId() == 0 ) {
487  fflush(stdout);
488  fprintf(stderr, "Connection \"%s\": attempt to set delay to %d, but the maximum allowed delay is %d. Exiting\n", this->getName(), maxdelay, allowedDelay);
489  }
490  exit(EXIT_FAILURE);
491  }
492 
493  // Make sure post-synaptic layer has enough channels.
494  int num_channels_check = 0;
495  int ch = (int) this->getChannel();
496  if (ch>=0) {
497  status = this->postSynapticLayer()->requireChannel(ch, &num_channels_check);
498  }
499  assert(status != PV_SUCCESS || num_channels_check > (int) this->getChannel()); // if requireChannel passes, layer's numChannels should be large enough for the connection's channel
500  if (status != PV_SUCCESS) {
501  if (parent->columnId()==0) {
502  fprintf(stderr, "%s \"%s\" error: postsynaptic layer \"%s\" failed to add channel %d\n",
503  this->getKeyword(), this->getName(), this->postSynapticLayer()->getName(), (int) this->getChannel());
504  }
505  }
506  return status;
507 }

514 int BaseConnection::initializeDelays(const float * fDelayArray, int size){
515  int status = PV_SUCCESS;
516  assert(!this->getParent()->parameters()->presentAndNotBeenRead(this->getName(), "numAxonalArbors"));
517  //Allocate delay data structure
518  delays = (int *) calloc(this->numberOfAxonalArborLists(), sizeof(int));
519  if( delays == NULL ) {
520  fprintf(stderr, "%s \"%s\": unable to allocate memory for %d delays: %s\n", this->getKeyword(), name, size, strerror(errno));
521  exit(EXIT_FAILURE);
522  }
523 
524  //Initialize delays for each arbor
525  //Using setDelay to convert ms to timesteps
526  for (int arborId=0;arborId<this->numberOfAxonalArborLists();arborId++) {
527  if (size == 0){
528  //No delay
529  setDelay(arborId, 0);
530  }
531  else if (size == 1){
532  setDelay(arborId, fDelayArray[0]);
533  }
534  else if (size == this->numberOfAxonalArborLists()){
535  setDelay(arborId, fDelayArray[arborId]);
536  }
537  else{
538  fprintf(stderr, "Delay must be either a single value or the same length as the number of arbors\n");
539  abort();
540  }
541  }
542  return status;
543 }

545 //Input delay is in ms
546 void BaseConnection::setDelay(int arborId, float delay) {
547  assert(arborId>=0 && arborId<this->numberOfAxonalArborLists());
548  int intDelay = round(delay/this->getParent()->getDeltaTime());
549  if (fmod(delay, this->getParent()->getDeltaTime()) != 0){
550  float actualDelay = intDelay * this->getParent()->getDeltaTime();
551  std::cerr << this->getName() << ": A delay of " << delay << " will be rounded to " << actualDelay << "\n";
552  }
553  delays[arborId] = (int)(intDelay);
554 }

556 int BaseConnection::initializeState() {
