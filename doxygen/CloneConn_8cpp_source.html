103 // parameters exist in params for theCloneKernelConn group, and whether they
104 // are consistent with the originalConn parameters.
105 // If consistent, issue a warning that the param is unnecessary and continue.
106 // If inconsistent, issue an error and quit.
107 // We can't do that in the read-method because we can't be sure originalConn
108 // has set its own parameter yet (or even if it's been instantiated),
109 // and in theory originalConn could be a subclass that determines
110 // the parameter some way other than reading its own parameter
111 // group's param directly.

168 int CloneConn::communicateInitInfo() {
169  // Need to set originalConn before calling HyPerConn::communicate, since HyPerConn::communicate calls setPatchSize, which needs originalConn.
170  BaseConnection * originalConnBase = parent->getConnFromName(originalConnName);
171  if (originalConnBase == NULL) {
172  if (parent->columnId()==0) {
173  fprintf(stderr, "CloneConn \"%s\" error: originalConnName \"%s\" is not a connection in the column.\n",
174  name, originalConnName);
175  }
176  MPI_Barrier(parent->icCommunicator()->communicator());
177  exit(EXIT_FAILURE);
178  }
179  originalConn = dynamic_cast<HyPerConn *>(originalConnBase);
180  if (originalConn == NULL) {
181  if (parent->columnId()==0) {
182  fprintf(stderr, "CloneConn \"%s\" error: originalConnName \"%s\" is not a HyPerConn or HyPerConn-derived class.\n",
183  name, originalConnName);
184  }
185  }
186  if (!originalConn->getInitInfoCommunicatedFlag()) {
187  if (parent->columnId()==0) {
188  const char * connectiontype = this->getKeyword();
189  printf("%s \"%s\" must wait until original connection \"%s\" has finished its communicateInitInfo stage.\n", connectiontype, name, originalConn->getName());
190  }
191  return PV_POSTPONE;
192  }
193 
194  // Copy some parameters from originalConn. Check if parameters exist is
195  // the clone's param group, and issue a warning (if the param has the right
196  // value) or an error (if it has the wrong value).
197  int status = cloneParameters();
198 
199  status = HyPerConn::communicateInitInfo();
200  if (status != PV_SUCCESS) return status;
201 
202  //Don't allocate post, just grab in allocate from orig
203  if(needPost){
204  originalConn->setNeedPost(true);
205  }
206 
207 #if defined(PV_USE_OPENCL) || defined(PV_USE_CUDA)
208  if((updateGSynFromPostPerspective && receiveGpu) || allocPostDeviceWeights){
209  originalConn->setAllocPostDeviceWeights();
210  }
211  if((!updateGSynFromPostPerspective && receiveGpu) || allocDeviceWeights){
212  originalConn->setAllocDeviceWeights();
213  }
214 #endif
215 
216 
217  // Presynaptic layers of the CloneConn and its original conn must have the same size, or the patches won't line up with each other.
218  const PVLayerLoc * preLoc = pre->getLayerLoc();
219  const PVLayerLoc * origPreLoc = originalConn->preSynapticLayer()->getLayerLoc();
220 
221  if (preLoc->nx != origPreLoc->nx || preLoc->ny != origPreLoc->ny || preLoc->nf != origPreLoc->nf ) {
222  if (parent->icCommunicator()->commRank()==0) {
223  const char * classname = this->getKeyword();
224  fprintf(stderr, "%s \"%s\" error in rank %d process: CloneConn and originalConn \"%s\" must have presynaptic layers with the same nx,ny,nf.\n",
225  classname, name, parent->columnId(), originalConn->getName());
226  fprintf(stderr, "{nx=%d, ny=%d, nf=%d} versus {nx=%d, ny=%d, nf=%d}\n",
227  preLoc->nx, preLoc->ny, preLoc->nf, origPreLoc->nx, origPreLoc->ny, origPreLoc->nf);
228  }
229  abort();
230  }
231 
232  // Make sure the original's and the clone's margin widths stay equal
233  originalConn->preSynapticLayer()->synchronizeMarginWidth(pre);
234  pre->synchronizeMarginWidth(originalConn->preSynapticLayer());
235 
237  //originalConn->postSynapticLayer()->synchronizeMarginWidth(post);
238  //post->synchronizeMarginWidth(originalConn->postSynapticLayer());
239 
240  //Redudant read in case it's a clone of a clone
241 
242  return status;
243 }
