94 // originalConn's values. communicateInitInfo will check if the associated
95 // parameters exist in params for theCloneKernelConn group, and whether they

98 // If inconsistent, issue an error and quit.

100 // has set its own parameter yet (or even if it's been instantiated),
101 // and in theory originalConn could be a subclass that determines
102 // the parameter some way other than reading its own parameter
103 // group's param directly.

175 int CloneConn::communicateInitInfo(CommunicateInitInfoMessage const *message) {
176  // Need to set originalConn before calling HyPerConn::communicate, since HyPerConn::communicate
177  // calls setPatchSize, which needs originalConn.
178  originalConn = message->lookup<HyPerConn>(std::string(originalConnName));
179  if (originalConn == NULL) {
180  if (parent->columnId() == 0) {
181  ErrorLog().printf(
182  "%s: originalConnName \"%s\" is not a HyPerConn or HyPerConn-derived object.\n",
183  getDescription_c(),
184  originalConnName);
185  }
186  }
187  if (!originalConn->getInitInfoCommunicatedFlag()) {
188  if (parent->columnId() == 0) {
189  InfoLog().printf(
190  "%s must wait until original connection \"%s\" has finished its communicateInitInfo "
191  "stage.\n",
192  getDescription_c(),
193  originalConn->getName());
194  }
195  return PV_POSTPONE;
196  }
197 
198  // Copy some parameters from originalConn. Check if parameters exist is
199  // the clone's param group, and issue a warning (if the param has the right
200  // value) or an error (if it has the wrong value).
201  int status = cloneParameters();
202 
203  status = HyPerConn::communicateInitInfo(message);
204  if (status != PV_SUCCESS)
205  return status;
206 
207  // Don't allocate post, just grab in allocate from orig
208  if (needPost) {
209  originalConn->setNeedPost();
210  }
211 
212 #ifdef PV_USE_CUDA
213  if ((updateGSynFromPostPerspective && receiveGpu) || allocPostDeviceWeights) {
214  originalConn->setAllocPostDeviceWeights();
215  }
216  if ((!updateGSynFromPostPerspective && receiveGpu) || allocDeviceWeights) {
217  originalConn->setAllocDeviceWeights();
218  }
219 #endif
220 
221  // Presynaptic layers of the CloneConn and its original conn must have the same size, or the
222  // patches won't line up with each other.
223  const PVLayerLoc *preLoc = pre->getLayerLoc();
224  const PVLayerLoc *origPreLoc = originalConn->preSynapticLayer()->getLayerLoc();
225 
226  if (preLoc->nx != origPreLoc->nx || preLoc->ny != origPreLoc->ny
227  || preLoc->nf != origPreLoc->nf) {
228  if (parent->getCommunicator()->commRank() == 0) {
229  ErrorLog(errorMessage);
230  errorMessage.printf(
231  "%s: CloneConn and originalConn \"%s\" must have presynaptic layers with the same "
232  "nx,ny,nf.\n",
233  getDescription_c(),
234  parent->columnId(),
235  originalConn->getName());
236  errorMessage.printf(
237  "{nx=%d, ny=%d, nf=%d} versus {nx=%d, ny=%d, nf=%d}\n",
238  preLoc->nx,
239  preLoc->ny,
240  preLoc->nf,
241  origPreLoc->nx,
242  origPreLoc->ny,
243  origPreLoc->nf);
244  }
245  MPI_Barrier(parent->getCommunicator()->communicator());
246  abort();
247  }
248 
249  // Make sure the original's and the clone's margin widths stay equal
250  originalConn->preSynapticLayer()->synchronizeMarginWidth(pre);
251  pre->synchronizeMarginWidth(originalConn->preSynapticLayer());
252 
253  // Make sure the original's and the clone's margin widths stay equal
254  // Only if this layer receives from post for patch to data LUT
255  if (getUpdateGSynFromPostPerspective()) {
256  originalConn->postSynapticLayer()->synchronizeMarginWidth(post);
257  post->synchronizeMarginWidth(originalConn->postSynapticLayer());
258  }
259  // Redudant read in case it's a clone of a clone
260 
261  return status;
262 }
