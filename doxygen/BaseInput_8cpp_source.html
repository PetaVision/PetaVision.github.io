420 int BaseInput::scatterInput(int batchIndex) {
421  int const rank = parent->columnId();
422  MPI_Comm mpi_comm = parent->icCommunicator()->communicator();
423  int const rootproc = 0;
424  pvadata_t * A = data + batchIndex * getNumExtended();
425  if (rank == rootproc) {
426  PVLayerLoc const * loc = getLayerLoc();
427  PVHalo const * halo = &loc->halo;
428  int const nxExt = loc->nx + halo->lt + halo->rt;
429  int const nyExt = loc->ny + halo->dn + halo->up;
430  int const nf = loc->nf;
431 
432  if (imageLoc.nf==1 && nf>1) {
433  convertGrayScaleToMultiBand(&imageData, imageLoc.nxGlobal, imageLoc.nyGlobal, nf);
434  imageLoc.nf = nf;
435  imageColorType = COLORTYPE_UNRECOGNIZED;
436  }
437  if (imageLoc.nf>1 && nf==1) {
438  convertToGrayScale(&imageData, imageLoc.nxGlobal, imageLoc.nyGlobal, imageLoc.nf, imageColorType);
439  imageLoc.nf = 1;
440  imageColorType = COLORTYPE_GRAYSCALE;
441  }
442  if (imageLoc.nf != nf) {
443  pvError().printf("%s \"%s\": imageLoc has %d features but layer has %d features.\n",
444  getKeyword(), name, imageLoc.nf, nf);
445  }
446 
447  resizeInput();
448  int dims[2] = {imageLoc.nxGlobal, imageLoc.nyGlobal};
449  MPI_Bcast(dims, 2, MPI_INT, rootproc, mpi_comm);
450  for (int r=0; r<parent->icCommunicator()->commSize(); r++) {
451  if (r==rootproc) { continue; } // Do root process last so that we don't clobber root process data by using the data buffer to send.
452  for (int n=0; n<getNumExtended(); n++) { A[n] = padValue; }
453  int dataLeft, dataTop, imageLeft, imageTop, width, height;
454  int status = calcLocalBox(r, &dataLeft, &dataTop, &imageLeft, &imageTop, &width, &height);
455  if (status==PV_SUCCESS) {
456  pvAssert(width>0 && height>0);
457  for (int y=0; y<height; y++) {
458  int imageIdx = kIndex(imageLeft, imageTop+y, 0, imageLoc.nxGlobal, imageLoc.nyGlobal, nf);
459  int dataIdx = kIndex(dataLeft, dataTop+y, 0, nxExt, nyExt, nf);
460  memcpy(&A[dataIdx], &imageData[imageIdx], sizeof(pvadata_t)*width*nf);
461  }
462  }
463  else {
464  pvAssert(width==0||height==0);
465  }
466  MPI_Send(A, getNumExtended(), MPI_FLOAT, r, 31, mpi_comm);
467  }
468  // Finally, do root process.
469  for (int n=0; n<getNumExtended(); n++) { A[n] = padValue; }
470  int status = calcLocalBox(rootproc, &dataLeft, &dataTop, &imageLeft, &imageTop, &dataWidth, &dataHeight);
471  if (status==PV_SUCCESS) {
472  pvAssert(dataWidth>0 && dataHeight>0);
473  for (int y=0; y<dataHeight; y++) {
474  int imageIdx = kIndex(imageLeft, imageTop+y, 0, imageLoc.nxGlobal, imageLoc.nyGlobal, nf);
475  int dataIdx = kIndex(dataLeft, dataTop+y, 0, nxExt, nyExt, nf);
476  assert(imageIdx>=0 && imageIdx<imageLoc.nxGlobal * imageLoc.nyGlobal * imageLoc.nf);
477  assert(dataIdx>=0 && dataIdx<getNumExtended());
478  memcpy(&A[dataIdx], &imageData[imageIdx], sizeof(pvadata_t)*dataWidth*nf);
479  }
480  }
481  else { assert(dataWidth==0||dataHeight==0); }
482  }
483  else {
484  int dims[2];
485  MPI_Bcast(dims, 2, MPI_INT, rootproc, mpi_comm);
486  imageLoc.nxGlobal = dims[0];
487  imageLoc.nyGlobal = dims[1];
488  imageLoc.nf = getLayerLoc()->nf;
489  MPI_Recv(A, getNumExtended(), MPI_FLOAT, rootproc, 31, mpi_comm, MPI_STATUS_IGNORE);
490  calcLocalBox(rank, &dataLeft, &dataTop, &imageLeft, &imageTop, &dataWidth, &dataHeight);
491  }
492  return PV_SUCCESS;
493 }

495 int BaseInput::resizeInput() {
496  pvAssert(parent->columnId()==0); // Should only be called by root process.
497  if (!autoResizeFlag) {
498  resizeFactor = 1.0f;
499  return PV_SUCCESS;
500  }
501  PVLayerLoc const * loc = getLayerLoc();
502  PVHalo const * halo = &loc->halo;
503  int const targetWidth = loc->nxGlobal + (useImageBCflag ? (halo->lt + halo->rt) : 0);
504  int const targetHeight = loc->nyGlobal + (useImageBCflag ? (halo->dn + halo->up) : 0);
505  float xRatio = (float) targetWidth/(float) imageLoc.nxGlobal;
506  float yRatio = (float) targetHeight/(float) imageLoc.nyGlobal;
507  int resizedWidth, resizedHeight;
508  if (!strcmp(aspectRatioAdjustment, "crop")) {
509  resizeFactor = xRatio < yRatio ? yRatio : xRatio;
510  // resizeFactor * width should be >= getLayerLoc()->nx; resizeFactor * height should be >= getLayerLoc()->ny,
511  // and one of these relations should be == (up to floating-point roundoff).
512  resizedWidth = (int) nearbyintf(resizeFactor * imageLoc.nxGlobal);
513  resizedHeight = (int) nearbyintf(resizeFactor * imageLoc.nyGlobal);
514  pvAssert(resizedWidth >= targetWidth);
515  pvAssert(resizedHeight >= targetHeight);
516  pvAssert(resizedWidth == targetWidth || resizedHeight == targetHeight);
517  }
518  else if (!strcmp(aspectRatioAdjustment, "pad")) {
519  resizeFactor = xRatio < yRatio ? xRatio : yRatio;
520  // resizeFactor * width should be <= getLayerLoc()->nx; resizeFactor * height should be <= getLayerLoc()->ny,
521  // and one of these relations should be == (up to floating-point roundoff).
522  resizedWidth = (int) nearbyintf(resizeFactor * imageLoc.nxGlobal);
523  resizedHeight = (int) nearbyintf(resizeFactor * imageLoc.nyGlobal);
524  pvAssert(resizedWidth <= getLayerLoc()->nxGlobal);
525  pvAssert(resizedHeight <= getLayerLoc()->nyGlobal);
526  pvAssert(resizedWidth == getLayerLoc()->nxGlobal || resizedHeight == getLayerLoc()->nxGlobal);
527  }
528  else {
529  assert(0);
530  }
531  float * newImageData = new float[resizedHeight*resizedWidth*imageLoc.nf];
532  switch(interpolationMethod) {
533  case INTERPOLATE_BICUBIC:
534  bicubicInterp(imageData, imageLoc.nxGlobal, imageLoc.nyGlobal, imageLoc.nf, imageLoc.nf, imageLoc.nf*imageLoc.nxGlobal, 1, newImageData, resizedWidth, resizedHeight);
535  break;
536  case INTERPOLATE_NEARESTNEIGHBOR:
537  nearestNeighborInterp(imageData, imageLoc.nxGlobal, imageLoc.nyGlobal, imageLoc.nf, imageLoc.nf, imageLoc.nf*imageLoc.nxGlobal, 1, newImageData, resizedWidth, resizedHeight);
538  break;
539  default:
540  assert(0);
541  break;
542  }
543  delete[] imageData;
544  imageData = newImageData;
545  imageLoc.nxGlobal = resizedWidth;
546  imageLoc.nyGlobal = resizedHeight;
547  return PV_SUCCESS;
548 }
549 int BaseInput::nearestNeighborInterp(pvadata_t const * bufferIn, int widthIn, int heightIn, int numBands, int xStrideIn, int yStrideIn, int bandStrideIn, pvadata_t * bufferOut, int widthOut, int heightOut) {
550  /* Interpolation using nearest neighbor interpolation */
551  int xinteger[widthOut];
552  float dx = (float) (widthIn-1)/(float) (widthOut-1);
553  for (int kx=0; kx<widthOut; kx++) {
554  float x = dx * (float) kx;
555  xinteger[kx] = (int) nearbyintf(x);
556  }
557 
558  int yinteger[heightOut];
559  float dy = (float) (heightIn-1)/(float) (heightOut-1);
560  for (int ky=0; ky<heightOut; ky++) {
561  float y = dy * (float) ky;
562  yinteger[ky] = (int) nearbyintf(y);
563  }
564 
565  for (int ky=0; ky<heightOut; ky++) {
566  float yfetch = yinteger[ky];
567  for (int kx=0; kx<widthOut; kx++) {
568  int xfetch = xinteger[kx];
569  for (int f=0; f<numBands; f++) {
570  int fetchIdx = yfetch * yStrideIn + xfetch * xStrideIn + f * bandStrideIn;
571  int outputIdx = kIndex(kx, ky, f, widthOut, heightOut, numBands);
572  bufferOut[outputIdx] = bufferIn[fetchIdx];
573  }
574  }
575  }

579 int BaseInput::bicubicInterp(pvadata_t const * bufferIn, int widthIn, int heightIn, int numBands, int xStrideIn, int yStrideIn, int bandStrideIn, pvadata_t * bufferOut, int widthOut, int heightOut) {
580  /* Interpolation using bicubic convolution with a=-1 (following Octave image toolbox's imremap function -- change this?). */
581  float xinterp[widthOut];
582  int xinteger[widthOut];
583  float xfrac[widthOut];
584  float dx = (float) (widthIn-1)/(float) (widthOut-1);
585  for (int kx=0; kx<widthOut; kx++) {
586  float x = dx * (float) kx;
587  xinterp[kx] = x;
588  float xfloor = floorf(x);
589  xinteger[kx] = (int) xfloor;
590  xfrac[kx] = x-xfloor;
591  }
592 
593  float yinterp[heightOut];
594  int yinteger[heightOut];
595  float yfrac[heightOut];
596  float dy = (float) (heightIn-1)/(float) (heightOut-1);
597  for (int ky=0; ky<heightOut; ky++) {
598  float y = dy * (float) ky;
599  yinterp[ky] = y;
600  float yfloor = floorf(y);
601  yinteger[ky] = (int) yfloor;
602  yfrac[ky] = y-yfloor;
603  }
604 
605  memset(bufferOut, 0, sizeof(*bufferOut)*size_t(widthOut*heightOut*numBands));
606  for (int xOff = 2; xOff > -2; xOff--) {
607  for (int yOff = 2; yOff > -2; yOff--) {
608  for (int ky=0; ky<heightOut; ky++) {
609  float ycoeff = bicubic(yfrac[ky]-(float) yOff);
610  int yfetch = yinteger[ky]+yOff;
611  if (yfetch < 0) yfetch = -yfetch;
612  if (yfetch >= heightIn) yfetch = heightIn - (yfetch - heightIn) - 1;
613  for (int kx=0; kx<widthOut; kx++) {
614  float xcoeff = bicubic(xfrac[kx]-(float) xOff);
615  int xfetch = xinteger[kx]+xOff;
616  if (xfetch < 0) xfetch = -xfetch;
617  if (xfetch >= widthIn) xfetch = widthIn - (xfetch - widthIn) - 1;
618  assert(xfetch >= 0 && xfetch < widthIn && yfetch >= 0 && yfetch < heightIn);
619  for (int f=0; f<numBands; f++) {
620  int fetchIdx = yfetch * yStrideIn + xfetch * xStrideIn + f * bandStrideIn;
621  pvadata_t p = bufferIn[fetchIdx];
622  int outputIdx = kIndex(kx, ky, f, widthOut, heightOut, numBands);
623  bufferOut[outputIdx] += xcoeff * ycoeff * p;
624  }
625  }
626  }
627  }
628  }
629  return PV_SUCCESS;

776 //Apply normalizeLuminanceFlag, normalizeStdDev, and inverseFlag, which can be done pixel-by-pixel
777 //after scattering.
778 int BaseInput::postProcess(double timef, double dt){
779  int numExtended = getNumExtended();
780 
781  // if normalizeLuminanceFlag == true:
782  // if normalizeStdDev is true, then scale so that average luminance to be 0 and std. dev. of luminance to be 1.
783  // if normalizeStdDev is false, then scale so that minimum is 0 and maximum is 1
784  // if normalizeLuminanceFlag == true and the image in buffer is completely flat, force all values to zero
785  for(int b = 0; b < parent->getNBatch(); b++){
786  float* buf = data + b * numExtended;
787  if(normalizeLuminanceFlag){
788  if (normalizeStdDev){
789  double image_sum = 0.0f;
790  double image_sum2 = 0.0f;
791  for (int k=0; k<numExtended; k++) {
792  image_sum += buf[k];
793  image_sum2 += buf[k]*buf[k];
794  }
795  double image_ave = image_sum / numExtended;
796  double image_ave2 = image_sum2 / numExtended;
797 #ifdef PV_USE_MPI
798  MPI_Allreduce(MPI_IN_PLACE, &image_ave, 1, MPI_DOUBLE, MPI_SUM, parent->icCommunicator()->communicator());
799  image_ave /= parent->icCommunicator()->commSize();
800  MPI_Allreduce(MPI_IN_PLACE, &image_ave2, 1, MPI_DOUBLE, MPI_SUM, parent->icCommunicator()->communicator());
801  image_ave2 /= parent->icCommunicator()->commSize();
802 #endif // PV_USE_MPI
803  // set mean to zero
804  for (int k=0; k<numExtended; k++) {
805  buf[k] -= image_ave;
806  }
807  // set std dev to 1
808  double image_std = sqrt(image_ave2 - image_ave*image_ave); // std = 1/N * sum((x[i]-sum(x[i])^2) ~ 1/N * sum(x[i]^2) - (sum(x[i])/N)^2 | Note: this permits running std w/o needing to first calc avg (although we already have avg)
809  if(image_std == 0){
810  for (int k=0; k<numExtended; k++) {
811  buf[k] = 0.0;
812  }
813  }
814  else{
815  for (int k=0; k<numExtended; k++) {
816  buf[k] /= image_std;
817  }
818  }
819  }
820  else{
821  float image_max = -FLT_MAX;
822  float image_min = FLT_MAX;
823  for (int k=0; k<numExtended; k++) {
824  image_max = buf[k] > image_max ? buf[k] : image_max;
825  image_min = buf[k] < image_min ? buf[k] : image_min;
826  }
827  MPI_Allreduce(MPI_IN_PLACE, &image_max, 1, MPI_FLOAT, MPI_MAX, parent->icCommunicator()->communicator());
828  MPI_Allreduce(MPI_IN_PLACE, &image_min, 1, MPI_FLOAT, MPI_MIN, parent->icCommunicator()->communicator());
829  if (image_max > image_min){
830  float image_stretch = 1.0f / (image_max - image_min);
831  for (int k=0; k<numExtended; k++) {
832  buf[k] -= image_min;
833  buf[k] *= image_stretch;
834  }
835  }
836  else{ // image_max == image_min, set to gray
837  for (int k=0; k<numExtended; k++) {
838  buf[k] = 0.0f;
839  }
840  }
841  }
842  } // normalizeLuminanceFlag
843  if( inverseFlag ) {
844  for (int k=0; k<numExtended; k++) {
845  buf[k] = 1 - buf[k]; // If normalizeLuminanceFlag is true, should the effect of inverseFlag be buf[k] = -buf[k]?
846  }
847  }
848  }
849  return PV_SUCCESS;
