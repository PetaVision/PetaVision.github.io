405 int BaseInput::postProcess(double timef, double dt){
406  int numExtended = getNumExtended();
407 
408  // if normalizeLuminanceFlag == true:
409  // if normalizeStdDev is true, then scale so that average luminance to be 0 and std. dev. of luminance to be 1.
410  // if normalizeStdDev is false, then scale so that minimum is 0 and maximum is 1
411  for(int b = 0; b < parent->getNBatch(); b++){
412  float* buf = data + b * numExtended;
413  if(normalizeLuminanceFlag){
414  if (normalizeStdDev){
415  double image_sum = 0.0f;
416  double image_sum2 = 0.0f;
417  for (int k=0; k<numExtended; k++) {
418  image_sum += buf[k];
419  image_sum2 += buf[k]*buf[k];
420  }
421  double image_ave = image_sum / numExtended;
422  double image_ave2 = image_sum2 / numExtended;
423 #ifdef PV_USE_MPI
424  MPI_Allreduce(MPI_IN_PLACE, &image_ave, 1, MPI_DOUBLE, MPI_SUM, parent->icCommunicator()->communicator());
425  image_ave /= parent->icCommunicator()->commSize();
426  MPI_Allreduce(MPI_IN_PLACE, &image_ave2, 1, MPI_DOUBLE, MPI_SUM, parent->icCommunicator()->communicator());
427  image_ave2 /= parent->icCommunicator()->commSize();
428 #endif // PV_USE_MPI
429  // set mean to zero
430  for (int k=0; k<numExtended; k++) {
431  buf[k] -= image_ave;
432  }
433  // set std dev to 1
434  double image_std = sqrt(image_ave2 - image_ave*image_ave); // std = 1/N * sum((x[i]-sum(x[i])^2) ~ 1/N * sum(x[i]^2) - (sum(x[i])/N)^2 | Note: this permits running std w/o needing to first calc avg (although we already have avg)
435  if(image_std == 0){
436  for (int k=0; k<numExtended; k++) {
437  buf[k] = .5;
438  }
439  }
440  else{
441  for (int k=0; k<numExtended; k++) {
442  buf[k] /= image_std;
443  }
444  }
445  }
446  else{
447  float image_max = -FLT_MAX;
448  float image_min = FLT_MAX;
449  for (int k=0; k<numExtended; k++) {
450  image_max = buf[k] > image_max ? buf[k] : image_max;
451  image_min = buf[k] < image_min ? buf[k] : image_min;
452  }
453  MPI_Allreduce(MPI_IN_PLACE, &image_max, 1, MPI_FLOAT, MPI_MAX, parent->icCommunicator()->communicator());
454  MPI_Allreduce(MPI_IN_PLACE, &image_min, 1, MPI_FLOAT, MPI_MIN, parent->icCommunicator()->communicator());
455  if (image_max > image_min){
456  float image_stretch = 1.0f / (image_max - image_min);
457  for (int k=0; k<numExtended; k++) {
458  buf[k] -= image_min;
459  buf[k] *= image_stretch;
460  }
461  }
462  else{ // image_max == image_min, set to gray
463  //float image_shift = 0.5f - image_ave;
464  for (int k=0; k<numExtended; k++) {
465  buf[k] = 0.5f; //image_shift;
466  }
467  }
468  }
469  } // normalizeLuminanceFlag
470  if( inverseFlag ) {
471  for (int k=0; k<numExtended; k++) {
472  buf[k] = 1 - buf[k];
473  }
474  }
475  }
476  return PV_SUCCESS;
477 }

560 int BaseInput::calcBias(int current_bias, int step, int sizeLength)
561 {
562  assert(jitterFlag);
563  double p;
564  int dbias = 0;
565  if (jitterType == RANDOM_WALK) {
566  p = randState->uniformRandom();
567  dbias = p < 0.5 ? step : -step;
568  } else if (jitterType == RANDOM_JUMP) {
569  p = randState->uniformRandom();
570  dbias = (int) floor(p*(double) step) + 1;
571  p = randState->uniformRandom();
572  if (p < 0.5) dbias = -dbias;
573  }
574  else {
575  assert(0); // Only allowable values of jitterType are RANDOM_WALK and RANDOM_JUMP
576  }
577 
578  int new_bias = current_bias + dbias;
579  new_bias = (new_bias < 0) ? -new_bias : new_bias;
580  new_bias = (new_bias > sizeLength) ? sizeLength - (new_bias-sizeLength) : new_bias;
581  return new_bias;
582 }

584 int BaseInput::calcNewBiases(int stepSize) {
585  assert(jitterFlag);
586  int step_radius = 0; // distance to step
587  switch (jitterType) {
588  case RANDOM_WALK:
589  step_radius = stepSize;
590  break;
591  case RANDOM_JUMP:
592  step_radius = 1 + (int) floor(randState->uniformRandom() * stepSize);
593  break;
594  default:
595  assert(0); // Only allowable values of jitterType are RANDOM_WALK and RANDOM_JUMP
596  break;
597  }
598  double p = randState->uniformRandom() * 2 * PI; // direction to step
599  int dx = (int) floor( step_radius * cos(p));
600  int dy = (int) floor( step_radius * sin(p));
601  assert(dx != 0 || dy != 0);

629 bool BaseInput::calcNewOffsets(int stepSize)
630 {
631  assert(jitterFlag);
632 
633  bool needNewImage = false;
634  double p = randState->uniformRandom();
635  if (timeSinceLastJitter >= jitterRefractoryPeriod) {
636  if (p > recurrenceProb) {
637  p = randState->uniformRandom();
638  if (p > persistenceProb) {
639  needNewImage = true;
640  int step_radius = 1 + (int) floor(randState->uniformRandom() * stepSize);
641  double p = randState->uniformRandom() * 2 * PI; // direction to step
642  int dx = (int) round( step_radius * cos(p));
643  int dy = (int) round( step_radius * sin(p));
644  assert(dx != 0 || dy != 0);
645  offsets[0] += dx;
646  offsets[1] += dy;
647  timeSinceLastJitter = 0;
648  }
649  }
650  else {
651  assert(sizeof(*offsets) == sizeof(*biases));
652  memcpy(offsets, biases, 2*sizeof(offsets));
653  timeSinceLastJitter = 0;
654  }
655  }
656  timeSinceLastJitter++;
657  return needNewImage;

660 bool BaseInput::constrainPoint(int * point, int min_x, int max_x, int min_y, int max_y, int method) {
661  bool moved_x = point[0] < min_x || point[0] > max_x;
662  bool moved_y = point[1] < min_y || point[1] > max_y;
663  if (moved_x) {
664  if (min_x > max_x) {
665  fprintf(stderr, "Image::constrainPoint error. min_x=%d is greater than max_x= %d\n", min_x, max_x);
666  exit(EXIT_FAILURE);
667  }
668  int size_x = max_x-min_x;
669  int new_x = point[0];
670  switch (method) {
671  case 0: // Ignore
672  break;
673  case 1: // Mirror
674  new_x -= min_x;
675  new_x %= (2*(size_x+1));
676  if (new_x<0) new_x++;
677  new_x = abs(new_x);
678  if (new_x>size_x) new_x = 2*size_x+1-new_x;
679  new_x += min_x;
680  break;
681  case 2: // Stick to wall
682  if (new_x<min_x) new_x = min_x;
683  if (new_x>max_x) new_x = max_x;
684  break;
685  case 3: // Circular
686  new_x -= min_x;
687  new_x %= size_x+1;
688  if (new_x<0) new_x += size_x+1;
689  new_x += min_x;
690  break;
691  default:
692  std::cout << "Method type " << method << " not understood\n";
693  assert(0);
694  break;
695  }
696  assert(new_x >= min_x && new_x <= max_x);
697  point[0] = new_x;
698  }
699  if (moved_y) {
700  if (min_y > max_y) {
701  fprintf(stderr, "Image::constrainPoint error. min_y=%d is greater than max_y=%d\n", min_y, max_y);
702  exit(EXIT_FAILURE);
703  }
704  int size_y = max_y-min_y;
705  int new_y = point[1];
706  switch (method) {
707  case 0: // Ignore
708  break;
709  case 1: // Mirror
710  new_y -= min_y;
711  new_y %= (2*(size_y+1));
712  if (new_y<0) new_y++;
713  new_y = abs(new_y);
714  if (new_y>size_y) new_y = 2*size_y+1-new_y;
715  new_y += min_y;
716  break;
717  case 2: // Stick to wall
718  if (new_y<min_y) new_y = min_y;
719  if (new_y>max_y) new_y = max_y;
720  break;
721  case 3: // Circular
722  new_y -= min_y;
723  new_y %= size_y+1;
724  if (new_y<0) new_y += size_y+1;
725  new_y += min_y;
726  break;
727  default:
728  assert(0);
729  break;
730  }
731  assert(new_y >= min_y && new_y <= max_y);
732  point[1] = new_y;
733  }
734  return moved_x || moved_y;
735 }
