134 int HyPerCol::initialize(const char *name, PV_Init *initObj) {
135  mPVInitObj = initObj;
136  mCommunicator = mPVInitObj->getCommunicator();
137  mParams = mPVInitObj->getParams();
138  if (mParams == nullptr) {
139  if (mCommunicator->globalCommRank() == 0) {
140  ErrorLog() << "HyPerCol::initialize: params have not been set." << std::endl;
141  MPI_Barrier(mCommunicator->communicator());
142  }
143  exit(EXIT_FAILURE);
144  }
145  std::string working_dir = mPVInitObj->getStringArgument("WorkingDirectory");
146  working_dir = expandLeadingTilde(working_dir);
147 
148  // Sep 27, 2016: handling --require-return has been moved to the Communicator
149  // constructor.
150 
151  mName = strdup(name);
152  setDescription();
153 
154  // mNumThreads will not be set, or used until HyPerCol::run.
155  // This means that threading cannot happen in the initialization or
156  // communicateInitInfo stages,
157  // but that should not be a problem.
158  char const *programName = mPVInitObj->getProgramName();
159 
160  if (columnId() == 0 && !working_dir.empty()) {
161  int status = chdir(working_dir.c_str());
162  if (status) {
163  Fatal(chdirMessage);
164  chdirMessage.printf("Unable to switch directory to \"%s\"\n", working_dir.c_str());
165  chdirMessage.printf("chdir error: %s\n", strerror(errno));
166  }
167  }
168 
169 #ifdef PV_USE_MPI // Fail if there was a parsing error, but make sure nonroot
170  // processes don't kill
171  // the root process before the root process reaches the syntax error
172  int parsedStatus;
173  int rootproc = 0;
174  if (globalRank() == rootproc) {
175  parsedStatus = this->mParams->getParseStatus();
176  }
177  MPI_Bcast(&parsedStatus, 1, MPI_INT, rootproc, getCommunicator()->globalCommunicator());
178 #else
179  int parsedStatus = this->mParams->getParseStatus();
180 #endif
181  if (parsedStatus != 0) {
182  exit(parsedStatus);
183  }
184 
185  mRandomSeed = mPVInitObj->getUnsignedIntArgument("RandomSeed");
186 
187  mCheckpointer = new Checkpointer(
188  std::string(mName), mCommunicator->getGlobalMPIBlock(), mPVInitObj->getArguments());
189  mCheckpointer->addObserver(this, BaseMessage{});
190  ioParams(PARAMS_IO_READ);
191  mSimTime = mStartTime;
192  mInitialStep = (long int)nearbyint(mStartTime / mDeltaTime);
193  mCurrentStep = mInitialStep;
194  mFinalStep = (long int)nearbyint(mStopTime / mDeltaTime);
195  mCheckpointer->provideFinalStep(mFinalStep);
196  mNextProgressTime = mStartTime;
197 
198  RandomSeed::instance()->initialize(mRandomSeed);
199 
200  mRunTimer = new Timer(mName, "column", "run ");
201  mCheckpointer->registerTimer(mRunTimer);
202  mCheckpointer->registerCheckpointData(
203  mName,
204  "nextProgressTime",
205  &mNextProgressTime,
206  (std::size_t)1,
207  true /*broadcast*/,
208  false /*not constant*/);
209 
210  mCheckpointReadFlag = !mCheckpointer->getCheckpointReadDirectory().empty();
211  return PV_SUCCESS;
212 }

316 void HyPerCol::ioParam_randomSeed(enum ParamsIOFlag ioFlag) {
317  switch (ioFlag) {
318  // randomSeed can be set on the command line, from the params file, or from
319  // the system clock
320  case PARAMS_IO_READ:
321  // set random seed if it wasn't set in the command line
322  if (!mRandomSeed) {
323  if (mParams->present(mName, "randomSeed")) {
324  mRandomSeed = (unsigned long)mParams->value(mName, "randomSeed");
325  }
326  else {
327  mRandomSeed = seedRandomFromWallClock();
328  }
329  }
330  if (mRandomSeed < RandomSeed::minSeed) {
331  Fatal().printf(
332  "Error: random seed %u is too small. Use a seed of at "
333  "least 10000000.\n",
334  mRandomSeed);
335  }
336  break;
337  case PARAMS_IO_WRITE: parameters()->writeParam("randomSeed", mRandomSeed); break;
338  default: assert(0); break;
339  }
340 }

392 void HyPerCol::allocateColumn() {
393  if (mReadyFlag) {
394  return;
395  }
396 
397 #ifdef PV_USE_CUDA
398  // Default to auto assign gpus
399  std::string const &gpu_devices = mPVInitObj->getStringArgument("GPUDevices");
400  initializeCUDA(gpu_devices);
401 #endif
402 
403  setNumThreads(false);
404  // When we call processParams, the communicateInitInfo stage will run, which
405  // can put out a lot of messages.
406  // So if there's a problem with the -t option setting, the error message can
407  // be hard to find.
408  // Instead of printing the error messages here, we will call setNumThreads a
409  // second time after processParams(), and only then print messages.
410 
411  // processParams function does communicateInitInfo stage, sets up adaptive
412  // time step, and prints params
413  pvAssert(mPrintParamsFilename && mPrintParamsFilename[0]);
414  if (mPrintParamsFilename[0] != '/') {
415  std::string printParamsFilename(mPrintParamsFilename);
416  std::string printParamsPath = mCheckpointer->makeOutputPathFilename(printParamsFilename);
417  processParams(printParamsPath.c_str());
418  }
419  else {
420  // If using absolute path, only global rank 0 writes, to avoid collisions.
421  if (mCheckpointer->getMPIBlock()->getGlobalRank() == 0) {
422  processParams(mPrintParamsFilename);
423  }
424  }
425 
426  int thread_status =
427  setNumThreads(true /*now, print messages related to setting number of threads*/);
428  MPI_Barrier(mCommunicator->globalCommunicator());
429  if (thread_status != PV_SUCCESS) {
430  exit(EXIT_FAILURE);
431  }
432 
433 #ifdef PV_USE_OPENMP_THREADS
434  pvAssert(mNumThreads > 0); // setNumThreads should fail if it sets
435  // mNumThreads less than or equal to zero
436  omp_set_num_threads(mNumThreads);
437 #endif // PV_USE_OPENMP_THREADS
438 
439  notify(std::make_shared<AllocateDataMessage>());
440 
441  mPhaseRecvTimers.clear();
442  for (int phase = 0; phase < mNumPhases; phase++) {
443  std::string timerTypeString("phRecv");
444  timerTypeString.append(std::to_string(phase));
445  Timer *phaseRecvTimer = new Timer(mName, "column", timerTypeString.c_str());
446  mPhaseRecvTimers.push_back(phaseRecvTimer);
447  mCheckpointer->registerTimer(phaseRecvTimer);
448  }
449 
450  notify(std::make_shared<RegisterDataMessage<Checkpointer>>(mCheckpointer));
451 
452 #ifdef DEBUG_OUTPUT
453  InfoLog().printf("[%d]: HyPerCol: running...\n", mCommunicator->globalCommRank());
454  InfoLog().flush();
455 #endif
456 
457  // Initialize the state of each object based on the params file,
458  // and then if reading from checkpoint, call the checkpointer.
459  // This needs to happen after initPublishers so that we can initialize
460  // the values in the data stores, and before the mLayers' publish calls
461  // so that the data in border regions gets copied correctly.
462  notify(std::make_shared<InitializeStateMessage>());
463  if (mCheckpointReadFlag) {
464  mCheckpointer->checkpointRead(&mSimTime, &mCurrentStep);
465  }
466  else {
467  mCheckpointer->readStateFromCheckpoint();
468  // readStateFromCheckpoint() does nothing if initializeFromCheckpointDir is empty or null.
469  }
470 // Note: ideally, if checkpointReadFlag is set, calling InitializeState should
471 // be unnecessary. However, currently initializeState does some CUDA kernel
472 // initializations that still need to happen when reading from checkpoint.
473 
474 #ifdef PV_USE_CUDA
475  notify(std::make_shared<CopyInitialStateToGPUMessage>());
476 #endif // PV_USE_CUDA
477 
478  // Initial normalization moved here to facilitate normalizations of groups
479  // of HyPerConns
480  normalizeWeights();
481  notify(std::make_shared<ConnectionFinalizeUpdateMessage>(mSimTime, mDeltaTime));
482 
483  // publish initial conditions
484  for (int phase = 0; phase < mNumPhases; phase++) {
485  notify(std::make_shared<LayerPublishMessage>(phase, mSimTime));
486  }
487 
488  // Feb 2, 2017: waiting and updating active indices have been moved into
489  // OutputState and CheckNotANumber, where they are called if needed.
490 
491  // output initial conditions
492  if (!mCheckpointReadFlag) {
493  notify(std::make_shared<ConnectionOutputMessage>(mSimTime));
494  for (int phase = 0; phase < mNumPhases; phase++) {
495  notify(std::make_shared<LayerOutputStateMessage>(phase, mSimTime));
496  }
497  }
498  mReadyFlag = true;
499 }

501 // typically called by buildandrun via HyPerCol::run()
502 int HyPerCol::run(double start_time, double stop_time, double dt) {
503  mStartTime = start_time;
504  mStopTime = stop_time;
505  mDeltaTime = dt;
506 
507  allocateColumn();
508 
509  bool dryRunFlag = mPVInitObj->getBooleanArgument("DryRun");
510  if (dryRunFlag) {
511  return PV_SUCCESS;
512  }
513 
514 #ifdef TIMER_ON
515  Clock runClock;
516  runClock.start_clock();
517 #endif
518 
519  advanceTimeLoop(runClock, 10 /*runClockStartingStep*/);
520 
521  notify(std::make_shared<CleanupMessage>());
522 
523 #ifdef DEBUG_OUTPUT
524  InfoLog().printf("[%d]: HyPerCol: done...\n", mCommunicator->globalCommRank());
525  InfoLog().flush();
526 #endif
527 
528  mCheckpointer->finalCheckpoint(mSimTime);
529 
530 #ifdef TIMER_ON
531  runClock.stop_clock();
532  runClock.print_elapsed(getOutputStream());
533 #endif
534 
535  return PV_SUCCESS;
536 }

542 // This routine sets the mNumThreads member variable. It should only be called
543 // by the run() method,
544 // and only inside the !ready if-statement.
545 // TODO: Instead of using the printMessagesFlag, why not use the same flag that
546 int HyPerCol::setNumThreads(bool printMessagesFlag) {
547  bool printMsgs0 = printMessagesFlag && globalRank() == 0;
548  int thread_status = PV_SUCCESS;
549  int num_threads = 0;
550 #ifdef PV_USE_OPENMP_THREADS
551  int max_threads = mPVInitObj->getMaxThreads();
552  int comm_size = mCommunicator->globalCommSize();
553  if (printMsgs0) {
554  InfoLog().printf(
555  "Maximum number of OpenMP threads%s is %d\n"
556  "Number of MPI processes is %d.\n",
557  comm_size == 1 ? "" : " (over all processes)",
558  max_threads,
559  comm_size);
560  }
561  Configuration::IntOptional numThreadsArg = mPVInitObj->getIntOptionalArgument("NumThreads");
562  if (numThreadsArg.mUseDefault) {
563  num_threads = max_threads / comm_size; // integer arithmetic
564  if (num_threads == 0) {
565  num_threads = 1;
566  if (printMsgs0) {
567  WarnLog().printf(
568  "Warning: more MPI processes than available threads. "
569  "Processors may be oversubscribed.\n");
570  }
571  }
572  }
573  else {
574  num_threads = numThreadsArg.mValue;
575  }
576  if (num_threads > 0) {
577  if (printMsgs0) {
578  InfoLog().printf("Number of threads used is %d\n", num_threads);
579  }
580  }
581  else if (num_threads == 0) {
582  thread_status = PV_FAILURE;
583  if (printMsgs0) {
584  ErrorLog().printf(
585  "%s: number of threads must be positive (was set to zero)\n",
586  mPVInitObj->getProgramName());
587  }
588  }
589  else {
590  assert(num_threads < 0);
591  thread_status = PV_FAILURE;
592  if (printMsgs0) {
593  ErrorLog().printf(
594  "%s was compiled with PV_USE_OPENMP_THREADS; "
595  "therefore the \"-t\" argument is "
596  "required.\n",
597  mPVInitObj->getProgramName());
598  }
599  }
600 #else // PV_USE_OPENMP_THREADS
601  Configuration::IntOptional numThreadsArg = mPVInitObj->getIntOptionalArgument("NumThreads");
602  if (numThreadsArg.mUseDefault) {
603  num_threads = 1;
604  if (printMsgs0) {
605  InfoLog().printf("Number of threads used is 1 (Compiled without OpenMP.\n");
606  }
607  }
608  else {
609  num_threads = numThreadsArg.mValue;
610  if (num_threads < 0) {
611  num_threads = 1;
612  }
613  if (num_threads != 1) {
614  thread_status = PV_FAILURE;
615  }
616  }
617  if (printMsgs0) {
618  if (thread_status != PV_SUCCESS) {
619  ErrorLog().printf(
620  "%s error: PetaVision must be compiled with "
621  "OpenMP to run with threads.\n",
622  mPVInitObj->getProgramName());
623  }
624  }
625 #endif // PV_USE_OPENMP_THREADS
626  mNumThreads = num_threads;
627  return thread_status;
628 }

654 int HyPerCol::normalizeWeights() {
655  int status = PV_SUCCESS;
656  for (int n = 0; n < mNormalizers.size(); n++) {

658  if (normalizer) {

690 int HyPerCol::advanceTime(double sim_time) {
691  if (mSimTime >= mNextProgressTime) {
692  mNextProgressTime += mProgressInterval;
693  if (mCommunicator->globalCommRank() == 0) {
694  std::ostream &progressStream = mWriteProgressToErr ? getErrorStream() : getOutputStream();
695  time_t current_time;
696  time(&current_time);
697  progressStream << " time==" << sim_time << " "
698  << ctime(&current_time); // ctime outputs an newline
699  }
700  }
701 
702  mRunTimer->start();
703 
704  // make sure mSimTime is updated even if HyPerCol isn't running time loop
705  // triggerOffset might fail if mSimTime does not advance uniformly because
706  // mSimTime could skip over tigger event
707  // !!!TODO: fix trigger layer to compute mTimeScale so as not to allow
708  // bypassing trigger event
709  mSimTime = sim_time + mDeltaTime;
710 
711  notify(std::make_shared<AdaptTimestepMessage>());
712  // Sep 26, 2016: Adaptive timestep routines and member variables have been
713  // moved to
714  // AdaptiveTimeScaleProbe.
715 
716  // At this point all activity from the previous time step has
717  // been delivered to the data store.
718  //
719 
720  int status = PV_SUCCESS;
721 
722  // update the connections (weights)
723  //
724  notify(std::make_shared<ConnectionUpdateMessage>(mSimTime, mDeltaTime));
725  normalizeWeights();
726  notify(std::make_shared<ConnectionFinalizeUpdateMessage>(mSimTime, mDeltaTime));
727  notify(std::make_shared<ConnectionOutputMessage>(mSimTime));
728 
729  // Each layer's phase establishes a priority for updating
730  for (int phase = 0; phase < mNumPhases; phase++) {
731  for (auto &l : mLayers) { // TODO: use notify/respond
732  l->clearProgressFlags();
733  }
734 
735 // nonblockingLayerUpdate allows for more concurrency than notify.
736 #ifdef PV_USE_CUDA
737  // Ordering needs to go recvGpu, if(recvGpu and upGpu)update, recvNoGpu,
738  // update rest
739  auto recvMessage = std::make_shared<LayerRecvSynapticInputMessage>(
740  phase, mPhaseRecvTimers.at(phase), true /*recvGpuFlag*/, mSimTime, mDeltaTime);
741  auto updateMessage = std::make_shared<LayerUpdateStateMessage>(
742  phase, true /*recvGpuFlag*/, true /*updateGpuFlag*/, mSimTime, mDeltaTime);
743  nonblockingLayerUpdate(recvMessage, updateMessage);
744 
745  recvMessage = std::make_shared<LayerRecvSynapticInputMessage>(
746  phase, mPhaseRecvTimers.at(phase), false /*recvGpuFlag*/, mSimTime, mDeltaTime);
747  updateMessage = std::make_shared<LayerUpdateStateMessage>(
748  phase, false /*recvGpuFlag*/, false /*updateGpuFlag*/, mSimTime, mDeltaTime);
749  nonblockingLayerUpdate(recvMessage, updateMessage);
750 
751  getDevice()->syncDevice();
752 
753  // Update for receiving on cpu and updating on gpu
754  notify(
755  std::make_shared<LayerUpdateStateMessage>(
756  phase, false /*recvOnGpuFlag*/, true /*updateOnGpuFlag*/, mSimTime, mDeltaTime));
757 
758  getDevice()->syncDevice();
759  notify(std::make_shared<LayerCopyFromGpuMessage>(phase, mPhaseRecvTimers.at(phase)));
760 
761  // Update for gpu recv and non gpu update
762  notify(
763  std::make_shared<LayerUpdateStateMessage>(
764  phase, true /*recvOnGpuFlag*/, false /*updateOnGpuFlag*/, mSimTime, mDeltaTime));
765 #else
766  auto recvMessage = std::make_shared<LayerRecvSynapticInputMessage>(
767  phase, mPhaseRecvTimers.at(phase), mSimTime, mDeltaTime);
768  auto updateMessage = std::make_shared<LayerUpdateStateMessage>(phase, mSimTime, mDeltaTime);
769  nonblockingLayerUpdate(recvMessage, updateMessage);
770 #endif
771  // Rotate DataStore ring buffers
772  notify(std::make_shared<LayerAdvanceDataStoreMessage>(phase));
773 
774  // copy activity buffer to DataStore, and do MPI exchange.
775  notify(std::make_shared<LayerPublishMessage>(phase, mSimTime));
776 
777  // Feb 2, 2017: waiting and updating active indices have been moved into
778  // OutputState and CheckNotANumber, where they are called if needed.
779  notify(std::make_shared<LayerOutputStateMessage>(phase, mSimTime));
780  if (mErrorOnNotANumber) {
781  notify(std::make_shared<LayerCheckNotANumberMessage>(phase));
782  }
783  }
784 
785  mRunTimer->stop();
786 
787  outputState(mSimTime);
788 
789  return status;
790 }

792 void HyPerCol::nonblockingLayerUpdate(
793  std::shared_ptr<LayerRecvSynapticInputMessage const> recvMessage,
794  std::shared_ptr<LayerUpdateStateMessage const> updateMessage) {
795  long int idleCounter = 0;
796  bool allUpdated = false;
797  int phase = recvMessage->mPhase;
798  pvAssert(phase == updateMessage->mPhase);
799  while (!allUpdated) {
800  bool anyUpdated = false;
801 
802  for (auto &l : mLayers) {
803 #ifdef PV_USE_CUDA
804  if (l->getRecvGpu() != recvMessage->mRecvOnGpuFlag) {
805  continue;
806  }
807 #endif // PV_USE_CUDA
808  if (l->getPhase() == phase && !l->getHasReceived() && l->isAllInputReady()) {
809  l->respond(recvMessage);
810  anyUpdated = true;
811  break;
812  }
813  }
814  if (anyUpdated) {
815  continue;
816  }
817 
818  for (auto &l : mLayers) {
819 #ifdef PV_USE_CUDA
820  if (l->getRecvGpu() != updateMessage->mRecvOnGpuFlag) {
821  continue;
822  }
823  if (l->getUpdateGpu() != updateMessage->mUpdateOnGpuFlag) {
824  continue;
825  }
826 #endif // PV_USE_CUDA
827  if (l->getPhase() == phase && !l->getHasUpdated() && l->getHasReceived()) {
828  l->respond(updateMessage);
829  anyUpdated = true;
830  break;
831  }
832  }
833  if (anyUpdated) {
834  continue;
835  }
836 
837  idleCounter++;
838  // TODO (maybe?) Waitany logic goes here.
839  allUpdated = true;
840  for (auto &l : mLayers) {
841 #ifdef PV_USE_CUDA
842  if (l->getRecvGpu() != updateMessage->mRecvOnGpuFlag) {
843  continue;
844  }
845  if (l->getUpdateGpu() != updateMessage->mUpdateOnGpuFlag) {
846  continue;
847  }
848 #endif // PV_USE_CUDA
849  if (l->getPhase() == phase) {
850  allUpdated &= l->getHasUpdated();
851  }
852  }
853  }
854  if (idleCounter > 1L) {
855  InfoLog() << "t = " << mSimTime << ", phase " << phase << ", idle count " << idleCounter
856  << "\n";
857  }
858 }

1046 int HyPerCol::getAutoGPUDevice() {
1047  int returnGpuIdx = -1;
1048 #ifdef PV_USE_CUDA
1049  int mpiRank = mCommunicator->globalCommRank();
1050  int numMpi = mCommunicator->globalCommSize();
1051  char hostNameStr[PV_PATH_MAX];
1052  gethostname(hostNameStr, PV_PATH_MAX);
1053  size_t hostNameLen = strlen(hostNameStr) + 1; //+1 for null terminator
1054 
1055  // Each rank communicates which host it is on
1056  // Root process
1057  if (mpiRank == 0) {
1058  // Allocate data structure for rank to host
1059  char rankToHost[numMpi][PV_PATH_MAX];
1060  assert(rankToHost);
1061  // Allocate data structure for rank to maxGpu
1062  int rankToMaxGpu[numMpi];
1063  // Allocate final data structure for rank to GPU index
1064  int rankToGpu[numMpi];
1065  assert(rankToGpu);
1066 
1067  for (int rank = 0; rank < numMpi; rank++) {
1068  if (rank == 0) {
1069  strcpy(rankToHost[rank], hostNameStr);
1070  rankToMaxGpu[rank] = PVCuda::CudaDevice::getNumDevices();
1071  }
1072  else {
1073  MPI_Recv(
1074  rankToHost[rank],
1075  PV_PATH_MAX,
1076  MPI_CHAR,
1077  rank,
1078  0,
1079  mCommunicator->globalCommunicator(),
1080  MPI_STATUS_IGNORE);
1081  MPI_Recv(
1082  &(rankToMaxGpu[rank]),
1083  1,
1084  MPI_INT,
1085  rank,
1086  0,
1087  mCommunicator->globalCommunicator(),
1088  MPI_STATUS_IGNORE);
1089  }
1090  }
1091 
1092  // rankToHost now is an array such that the index is the rank, and the value
1093  // is the host
1094  // Convert to a map of vectors, such that the key is the host name and the
1095  // value
1096  // is a vector of mpi ranks that is running on that host
1097  std::map<std::string, std::vector<int>> hostMap;
1098  for (int rank = 0; rank < numMpi; rank++) {
1099  hostMap[std::string(rankToHost[rank])].push_back(rank);
1100  }
1101 
1102  // Determine what gpus to use per mpi
1103  for (auto &host : hostMap) {
1104  std::vector<int> rankVec = host.second;
1105  int numRanksPerHost = rankVec.size();
1106  assert(numRanksPerHost > 0);
1107  // Grab maxGpus of current host
1108  int maxGpus = rankToMaxGpu[rankVec[0]];
1109  // Warnings for overloading/underloading gpus
1110  if (numRanksPerHost != maxGpus) {
1111  WarnLog(assignGpuWarning);
1112  assignGpuWarning.printf(
1113  "HyPerCol::getAutoGPUDevice: Host \"%s\" (rank[s] ", host.first.c_str());
1114  for (int v_i = 0; v_i < numRanksPerHost; v_i++) {
1115  if (v_i != numRanksPerHost - 1) {
1116  assignGpuWarning.printf("%d, ", rankVec[v_i]);
1117  }
1118  else {
1119  assignGpuWarning.printf("%d", rankVec[v_i]);
1120  }
1121  }
1122  assignGpuWarning.printf(
1123  ") is being %s, with %d mpi processes mapped to %d total GPU[s]\n",
1124  numRanksPerHost < maxGpus ? "underloaded" : "overloaded",
1125  numRanksPerHost,
1126  maxGpus);
1127  }
1128 
1129  // Match a rank to a gpu
1130  for (int v_i = 0; v_i < numRanksPerHost; v_i++) {
1131  rankToGpu[rankVec[v_i]] = v_i % maxGpus;
1132  }
1133  }
1134 
1135  // MPI sends to each process to specify which gpu the rank should use
1136  for (int rank = 0; rank < numMpi; rank++) {
1137  InfoLog() << "Rank " << rank << " on host \"" << rankToHost[rank] << "\" ("
1138  << rankToMaxGpu[rank] << " GPU[s]) using GPU index " << rankToGpu[rank] << "\n";
1139  if (rank == 0) {
1140  returnGpuIdx = rankToGpu[rank];
1141  }
1142  else {
1143  MPI_Send(&(rankToGpu[rank]), 1, MPI_INT, rank, 0, mCommunicator->globalCommunicator());
1144  }
1145  }
1146  }
1147  // Non root process
1148  else {
1149  // Send host name
1150  MPI_Send(hostNameStr, hostNameLen, MPI_CHAR, 0, 0, mCommunicator->globalCommunicator());
1151  // Send max gpus for that host
1152  int maxGpu = PVCuda::CudaDevice::getNumDevices();
1153  MPI_Send(&maxGpu, 1, MPI_INT, 0, 0, mCommunicator->globalCommunicator());
1154  // Recv gpu idx
1155  MPI_Recv(
1156  &(returnGpuIdx),
1157  1,
1158  MPI_INT,
1159  0,
1160  0,
1161  mCommunicator->globalCommunicator(),
1162  MPI_STATUS_IGNORE);
1163  }
1164  assert(returnGpuIdx >= 0 && returnGpuIdx < PVCuda::CudaDevice::getNumDevices());
1165 #else
1166  // This function should never be called when not running with GPUs
1167  assert(false);
1168 #endif
1169  return returnGpuIdx;
1170 }
