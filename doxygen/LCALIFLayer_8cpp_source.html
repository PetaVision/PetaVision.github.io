107 int LCALIFLayer::initialize(const char * name, HyPerCol * hc, const char * kernel_name){
108  LIFGap::initialize(name, hc, kernel_name);
109  PVParams * params = hc->parameters();
110 
111  float defaultDynVthScale = lParams.VthRest-lParams.Vrest;
112  Vscale = defaultDynVthScale > 0 ? defaultDynVthScale : DEFAULT_DYNVTHSCALE;
113  if (Vscale <= 0) {
114  if (hc->columnId()==0) {
115  pvErrorNoExit().printf("LCALIFLayer \"%s\": Vscale must be positive (value in params is %f).\n", name, Vscale);
116  }
117  abort();
118  }
119 
120  return PV_SUCCESS;
121 }

193 int LCALIFLayer::updateState(double timed, double dt)
194 {
195  //Calculate_state kernel
196  for (int k=0; k<getNumNeuronsAllBatches(); k++) {
197  G_Norm[k] = GSyn[CHANNEL_NORM][k]; // Copy GSyn buffer on normalizing channel for checkpointing, since LCALIF_update_state will blank the GSyn's
198  }
199  LCALIF_update_state(clayer->loc.nbatch, getNumNeurons(), timed, dt, clayer->loc.nx, clayer->loc.ny, clayer->loc.nf,
200  clayer->loc.halo.lt, clayer->loc.halo.rt, clayer->loc.halo.dn, clayer->loc.halo.up, Vscale, Vadpt, tauTHR, targetRateHz, integratedSpikeCount, &lParams,
201  randState->getRNG(0), clayer->V, Vth, G_E, G_I, G_IB, GSyn[0], clayer->activity->data, getGapStrength(), Vattained, Vmeminf, (int) normalizeInputFlag,
202  GSynExcEffective, GSynInhEffective, excitatoryNoise, inhibitoryNoise, inhibNoiseB);
203  return PV_SUCCESS;
204 }
