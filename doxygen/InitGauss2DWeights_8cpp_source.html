59 int InitGauss2DWeights::gauss2DCalcWeights(pvdata_t * dataStart, InitGauss2DWeightsParams * weightParamPtr) {
60 
61  //load necessary params:
62  int nfPatch_tmp = weightParamPtr->getnfPatch();
63  int nyPatch_tmp = weightParamPtr->getnyPatch();
64  int nxPatch_tmp = weightParamPtr->getnxPatch();
65  float strength=weightParamPtr->getStrength();
66  float aspect=weightParamPtr->getAspect();
67  float shift=weightParamPtr->getShift();
68  int numFlanks=weightParamPtr->getNumFlanks();
69  float sigma=weightParamPtr->getSigma();
70  int sx_tmp=weightParamPtr->getsx();
71  int sy_tmp=weightParamPtr->getsy();
72  int sf_tmp=weightParamPtr->getsf();
73  double r2Max=weightParamPtr->getr2Max();
74  double r2Min=weightParamPtr->getr2Min();
75 
76  pvdata_t * w_tmp = dataStart;
77 
78 
79 
80  // loop over all post-synaptic cells in temporary patch
81  for (int fPost = 0; fPost < nfPatch_tmp; fPost++) {
82  float thPost = weightParamPtr->calcThPost(fPost);
83  //TODO: add additional weight factor for difference between thPre and thPost
84  if(weightParamPtr->checkThetaDiff(thPost)) continue;
85  if(weightParamPtr->checkColorDiff(fPost)) continue;
86  for (int jPost = 0; jPost < nyPatch_tmp; jPost++) {
87  float yDelta = weightParamPtr->calcYDelta(jPost);
88  for (int iPost = 0; iPost < nxPatch_tmp; iPost++) {
89  float xDelta = weightParamPtr->calcXDelta(iPost);
90 
91  if(weightParamPtr->isSameLocOrSelf(xDelta, yDelta, fPost)) continue;
92 
93  // rotate the reference frame by th (change sign of thPost?)
94  float xp = +xDelta * cos(thPost) + yDelta * sin(thPost);
95  float yp = -xDelta * sin(thPost) + yDelta * cos(thPost);
96 
97  if(weightParamPtr->checkBowtieAngle(yp, xp)) continue;
98 
99 
100  // include shift to flanks
101  float d2 = xp * xp + (aspect * (yp - shift) * aspect * (yp - shift));
102  int index = iPost * sx_tmp + jPost * sy_tmp + fPost * sf_tmp;
103  w_tmp[index] = 0;
104  if ((d2 <= r2Max) && (d2 >= r2Min)) {
105  w_tmp[index] += strength*exp(-d2 / (2.0f * sigma * sigma));
106  }
107  if (numFlanks > 1) {
108  // shift in opposite direction
109  d2 = xp * xp + (aspect * (yp + shift) * aspect * (yp + shift));
110  if ((d2 <= r2Max) && (d2 >= r2Min)) {
111  w_tmp[index] += strength*exp(-d2 / (2.0f * sigma * sigma));
112  }
113  }
114  }
115  }
116  }
117 
118  return PV_SUCCESS;
119 }
