96 PoolingConn::AccumulateType PoolingConn::parseAccumulateTypeString(char const *poolingTypeString) {
97  if (poolingTypeString == nullptr) {
98  return UNDEFINED;
99  }
100  PoolingConn::AccumulateType accType;
101  std::string str(poolingTypeString);
102  // Convert string to lowercase so that capitalization doesn't matter.
103  for (auto &c : str) {
104  c = std::tolower(c, std::locale());
105  }
106  // "max_pooling", "max pooling", "maxpooling" are equally acceptable (same for
107  // sum and avg)
108  if (str.size() >= 4 && (str[3] == ' ' || str[3] == '_')) {
109  str.erase(3, 1);
110  }
111 
112  if (strcmp(str.c_str(), "maxpooling") == 0) {
113  accType = MAX;
114  }
115  else if (strcmp(str.c_str(), "sumpooling") == 0) {
116  accType = SUM;
117  }
118  else if (strcmp(str.c_str(), "avgpooling") == 0) {
119  accType = AVG;
120  }
121  else {
122  accType = UNDEFINED;
123  }
124  return accType;
125 }

181 int PoolingConn::initialize(
182  const char *name,
183  HyPerCol *hc,
184  InitWeights *weightInitializer,
185  NormalizeBase *weightNormalizer) {
186  // It is okay for either of weightInitializer or weightNormalizer to be null
187  // at this point,
188  // either because we're in a subclass that doesn't need it, or because we are
189  // allowing for
190  // backward compatibility.
191  // The two lines needs to be before the call to BaseConnection::initialize,
192  // because that function
193  // calls ioParamsFillGroup,
194  // which will call ioParam_weightInitType and ioParam_normalizeMethod, which
195  // for reasons of
196  // backward compatibility
197  // will create the initializer and normalizer if those member variables are
198  // null.
199  this->weightInitializer = weightInitializer;
200  this->normalizer = weightNormalizer;
201 
202  int status = BaseConnection::initialize(name, hc); // BaseConnection should *NOT* take
203 // weightInitializer or weightNormalizer as
204 // arguments, as it does not know about
205 // InitWeights or NormalizeBase
206 
207 #ifdef PV_USE_CUDA
208  if (needPostIndexLayer && receiveGpu) {
209  if (parent->getCommunicator()->commRank() == 0) {
210  Fatal() << getDescription() << ": receiveGpu and needPostIndexLayer both set. The GPU "
211  "version does not currently compute the post index "
212  "layer.";
213  }
214  MPI_Barrier(parent->getCommunicator()->communicator());
215  exit(EXIT_FAILURE);
216  }
217 #endif // PV_USE_CUDA
218 
219  assert(parent);
220  PVParams *inputParams = parent->parameters();
221 
222  // set accumulateFunctionPointer
223  assert(!inputParams->presentAndNotBeenRead(name, "pvpatchAccumulateType"));
224  switch (poolingType) {
225  case MAX:
226  accumulateFunctionPointer = &pvpatch_max_pooling;
227  accumulateFunctionFromPostPointer = &pvpatch_max_pooling_from_post;
228  break;
229  case SUM:
230  accumulateFunctionPointer = &pvpatch_sum_pooling;
231  accumulateFunctionFromPostPointer = &pvpatch_sumpooling_from_post;
232  break;
233  case AVG:
234  accumulateFunctionPointer = &pvpatch_sum_pooling;
235  accumulateFunctionFromPostPointer = &pvpatch_sumpooling_from_post;
236  break;
237  default:
238  pvAssert(0); // Only MAX, SUM, and AVG are defined in PoolingConn; other
239  // methods should be
240  // handled in other classes.
241  break;
242  }
243 
244  this->io_timer = new Timer(getName(), "conn", "io ");
245  this->update_timer = new Timer(getName(), "conn", "update ");
246 
247  return status;
248 }

250 int PoolingConn::communicateInitInfo(CommunicateInitInfoMessage const *message) {
251  int status = HyPerConn::communicateInitInfo(message);
252 
253  // Check pre/post connections here
254  const PVLayerLoc *preLoc = pre->getLayerLoc();
255  const PVLayerLoc *postLoc = post->getLayerLoc();
256 
257  if (preLoc->nf != postLoc->nf) {
258  Fatal() << "Pooling Layer " << name << ": preLayer " << pre->getName() << " nf of "
259  << preLoc->nf << " does not match postLayer " << post->getName() << " nf of "
260  << postLoc->nf << ". Features must match\n";
261  }
262 
263  float preToPostScaleX = (float)preLoc->nx / postLoc->nx;
264  float preToPostScaleY = (float)preLoc->ny / postLoc->ny;
265  if (preToPostScaleX < 1 || preToPostScaleY < 1) {
266  Fatal() << "Pooling Layer " << name << ": preLayer to postLayer must be "
267  "a many to one or one to one "
268  "conection\n";
269  }
270 
271  if (needPostIndexLayer) {
272  postIndexLayer = message->lookup<PoolingIndexLayer>(std::string(this->postIndexLayerName));
273  if (postIndexLayer == NULL) {
274  if (parent->columnId() == 0) {
275  ErrorLog().printf(
276  "%s: postIndexLayerName \"%s\" is not a PoolingIndexLayer.\n",
277  getDescription_c(),
278  this->postIndexLayerName);
279  }
280  MPI_Barrier(parent->getCommunicator()->communicator());
281  exit(EXIT_FAILURE);
282  }
283 
284  if (postIndexLayer->getDataType() != PV_INT) {
285  if (parent->columnId() == 0) {
286  ErrorLog().printf(
287  "%s: postIndexLayer \"%s\" must have data type "
288  "of int. Specify parameter "
289  "dataType in this layer to be \"int\".\n",
290  getDescription_c(),
291  this->postIndexLayerName);
292  }
293  MPI_Barrier(parent->getCommunicator()->communicator());
294  exit(EXIT_FAILURE);
295  }
296 
297  const PVLayerLoc *idxLoc = postIndexLayer->getLayerLoc();
298  // postIndexLayer must be the same size as the post layer
299  //(margins doesnt matter)
300  if (idxLoc->nxGlobal != postLoc->nxGlobal || idxLoc->nyGlobal != postLoc->nyGlobal
301  || idxLoc->nf != postLoc->nf) {
302  if (parent->columnId() == 0) {
303  ErrorLog().printf(
304  "%s: postIndexLayer \"%s\" must have the same "
305  "dimensions as the post pooling "
306  "layer \"%s\".\n",
307  getDescription_c(),
308  this->postIndexLayerName,
309  this->postLayerName);
310  }
311  MPI_Barrier(parent->getCommunicator()->communicator());
312  exit(EXIT_FAILURE);
313  }
314 
315  // TODO this is currently a hack, need to properly implement data types.
316  assert(sizeof(int) == sizeof(float));
317  }
318 
319  if (getUpdateGSynFromPostPerspective()) {
320  setNeedPost();
321  needAllocPostWeights = false;
322  }
323 
324  return status;
325 }

338 int PoolingConn::allocateDataStructures() {
339  if (postIndexLayer && postIndexLayer->getDataStructuresAllocatedFlag() == false) {
340  if (parent->columnId() == 0) {
341  InfoLog().printf(
342  "%s must wait until postIndexLayer layer \"%s\" has finished its "
343  "allocateDataStructures stage.\n",
344  getDescription_c(),
345  postIndexLayer->getName());
346  }
347  return PV_POSTPONE;
348  }
349  int status = HyPerConn::allocateDataStructures();
350  if (status == PV_POSTPONE) {
351  return status;
352  }
353  assert(status == PV_SUCCESS);
354 
355  if (needPostIndexLayer) {
356  // Allocate temp buffers if needed, 1 for each thread
357  if (parent->getNumThreads() > 1) {
358  thread_gateIdxBuffer = (float **)malloc(sizeof(int *) * parent->getNumThreads());
359  assert(thread_gateIdxBuffer);
360  // Assign thread_gSyn to different points of tempMem
361  for (int i = 0; i < parent->getNumThreads(); i++) {
362  float *thread_buffer = (float *)malloc(sizeof(float) * post->getNumNeurons());
363  if (!thread_buffer) {
364  Fatal().printf(
365  "HyPerLayer \"%s\" error: rank %d unable to "
366  "allocate %zu memory for "
367  "thread_gateIdxBuffer: %s\n",
368  name,
369  parent->columnId(),
370  sizeof(int) * post->getNumNeurons(),
371  strerror(errno));
372  }
373  thread_gateIdxBuffer[i] = thread_buffer;
374  }
375  }
376 
377  if (thread_gateIdxBuffer) {
378 #ifdef PV_USE_OPENMP_THREADS
379 #pragma omp parallel for
380 #endif
381  for (int i = 0; i < parent->getNumThreads() * post->getNumNeurons(); i++) {
382  int ti = i / post->getNumNeurons();
383  int ni = i % post->getNumNeurons();
384  thread_gateIdxBuffer[ti][ni] = -1;
385  }
386  }
387 
388  clearGateIdxBuffer();
389  }
390  return PV_SUCCESS;
391 }

442 int PoolingConn::constructWeights() {
443  int sx = nfp;
444  int sy = sx * nxp;
445  int sp = sy * nyp;
446  int nPatches = getNumDataPatches();
447  int status = PV_SUCCESS;
448 
449  assert(!parent->parameters()->presentAndNotBeenRead(name, "shrinkPatches"));
450  // createArbors() uses the value of shrinkPatches. It should have already
451  // been read in
452  // ioParamsFillGroup.
453  // allocate the arbor arrays:
454  createArbors();
455 
456  setPatchStrides();
457 
458  for (int arborId = 0; arborId < numAxonalArborLists; arborId++) {
459  PVPatch ***wPatches = get_wPatches();
460  status = createWeights(wPatches, arborId);
461  assert(wPatches[arborId] != NULL);
462  if (shrinkPatches_flag || arborId == 0) {
463  status |= adjustAxonalArbors(arborId);
464  }
465  } // arborId
466 
467  // call to initializeWeights moved to BaseConnection::initializeState()
468  status |= initPlasticityPatches();
469  assert(status == 0);
470  if (shrinkPatches_flag) {
471  for (int arborId = 0; arborId < numAxonalArborLists; arborId++) {
472  shrinkPatches(arborId);
473  }
474  }
475 
476  return status;
477 }

479 float PoolingConn::minWeight(int arborId) {
480  if (getPoolingType() == MAX) {
481  return 1.0;
482  }
483  else if (getPoolingType() == SUM) {

488  int relative_YScale = (int)pow(2, pre->getYScale() - post->getYScale());
489  return (1.0 / (nxp * nyp * relative_XScale * relative_YScale));
490  }
491  else {
492  assert(0); // only possibilities are PoolingConn::MAX, PoolingConn::SUM,
493  // PoolingConn::AVG

517 int PoolingConn::deliverPresynapticPerspective(PVLayerCube const *activity, int arborID) {
518 
519  // Check if we need to update based on connection's channel
520  if (getChannel() == CHANNEL_NOUPDATE) {
521  return PV_SUCCESS;
522  }
523  assert(post->getChannel(getChannel()));
524 
525  float dt_factor = getConvertToRateDeltaTimeFactor();
526 
527  const PVLayerLoc *preLoc = preSynapticLayer()->getLayerLoc();
528  const PVLayerLoc *postLoc = postSynapticLayer()->getLayerLoc();
529 
530  assert(arborID >= 0);
531  const int numExtended = activity->numItems;
532 
533  float resetVal = 0;
534  if (getPoolingType() == MAX) {
535  resetVal = -INFINITY;
536  float *gSyn = post->getChannel(getChannel());
537 // gSyn is res
538 #ifdef PV_USE_OPENMP_THREADS
539 #pragma omp parallel for
540 #endif
541  for (int i = 0; i < post->getNumNeuronsAllBatches(); i++) {
542  gSyn[i] = resetVal;
543  }
544  }
545 
546  clearGateIdxBuffer();
547 
548  for (int b = 0; b < parent->getNBatch(); b++) {
549  float *activityBatch = activity->data
550  + b * (preLoc->nx + preLoc->halo.rt + preLoc->halo.lt)
551  * (preLoc->ny + preLoc->halo.up + preLoc->halo.dn)
552  * preLoc->nf;
553  float *gSynPatchHeadBatch =
554  post->getChannel(getChannel()) + b * postLoc->nx * postLoc->ny * postLoc->nf;
555  float *gatePatchHeadBatch = NULL;
556  if (needPostIndexLayer) {
557  gatePatchHeadBatch =
558  postIndexLayer->getChannel(CHANNEL_EXC) + b * postIndexLayer->getNumNeurons();
559  }
560 
561  SparseList<float>::Entry const *activeIndicesBatch = NULL;
562  if (activity->isSparse) {
563  activeIndicesBatch = (SparseList<float>::Entry *)activity->activeIndices
564  + b * (preLoc->nx + preLoc->halo.rt + preLoc->halo.lt)
565  * (preLoc->ny + preLoc->halo.up + preLoc->halo.dn)
566  * preLoc->nf;
567  }
568  int numLoop;
569  if (activity->isSparse) {
570  numLoop = activity->numActive[b];
571  }
572  else {
573  numLoop = numExtended;
574  }
575 
576  if (thread_gateIdxBuffer) {
577 #ifdef PV_USE_OPENMP_THREADS
578 #pragma omp parallel for
579 #endif
580  for (int i = 0; i < parent->getNumThreads() * post->getNumNeurons(); i++) {
581  int ti = i / post->getNumNeurons();
582  int ni = i % post->getNumNeurons();
583  thread_gateIdxBuffer[ti][ni] = -1;
584  }
585  }
586 
587 #ifdef PV_USE_OPENMP_THREADS
588  // Clear all gsyn buffers
589  if (thread_gSyn) {
590  int numNeurons = post->getNumNeurons();
591 #ifdef PV_USE_OPENMP_THREADS
592 #pragma omp parallel for
593 #endif
594  for (int i = 0; i < parent->getNumThreads() * numNeurons; i++) {
595  int ti = i / numNeurons;
596  int ni = i % numNeurons;
597  thread_gSyn[ti][ni] = resetVal;
598  }
599  }
600 #endif // PV_USE_OPENMP_THREADS
601 
602 #ifdef PV_USE_OPENMP_THREADS
603 #pragma omp parallel for schedule(static)
604 #endif
605  for (int loopIndex = 0; loopIndex < numLoop; loopIndex++) {
606  int kPreExt;
607  float a = dt_factor;
608  if (activity->isSparse) {
609  kPreExt = activeIndicesBatch[loopIndex].index;
610  a *= activeIndicesBatch[loopIndex].value;
611  }
612  else {
613  kPreExt = loopIndex;
614  a *= activityBatch[kPreExt];
615  }
616 
617  // If we're using thread_gSyn, set this here
618  float *gSynPatchHead;
619  float *gatePatchHead = NULL;
620 #ifdef PV_USE_OPENMP_THREADS
621  if (thread_gSyn) {
622  int ti = omp_get_thread_num();
623  gSynPatchHead = thread_gSyn[ti];
624  }
625  else {
626  gSynPatchHead = gSynPatchHeadBatch;
627  }
628 
629  if (needPostIndexLayer) {
630  if (thread_gateIdxBuffer) {
631  int ti = omp_get_thread_num();
632  gatePatchHead = thread_gateIdxBuffer[ti];
633  }
634  else {
635  gatePatchHead = gatePatchHeadBatch;
636  }
637  }
638 #else // PV_USE_OPENMP_THREADS
639  gSynPatchHead = gSynPatchHeadBatch;
640  if (needPostIndexLayer) {
641  gatePatchHead = gatePatchHeadBatch;
642  }
643 #endif // PV_USE_OPENMP_THREADS
644 
645  PVPatch *weights = getWeights(kPreExt, arborID);
646  const int nk = weights->nx * fPatchSize();
647  const int ny = weights->ny;
648  const int sy = getPostNonextStrides()->sy; // stride in layer
649  float *weightDataStart = NULL;
650  float *postPatchStart = gSynPatchHead + getGSynPatchStart(kPreExt, arborID);
651  float *postGatePatchStart = gatePatchHead + getGSynPatchStart(kPreExt, arborID);
652 
653  const int kxPreExt =
654  kxPos(kPreExt,
655  preLoc->nx + preLoc->halo.lt + preLoc->halo.rt,
656  preLoc->ny + preLoc->halo.dn + preLoc->halo.up,
657  preLoc->nf);
658  const int kyPreExt =
659  kyPos(kPreExt,
660  preLoc->nx + preLoc->halo.lt + preLoc->halo.rt,
661  preLoc->ny + preLoc->halo.dn + preLoc->halo.up,
662  preLoc->nf);
663  const int kfPre = featureIndex(
664  kPreExt,
665  preLoc->nx + preLoc->halo.lt + preLoc->halo.rt,
