78 int privateTransposeConn::setPatchSize() {
79  // If postConn is many-to-one, the transpose connection is one-to-many; then xscaleDiff > 0.
80  // Similarly, if postConn is one-to-many, xscaleDiff < 0.
81 
82  // Some of the code duplication might be eliminated by adding some functions to convert.h
83 
84  assert(pre && post);
85  assert(postConn);
86 
87  int xscaleDiff = pre->getXScale() - post->getXScale();
88  int nxp_orig = postConn->xPatchSize();
89  int nyp_orig = postConn->yPatchSize();
90  nxp = nxp_orig;
91  if(xscaleDiff > 0 ) {
92  nxp *= (int) pow( 2, xscaleDiff );
93  }
94  else if(xscaleDiff < 0) {
95  nxp /= (int) pow(2,-xscaleDiff);
96  assert(nxp_orig==nxp*pow( 2, (float) (-xscaleDiff) ));
97  }
98 
99  int yscaleDiff = pre->getYScale() - post->getYScale();
100  nyp = nyp_orig;
101  if(yscaleDiff > 0 ) {
102  nyp *= (int) pow( 2, yscaleDiff );
103  }
104  else if(yscaleDiff < 0) {
105  nyp /= (int) pow(2,-yscaleDiff);
106  assert(nyp_orig==nyp*pow( 2, (float) (-yscaleDiff) ));
107  }
108 
109  nfp = post->getLayerLoc()->nf;
110  // post->getLayerLoc()->nf must be the same as postConn->preSynapticLayer()->getLayerLoc()->nf.
111  // This requirement is checked in communicateInitInfo
112 
113  //parent->parameters()->handleUnnecessaryParameter(name, "nxp", nxp);
114  //parent->parameters()->handleUnnecessaryParameter(name, "nyp", nyp);
115  //parent->parameters()->handleUnnecessaryParameter(name, "nfp", nfp);
116  return PV_SUCCESS;

443 int privateTransposeConn::mpiexchangesize(int neighbor, int * size, int * startx, int * stopx, int * starty, int * stopy, int * blocksize, size_t * buffersize) {
444  const PVLayerLoc * preLocOrig = postConn->preSynapticLayer()->getLayerLoc();
445  PVHalo const * halo = &preLocOrig->halo;
446  const int nx = preLocOrig->nx;
447  const int ny = preLocOrig->ny;
448  switch(neighbor) {
449  case LOCAL:
450  assert(0);
451  break;
452  case NORTHWEST:
453  *startx = 0; *stopx = halo->lt;
454  *starty = 0; *stopy = halo->up;
455  break;
456  case NORTH:
457  *startx = halo->lt; *stopx = halo->lt + nx;
458  *starty = 0; *stopy = halo->up;
459  break;
460  case NORTHEAST:
461  *startx = nx + halo->lt; *stopx = nx + halo->lt + halo->rt;
462  *starty = 0; *stopy = halo->up;
463  break;
464  case WEST:
465  *startx = 0; *stopx = halo->lt;
466  *starty = halo->up; *stopy = halo->up + ny;
467  break;
468  case EAST:
469  *startx = nx + halo->lt; *stopx = nx + halo->lt + halo->rt;
470  *starty = halo->up; *stopy = halo->up + ny;
471  break;
472  case SOUTHWEST:
473  *startx = 0; *stopx = halo->lt;
474  *starty = ny + halo->up; *stopy = ny + halo->dn + halo->up;
475  break;
476  case SOUTH:
477  *startx = halo->lt; *stopx = halo->lt + nx;
478  *starty = ny + halo->up; *stopy = ny + halo->dn + halo->up;
479  break;
480  case SOUTHEAST:
481  *startx = nx + halo->lt; *stopx = nx + halo->lt + halo->rt;

496 int privateTransposeConn::transposeSharedWeights(int arborId) {
497  // compute the transpose of postConn->kernelPatches and
498  // store into this->kernelPatches
499  // assume scale factors are 1 and that nxp, nyp are odd.
500 
501  int xscalediff = pre->getXScale()-post->getXScale();
502  int yscalediff = pre->getYScale()-post->getYScale();
503  // scalediff>0 means privateTransposeConn's post--that is, the postConn's pre--has a higher neuron density
504 
505  int numFBKernelPatches = getNumDataPatches();
506  int numFFKernelPatches = postConn->getNumDataPatches();
507 
508  if( xscalediff <= 0 && yscalediff <= 0) {
509  int xscaleq = (int) pow(2,-xscalediff);
510  int yscaleq = (int) pow(2,-yscalediff);
511 
512  int kerneloffsetx = 0;
513  int kerneloffsety = 0;
514  if(nxp%2 == 0){
515  kerneloffsetx = xscaleq/2;
516  }
517  if(nyp%2 == 0){
518  kerneloffsety = yscaleq/2;
519  }
520 
521  for( int kernelnumberFB = 0; kernelnumberFB < numFBKernelPatches; kernelnumberFB++ ) {
522  // PVPatch * kpFB = getKernelPatch(0, kernelnumberFB);
523  pvwdata_t * dataStartFB = get_wDataHead(arborId, kernelnumberFB);
524  int nfFB = nfp;
525  assert(numFFKernelPatches == nfFB);
526  int nxFB = nxp; // kpFB->nx;
527  int nyFB = nyp; // kpFB->ny;
528  for( int kyFB = 0; kyFB < nyFB; kyFB++ ) {
529  for( int kxFB = 0; kxFB < nxFB; kxFB++ ) {
530  for( int kfFB = 0; kfFB < nfFB; kfFB++ ) {
531  int kIndexFB = kIndex(kxFB,kyFB,kfFB,nxFB,nyFB,nfFB);
532  int kernelnumberFF = kfFB;
533  // PVPatch * kpFF = postConn>getKernelPatch(0, kernelnumberFF);
534  pvwdata_t * dataStartFF = postConn->get_wDataHead(arborId, kernelnumberFF);
535  int nxpFF = postConn->xPatchSize();
536  int nypFF = postConn->yPatchSize();
537  assert(numFBKernelPatches == postConn->fPatchSize() * xscaleq * yscaleq);
538  int kfFF = featureIndex(kernelnumberFB, xscaleq, yscaleq, postConn->fPatchSize());
539 
540  //Calculate x and y position of the FB kernel, with an offset for the case of even patches
541  int kxFFoffset = (kxPos(kernelnumberFB, xscaleq, yscaleq, postConn->fPatchSize()) + kerneloffsetx) % xscaleq;
542  int kxFF = (nxp - 1 - kxFB) * xscaleq + kxFFoffset;
543 
544  int kyFFoffset = (kyPos(kernelnumberFB, xscaleq, yscaleq, postConn->fPatchSize()) + kerneloffsety) % yscaleq;
545  int kyFF = (nyp - 1 - kyFB) * yscaleq + kyFFoffset;
546 
547  int kIndexFF = kIndex(kxFF, kyFF, kfFF, nxpFF, nypFF, postConn->fPatchSize());
548 
549  // can the calls to kxPos, kyPos, featureIndex be replaced by one call to patchIndexToKernelIndex?
550  dataStartFB[kIndexFB] = dataStartFF[kIndexFF];
551  // kpFB->data[kIndexFB] = kpFF->data[kIndexFF];
552  }
553  }
554  }
555  }
556  }
557  else if( xscalediff >= 0 && yscalediff >= 0) {
558  int xscaleq = (int) pow(2,xscalediff);
559  int yscaleq = (int) pow(2,yscalediff);
560 
561  int kerneloffsetx = 0;
562  int kerneloffsety = 0;
563  if((nxp/xscaleq)%2 == 0){
564  kerneloffsetx = xscaleq/2;
565  }
566  if((nyp/yscaleq)%2 == 0){
567  kerneloffsety = yscaleq/2;
568  }
569 
570  for( int kernelnumberFB = 0; kernelnumberFB < numFBKernelPatches; kernelnumberFB++ ) {
571  // PVPatch * kpFB = getKernelPatch(0, kernelnumberFB);
572  pvwdata_t * dataStartFB = get_wDataHead(arborId, kernelnumberFB);
573  int nxFB = nxp; // kpFB->nx;
574  int nyFB = nyp; // kpFB->ny;
575  int nfFB = nfp;
576  for( int kyFB = 0; kyFB < nyFB; kyFB++ ) {
577  int precelloffsety = (kyFB + kerneloffsety) % yscaleq;
578  for( int kxFB = 0; kxFB < nxFB; kxFB++ ) {
579  int precelloffsetx = (kxFB + kerneloffsetx) % xscaleq;
580  for( int kfFB = 0; kfFB < nfFB; kfFB++ ) {
581  int kernelnumberFF = (precelloffsety*xscaleq + precelloffsetx)*nfFB + kfFB;
582  pvwdata_t * dataStartFF = postConn->get_wDataHead(arborId, kernelnumberFF);
583  int nxpFF = postConn->xPatchSize();
584  int nypFF = postConn->yPatchSize();
585  int kxFF = (nxp-kxFB-1)/xscaleq;
586  assert(kxFF >= 0 && kxFF < postConn->xPatchSize());
587  int kyFF = (nyp-kyFB-1)/yscaleq;
588  assert(kyFF >= 0 && kyFF < postConn->yPatchSize());
589  int kfFF = kernelnumberFB;
590  assert(kfFF >= 0 && kfFF < postConn->fPatchSize());
591  int kIndexFF = kIndex(kxFF, kyFF, kfFF, nxpFF, nypFF, postConn->fPatchSize());
592  int kIndexFB = kIndex(kxFB, kyFB, kfFB, nxFB, nyFB, nfFB);
593  dataStartFB[kIndexFB] = dataStartFF[kIndexFF];
594  }
595  }
596  }
597  }
598  }
599  else {
600  fprintf(stderr,"xscalediff = %d, yscalediff = %d: the case of many-to-one in one dimension and one-to-many in the other"
601  "has not yet been implemented.\n", xscalediff, yscalediff);
602  exit(1);
603  }
604
