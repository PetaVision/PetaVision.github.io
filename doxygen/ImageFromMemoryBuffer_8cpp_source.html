58  int oldSize = imageLoc.nxGlobal * imageLoc.nyGlobal * imageLoc.nf;
59  int newSize = height * width * numbands;
60 
61 
62  if (parent->columnId()==0) {
63  if (oldSize != newSize) {
64  delete[] imageData;
65  imageData = new pvadata_t[newSize];
66  imageLoc.nxGlobal = width;
67  imageLoc.nyGlobal = height;
68  imageLoc.nf = numbands;
69  }
70 #ifdef PV_USE_OPENMP_THREADS
71 #pragma omp parallel for
72 #endif // PV_USE_OPENMP_THREADS
73  for (int k=0; k<newSize; k++) {

75  int y=kyPos(k, imageLoc.nxGlobal, imageLoc.nyGlobal, imageLoc.nf);
76  int f=featureIndex(k, imageLoc.nxGlobal, imageLoc.nyGlobal, imageLoc.nf);
77  int externalIndex = x*xstride + y*ystride + f*bandstride;
78  imageData[k] = pixelTypeConvert(externalBuffer[externalIndex], zeroval, oneval);
79  }
80  }
