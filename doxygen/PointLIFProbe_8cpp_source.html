62 int PointLIFProbe::calcValues(double timevalue) {
63  // TODO: Reduce duplicated code between PointProbe::calcValues and PointLIFProbe::calcValues.
64  assert(this->getNumValues()==NUMBER_OF_VALUES);
65  LIF * LIF_layer = dynamic_cast<LIF *>(getTargetLayer());
66  assert(LIF_layer != NULL);
67  pvconductance_t const * G_E = LIF_layer->getConductance(CHANNEL_EXC) + batchLoc * LIF_layer->getNumNeurons();
68  pvconductance_t const * G_I = LIF_layer->getConductance(CHANNEL_INH) + batchLoc * LIF_layer->getNumNeurons();
69  pvconductance_t const * G_IB = LIF_layer->getConductance(CHANNEL_INHB) + batchLoc * LIF_layer->getNumNeurons();
70  pvdata_t const * V = getTargetLayer()->getV();
71  pvdata_t const * Vth = LIF_layer->getVth();
72  pvdata_t const * activity = getTargetLayer()->getLayerData();
73  assert(V && activity && G_E && G_I && G_IB && Vth);
74  double * valuesBuffer = this->getValuesBuffer();
75  //We need to calculate which mpi process contains the target point, and send that info to the root process
76  //Each process calculates local index
77  const PVLayerLoc * loc = getTargetLayer()->getLayerLoc();
78  //Calculate local cords from global
79  const int kx0 = loc->kx0;
80  const int ky0 = loc->ky0;
81  const int kb0 = loc->kb0;
82  const int nx = loc->nx;
83  const int ny = loc->ny;
84  const int nf = loc->nf;
85  const int nbatch = loc->nbatch;
86  const int xLocLocal = xLoc - kx0;
87  const int yLocLocal = yLoc - ky0;
88  const int nbatchLocal = batchLoc - kb0;
89 
90  //if in bounds
91  if( xLocLocal >= 0 && xLocLocal < nx &&
92  yLocLocal >= 0 && yLocLocal < ny &&
93  nbatchLocal >= 0 && nbatchLocal < nbatch){
94  const pvdata_t * V = getTargetLayer()->getV();
95  const pvdata_t * activity = getTargetLayer()->getLayerData();
96  //Send V and A to root
97  const int k = kIndex(xLocLocal, yLocLocal, fLoc, nx, ny, nf);
98  const int kbatch = k + nbatchLocal*getTargetLayer()->getNumNeurons();
99  valuesBuffer[0] = G_E[kbatch];
100  valuesBuffer[1] = G_I[kbatch];
101  valuesBuffer[2] = G_IB[kbatch];
102  valuesBuffer[3] = V[kbatch];
103  valuesBuffer[4] = Vth[kbatch];
104  const int kex = kIndexExtended(k, nx, ny, nf, loc->halo.lt, loc->halo.rt, loc->halo.dn, loc->halo.up);
105  valuesBuffer[5] = activity[kex + nbatchLocal * getTargetLayer()->getNumExtended()];
106  //If not in root process, send to root process
107  if(parent->columnId()!=0){
108  MPI_Send(&valuesBuffer, NUMBER_OF_VALUES, MPI_DOUBLE, 0, 0, parent->icCommunicator()->communicator());
109  }
110  }
111 
112  //Root process
113  if(parent->columnId()==0){
114  //Calculate which rank target neuron is
115  //TODO we need to calculate rank from batch as well
116  int xRank = xLoc/nx;
117  int yRank = yLoc/ny;
118 
119  int srcRank = rankFromRowAndColumn(yRank, xRank, parent->icCommunicator()->numCommRows(), parent->icCommunicator()->numCommColumns());
120 
121  //If srcRank is not root process, MPI_Recv from that rank
122  if(srcRank != 0){
123  MPI_Recv(&valuesBuffer, NUMBER_OF_VALUES, MPI_DOUBLE, srcRank, 0, parent->icCommunicator()->communicator(), MPI_STATUS_IGNORE);
124  }
125  }
126  return PV_SUCCESS;
127 }
